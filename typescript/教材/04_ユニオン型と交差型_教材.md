# 第4章: ユニオン型と交差型

## 4.1 ユニオン型（Union Types）

ユニオン型は「AまたはB」を表す型です。`|`（パイプ）で複数の型を結合します。

```typescript
// 基本的なユニオン型
let id: string | number;
id = "abc-123";  // OK
id = 456;         // OK
// id = true;     // エラー！booleanは含まれない

// 関数パラメータでのユニオン型
function printId(id: string | number): void {
  console.log(`ID: ${id}`);
}

printId("abc");
printId(123);

// ユニオン型の変数を使う場合、共通のメソッドのみ安全に呼べる
function getLength(value: string | number): number {
  // value.toUpperCase(); // エラー！numberにはtoUpperCaseがない
  return value.toString().length; // toString()は両方にある
}
```

## 4.2 型の絞り込み（Type Narrowing）

ユニオン型の値を安全に使うには、型を絞り込む必要があります。

### typeof による絞り込み

```typescript
function formatValue(value: string | number | boolean): string {
  if (typeof value === "string") {
    // ここではvalueはstring型
    return value.toUpperCase();
  } else if (typeof value === "number") {
    // ここではvalueはnumber型
    return value.toFixed(2);
  } else {
    // ここではvalueはboolean型
    return value ? "はい" : "いいえ";
  }
}

formatValue("hello");  // "HELLO"
formatValue(3.14);      // "3.14"
formatValue(true);      // "はい"
```

### instanceof による絞り込み

```typescript
function getDateInfo(value: string | Date): string {
  if (value instanceof Date) {
    // ここではvalueはDate型
    return value.toISOString();
  } else {
    // ここではvalueはstring型
    return new Date(value).toISOString();
  }
}

class Dog {
  bark() { return "ワン！"; }
}

class Cat {
  meow() { return "ニャー！"; }
}

function makeSound(animal: Dog | Cat): string {
  if (animal instanceof Dog) {
    return animal.bark();
  } else {
    return animal.meow();
  }
}
```

### in 演算子による絞り込み

```typescript
interface Fish {
  swim: () => void;
  name: string;
}

interface Bird {
  fly: () => void;
  name: string;
}

function move(animal: Fish | Bird): void {
  if ("swim" in animal) {
    // animalはFish型
    animal.swim();
  } else {
    // animalはBird型
    animal.fly();
  }
}
```

## 4.3 リテラル型（Literal Types）

リテラル型は、特定の値だけを許容する型です。

```typescript
// 文字列リテラル型
type Direction = "north" | "south" | "east" | "west";

let dir: Direction = "north"; // OK
// dir = "up"; // エラー！"up"はDirection型に含まれない

// 数値リテラル型
type DiceValue = 1 | 2 | 3 | 4 | 5 | 6;

let dice: DiceValue = 3; // OK
// dice = 7; // エラー！

// booleanリテラル型
type True = true;
type False = false;

// テンプレートリテラル型
type EventName = `on${string}`;
let event: EventName = "onClick";   // OK
let event2: EventName = "onSubmit"; // OK
// let event3: EventName = "click"; // エラー！"on"で始まっていない

// HTTPメソッド
type HttpMethod = "GET" | "POST" | "PUT" | "DELETE" | "PATCH";

function request(url: string, method: HttpMethod): void {
  // ...
}

request("/api/users", "GET");    // OK
// request("/api/users", "GETS"); // エラー！
```

## 4.4 判別共用体（Discriminated Unions）

判別共用体は、共通の「判別プロパティ（discriminant）」を持つユニオン型です。TypeScriptの最も強力なパターンの一つです。

```typescript
// 各型に共通のkindプロパティ（判別プロパティ）を持たせる
interface Circle {
  kind: "circle";
  radius: number;
}

interface Rectangle {
  kind: "rectangle";
  width: number;
  height: number;
}

interface Triangle {
  kind: "triangle";
  base: number;
  height: number;
}

type Shape = Circle | Rectangle | Triangle;

// kindプロパティで型を絞り込める
function getArea(shape: Shape): number {
  switch (shape.kind) {
    case "circle":
      // shapeはCircle型
      return Math.PI * shape.radius ** 2;
    case "rectangle":
      // shapeはRectangle型
      return shape.width * shape.height;
    case "triangle":
      // shapeはTriangle型
      return (shape.base * shape.height) / 2;
  }
}

// 使用例
const circle: Circle = { kind: "circle", radius: 5 };
const rect: Rectangle = { kind: "rectangle", width: 10, height: 20 };
const tri: Triangle = { kind: "triangle", base: 6, height: 8 };

console.log(getArea(circle)); // 78.54
console.log(getArea(rect));   // 200
console.log(getArea(tri));    // 24
```

### 実践例: APIレスポンスの型

```typescript
// 成功と失敗を型で表現
interface SuccessResponse<T> {
  status: "success";
  data: T;
}

interface ErrorResponse {
  status: "error";
  error: {
    code: number;
    message: string;
  };
}

interface LoadingState {
  status: "loading";
}

type ApiState<T> = SuccessResponse<T> | ErrorResponse | LoadingState;

// 型安全にレスポンスを処理
function handleResponse<T>(state: ApiState<T>): string {
  switch (state.status) {
    case "success":
      return `データ: ${JSON.stringify(state.data)}`;
    case "error":
      return `エラー ${state.error.code}: ${state.error.message}`;
    case "loading":
      return "読み込み中...";
  }
}
```

### 実践例: Reduxスタイルのアクション

```typescript
interface AddTodoAction {
  type: "ADD_TODO";
  payload: {
    id: number;
    text: string;
  };
}

interface ToggleTodoAction {
  type: "TOGGLE_TODO";
  payload: {
    id: number;
  };
}

interface DeleteTodoAction {
  type: "DELETE_TODO";
  payload: {
    id: number;
  };
}

type TodoAction = AddTodoAction | ToggleTodoAction | DeleteTodoAction;

interface Todo {
  id: number;
  text: string;
  completed: boolean;
}

function todoReducer(state: Todo[], action: TodoAction): Todo[] {
  switch (action.type) {
    case "ADD_TODO":
      return [...state, {
        id: action.payload.id,
        text: action.payload.text,
        completed: false,
      }];
    case "TOGGLE_TODO":
      return state.map((todo) =>
        todo.id === action.payload.id
          ? { ...todo, completed: !todo.completed }
          : todo
      );
    case "DELETE_TODO":
      return state.filter((todo) => todo.id !== action.payload.id);
  }
}
```

## 4.5 交差型（Intersection Types）

交差型は「AかつB」を表す型です。`&` で複数の型を結合します。

```typescript
// 基本的な交差型
interface HasName {
  name: string;
}

interface HasAge {
  age: number;
}

interface HasEmail {
  email: string;
}

// 3つの型を全て満たす型
type Person = HasName & HasAge & HasEmail;

const person: Person = {
  name: "太郎",
  age: 25,
  email: "taro@example.com",
  // 3つのプロパティが全て必要
};
```

### ミックスイン（Mixin）パターン

```typescript
// タイムスタンプを追加
type Timestamped<T> = T & {
  createdAt: Date;
  updatedAt: Date;
};

// ソフトデリートを追加
type SoftDeletable<T> = T & {
  deletedAt: Date | null;
  isDeleted: boolean;
};

// 基本のユーザー型
interface BaseUser {
  id: number;
  name: string;
  email: string;
}

// タイムスタンプ付きユーザー
type User = Timestamped<BaseUser>;

// ソフトデリート可能なタイムスタンプ付きユーザー
type DeletableUser = SoftDeletable<Timestamped<BaseUser>>;

const user: DeletableUser = {
  id: 1,
  name: "太郎",
  email: "taro@example.com",
  createdAt: new Date(),
  updatedAt: new Date(),
  deletedAt: null,
  isDeleted: false,
};
```

## 4.6 ユニオン型と交差型の関係

```typescript
// ユニオン型: AまたはB → プロパティはAとBの共通部分のみ安全に使える
type AB = { a: string; common: string } | { b: number; common: string };
function useAB(value: AB) {
  value.common; // OK: 共通プロパティ
  // value.a;   // エラー！aがない場合がある
}

// 交差型: AかつB → プロパティはAとBの全てが使える
type CD = { c: string; common: string } & { d: number; common: string };
function useCD(value: CD) {
  value.c;      // OK
  value.d;      // OK
  value.common; // OK
}

// 矛盾する交差型はnever型になる
type Impossible = string & number; // never型
```

## 4.7 型ガードのパターン集

```typescript
// 1. typeof ガード
function isString(value: unknown): boolean {
  return typeof value === "string";
}

// 2. instanceof ガード
function isDate(value: unknown): boolean {
  return value instanceof Date;
}

// 3. in ガード
function hasProp(obj: object, key: string): boolean {
  return key in obj;
}

// 4. 等値チェック
function handle(value: "a" | "b" | "c") {
  if (value === "a") {
    // valueは"a"型
  }
}

// 5. 真偽値チェック（truthiness narrowing）
function printName(name: string | null | undefined) {
  if (name) {
    // nameはstring型（null, undefined, ""は除外）
    console.log(name.toUpperCase());
  }
}

// 6. Array.isArray
function processArray(value: string | string[]) {
  if (Array.isArray(value)) {
    // valueはstring[]型
    value.forEach((s) => console.log(s));
  } else {
    // valueはstring型
    console.log(value);
  }
}
```

## 4.8 網羅性チェック（Exhaustive Check）

switch文で全てのケースを処理したことを型レベルで保証できます。

```typescript
type Color = "red" | "green" | "blue";

function colorToHex(color: Color): string {
  switch (color) {
    case "red":
      return "#FF0000";
    case "green":
      return "#00FF00";
    case "blue":
      return "#0000FF";
    default:
      // 全てのケースを処理していれば、ここはnever型
      const _exhaustive: never = color;
      return _exhaustive;
  }
}

// 新しい色を追加した場合
type Color2 = "red" | "green" | "blue" | "yellow";

function colorToHex2(color: Color2): string {
  switch (color) {
    case "red":
      return "#FF0000";
    case "green":
      return "#00FF00";
    case "blue":
      return "#0000FF";
    // "yellow" のケースがない！
    default:
      // エラー！型 'string' を型 'never' に割り当てることはできません
      // const _exhaustive: never = color;
      // return _exhaustive;
      return ""; // これだと型安全性が失われる
  }
}

// ヘルパー関数を作ると便利
function assertNever(value: never, message?: string): never {
  throw new Error(message ?? `Unexpected value: ${value}`);
}
```

## 4.9 実践パターン: Result型

Rust言語のResult型に着想を得た、エラーハンドリングのパターンです。

```typescript
// 成功と失敗を型で表現
type Result<T, E = Error> =
  | { ok: true; value: T }
  | { ok: false; error: E };

// ヘルパー関数
function ok<T>(value: T): Result<T, never> {
  return { ok: true, value };
}

function err<E>(error: E): Result<never, E> {
  return { ok: false, error };
}

// 使用例：パース関数
function parseJSON(input: string): Result<unknown, string> {
  try {
    return ok(JSON.parse(input));
  } catch (e) {
    return err(`JSON解析エラー: ${(e as Error).message}`);
  }
}

// 使用例
const result = parseJSON('{"name": "太郎"}');

if (result.ok) {
  console.log("成功:", result.value);
} else {
  console.log("失敗:", result.error);
}

// チェイン可能にする
function map<T, U, E>(
  result: Result<T, E>,
  fn: (value: T) => U
): Result<U, E> {
  if (result.ok) {
    return ok(fn(result.value));
  }
  return result;
}
```

## 4.10 まとめ

| トピック | 内容 |
|---------|------|
| ユニオン型 (`\|`) | 「AまたはB」を表す型 |
