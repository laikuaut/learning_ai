## 読解対象コード

```python
import os
from datetime import datetime


# ==================== ステージ1: データ取得 ====================

def generate_sample_log(filename):
    """サンプルのアクセスログを生成する"""
    log_entries = [
        "2024-03-01 09:15:23 GET /index.html 200 1234",
        "2024-03-01 09:15:45 GET /about.html 200 2345",
        "2024-03-01 09:16:01 POST /api/login 200 567",
        "2024-03-01 09:16:15 GET /index.html 200 1234",
        "2024-03-01 09:17:00 GET /products.html 404 0",
        "2024-03-01 10:00:12 GET /index.html 200 1234",
        "2024-03-01 10:05:30 POST /api/login 401 234",
        "2024-03-01 10:05:45 POST /api/login 401 234",
        "2024-03-01 10:06:00 POST /api/login 200 567",
        "2024-03-01 11:00:00 GET /about.html 200 2345",
        "2024-03-01 11:30:00 GET /dashboard.html 500 0",
        "2024-03-01 11:30:05 GET /dashboard.html 500 0",
        "2024-03-01 14:00:00 GET /index.html 200 1234",
        "2024-03-01 14:15:00 DELETE /api/users/5 403 0",
        "2024-03-01 15:00:00 GET /products.html 200 3456",
    ]
    with open(filename, "w") as f:
        f.write("\n".join(log_entries))
    return filename


# ==================== ステージ2: パース ====================

def parse_log_line(line):
    """1行のログをパースして辞書にする"""
    parts = line.strip().split()
    if len(parts) != 6:
        return None
    return {
        "date": parts[0],
        "time": parts[1],
        "method": parts[2],
        "path": parts[3],
        "status": int(parts[4]),
        "size": int(parts[5]),
    }


def load_and_parse(filename):
    """ファイルを読み込み、全行をパースする"""
    with open(filename, "r") as f:
        lines = f.readlines()
    return [entry for entry in (parse_log_line(line) for line in lines) if entry]


# ==================== ステージ3: フィルタ・変換 ====================

def filter_by_status(entries, status_range):
    """ステータスコードの範囲でフィルタする"""
    low, high = status_range
    return [e for e in entries if low <= e["status"] <= high]


def add_hour_field(entries):
    """各エントリに 'hour' フィールドを追加する"""
    for entry in entries:
        entry["hour"] = int(entry["time"].split(":")[0])
    return entries


def classify_status(entries):
    """ステータスコードを分類ラベルに変換する"""
    labels = {2: "success", 3: "redirect", 4: "client_error", 5: "server_error"}
    for entry in entries:
        category = entry["status"] // 100
        entry["status_label"] = labels.get(category, "unknown")
    return entries


# ==================== ステージ4: 集計 ====================

def count_by_key(entries, key):
    """指定キーでグループ化してカウントする"""
    counts = {}
    for entry in entries:
        value = entry[key]
        counts[value] = counts.get(value, 0) + 1
    return counts


def hourly_traffic(entries):
    """時間帯ごとのアクセス数を集計する"""
    return count_by_key(entries, "hour")


def top_paths(entries, n=5):
    """アクセス数の多いパスをランキングする"""
    path_counts = count_by_key(entries, "path")
    sorted_paths = sorted(path_counts.items(), key=lambda x: x[1], reverse=True)
    return sorted_paths[:n]


def error_summary(entries):
    """エラー（4xx, 5xx）の内訳をまとめる"""
    errors = [e for e in entries if e["status"] >= 400]
    summary = {}
    for e in errors:
        key = f"{e['status']} {e['method']} {e['path']}"
        summary[key] = summary.get(key, 0) + 1
    return dict(sorted(summary.items(), key=lambda x: x[1], reverse=True))


# ==================== ステージ5: レポート生成 ====================

def generate_report(entries):
    """全集計結果をまとめたレポート辞書を返す"""
    entries = add_hour_field(entries)
    entries = classify_status(entries)

    status_counts = count_by_key(entries, "status_label")
    total = len(entries)

    return {
        "total_requests": total,
        "status_breakdown": {
            label: {"count": count, "percent": round(count / total * 100, 1)}
            for label, count in status_counts.items()
        },
        "hourly_traffic": dict(sorted(hourly_traffic(entries).items())),
        "top_paths": top_paths(entries),
        "error_details": error_summary(entries),
    }


def print_report(report):
    """レポートをコンソールに出力する"""
    print(f"\n{'=' * 50}")
    print(f"  アクセスログ解析レポート")
    print(f"  総リクエスト数: {report['total_requests']}")
    print(f"{'=' * 50}")

    print("\n■ ステータス内訳")
    for label, data in report["status_breakdown"].items():
        print(f"  {label:<14}: {data['count']:>3}件 ({data['percent']}%)")

    print("\n■ 時間帯別アクセス数")
    for hour, count in report["hourly_traffic"].items():
        bar = "█" * count
        print(f"  {hour:02d}時: {count:>2}件 {bar}")

    print("\n■ アクセス数 TOP5 パス")
    for i, (path, count) in enumerate(report["top_paths"], 1):
        print(f"  {i}. {path:<20} {count}件")

    print("\n■ エラー詳細")
    for key, count in report["error_details"].items():
        print(f"  {key}: {count}件")


# ==================== パイプライン実行 ====================

def run_pipeline():
    """パイプライン全体を実行する"""
    filename = "access.log"

    # ステージ1: データ取得
    generate_sample_log(filename)

    # ステージ2: パース
    entries = load_and_parse(filename)

    # ステージ3: 変換
    entries = classify_status(add_hour_field(entries))

    # ステージ4 & 5: 集計 & レポート
    report = generate_report(entries)
    print_report(report)

    # クリーンアップ
    os.remove(filename)


if __name__ == "__main__":
    run_pipeline()
```

<summary>設問1〜5の解答</summary>

### 設問1
ログ行を空白で分割した結果が **6要素でないとき** に `None` を返します。
理由は、正しいログ形式（日付・時刻・メソッド・パス・ステータス・サイズ）でないデータを除外するためです。空行や不正なフォーマットの行を安全にスキップできます。

### 設問2
**ジェネレータ式** と呼ばれます。リスト内包表記 `[...]` はすべての結果を一度にメモリに保持しますが、ジェネレータ式 `(...)` は1要素ずつ遅延生成します。外側のリスト内包表記で `if entry` フィルタと組み合わせることで、`None` を除外しつつメモリ効率を保てます。

### 設問4
`401 // 100 = 4` → `labels` 辞書のキー `4` に対応 → `"client_error"` というラベルが付きます。
整数除算 `//` でステータスコードの百の位を取り出し、それをキーにしてラベルを引いています。

### 設問5
呼び出し箇所：
1. `hourly_traffic` → キー `"hour"` で時間帯別カウント
2. `top_paths` → キー `"path"` でパス別カウント
3. `generate_report` → キー `"status_label"` でステータス別カウント

汎用化の理由は、「辞書のリストを特定キーで集計する」という処理パターンが共通しているからです。キー名を引数にすることで、同じロジックを3回書かずに済みます。

### 設問6
**パイプライン設計のメリット：**
- **関心の分離**: 各ステージが独立 → 理解・テスト・修正が容易
- **再利用性**: フィルタ関数や集計関数を別の分析に流用できる
- **拡張性**: 新しいステージを追加するとき、他のステージを変更不要
- **デバッグ容易性**: 各ステージの出力を確認できる

### 設問7
**意図的ではありません（バグです）。**

`run_pipeline` の中で：
```python
entries = classify_status(add_hour_field(entries))  # 1回目
report = generate_report(entries)  # generate_report内で2回目
```

`generate_report` の冒頭でも `add_hour_field` と `classify_status` を呼んでいます。
結果として `hour` と `status_label` が2回セットされます。この場合は同じ値が上書きされるだけなので結果は変わりませんが、無駄な処理です。

修正方法：`run_pipeline` で変換を行うか、`generate_report` で行うかに統一すべきです。

### 設問9

```python
def error_summary(entries):
    errors = [e for e in entries if e["status_label"] in ("client_error", "server_error")]
    summary = {}
    for e in errors:
        key = f"{e['status']} {e['method']} {e['path']}"
        summary[key] = summary.get(key, 0) + 1

```

`status_label` を使うことで、マジックナンバー `400` が消え、ラベルの定義を一箇所（`classify_status`）で管理できます。

### 設問10
**ステージ4（集計）に追加** するのが適切です。

```python
def detect_rapid_access(entries, path, threshold_minutes=1, min_count=3):
    """
    指定パスへの連続アクセス（threshold_minutes以内にmin_count回以上）を検出する

    Args:
        entries: パース済みエントリのリスト（hour付き）
        path: 対象パス（Noneの場合は全パスを対象）
        threshold_minutes: 連続とみなす時間の閾値（分）
        min_count: 連続とみなす最小アクセス回数

    Returns:
        検出された連続アクセスのリスト [{"path": ..., "times": [...], "count": ...}]
    """
```

理由：フィルタ・変換（ステージ3）はデータの加工で、集計（ステージ4）はデータの分析です。連続アクセスの検出は分析に該当するため、ステージ4が適切です。
