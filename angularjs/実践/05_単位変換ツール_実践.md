# 実践課題05：単位変換ツール ★2

> **難易度**: ★★☆☆☆（基礎）
> **前提知識**: 第2章（データバインディングとスコープ）、第4章（フィルターとサービス）
> **課題の種類**: ミニプロジェクト
> **学習目標**: カスタムサービス（`factory`）を使ったロジックの分離と依存性注入（Dependency Injection / DI）の仕組みを体感する

---

## 完成イメージ

複数種類の単位変換ができるツールです。変換ロジックはサービスに分離します。

```
┌──────────────────────────────────────┐
│  単位変換ツール                        │
│                                      │
│  変換タイプ: [長さ ▼]                  │
│                                      │
│  ┌─────────────────────────────────┐ │
│  │  [100    ] [メートル ▼]          │ │
│  │      ↕ 双方向変換                 │ │
│  │  [328.084] [フィート ▼]          │ │
│  └─────────────────────────────────┘ │
│                                      │
│  変換式: 1 メートル = 3.28084 フィート  │
│                                      │
│  履歴:                               │
│  ・100 メートル → 328.084 フィート     │
│  ・72 キログラム → 158.733 ポンド      │
└──────────────────────────────────────┘
```

---

## 課題の要件

1. 長さ・重さ・温度の3種類の変換に対応する
2. 変換ロジックは `factory` サービスとして分離する
3. コントローラーはサービスを DI（依存性注入）で受け取る
4. 入力値の変更でリアルタイムに変換結果が更新される
5. 変換の単位を双方向（例：メートル→フィート / フィート→メートル）で切り替え可能にする
6. 変換履歴を表示する

---

## ステップガイド

<details>
<summary>ステップ1：変換サービスを作る</summary>

```javascript
app.factory('ConvertService', function() {
  var conversions = {
    length: {
      units: ['メートル', 'フィート', 'インチ', 'センチメートル'],
      convert: function(value, from, to) {
        // まずメートルに変換してから目的の単位に変換
      }
    }
  };
  return conversions;
});
```

</details>

<details>
<summary>ステップ2：コントローラーでサービスを注入する</summary>

```javascript
app.controller('ConvertCtrl', function($scope, ConvertService) {
  $scope.result = ConvertService.length.convert(100, 'メートル', 'フィート');
});
```

</details>

<details>
<summary>ステップ3：$scope.$watch で入力値の変化を監視する</summary>

```javascript
$scope.$watch('inputValue', function(newVal) {
  if (newVal !== undefined && newVal !== '') {
    $scope.outputValue = ConvertService[$scope.type].convert(
      parseFloat(newVal), $scope.fromUnit, $scope.toUnit
    );
  }
});
```

    var app = angular.module('convertApp', []);

    // 変換サービス
    app.factory('ConvertService', function() {
      // 基準単位への変換レート
      var rates = {
        length: {
          'メートル': 1,
          'キロメートル': 1000,
          'センチメートル': 0.01,
          'フィート': 0.3048,
          'インチ': 0.0254
        },
        weight: {
          'キログラム': 1,
          'グラム': 0.001,
          'ポンド': 0.453592,
          'オンス': 0.0283495
        }
      };

      return {
        getUnits: function(type) {
          if (type === 'temperature') {
            return ['摂氏(℃)', '華氏(℉)', 'ケルビン(K)'];
          }
          return Object.keys(rates[type]);
        },
        convert: function(type, value, from, to) {
          if (type === 'temperature') {
            return this.convertTemperature(value, from, to);
          }
          // 基準単位経由で変換
          var baseValue = value * rates[type][from];
          return baseValue / rates[type][to];
        },
        convertTemperature: function(value, from, to) {
          // まず摂氏に変換
          var celsius;
          if (from === '摂氏(℃)') celsius = value;
          else if (from === '華氏(℉)') celsius = (value - 32) * 5 / 9;
          else celsius = value - 273.15; // ケルビン

          // 摂氏から目的の単位に変換
          if (to === '摂氏(℃)') return celsius;
          if (to === '華氏(℉)') return celsius * 9 / 5 + 32;
          return celsius + 273.15; // ケルビン
        }
      };
    });

    app.controller('ConvertCtrl', function($scope, ConvertService) {
      $scope.types = [
        { key: 'length', label: '長さ' },
        { key: 'weight', label: '重さ' },
        { key: 'temperature', label: '温度' }
      ];
      $scope.selectedType = 'length';
      $scope.currentUnits = ConvertService.getUnits('length');
      $scope.fromUnit = $scope.currentUnits[0];
      $scope.toUnit = $scope.currentUnits[1];
      $scope.inputValue = 1;
      $scope.outputValue = 0;
      $scope.formula = '';
      $scope.history = [];

      $scope.changeType = function(type) {
        $scope.selectedType = type;
        $scope.currentUnits = ConvertService.getUnits(type);
        $scope.fromUnit = $scope.currentUnits[0];
        $scope.toUnit = $scope.currentUnits[1];
        $scope.doConvert();
      };

      $scope.doConvert = function() {
        if ($scope.inputValue === undefined || $scope.inputValue === '') return;
        var result = ConvertService.convert(
          $scope.selectedType, parseFloat($scope.inputValue), $scope.fromUnit, $scope.toUnit
        );
        $scope.outputValue = Math.round(result * 10000) / 10000;
        $scope.updateFormula();
      };

      $scope.doReverseConvert = function() {
        if ($scope.outputValue === undefined || $scope.outputValue === '') return;
        var result = ConvertService.convert(
          $scope.selectedType, parseFloat($scope.outputValue), $scope.toUnit, $scope.fromUnit
        );
        $scope.inputValue = Math.round(result * 10000) / 10000;
        $scope.updateFormula();
      };

      $scope.updateFormula = function() {
        var oneUnit = ConvertService.convert($scope.selectedType, 1, $scope.fromUnit, $scope.toUnit);
        $scope.formula = '1 ' + $scope.fromUnit + ' = ' + (Math.round(oneUnit * 10000) / 10000) + ' ' + $scope.toUnit;

        // 履歴に追加
        var entry = $scope.inputValue + ' ' + $scope.fromUnit + ' → ' + $scope.outputValue + ' ' + $scope.toUnit;
        if ($scope.history[0] !== entry) {
          $scope.history.unshift(entry);
          if ($scope.history.length > 10) {
            $scope.history.pop();
          }
        }
      };

      // 初期変換
      $scope.doConvert();
    });
  </script>

</body>

  <script>
    var app = angular.module('convertApp', []);

    // service メソッドで定義（new でインスタンス化される）
    app.service('UnitConverter', function() {
      var self = this;

      var rates = {
        length: { 'メートル': 1, 'キロメートル': 1000, 'センチメートル': 0.01, 'ミリメートル': 0.001, 'フィート': 0.3048, 'インチ': 0.0254 },
        weight: { 'キログラム': 1, 'グラム': 0.001, 'ミリグラム': 0.000001, 'ポンド': 0.453592, 'オンス': 0.0283495 },
        area: { '平方メートル': 1, '平方キロメートル': 1000000, 'ヘクタール': 10000, '坪': 3.30579, 'エーカー': 4046.86 }
      };

      self.getUnits = function(type) {
        if (type === 'temperature') return ['摂氏(℃)', '華氏(℉)', 'ケルビン(K)'];
        return Object.keys(rates[type] || {});
      };

      self.convert = function(type, value, from, to) {
        if (isNaN(value)) return 0;
        if (type === 'temperature') return self._convertTemp(value, from, to);
        var base = value * (rates[type][from] || 1);
        return base / (rates[type][to] || 1);
      };

      self._convertTemp = function(value, from, to) {
        var c;
        if (from === '摂氏(℃)') c = value;
        else if (from === '華氏(℉)') c = (value - 32) * 5 / 9;
        else c = value - 273.15;

        if (to === '摂氏(℃)') return c;
        if (to === '華氏(℉)') return c * 9 / 5 + 32;
        return c + 273.15;
      };

      self.round = function(v, digits) {
        var f = Math.pow(10, digits || 4);
        return Math.round(v * f) / f;
      };
    });

    app.controller('ConvertCtrl', ['UnitConverter', function(UnitConverter) {
      var vm = this;

      vm.types = [
        { key: 'length', label: '長さ' },
        { key: 'weight', label: '重さ' },
        { key: 'temperature', label: '温度' },
        { key: 'area', label: '面積' }
      ];
      vm.type = 'length';
      vm.units = UnitConverter.getUnits('length');
      vm.from = vm.units[0];
      vm.to = vm.units[1];
      vm.input = 1;
      vm.output = 0;
      vm.history = [];
      vm.favorites = [];

      vm.setType = function(type) {
        vm.type = type;
        vm.units = UnitConverter.getUnits(type);
        vm.from = vm.units[0];
        vm.to = vm.units[1];
        vm.calc();
      };

      vm.calc = function() {
        vm.output = UnitConverter.round(UnitConverter.convert(vm.type, vm.input, vm.from, vm.to));
        vm.addHistory();
      };

      vm.calcReverse = function() {
        vm.input = UnitConverter.round(UnitConverter.convert(vm.type, vm.output, vm.to, vm.from));
        vm.addHistory();
      };

      vm.swapUnits = function() {
        var tmp = vm.from;
        vm.from = vm.to;
        vm.to = tmp;
        vm.calc();
      };

      vm.getRate = function() {
        return UnitConverter.round(UnitConverter.convert(vm.type, 1, vm.from, vm.to));
      };

      vm.addHistory = function() {
        if (vm.input === undefined || vm.input === '') return;
        var text = vm.input + ' ' + vm.from + ' → ' + vm.output + ' ' + vm.to;
        var entry = { text: text, type: vm.type, input: vm.input, from: vm.from, to: vm.to };
        if (!vm.history[0] || vm.history[0].text !== text) {
          vm.history.unshift(entry);
          if (vm.history.length > 10) vm.history.pop();
        }
      };

      vm.loadHistory = function(h) {
        vm.type = h.type;
        vm.units = UnitConverter.getUnits(h.type);
        vm.from = h.from;
        vm.to = h.to;
        vm.input = h.input;
        vm.calc();
      };

      vm.addFavorite = function() {
        var label = vm.from + ' ⇄ ' + vm.to;
        var exists = vm.favorites.some(function(f) { return f.label === label && f.type === vm.type; });
        if (!exists) {
          vm.favorites.push({ label: label, type: vm.type, from: vm.from, to: vm.to });
        }
      };

      vm.removeFavorite = function(index) {
        vm.favorites.splice(index, 1);
      };

      vm.loadFavorite = function(f) {
        vm.type = f.type;
        vm.units = UnitConverter.getUnits(f.type);
        vm.from = f.from;
        vm.to = f.to;
        vm.calc();
      };

      vm.calc();
    }]);
  </script>

</body>

**初心者向けとの違い:**
- `factory` ではなく `service` メソッドで定義（`this` を使うパターン）
- 明示的な DI アノテーション `['UnitConverter', function(UnitConverter) {...}]` でミニファイ対策
- 面積の変換カテゴリを追加
- 単位の入れ替え（スワップ）ボタンで直感的な操作
- お気に入り変換の保存・呼び出し機能
- 履歴クリックで過去の変換を再現
