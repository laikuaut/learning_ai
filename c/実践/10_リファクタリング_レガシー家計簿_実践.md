> **難易度**: ★★★★☆（上級）

> **課題の種類**: リファクタリング
> **学習目標**: 「動くが読みにくい」コードを、保守しやすい構造に段階的に改善する力を養う

## 課題の説明

以下は「家計簿アプリケーション」の全コードです。**正しく動作します**が、以下の問題を抱えています。

- すべての処理が `main` 関数に詰め込まれている（約200行の巨大関数）
- グローバル変数を多用している
- マジックナンバー（意味不明な数値）がある
- 同じようなコードがコピー&ペーストで繰り返されている
- データ構造が配列の配列で管理されており、意味が分かりにくい
- エラー処理がない

**このコードを段階的にリファクタリング（構造改善）してください。**

### ゴール

1. **ステップ1**: マジックナンバーを定数化し、構造体を導入する
2. **ステップ2**: 処理を関数に切り出す
3. **ステップ3**: エラー処理とファイル保存を追加する

/* 家計簿アプリケーション（レガシー版） */
/* 動作するが保守性が低いコード。リファクタリングしてください。 */

#include <stdio.h>
#include <string.h>

char names[50][30];
int amounts[50];
int types[50];   /* 1:収入, 2:食費, 3:交通費, 4:娯楽, 5:その他 */
int cnt = 0;

int main(void) {
    char line[10];
    int c;

    printf("===== 家計簿 =====\n");

    while (1) {
        printf("\n1.追加 2.一覧 3.集計 4.検索 5.削除 0.終了\n> ");
        fgets(line, sizeof(line), stdin);
        c = line[0] - '0';

        if (c == 0) {
            printf("終了します。\n");
            break;
        }

        if (c == 1) {
            if (cnt >= 50) {
                printf("これ以上登録できません。\n");
            } else {
                printf("項目名: ");
                fgets(names[cnt], 30, stdin);
                names[cnt][strcspn(names[cnt], "\n")] = '\0';

                printf("金額: ");
                fgets(line, sizeof(line), stdin);
                amounts[cnt] = 0;
                int k;
                for (k = 0; line[k] >= '0' && line[k] <= '9'; k++) {
                    amounts[cnt] = amounts[cnt] * 10 + (line[k] - '0');
                }

                printf("種類(1:収入 2:食費 3:交通費 4:娯楽 5:その他): ");
                fgets(line, sizeof(line), stdin);
                types[cnt] = line[0] - '0';
                if (types[cnt] < 1 || types[cnt] > 5) {
                    types[cnt] = 5;
                }
                cnt++;
                printf("追加しました。\n");
            }
        }

        if (c == 2) {
            if (cnt == 0) {
                printf("データがありません。\n");
            } else {
                printf("\n--- 一覧 ---\n");
                int i;
                for (i = 0; i < cnt; i++) {
                    printf("%2d. ", i + 1);
                    if (types[i] == 1) printf("[収入  ] ");
                    if (types[i] == 2) printf("[食費  ] ");
                    if (types[i] == 3) printf("[交通費] ");
                    if (types[i] == 4) printf("[娯楽  ] ");
                    if (types[i] == 5) printf("[その他] ");
                    printf("%-20s %7d円\n", names[i], amounts[i]);
                }
            }
        }

        if (c == 3) {
            int income = 0;
            int food = 0;
            int transport = 0;
            int entertainment = 0;
            int other = 0;
            int i;
            for (i = 0; i < cnt; i++) {
                if (types[i] == 1) income += amounts[i];
                if (types[i] == 2) food += amounts[i];
                if (types[i] == 3) transport += amounts[i];
                if (types[i] == 4) entertainment += amounts[i];
                if (types[i] == 5) other += amounts[i];
            }
            printf("\n--- 集計 ---\n");
            printf("収入  : %8d円\n", income);
            printf("食費  : %8d円\n", food);
            printf("交通費: %8d円\n", transport);
            printf("娯楽  : %8d円\n", entertainment);
            printf("その他: %8d円\n", other);
            int expense = food + transport + entertainment + other;
            printf("---\n");
            printf("支出合計: %8d円\n", expense);
            printf("収支    : %8d円\n", income - expense);
        }

        if (c == 4) {
            char kw[30];
            printf("検索キーワード: ");
            fgets(kw, 30, stdin);
            kw[strcspn(kw, "\n")] = '\0';
            int found = 0;
            int i;
            for (i = 0; i < cnt; i++) {
                if (strstr(names[i], kw) != NULL) {
                    printf("  %d. ", i + 1);
                    if (types[i] == 1) printf("[収入  ] ");
                    if (types[i] == 2) printf("[食費  ] ");
                    if (types[i] == 3) printf("[交通費] ");
                    if (types[i] == 4) printf("[娯楽  ] ");
                    if (types[i] == 5) printf("[その他] ");
                    printf("%-20s %7d円\n", names[i], amounts[i]);
                    found++;
                }
            }
            if (found == 0) {
                printf("見つかりませんでした。\n");
            }
        }

        if (c == 5) {
            int num;
            printf("削除する番号: ");
            fgets(line, sizeof(line), stdin);
            num = line[0] - '0';
            if (num < 1 || num > cnt) {
                printf("無効な番号です。\n");
            } else {
                int i;
                for (i = num - 1; i < cnt - 1; i++) {
                    strcpy(names[i], names[i + 1]);
                    amounts[i] = amounts[i + 1];
                    types[i] = types[i + 1];
                }
                cnt--;
                printf("削除しました。\n");
            }
        }
    }

    return 0;
}

## ステップ1：構造体とマジックナンバーの定数化

<details>
<summary>ヒント</summary>

以下の点を改善します。

- `50`, `30` などのマジックナンバーを `#define` で定数化する
- 種類の数値 `1〜5` を `enum` または `#define` で名前をつける
- `names[]`, `amounts[]`, `types[]` の3つの配列を1つの構造体にまとめる

```c
#define MAX_ENTRIES 50
#define NAME_LEN 30
#define NUM_CATEGORIES 5

enum Category {
    CAT_INCOME = 1,
    CAT_FOOD,
    CAT_TRANSPORT,
    CAT_ENTERTAINMENT,
    CAT_OTHER
};

typedef struct {
    char name[NAME_LEN];
    int amount;
    int category;
} KakeiboEntry;
```

## ステップ2：関数への切り出し

<details>
<summary>ヒント</summary>

以下のような関数に分割します。

- `get_category_name(int cat)` → 種類番号をラベル文字列に変換
- `add_entry(KakeiboEntry entries[], int *count)` → エントリ追加
- `show_all(KakeiboEntry entries[], int count)` → 一覧表示
- `show_summary(KakeiboEntry entries[], int count)` → 集計表示
- `search_entry(KakeiboEntry entries[], int count)` → 検索
- `delete_entry(KakeiboEntry entries[], int *count)` → 削除

種類ラベルの表示で `if` が連続しているコードは、配列で一括管理できます。

```c
const char *category_names[] = {
    "",        /* 0: 未使用 */
    "収入  ",
    "食費  ",
    "交通費",
    "娯楽  ",
    "その他"
};

const char *get_category_name(int cat) {
    if (cat >= 1 && cat <= NUM_CATEGORIES) {
        return category_names[cat];
    }
    return "不明";
}
```

## 解答例

<details>
<summary>解答例（リファクタリング後の完全なコード）</summary>

```c
/*
 * 家計簿アプリケーション（リファクタリング版）
 * 学べる内容：構造体、enum、関数分割、ファイルI/O、定数管理
 * コンパイル：gcc -o kakeibo kakeibo.c
 * 実行：./kakeibo
 */
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

/* --- 定数定義 --- */
#define MAX_ENTRIES 50
#define NAME_LEN 30
#define LINE_LEN 64
#define FILENAME "kakeibo.csv"
#define NUM_CATEGORIES 5

/* --- カテゴリ定義 --- */
enum Category {
    CAT_INCOME = 1,
    CAT_FOOD,
    CAT_TRANSPORT,
    CAT_ENTERTAINMENT,
    CAT_OTHER
};

const char *category_labels[] = {
    "",
    "収入  ",
    "食費  ",
    "交通費",
    "娯楽  ",
    "その他"
};

/* --- 構造体 --- */
typedef struct {
    char name[NAME_LEN];
    int amount;
    int category;
} KakeiboEntry;

/* --- カテゴリ名を取得 --- */
const char *get_category_label(int cat) {
    if (cat >= 1 && cat <= NUM_CATEGORIES) {
        return category_labels[cat];
    }
    return "不明  ";
}

/* --- エントリを1件表示 --- */
void print_entry(int index, const KakeiboEntry *entry) {
    printf("%2d. [%s] %-20s %7d円\n",
           index + 1,
           get_category_label(entry->category),
           entry->name,
           entry->amount);
}

/* --- 追加 --- */
void add_entry(KakeiboEntry entries[], int *count) {
    char line[LINE_LEN];
    int amount;

    if (*count >= MAX_ENTRIES) {
        printf("これ以上登録できません（最大%d件）。\n", MAX_ENTRIES);
        return;
    }

    printf("項目名: ");
    fgets(entries[*count].name, NAME_LEN, stdin);
    entries[*count].name[strcspn(entries[*count].name, "\n")] = '\0';

    printf("金額: ");
    fgets(line, sizeof(line), stdin);
    if (sscanf(line, "%d", &amount) != 1 || amount < 0) {
        printf("エラー: 正の整数を入力してください。\n");
        return;
    }
    entries[*count].amount = amount;

    printf("種類 (1:収入 2:食費 3:交通費 4:娯楽 5:その他): ");
    fgets(line, sizeof(line), stdin);
    int cat;
    if (sscanf(line, "%d", &cat) != 1 || cat < 1 || cat > NUM_CATEGORIES) {
        cat = CAT_OTHER;
    }
    entries[*count].category = cat;

    (*count)++;
    printf("追加しました。（現在 %d 件）\n", *count);
}

/* --- 一覧表示 --- */
void show_all(const KakeiboEntry entries[], int count) {
    int i;
    if (count == 0) {
        printf("データがありません。\n");
        return;
    }
    printf("\n--- 一覧 ---\n");
    for (i = 0; i < count; i++) {
        print_entry(i, &entries[i]);
    }
}

/* --- 集計 --- */
void show_summary(const KakeiboEntry entries[], int count) {
    int totals[NUM_CATEGORIES + 1] = {0};  /* インデックス1〜5を使用 */
    int i;

    if (count == 0) {
        printf("データがありません。\n");
        return;
    }

    for (i = 0; i < count; i++) {
        int cat = entries[i].category;
        if (cat >= 1 && cat <= NUM_CATEGORIES) {
            totals[cat] += entries[i].amount;
        }
    }

    printf("\n--- 集計 ---\n");
    for (i = 1; i <= NUM_CATEGORIES; i++) {
        printf("%s: %8d円\n", get_category_label(i), totals[i]);
    }

    int expense = totals[CAT_FOOD] + totals[CAT_TRANSPORT]
                + totals[CAT_ENTERTAINMENT] + totals[CAT_OTHER];
    printf("---\n");
    printf("支出合計: %8d円\n", expense);
    printf("収支    : %8d円\n", totals[CAT_INCOME] - expense);
}

/* --- 検索 --- */
void search_entry(const KakeiboEntry entries[], int count) {
    char keyword[NAME_LEN];
    int i, found = 0;

    printf("検索キーワード: ");
    fgets(keyword, sizeof(keyword), stdin);
    keyword[strcspn(keyword, "\n")] = '\0';

    for (i = 0; i < count; i++) {
        if (strstr(entries[i].name, keyword) != NULL) {
            print_entry(i, &entries[i]);
            found++;
        }
    }
    if (found == 0) {
        printf("見つかりませんでした。\n");
    }
}

/* --- 削除 --- */
void delete_entry(KakeiboEntry entries[], int *count) {
    char line[LINE_LEN];
    int num, i;

    printf("削除する番号: ");
    fgets(line, sizeof(line), stdin);
    if (sscanf(line, "%d", &num) != 1 || num < 1 || num > *count) {
        printf("無効な番号です。\n");
        return;
    }

    for (i = num - 1; i < *count - 1; i++) {
        entries[i] = entries[i + 1];
    }
    (*count)--;
    printf("削除しました。（残り %d 件）\n", *count);
}

/* --- ファイル保存 --- */
void save_to_file(const KakeiboEntry entries[], int count) {
    FILE *fp = fopen(FILENAME, "w");
    int i;
    if (fp == NULL) {
        printf("ファイルを開けませんでした。\n");
        return;
    }
    for (i = 0; i < count; i++) {
        fprintf(fp, "%s,%d,%d\n",
                entries[i].name,
                entries[i].amount,
                entries[i].category);
    }
    fclose(fp);
    printf("%s に %d 件保存しました。\n", FILENAME, count);
}

/* --- ファイル読み込み --- */
void load_from_file(KakeiboEntry entries[], int *count) {
    FILE *fp = fopen(FILENAME, "r");
    char line[256];
    if (fp == NULL) {
        printf("ファイルが見つかりません: %s\n", FILENAME);
        return;
    }
    *count = 0;
    while (fgets(line, sizeof(line), fp) != NULL && *count < MAX_ENTRIES) {
        line[strcspn(line, "\n")] = '\0';
        if (sscanf(line, "%[^,],%d,%d",
                   entries[*count].name,
                   &entries[*count].amount,
                   &entries[*count].category) == 3) {
            (*count)++;
        }
    }
    fclose(fp);
    printf("%s から %d 件読み込みました。\n", FILENAME, *count);
}

/* --- メイン --- */
int main(void) {
    KakeiboEntry entries[MAX_ENTRIES];
    int count = 0;
    char line[LINE_LEN];
    int choice;

    printf("===== 家計簿 =====\n");

    while (1) {
        printf("\n1.追加 2.一覧 3.集計 4.検索 5.削除 6.保存 7.読込 0.終了\n> ");
        fgets(line, sizeof(line), stdin);
        if (sscanf(line, "%d", &choice) != 1) {
            printf("数字を入力してください。\n");
            continue;
        }

        switch (choice) {
            case 0:
                printf("終了します。\n");
                return 0;
            case 1: add_entry(entries, &count); break;
            case 2: show_all(entries, count); break;
            case 3: show_summary(entries, count); break;
            case 4: search_entry(entries, count); break;
            case 5: delete_entry(entries, &count); break;
            case 6: save_to_file(entries, count); break;
            case 7: load_from_file(entries, &count); break;
            default:
                printf("無効な選択です。\n");
                break;
        }
    }

    return 0;
}
```

<details>
<summary>リファクタリングのポイント解説</summary>

### 1. マジックナンバーの除去

| 修正前 | 修正後 | 理由 |
|---|---|---|
| `50` | `MAX_ENTRIES` | 配列サイズの意味が明確 |
| `30` | `NAME_LEN` | 文字列長の意味が明確 |
| `1〜5` | `CAT_INCOME` 等 | 数値の意味が名前から分かる |

### 2. 構造体による統合

修正前は3つの並列配列 (`names[]`, `amounts[]`, `types[]`) でデータを管理していました。これでは「同じインデックスの要素が1つのエントリを構成する」という暗黙の前提があり、バグの温床です。構造体にまとめることで、1つのエントリが1つの変数で表現できます。

### 3. 関数分割

`main` 関数が約200行 → 約30行に縮小しました。各関数は1つの責任だけを持ち、独立してテストできます。

### 4. 重複コードの排除

種類ラベルの表示は `if` の連続から配列ルックアップに変更しました。エントリの表示処理も `print_entry` 関数に統一し、一覧表示と検索結果表示で共用しています。

### 5. エラー処理の追加

- 金額入力を `sscanf` でパースし、失敗時にエラーメッセージを表示
- ファイルオープンの失敗をチェック
- 入力値の範囲チェックを追加

</details>
