## 問題1：基本的なジェネリック関数（基本）

以下のジェネリック関数を実装してください。

1. `identity<T>(value: T): T` — 受け取った値をそのまま返す
2. `toArray<T>(value: T): T[]` — 受け取った値を1要素の配列にして返す
3. `swap<T, U>(pair: [T, U]): [U, T]` — タプルの要素を入れ替えて返す

**期待される出力：**
```
identity("hello") = hello
identity(42) = 42
toArray("abc") = ["abc"]
toArray(100) = [100]
swap(["hello", 42]) = [42, "hello"]
```

<details>
<summary>解答例</summary>

```typescript
// 基本的なジェネリック関数
function identity<T>(value: T): T {
  return value;
}

function toArray<T>(value: T): T[] {
  return [value];
}

function swap<T, U>(pair: [T, U]): [U, T] {
  return [pair[1], pair[0]];
}

console.log(`identity("hello") = ${identity("hello")}`);
console.log(`identity(42) = ${identity(42)}`);
console.log(`toArray("abc") = ${JSON.stringify(toArray("abc"))}`);
console.log(`toArray(100) = ${JSON.stringify(toArray(100))}`);
console.log(`swap(["hello", 42]) = ${JSON.stringify(swap(["hello", 42]))}`);
```

</details>

## 問題2：配列ユーティリティ関数（基本）

以下のジェネリックな配列ユーティリティ関数を実装してください。

1. `first<T>(arr: T[]): T | undefined` — 最初の要素を返す
2. `last<T>(arr: T[]): T | undefined` — 最後の要素を返す
3. `unique<T>(arr: T[]): T[]` — 重複を除いた配列を返す

**期待される出力：**
```
first([10, 20, 30]) = 10
last([10, 20, 30]) = 30
first(["a", "b", "c"]) = a
unique([1, 2, 2, 3, 3, 3]) = [1, 2, 3]
unique(["a", "b", "a", "c"]) = ["a", "b", "c"]
```

<details>
<summary>解答例</summary>

```typescript
function first<T>(arr: T[]): T | undefined {
  return arr[0];
}

function last<T>(arr: T[]): T | undefined {
  return arr[arr.length - 1];
}

function unique<T>(arr: T[]): T[] {
  return [...new Set(arr)];
}

console.log(`first([10, 20, 30]) = ${first([10, 20, 30])}`);
console.log(`last([10, 20, 30]) = ${last([10, 20, 30])}`);
console.log(`first(["a", "b", "c"]) = ${first(["a", "b", "c"])}`);
console.log(`unique([1, 2, 2, 3, 3, 3]) = ${JSON.stringify(unique([1, 2, 2, 3, 3, 3]))}`);
console.log(`unique(["a", "b", "a", "c"]) = ${JSON.stringify(unique(["a", "b", "a", "c"]))}`);
```

</details>

## 問題3：ジェネリック制約（基本）

`extends` を使って型パラメータに制約を付けた関数を実装してください。

1. `getLength<T extends { length: number }>(value: T): number` — `length` プロパティを持つ値の長さを返す
2. `getProperty<T, K extends keyof T>(obj: T, key: K): T[K]` — オブジェクトの指定キーの値を返す

**期待される出力：**
```
getLength("hello") = 5
getLength([1, 2, 3]) = 3
getProperty({name: "太郎", age: 25}, "name") = 太郎
getProperty({name: "太郎", age: 25}, "age") = 25
```

<details>
<summary>ヒント</summary>

- `T extends { length: number }` で「lengthプロパティを持つ型」に制約します。
- `K extends keyof T` で「Tのキーの一つ」に制約します。

</details>

<details>
<summary>解答例</summary>

```typescript
// lengthプロパティを持つ型に制約
function getLength<T extends { length: number }>(value: T): number {
  return value.length;
}

// オブジェクトのキーに制約
function getProperty<T, K extends keyof T>(obj: T, key: K): T[K] {
  return obj[key];
}

console.log(`getLength("hello") = ${getLength("hello")}`);
console.log(`getLength([1, 2, 3]) = ${getLength([1, 2, 3])}`);

const user = { name: "太郎", age: 25 };
console.log(`getProperty({name: "太郎", age: 25}, "name") = ${getProperty(user, "name")}`);
console.log(`getProperty({name: "太郎", age: 25}, "age") = ${getProperty(user, "age")}`);

// getProperty(user, "email"); // エラー！"email"はuserのキーではない
```

</details>

## 問題4：ジェネリックインターフェース（応用）

`Stack<T>` インターフェースを定義し、実装してください。

- `push(item: T): void`
- `pop(): T | undefined`
- `peek(): T | undefined`
- `size(): number`
- `isEmpty(): boolean`

**期待される出力：**
```
numberスタック: push 1, 2, 3
peek = 3
pop = 3
pop = 2
size = 1, isEmpty = false

stringスタック: push "a", "b"
peek = b
size = 2
```

<details>
<summary>解答例</summary>

```typescript
// ジェネリックインターフェース
interface Stack<T> {
  push(item: T): void;
  pop(): T | undefined;
  peek(): T | undefined;
  size(): number;
  isEmpty(): boolean;
}

// ファクトリ関数で実装
function createStack<T>(): Stack<T> {
  const items: T[] = [];

  return {
    push(item: T): void {
      items.push(item);
    },
    pop(): T | undefined {
      return items.pop();
    },
    peek(): T | undefined {
      return items[items.length - 1];
    },
    size(): number {
      return items.length;
    },
    isEmpty(): boolean {
      return items.length === 0;
    },
  };
}

// numberスタック
const numStack = createStack<number>();
numStack.push(1);
numStack.push(2);
numStack.push(3);
console.log("numberスタック: push 1, 2, 3");
console.log(`peek = ${numStack.peek()}`);
console.log(`pop = ${numStack.pop()}`);
console.log(`pop = ${numStack.pop()}`);
console.log(`size = ${numStack.size()}, isEmpty = ${numStack.isEmpty()}`);

// stringスタック
const strStack = createStack<string>();
strStack.push("a");
strStack.push("b");
console.log(`\nstringスタック: push "a", "b"`);
console.log(`peek = ${strStack.peek()}`);
console.log(`size = ${strStack.size()}`);
```

</details>

## 問題5：ジェネリック型エイリアス（応用）

以下のジェネリック型エイリアスを定義し、使用してください。

1. `Nullable<T>` — `T | null`
2. `Optional<T>` — `T | undefined`
3. `AsyncResult<T, E>` — `{ data: T | null; error: E | null; loading: boolean }`

`AsyncResult` を使ってAPI呼び出しの状態を管理する例を実装してください。

**期待される出力：**
```
状態1: 読み込み中...
状態2: ユーザー一覧 - 太郎, 花子
状態3: エラー - データ取得に失敗しました
```

<details>
<summary>解答例</summary>

```typescript
// ジェネリック型エイリアス
type Nullable<T> = T | null;
type Optional<T> = T | undefined;
type AsyncResult<T, E = string> = {
  data: T | null;
  error: E | null;
  loading: boolean;
};

// AsyncResultの表示関数
function renderResult<T>(
  result: AsyncResult<T>,
  formatData: (data: T) => string
): string {
  if (result.loading) {
    return "読み込み中...";
  }
  if (result.error) {
    return `エラー - ${result.error}`;
  }
  if (result.data !== null) {
    return formatData(result.data);
  }
  return "データなし";
}

// 各状態のテスト
const loading: AsyncResult<string[]> = { data: null, error: null, loading: true };
const success: AsyncResult<string[]> = {
  data: ["太郎", "花子"], error: null, loading: false,
};
const failure: AsyncResult<string[]> = {
  data: null, error: "データ取得に失敗しました", loading: false,
};

const formatter = (users: string[]) => `ユーザー一覧 - ${users.join(", ")}`;

console.log(`状態1: ${renderResult(loading, formatter)}`);
console.log(`状態2: ${renderResult(success, formatter)}`);
console.log(`状態3: ${renderResult(failure, formatter)}`);
```

</details>

## 問題6：条件型（応用）

以下の条件型を定義してください。

1. `IsString<T>` — `T` が `string` なら `"yes"`、そうでなければ `"no"`
2. `ArrayElement<T>` — `T` が配列なら要素の型、そうでなければ `never`
3. `UnwrapPromise<T>` — `T` が `Promise<U>` なら `U`、そうでなければ `T`

型がコンパイル時に正しく推論されることを確認するコードを書いてください。

**期待されるコード（コンパイルが通ること）：**
```typescript
const a: IsString<string> = "yes";
const b: IsString<number> = "no";
const c: ArrayElement<string[]> = "hello";
const d: UnwrapPromise<Promise<number>> = 42;
```

<details>
<summary>解答例</summary>

```typescript
// 条件型の定義
type IsString<T> = T extends string ? "yes" : "no";

type ArrayElement<T> = T extends (infer E)[] ? E : never;

type UnwrapPromise<T> = T extends Promise<infer U> ? U : T;

// 型の確認（コンパイル時に検証される）
const a: IsString<string> = "yes";    // OK
const b: IsString<number> = "no";     // OK
// const c: IsString<string> = "no";  // エラー！

const d: ArrayElement<string[]> = "hello";  // OK: string型
const e: ArrayElement<number[]> = 42;       // OK: number型

const f: UnwrapPromise<Promise<number>> = 42;       // OK: number型
const g: UnwrapPromise<Promise<string>> = "hello";  // OK: string型
const h: UnwrapPromise<boolean> = true;              // OK: boolean型（Promiseでない）

console.log("全ての型チェックが通りました");
console.log(`a = ${a}, b = ${b}`);
console.log(`d = ${d}, e = ${e}`);
console.log(`f = ${f}, g = ${g}, h = ${h}`);
```

</details>

## 問題7：マップ型（チャレンジ）

以下のマップ型を自作してください。

1. `MyPartial<T>` — 全プロパティをオプショナルにする
2. `MyReadonly<T>` — 全プロパティをreadonlyにする
3. `Mutable<T>` — readonlyを除去する

さらに、`User` 型を使ってそれぞれの型が正しく動作することを確認してください。

**期待されるコード（コンパイルが通ること）：**
```typescript
const partial: MyPartial<User> = { name: "太郎" }; // age, emailは省略OK
const readonly_: MyReadonly<User> = { name: "太郎", age: 25, email: "..." };
// readonly_.name = "花子"; // エラーになること
```

<details>
<summary>解答例</summary>

```typescript
// マップ型の自作
type MyPartial<T> = {
  [K in keyof T]?: T[K];
};

type MyReadonly<T> = {
  readonly [K in keyof T]: T[K];
};

type Mutable<T> = {
  -readonly [K in keyof T]: T[K];
};

// テスト用の型
interface User {
  name: string;
  age: number;
  email: string;
}

// MyPartialの動作確認（全プロパティが省略可能）
const partial: MyPartial<User> = { name: "太郎" };
console.log(`partial.name = ${partial.name}`);
console.log(`partial.age = ${partial.age}`); // undefined

// MyReadonlyの動作確認
const readonlyUser: MyReadonly<User> = {
  name: "太郎",
  age: 25,
  email: "taro@example.com",
};
// readonlyUser.name = "花子"; // エラー！readonlyプロパティ
console.log(`readonlyUser.name = ${readonlyUser.name}`);

// Mutableの動作確認
interface ReadonlyConfig {
  readonly host: string;
  readonly port: number;
}

const mutableConfig: Mutable<ReadonlyConfig> = { host: "localhost", port: 3000 };
mutableConfig.host = "0.0.0.0"; // OK！readonlyが除去されている
mutableConfig.port = 8080;
console.log(`mutableConfig = ${mutableConfig.host}:${mutableConfig.port}`);
```

</details>

## 問題8：総合問題 — ジェネリックなリポジトリ（チャレンジ）

データベースのCRUD操作を模倣するジェネリックなリポジトリを実装してください。

1. `Entity` インターフェース：`{ id: number }`
2. `Repository<T extends Entity>` インターフェース
   - `findAll(): T[]`
   - `findById(id: number): T | undefined`
   - `create(data: Omit<T, "id">): T`
   - `delete(id: number): boolean`
3. `createRepository<T extends Entity>()` ファクトリ関数で実装

**期待される出力：**
```
=== ユーザーリポジトリ ===
作成: {id: 1, name: "太郎", email: "taro@example.com"}
作成: {id: 2, name: "花子", email: "hanako@example.com"}
全件取得: 2件
ID検索(1): 太郎
削除(1): true
残り: 1件
```

<details>
<summary>解答例</summary>

```typescript
// エンティティの基底型
interface Entity {
  id: number;
}

// ジェネリックなリポジトリインターフェース
interface Repository<T extends Entity> {
  findAll(): T[];
  findById(id: number): T | undefined;
  create(data: Omit<T, "id">): T;
  delete(id: number): boolean;
}

// ファクトリ関数
function createRepository<T extends Entity>(): Repository<T> {
  let items: T[] = [];
  let nextId = 1;

  return {
    findAll(): T[] {
      return [...items];
    },
    findById(id: number): T | undefined {
      return items.find((item) => item.id === id);
    },
    create(data: Omit<T, "id">): T {
      const newItem = { ...data, id: nextId++ } as T;
      items.push(newItem);
      return newItem;
    },
    delete(id: number): boolean {
      const initialLength = items.length;
      items = items.filter((item) => item.id !== id);
      return items.length < initialLength;
    },
  };
}

// ユーザー型
interface User extends Entity {
  name: string;
  email: string;
}

// 使用例
console.log("=== ユーザーリポジトリ ===");
const userRepo = createRepository<User>();

const user1 = userRepo.create({ name: "太郎", email: "taro@example.com" });
console.log(`作成: ${JSON.stringify(user1)}`);

const user2 = userRepo.create({ name: "花子", email: "hanako@example.com" });
console.log(`作成: ${JSON.stringify(user2)}`);

console.log(`全件取得: ${userRepo.findAll().length}件`);

const found = userRepo.findById(1);
console.log(`ID検索(1): ${found?.name}`);

const deleted = userRepo.delete(1);
console.log(`削除(1): ${deleted}`);

console.log(`残り: ${userRepo.findAll().length}件`);
```

</details>
