# 実践課題04：イベント管理システム ★2

> **難易度**: ★★☆☆☆（基礎）
> **前提知識**: 第2章（インターフェースと型エイリアス）、第4章（ユニオン型と交差型）
> **課題の種類**: ミニプロジェクト
> **学習目標**: ユニオン型（union type）と交差型（intersection type）を使って、現実のデータモデルを柔軟に表現する

## 課題の要件

1. イベントの種類を3つ定義する（セミナー、ワークショップ、オンライン）
2. 各種類は共通のプロパティ（名前、日時、参加者数）と種類固有のプロパティを持つ
3. ユニオン型でイベント型を定義し、種別に応じた処理を行う
4. 交差型を使って共通プロパティを定義する
5. 満席判定、種別フィルタ、統計集計の機能を作る

---

## ステップガイド

<details>
<summary>ステップ1：共通型と種別型を定義する</summary>

交差型（`&`）を使って、共通部分と固有部分を組み合わせます。

```typescript
// 共通プロパティ
type EventBase = {
  id: number;
  name: string;
  date: string;
  participants: number;
};

// セミナー固有のプロパティ
type Seminar = EventBase & {
  type: "seminar";
  venue: string;
  capacity: number;
};

// ワークショップ固有
type Workshop = EventBase & {
  type: "workshop";
  venue: string;
  capacity: number;
  materials: string[];
};

// オンラインイベント固有
type OnlineEvent = EventBase & {
  type: "online";
  url: string;
};
```

`type: "seminar"` のような文字列リテラル型は**判別プロパティ**（discriminant property）と呼ばれ、ユニオン型の判別に使います。

</details>

<details>
<summary>ステップ2：ユニオン型でまとめる</summary>

```typescript
type AppEvent = Seminar | Workshop | OnlineEvent;
```

この型は「セミナー、ワークショップ、またはオンラインイベントのいずれか」を表します。

</details>

<details>
<summary>ステップ3：種別に応じた処理を書く</summary>

`type` プロパティで条件分岐すると、TypeScriptが自動的に型を絞り込みます。

```typescript
function displayEvent(event: AppEvent): void {
  if (event.type === "seminar") {
    // ここでは event は Seminar 型として扱える
    console.log(`場所: ${event.venue}  定員: ${event.capacity}名`);
  } else if (event.type === "online") {
    // ここでは event は OnlineEvent 型
    console.log(`URL: ${event.url}`);
  }
}
```

</details>

## 解答例

<details>
<summary>解答例（初心者向け）</summary>

```typescript
// イベント管理システム
// 学べる内容：ユニオン型、交差型、判別プロパティ
// 実行方法：npx ts-node 04_event_manager.ts

// --- 型定義 ---
type EventBase = {
  id: number;
  name: string;
  date: string;
  participants: number;
};

type Seminar = EventBase & {
  type: "seminar";
  venue: string;
  capacity: number;
};

type Workshop = EventBase & {
  type: "workshop";
  venue: string;
  capacity: number;
  materials: string[];
};

type OnlineEvent = EventBase & {
  type: "online";
  url: string;
};

type AppEvent = Seminar | Workshop | OnlineEvent;

// --- イベントデータ ---
const events: AppEvent[] = [
  {
    id: 1,
    type: "seminar",
    name: "TypeScript勉強会",
    date: "2026-05-15",
    venue: "東京ホールA",
    capacity: 100,
    participants: 45,
  },
  {
    id: 2,
    type: "workshop",
    name: "チームビルディング",
    date: "2026-05-20",
    venue: "会議室B",
    capacity: 20,
    participants: 18,
    materials: ["付箋", "ホワイトボード", "マーカー"],
  },
  {
    id: 3,
    type: "online",
    name: "React入門",
    date: "2026-06-01",
    url: "https://meet.example.com/react",
    participants: 200,
  },
  {
    id: 4,
    type: "seminar",
    name: "AI最新トレンド",
    date: "2026-06-10",
    venue: "大阪ホールC",
    capacity: 150,
    participants: 150,
  },
];

// --- イベント種別の絵文字 ---
function getIcon(event: AppEvent): string {
  if (event.type === "seminar") return "🎤";
  if (event.type === "workshop") return "🏋️";
  return "💻";
}

// --- イベント種別の日本語名 ---
function getTypeName(event: AppEvent): string {
  if (event.type === "seminar") return "セミナー";
  if (event.type === "workshop") return "ワークショップ";
  return "オンライン";
}

// --- 満席判定 ---
function isFull(event: AppEvent): boolean {
  if (event.type === "seminar" || event.type === "workshop") {
    return event.participants >= event.capacity;
  }
  // オンラインは定員なし
  return false;
}

// --- 全イベント表示 ---
function displayAll(events: AppEvent[]): void {
  console.log("--- 全イベント一覧 ---");
  for (const event of events) {
    const icon: string = getIcon(event);
    const typeName: string = getTypeName(event);
    const fullLabel: string = isFull(event) ? " 【満席】" : "";
    console.log(`[${event.id}] ${icon} ${event.name}（${typeName}）`);

    if (event.type === "seminar") {
      console.log(
        `    日時: ${event.date}  場所: ${event.venue.padEnd(10)}  定員: ${event.capacity}名  参加: ${event.participants}名${fullLabel}`
      );
    } else if (event.type === "workshop") {
      console.log(
        `    日時: ${event.date}  場所: ${event.venue.padEnd(10)}  定員: ${event.capacity}名  参加: ${event.participants}名${fullLabel}`
      );
    } else if (event.type === "online") {
      console.log(
        `    日時: ${event.date}  URL: ${event.url}  参加: ${event.participants}名`
      );
    }
  }
}

// --- フィルタ関数 ---
function filterOnline(events: AppEvent[]): OnlineEvent[] {
  const result: OnlineEvent[] = [];
  for (const event of events) {
    if (event.type === "online") {
      result.push(event);
    }
  }
  return result;
}

function filterFull(events: AppEvent[]): AppEvent[] {
  const result: AppEvent[] = [];
  for (const event of events) {
    if (isFull(event)) {
      result.push(event);
    }
  }
  return result;
}

// --- 統計 ---
function showStatistics(events: AppEvent[]): void {
  let seminarCount: number = 0;
  let workshopCount: number = 0;
  let onlineCount: number = 0;
  let totalParticipants: number = 0;

  for (const event of events) {
    totalParticipants += event.participants;
    if (event.type === "seminar") seminarCount++;
    else if (event.type === "workshop") workshopCount++;
    else onlineCount++;
  }

  console.log("--- イベント統計 ---");
  console.log(`セミナー: ${seminarCount}件`);
  console.log(`ワークショップ: ${workshopCount}件`);
  console.log(`オンライン: ${onlineCount}件`);
  console.log(`総参加者数: ${totalParticipants}名`);
}

// --- メイン処理 ---
console.log("===== イベント管理システム =====\n");

displayAll(events);

console.log("\n--- オンラインイベントのみ ---");
const onlineEvents: OnlineEvent[] = filterOnline(events);
for (const event of onlineEvents) {
  console.log(`[${event.id}] 💻 ${event.name}`);
  console.log(`    URL: ${event.url}`);
}

console.log("\n--- 満席イベント ---");
const fullEvents: AppEvent[] = filterFull(events);
for (const event of fullEvents) {
  if (event.type === "seminar" || event.type === "workshop") {
    console.log(`[${event.id}] ${getIcon(event)} ${event.name}（${event.participants}/${event.capacity}名）`);
  }
}

console.log("\n" );
showStatistics(events);
```

<summary>解答例（改良版 ─ switch文による網羅性チェック）</summary>

```typescript
// イベント管理システム（改良版）
// 学べる内容：網羅性チェック（exhaustive check）、never型、型ガードの先取り
// 実行方法：npx ts-node 04_event_manager_v2.ts

// --- 型定義 ---
type EventBase = {
  readonly id: number;
  readonly name: string;
  readonly date: string;
  readonly participants: number;
};

type Seminar = EventBase & {
  readonly type: "seminar";
  readonly venue: string;
  readonly capacity: number;
};

type Workshop = EventBase & {
  readonly type: "workshop";
  readonly venue: string;
  readonly capacity: number;
  readonly materials: readonly string[];
};

type OnlineEvent = EventBase & {
  readonly type: "online";
  readonly url: string;
};

type AppEvent = Seminar | Workshop | OnlineEvent;

// --- 網羅性チェック用ヘルパー ---
// すべてのケースが処理されていることをコンパイル時に保証する
function assertNever(value: never): never {
  throw new Error(`Unexpected value: ${value}`);
}

// --- イベントデータ ---
const events: AppEvent[] = [
  { id: 1, type: "seminar", name: "TypeScript勉強会", date: "2026-05-15", venue: "東京ホールA", capacity: 100, participants: 45 },
  { id: 2, type: "workshop", name: "チームビルディング", date: "2026-05-20", venue: "会議室B", capacity: 20, participants: 18, materials: ["付箋", "ホワイトボード"] },
  { id: 3, type: "online", name: "React入門", date: "2026-06-01", url: "https://meet.example.com/react", participants: 200 },
  { id: 4, type: "seminar", name: "AI最新トレンド", date: "2026-06-10", venue: "大阪ホールC", capacity: 150, participants: 150 },
];

// --- switch文による型安全な処理分岐 ---
function getEventDetail(event: AppEvent): string {
  switch (event.type) {
    case "seminar":
      return `場所: ${event.venue}  定員: ${event.capacity}名  参加: ${event.participants}名`;
    case "workshop":
      return `場所: ${event.venue}  定員: ${event.capacity}名  参加: ${event.participants}名  教材: ${event.materials.join(", ")}`;
    case "online":
      return `URL: ${event.url}  参加: ${event.participants}名`;
    default:
      // もし新しいイベント種別が追加されてここに到達したら、コンパイルエラーになる
      return assertNever(event);
  }
}

function getIcon(event: AppEvent): string {
  switch (event.type) {
    case "seminar": return "🎤";
    case "workshop": return "🏋️";
    case "online": return "💻";
    default: return assertNever(event);
  }
}

function getTypeName(event: AppEvent): string {
  switch (event.type) {
    case "seminar": return "セミナー";
    case "workshop": return "ワークショップ";
    case "online": return "オンライン";
    default: return assertNever(event);
  }
}

// --- 会場ありイベントかどうかの型ガード ---
function hasVenue(event: AppEvent): event is Seminar | Workshop {
  return event.type === "seminar" || event.type === "workshop";
}

function isFull(event: AppEvent): boolean {
  return hasVenue(event) && event.participants >= event.capacity;
}

// --- 汎用フィルタ ---
function filterEvents<T extends AppEvent>(
  events: AppEvent[],
  predicate: (e: AppEvent) => e is T
): T[] {
  return events.filter(predicate);
}

function isOnline(event: AppEvent): event is OnlineEvent {
  return event.type === "online";
}

// --- 統計集計 ---
function getStatistics(events: AppEvent[]): Map<string, number> {
  const counts: Map<string, number> = new Map();
  for (const event of events) {
    const typeName: string = getTypeName(event);
    counts.set(typeName, (counts.get(typeName) ?? 0) + 1);
  }
  return counts;
}

// --- メイン処理 ---
console.log("===== イベント管理システム =====\n");

console.log("--- 全イベント一覧 ---");
for (const event of events) {
  const fullLabel: string = isFull(event) ? " 【満席】" : "";
  console.log(`[${event.id}] ${getIcon(event)} ${event.name}（${getTypeName(event)}）`);
  console.log(`    日時: ${event.date}  ${getEventDetail(event)}${fullLabel}`);
}

console.log("\n--- オンラインイベントのみ ---");
const onlineOnly: OnlineEvent[] = filterEvents(events, isOnline);
onlineOnly.forEach((e: OnlineEvent): void => {
  console.log(`[${e.id}] 💻 ${e.name}`);
  console.log(`    URL: ${e.url}`);
});

console.log("\n--- 満席イベント ---");
events.filter(isFull).forEach((event: AppEvent): void => {
  if (hasVenue(event)) {
    console.log(`[${event.id}] ${getIcon(event)} ${event.name}（${event.participants}/${event.capacity}名）`);
  }
});

console.log("\n--- イベント統計 ---");
const stats: Map<string, number> = getStatistics(events);
stats.forEach((count: number, typeName: string): void => {
  console.log(`${typeName}: ${count}件`);
});
const totalParticipants: number = events.reduce(
  (sum: number, e: AppEvent): number => sum + e.participants, 0
);
console.log(`総参加者数: ${totalParticipants}名`);
```

**初心者向けとの違い:**
- `assertNever` を使った網羅性チェック（exhaustive check）: 新しいイベント種別を追加した際に処理漏れがコンパイルエラーで検出される
- `event is Seminar | Workshop` のような型ガード（type guard）関数
- `filterEvents` にジェネリクスを使って型安全なフィルタを実現
- `Map` による集計、`reduce` による合計計算

</details>
