## ステップガイド

<details>
<summary>ステップ1：手計算の解き方を確認する</summary>

**10進数 → 2進数** は「繰り返し2で割り、余りを逆順に並べる」方法で解きます。

```
45 ÷ 2 = 22 余り 1  ↑
22 ÷ 2 = 11 余り 0  ↑
11 ÷ 2 = 5  余り 1  ↑
 5 ÷ 2 = 2  余り 1  ↑
 2 ÷ 2 = 1  余り 0  ↑
 1 ÷ 2 = 0  余り 1  ↑（読む方向）
→ (101101)₂
```

**2進数 → 16進数** は「4桁ずつ区切って変換」します。

```
1011 0011
  B    3
→ (B3)₁₆
```

```python
# 基数変換ドリル
# 学べる内容：基数変換、bin()/hex()/int()の使い方、文字列処理
# 実行方法：python 01_radix_drill.py

import random

def main():
    print("===== 基数変換ドリル =====\n")

    correct_count = 0
    total = 8

    for i in range(1, total + 1):
        num = random.randint(1, 255)

        # 変換タイプをランダムに選ぶ
        q_type = random.choice(["dec2bin", "bin2dec", "dec2hex", "hex2dec", "bin2hex"])

        if q_type == "dec2bin":
            question = f"10進数 {num} を2進数に変換してください。"
            correct_answer = bin(num)[2:]
        elif q_type == "bin2dec":
            bin_str = bin(num)[2:]
            question = f"2進数 {bin_str} を10進数に変換してください。"
            correct_answer = str(num)
        elif q_type == "dec2hex":
            question = f"10進数 {num} を16進数に変換してください。"
            correct_answer = hex(num)[2:].upper()
        elif q_type == "hex2dec":
            hex_str = hex(num)[2:].upper()
            question = f"16進数 {hex_str} を10進数に変換してください。"
            correct_answer = str(num)
        else:  # bin2hex
            bin_str = bin(num)[2:]
            question = f"2進数 {bin_str} を16進数に変換してください。"
            correct_answer = hex(num)[2:].upper()

        print(f"【問題{i}】")
        print(f"  {question}")
        user_answer = input("\nあなたの回答: ").strip().upper()

        if user_answer == correct_answer.upper():
            print("✓ 正解！\n")
            correct_count += 1
        else:
            print(f"✗ 不正解... 正解は {correct_answer} でした。\n")

    # 結果表示
    rate = int(correct_count / total * 100)
    print("===== 結果 =====")
    print(f"{total}問中 {correct_count}問正解（正答率: {rate}%）")

main()
```

<summary>解答例（改良版 ─ 出題タイプ表示・補数問題対応）</summary>

```python
# 基数変換ドリル（改良版）
# 補数問題も含む高機能バージョン

import random


def generate_basic_question(num):
    """基本的な基数変換問題を生成する"""
    types = [
        ("10進数→2進数", f"10進数 {num} を2進数に変換", bin(num)[2:]),
        ("2進数→10進数", f"2進数 {bin(num)[2:]} を10進数に変換", str(num)),
        ("10進数→16進数", f"10進数 {num} を16進数に変換", hex(num)[2:].upper()),
        ("16進数→10進数", f"16進数 {hex(num)[2:].upper()} を10進数に変換", str(num)),
        ("2進数→16進数", f"2進数 {bin(num)[2:]} を16進数に変換", hex(num)[2:].upper()),
    ]
    return random.choice(types)


def generate_complement_question():
    """2の補数問題を生成する"""
    num = random.randint(1, 127)
    # 8ビットの2の補数表現
    complement = (256 - num) & 0xFF
    complement_bin = format(complement, '08b')
    return (
        "2の補数",
        f"8ビットで -{num} を2の補数で表現",
        complement_bin,
    )


def run_drill():
    """ドリルのメインループ"""
    print("===== 基数変換ドリル（改良版） =====")
    print("※ 16進数の回答は大文字・小文字どちらでもOKです\n")

    total = 10
    correct = 0
    wrong_list = []

    for i in range(1, total + 1):
        # 8割は基本問題、2割は補数問題
        if random.random() < 0.8:
            num = random.randint(1, 255)
            category, question, answer = generate_basic_question(num)
        else:
            category, question, answer = generate_complement_question()

        print(f"【問題{i}】[{category}]")
        print(f"  {question}")
        user = input("  回答: ").strip().upper()

        if user == answer.upper():
            print("  ✓ 正解！\n")
            correct += 1
        else:
            print(f"  ✗ 不正解... 正解は {answer} でした。\n")
            wrong_list.append((i, category, question, answer))

    # --- 結果表示 ---
    rate = int(correct / total * 100)
    print(f"\n===== 結果: {total}問中 {correct}問正解（{rate}%） =====")

    if wrong_list:
        print("\n--- 間違えた問題の復習 ---")
        for num, cat, q, a in wrong_list:
            print(f"  問題{num} [{cat}] {q} → {a}")

    return rate


# --- メイン ---
while True:
    score = run_drill()
    if score == 100:
        print("\nパーフェクト！ 基数変換をマスターしました！")
    retry = input("\nもう一度挑戦しますか？ (y/n): ")
    if retry.lower() != "y":
        print("お疲れさまでした！")
        break
```

**初心者向けとの違い:**
- 出題カテゴリ（変換方向）を画面に表示 → 弱点分野が一目でわかる
- 2の補数問題も出題 → 試験範囲をより広くカバー
- 間違えた問題の復習表示 → 効率的な学習
- リトライ機能付き → 繰り返し練習が可能
