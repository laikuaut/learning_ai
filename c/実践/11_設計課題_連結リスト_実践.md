# 実践課題11：設計課題 ─ 連結リスト ★4

> **難易度**: ★★★★☆（上級）
> **前提知識**: 第5章（関数）、第6章（ポインタ）、第7章（構造体）
> **課題の種類**: 設計課題
> **学習目標**: ポインタと動的メモリ確保を駆使して、連結リスト（linked list）を一から設計・実装する力を養う

---

## 課題の説明

固定サイズの配列とは異なり、**連結リスト（linked list）**は要素数を事前に決める必要がないデータ構造です。各要素（ノード）がポインタで次の要素を指すことで、鎖のようにデータをつなぎます。

```
連結リストのイメージ:

  head
   ↓
  [10|→]─→[20|→]─→[30|→]─→[40|NULL]

  各ノードは「データ」と「次のノードへのポインタ」を持つ
```

以下の仕様を満たす整数の単方向連結リストを設計・実装してください。

---

## 要件

### データ構造

```c
typedef struct Node {
    int data;
    struct Node *next;
} Node;

typedef struct {
    Node *head;
    int size;
} LinkedList;
```

### 実装すべき関数

| 関数名 | 説明 |
|---|---|
| `LinkedList *list_create(void)` | 空のリストを作成する |
| `void list_destroy(LinkedList *list)` | リスト全体のメモリを解放する |
| `int list_push_front(LinkedList *list, int data)` | 先頭に要素を追加する |
| `int list_push_back(LinkedList *list, int data)` | 末尾に要素を追加する |
| `int list_pop_front(LinkedList *list, int *out)` | 先頭の要素を取り出す |
| `int list_insert_at(LinkedList *list, int index, int data)` | 指定位置に挿入する |
| `int list_remove_at(LinkedList *list, int index, int *out)` | 指定位置の要素を削除する |
| `int list_get(const LinkedList *list, int index, int *out)` | 指定位置の値を取得する |
| `int list_find(const LinkedList *list, int data)` | 値を検索してインデックスを返す |
| `void list_print(const LinkedList *list)` | リストの内容を表示する |
| `int list_size(const LinkedList *list)` | 要素数を返す |

### テストプログラム

上記の関数をすべて使うテスト用の `main` 関数を書いて動作確認してください。

## ステップガイド

<details>
<summary>ステップ1：ノードの作成と list_push_front</summary>

新しいノードを `malloc` で作成し、先頭に挿入します。

```c
Node *create_node(int data) {
    Node *node = (Node *)malloc(sizeof(Node));
    if (node == NULL) return NULL;
    node->data = data;
    node->next = NULL;
    return node;
}

int list_push_front(LinkedList *list, int data) {
    Node *node = create_node(data);
    if (node == NULL) return 0;
    node->next = list->head;  /* 新ノードの next を現在の先頭に */
    list->head = node;         /* 先頭を新ノードに更新 */
    list->size++;
    return 1;
}
```

</details>

<details>
<summary>ステップ2：list_push_back と list_print</summary>

末尾追加は、リストの最後のノード（`next == NULL` のノード）を見つけてからつなぎます。

```c
int list_push_back(LinkedList *list, int data) {
    Node *node = create_node(data);
    if (node == NULL) return 0;

    if (list->head == NULL) {
        list->head = node;  /* リストが空なら先頭に */
    } else {
        Node *cur = list->head;
        while (cur->next != NULL) {
            cur = cur->next;
        }
        cur->next = node;  /* 末尾ノードの next に新ノードを接続 */
    }
    list->size++;
    return 1;
}
```

</details>

<details>
<summary>ステップ3：list_destroy でメモリリークを防ぐ</summary>

リストの全ノードを順番に `free` します。現在のノードを解放する前に、次のノードのアドレスを保存しておくことが重要です。

```c
void list_destroy(LinkedList *list) {
    Node *cur = list->head;
    while (cur != NULL) {
        Node *next = cur->next;  /* 先に次を保存 */
        free(cur);                /* 現在のノードを解放 */
        cur = next;
    }
    free(list);  /* リスト構造体自体も解放 */
}
```

</details>

## 解答例

<details>
<summary>解答例（初心者向け）</summary>

```c
/*
 * 連結リストの実装
 * 学べる内容：ポインタ、malloc/free、自己参照構造体、データ構造設計
 * コンパイル：gcc -o linked_list linked_list.c
 * 実行：./linked_list
 */
#include <stdio.h>
#include <stdlib.h>

/* --- データ構造 --- */

typedef struct Node {
    int data;
    struct Node *next;
} Node;

typedef struct {
    Node *head;
    int size;
} LinkedList;

/* --- ノード作成（内部関数）--- */

Node *create_node(int data) {
    Node *node = (Node *)malloc(sizeof(Node));
    if (node == NULL) return NULL;
    node->data = data;
    node->next = NULL;
    return node;
}

/* --- リスト作成・破棄 --- */

LinkedList *list_create(void) {
    LinkedList *list = (LinkedList *)malloc(sizeof(LinkedList));
    if (list == NULL) return NULL;
    list->head = NULL;
    list->size = 0;
    return list;
}

void list_destroy(LinkedList *list) {
    if (list == NULL) return;
    Node *cur = list->head;
    while (cur != NULL) {
        Node *next = cur->next;
        free(cur);
        cur = next;
    }
    free(list);
}

/* --- 追加 --- */

int list_push_front(LinkedList *list, int data) {
    Node *node = create_node(data);
    if (node == NULL) return 0;
    node->next = list->head;
    list->head = node;
    list->size++;
    return 1;
}

int list_push_back(LinkedList *list, int data) {
    Node *node = create_node(data);
    if (node == NULL) return 0;

    if (list->head == NULL) {
        list->head = node;
    } else {
        Node *cur = list->head;
        while (cur->next != NULL) {
            cur = cur->next;
        }
        cur->next = node;
    }
    list->size++;
    return 1;
}

int list_insert_at(LinkedList *list, int index, int data) {
    if (index < 0 || index > list->size) return 0;

    if (index == 0) {
        return list_push_front(list, data);
    }

    Node *node = create_node(data);
    if (node == NULL) return 0;

    Node *cur = list->head;
    int i;
    for (i = 0; i < index - 1; i++) {
        cur = cur->next;
    }
    node->next = cur->next;
    cur->next = node;
    list->size++;
    return 1;
}

/* --- 削除 --- */

int list_pop_front(LinkedList *list, int *out) {
    if (list->head == NULL) return 0;

    Node *removed = list->head;
    *out = removed->data;
    list->head = removed->next;
    free(removed);
    list->size--;
    return 1;
}

int list_remove_at(LinkedList *list, int index, int *out) {
    if (index < 0 || index >= list->size) return 0;

    if (index == 0) {
        return list_pop_front(list, out);
    }

    Node *cur = list->head;
    int i;
    for (i = 0; i < index - 1; i++) {
        cur = cur->next;
    }
    Node *removed = cur->next;
    *out = removed->data;
    cur->next = removed->next;
    free(removed);
    list->size--;
    return 1;
}

/* --- 参照・検索 --- */

int list_get(const LinkedList *list, int index, int *out) {
    if (index < 0 || index >= list->size) return 0;

    Node *cur = list->head;
    int i;
    for (i = 0; i < index; i++) {
        cur = cur->next;
    }
    *out = cur->data;
    return 1;
}

int list_find(const LinkedList *list, int data) {
    Node *cur = list->head;
    int index = 0;
    while (cur != NULL) {
        if (cur->data == data) {
            return index;
        }
        cur = cur->next;
        index++;
    }
    return -1;  /* 見つからない */
}

int list_size(const LinkedList *list) {
    return list->size;
}

/* --- 表示 --- */

void list_print(const LinkedList *list) {
    Node *cur = list->head;
    printf("リスト: ");
    while (cur != NULL) {
        printf("[%d] -> ", cur->data);
        cur = cur->next;
    }
    printf("NULL (size=%d)\n", list->size);
}

/* --- テスト --- */

int main(void) {
    int val;

    printf("=== 連結リスト テスト ===\n\n");

    LinkedList *list = list_create();
    if (list == NULL) {
        printf("リスト作成に失敗しました。\n");
        return 1;
    }

    /* push_front */
    list_push_front(list, 30);
    list_push_front(list, 20);
    list_push_front(list, 10);
    printf("push_front: 30, 20, 10\n");
    list_print(list);

    /* push_back */
    list_push_back(list, 40);
    list_push_back(list, 50);
    printf("\npush_back: 40, 50\n");
    list_print(list);

    /* insert_at */
    list_insert_at(list, 2, 25);
    printf("\ninsert_at(2, 25):\n");
    list_print(list);

    /* pop_front */
    list_pop_front(list, &val);
    printf("\npop_front: %d\n", val);
    list_print(list);

    /* remove_at */
    list_remove_at(list, 2, &val);
    printf("\nremove_at(2): %d\n", val);
    list_print(list);

    /* get */
    if (list_get(list, 1, &val)) {
        printf("\nget(1): %d\n", val);
    }

    /* find */
    int idx = list_find(list, 40);
    printf("find(40): index=%d\n", idx);
    idx = list_find(list, 99);
    printf("find(99): %s\n", idx < 0 ? "not found" : "found");

    /* 破棄 */
    list_destroy(list);
    printf("\nリストを破棄しました。\n");

    return 0;
}
```

<summary>解答例（改良版 ─ 双方向連結リスト）</summary>

双方向連結リスト（doubly linked list）は各ノードが前後両方のポインタを持ち、末尾からの操作も O(1) で行えます。

```c
/*
 * 双方向連結リスト
 * 学べる内容：双方向ポインタ、末尾ポインタの管理
 * コンパイル：gcc -o dlinked_list dlinked_list.c
 * 実行：./dlinked_list
 */
#include <stdio.h>
#include <stdlib.h>

typedef struct DNode {
    int data;
    struct DNode *prev;
    struct DNode *next;
} DNode;

typedef struct {
    DNode *head;
    DNode *tail;
    int size;
} DLinkedList;

DNode *dnode_create(int data) {
    DNode *node = (DNode *)malloc(sizeof(DNode));
    if (node == NULL) return NULL;
    node->data = data;
    node->prev = NULL;
    node->next = NULL;
    return node;
}

DLinkedList *dlist_create(void) {
    DLinkedList *list = (DLinkedList *)malloc(sizeof(DLinkedList));
    if (list == NULL) return NULL;
    list->head = NULL;
    list->tail = NULL;
    list->size = 0;
    return list;
}

void dlist_destroy(DLinkedList *list) {
    if (list == NULL) return;
    DNode *cur = list->head;
    while (cur != NULL) {
        DNode *next = cur->next;
        free(cur);
        cur = next;
    }
    free(list);
}

int dlist_push_front(DLinkedList *list, int data) {
    DNode *node = dnode_create(data);
    if (node == NULL) return 0;

    if (list->head == NULL) {
        list->head = list->tail = node;
    } else {
        node->next = list->head;
        list->head->prev = node;
        list->head = node;
    }
    list->size++;
    return 1;
}

int dlist_push_back(DLinkedList *list, int data) {
    DNode *node = dnode_create(data);
    if (node == NULL) return 0;

    if (list->tail == NULL) {
        list->head = list->tail = node;
    } else {
        node->prev = list->tail;
        list->tail->next = node;
        list->tail = node;
    }
    list->size++;
    return 1;
}

int dlist_pop_front(DLinkedList *list, int *out) {
    if (list->head == NULL) return 0;
    DNode *removed = list->head;
    *out = removed->data;
    list->head = removed->next;
    if (list->head != NULL) {
        list->head->prev = NULL;
    } else {
        list->tail = NULL;
    }
    free(removed);
    list->size--;
    return 1;
}

int dlist_pop_back(DLinkedList *list, int *out) {
    if (list->tail == NULL) return 0;
    DNode *removed = list->tail;
    *out = removed->data;
    list->tail = removed->prev;
    if (list->tail != NULL) {
        list->tail->next = NULL;
    } else {
        list->head = NULL;
    }
    free(removed);
    list->size--;
    return 1;
}

void dlist_print_forward(const DLinkedList *list) {
    DNode *cur = list->head;
    printf("前方: ");
    while (cur != NULL) {
        printf("[%d] <-> ", cur->data);
        cur = cur->next;
    }
    printf("NULL (size=%d)\n", list->size);
}

void dlist_print_backward(const DLinkedList *list) {
    DNode *cur = list->tail;
    printf("後方: ");
    while (cur != NULL) {
        printf("[%d] <-> ", cur->data);
        cur = cur->prev;
    }
    printf("NULL\n");
}

int main(void) {
    int val;
    DLinkedList *list = dlist_create();

    printf("=== 双方向連結リスト テスト ===\n\n");

    dlist_push_back(list, 10);
    dlist_push_back(list, 20);
    dlist_push_back(list, 30);
    dlist_push_front(list, 5);

    dlist_print_forward(list);
    dlist_print_backward(list);

    dlist_pop_front(list, &val);
    printf("\npop_front: %d\n", val);

    dlist_pop_back(list, &val);
    printf("pop_back: %d\n", val);

    dlist_print_forward(list);

    dlist_destroy(list);
    printf("\nリストを破棄しました。\n");

    return 0;
}
```

**初心者向け（単方向）との違い:**
- 各ノードが `prev` ポインタも持つ → 後方への走査が可能
- `tail` ポインタを管理 → 末尾への追加・削除が O(1)
- `pop_back` が効率的に実装できる → 単方向リストでは O(n) かかる操作が O(1) に
- ポインタの更新箇所が増えるため、バグが入りやすい → 慎重な実装が必要

</details>
