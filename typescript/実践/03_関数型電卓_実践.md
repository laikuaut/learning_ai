# 実践課題03：関数型電卓 ★2

> **難易度**: ★★☆☆☆（基礎）
> **前提知識**: 第1章（TypeScriptの基本）、第2章（インターフェースと型エイリアス）、第3章（関数の型）
> **課題の種類**: ミニプロジェクト
> **学習目標**: 関数の型注釈、コールバック関数（callback function）、オーバーロード（overload）を実践的に使えるようになる

---

## 完成イメージ

```
===== 関数型電卓 =====

--- 基本演算 ---
10 + 3 = 13
10 - 3 = 7
10 * 3 = 30
10 / 3 = 3.3333

--- 計算履歴 ---
[1] 10 + 3 = 13
[2] 10 - 3 = 7
[3] 10 * 3 = 30
[4] 10 / 3 = 3.3333

--- 連鎖計算 ---
初期値: 100
  → +50 → 150
  → *2  → 300
  → -80 → 220
  → /4  → 55
最終結果: 55

--- カスタム演算の登録 ---
10 % 3 = 1（剰余）
2 ^ 8 = 256（べき乗）

--- エラーハンドリング ---
0での除算: エラー - 0で割ることはできません
```

---

## 課題の要件

1. 四則演算を行う関数を型付きで定義する
2. 演算関数の型を型エイリアスで定義する（`type Operation = ...`）
3. 計算履歴を管理し、一覧表示できるようにする
4. 連鎖計算（メソッドチェーンのように値を引き継ぐ計算）を実装する
5. カスタム演算を登録できるようにする（コールバック関数の活用）
6. 0での除算のエラーハンドリングを行う

## 解答例

<details>
<summary>解答例（初心者向け）</summary>

```typescript
// 関数型電卓
// 学べる内容：関数の型、型エイリアス、コールバック関数
// 実行方法：npx ts-node 03_calculator.ts

// --- 型定義 ---
type BinaryOperation = (a: number, b: number) => number;

interface CalculationRecord {
  leftOperand: number;
  operator: string;
  rightOperand: number;
  result: number;
}

// --- 四則演算の関数 ---
const add: BinaryOperation = (a: number, b: number): number => a + b;
const subtract: BinaryOperation = (a: number, b: number): number => a - b;
const multiply: BinaryOperation = (a: number, b: number): number => a * b;
const divide: BinaryOperation = (a: number, b: number): number => {
  if (b === 0) {
    throw new Error("0で割ることはできません");
  }
  return a / b;
};

// --- 演算子とデータの対応表 ---
const operators: { [key: string]: BinaryOperation } = {
  "+": add,
  "-": subtract,
  "*": multiply,
  "/": divide,
};

// --- 計算履歴 ---
const history: CalculationRecord[] = [];

// --- 計算実行関数 ---
function calculate(a: number, operator: string, b: number): number {
  const operation: BinaryOperation | undefined = operators[operator];
  if (operation === undefined) {
    throw new Error(`未知の演算子: ${operator}`);
  }
  const result: number = operation(a, b);

  // 履歴に記録
  history.push({
    leftOperand: a,
    operator: operator,
    rightOperand: b,
    result: result,
  });

  return result;
}

// --- 履歴表示 ---
function showHistory(): void {
  console.log("\n--- 計算履歴 ---");
  for (let i = 0; i < history.length; i++) {
    const rec: CalculationRecord = history[i];
    const resultStr: string =
      Number.isInteger(rec.result)
        ? rec.result.toString()
        : rec.result.toFixed(4);
    console.log(
      `[${i + 1}] ${rec.leftOperand} ${rec.operator} ${rec.rightOperand} = ${resultStr}`
    );
  }
}

// --- 連鎖計算 ---
function chainCalculate(
  initial: number,
  steps: { operator: string; operand: number }[]
): number {
  let result: number = initial;
  console.log(`初期値: ${initial}`);

  for (const step of steps) {
    const prev: number = result;
    result = calculate(prev, step.operator, step.operand);
    console.log(`  → ${step.operator}${step.operand} → ${result}`);
  }

  console.log(`最終結果: ${result}`);
  return result;
}

// --- カスタム演算登録 ---
function registerOperation(symbol: string, operation: BinaryOperation): void {
  operators[symbol] = operation;
}

// --- メイン処理 ---
console.log("===== 関数型電卓 =====\n");

// 基本演算
console.log("--- 基本演算 ---");
const a: number = 10;
const b: number = 3;
console.log(`${a} + ${b} = ${calculate(a, "+", b)}`);
console.log(`${a} - ${b} = ${calculate(a, "-", b)}`);
console.log(`${a} * ${b} = ${calculate(a, "*", b)}`);
console.log(`${a} / ${b} = ${calculate(a, "/", b).toFixed(4)}`);

// 履歴表示
showHistory();

// 連鎖計算
console.log("\n--- 連鎖計算 ---");
chainCalculate(100, [
  { operator: "+", operand: 50 },
  { operator: "*", operand: 2 },
  { operator: "-", operand: 80 },
  { operator: "/", operand: 4 },
]);

// カスタム演算
console.log("\n--- カスタム演算の登録 ---");
registerOperation("%", (a: number, b: number): number => a % b);
registerOperation("^", (a: number, b: number): number => Math.pow(a, b));
console.log(`10 % 3 = ${calculate(10, "%", 3)}（剰余）`);
console.log(`2 ^ 8 = ${calculate(2, "^", 8)}（べき乗）`);

// エラーハンドリング
console.log("\n--- エラーハンドリング ---");
try {
  calculate(10, "/", 0);
} catch (error) {
  if (error instanceof Error) {
    console.log(`0での除算: エラー - ${error.message}`);
  }
}
```

<summary>解答例（改良版 ─ オプショナル引数とレストパラメータ）</summary>

```typescript
// 関数型電卓（改良版）
// 学べる内容：オプショナル引数、デフォルト引数、レストパラメータ、関数オーバーロード
// 実行方法：npx ts-node 03_calculator_v2.ts

// --- 型定義 ---
type BinaryOperation = (a: number, b: number) => number;
type UnaryOperation = (a: number) => number;

interface CalculationRecord {
  expression: string;
  result: number;
  timestamp: Date;
}

interface Calculator {
  history: CalculationRecord[];
  operations: Map<string, BinaryOperation>;
  calculate: (a: number, op: string, b: number) => number;
  registerOp: (symbol: string, fn: BinaryOperation) => void;
  showHistory: (limit?: number) => void;
}

// --- 電卓ファクトリ関数 ---
function createCalculator(): Calculator {
  const history: CalculationRecord[] = [];
  const operations: Map<string, BinaryOperation> = new Map<string, BinaryOperation>([
    ["+", (a, b) => a + b],
    ["-", (a, b) => a - b],
    ["*", (a, b) => a * b],
    ["/", (a, b) => {
      if (b === 0) throw new Error("0で割ることはできません");
      return a / b;
    }],
  ]);

  function calculate(a: number, op: string, b: number): number {
    const fn: BinaryOperation | undefined = operations.get(op);
    if (!fn) throw new Error(`未知の演算子: ${op}`);

    const result: number = fn(a, b);
    history.push({
      expression: `${a} ${op} ${b} = ${Number.isInteger(result) ? result : result.toFixed(4)}`,
      result,
      timestamp: new Date(),
    });
    return result;
  }

  function registerOp(symbol: string, fn: BinaryOperation): void {
    operations.set(symbol, fn);
  }

  // limit はオプショナル引数（デフォルトは全件表示）
  function showHistory(limit: number = history.length): void {
    console.log("\n--- 計算履歴 ---");
    const start: number = Math.max(0, history.length - limit);
    for (let i = start; i < history.length; i++) {
      console.log(`[${i + 1}] ${history[i].expression}`);
    }
  }

  return { history, operations, calculate, registerOp, showHistory };
}

// --- 連鎖計算（レストパラメータ版） ---
function chainCalc(
  calc: Calculator,
  initial: number,
  ...steps: [string, number][]
): number {
  let result: number = initial;
  console.log(`初期値: ${initial}`);
  for (const [op, operand] of steps) {
    result = calc.calculate(result, op, operand);
    console.log(`  → ${op}${operand} → ${result}`);
  }
  console.log(`最終結果: ${result}`);
  return result;
}

// --- 複数の値をまとめて演算（レストパラメータ） ---
function sum(...numbers: number[]): number {
  return numbers.reduce((acc: number, n: number): number => acc + n, 0);
}

// --- メイン処理 ---
console.log("===== 関数型電卓 =====\n");

const calc: Calculator = createCalculator();

console.log("--- 基本演算 ---");
console.log(`10 + 3 = ${calc.calculate(10, "+", 3)}`);
console.log(`10 - 3 = ${calc.calculate(10, "-", 3)}`);
console.log(`10 * 3 = ${calc.calculate(10, "*", 3)}`);
console.log(`10 / 3 = ${calc.calculate(10, "/", 3).toFixed(4)}`);

calc.showHistory();

console.log("\n--- 連鎖計算 ---");
chainCalc(calc, 100, ["+", 50], ["*", 2], ["-", 80], ["/", 4]);

console.log("\n--- カスタム演算 ---");
calc.registerOp("%", (a, b) => a % b);
calc.registerOp("^", (a, b) => Math.pow(a, b));
console.log(`10 % 3 = ${calc.calculate(10, "%", 3)}（剰余）`);
console.log(`2 ^ 8 = ${calc.calculate(2, "^", 8)}（べき乗）`);

console.log("\n--- レストパラメータで合計 ---");
console.log(`sum(1,2,3,4,5) = ${sum(1, 2, 3, 4, 5)}`);

console.log("\n--- エラーハンドリング ---");
try {
  calc.calculate(10, "/", 0);
} catch (error) {
  if (error instanceof Error) {
    console.log(`0での除算: エラー - ${error.message}`);
  }
}

// 直近3件だけ表示
calc.showHistory(3);
```

**初心者向けとの違い:**
- `Map` を使って演算子を管理（`Map<string, BinaryOperation>`）
- `createCalculator` ファクトリ関数でカプセル化
- オプショナル引数（`limit?: number`）とデフォルト引数の実用例
- レストパラメータ（`...steps`）でタプル配列を受け取るパターン
- `sum(...numbers: number[])` でレストパラメータの基本パターン

</details>
