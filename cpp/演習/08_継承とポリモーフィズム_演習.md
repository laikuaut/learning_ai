## 問題1：図形クラスの階層（基本）

以下のクラス階層を作成してください。

- **基底クラス `Shape`**: 名前を返す `name()` と面積を返す `area()` を純粋仮想関数として定義
- **派生クラス `Circle`**: 半径を持ち、面積は π×r²
- **派生クラス `Rectangle`**: 幅と高さを持ち、面積は width×height
- **派生クラス `Triangle`**: 底辺と高さを持ち、面積は base×height/2

基底クラスのポインタ配列を使い、ポリモーフィズムで面積を計算してください。

**期待される出力例：**
```
円 の面積: 78.5398
長方形 の面積: 12
三角形 の面積: 10
合計面積: 100.54
```

<details>
<summary>ヒント</summary>

- 純粋仮想関数は `virtual double area() const = 0;` のように書きます
- 基底クラスに仮想デストラクタ `virtual ~Shape() = default;` を忘れずに定義してください
- ポインタ配列は `Shape* shapes[]` または `vector<Shape*>` を使います
- `M_PI` は `<cmath>` をインクルードすると使えます

</details>

<details>
<summary>解答例</summary>

```cpp
#include <iostream>
#include <cmath>
#include <vector>
#include <memory>
using namespace std;

// 基底クラス（抽象クラス）
class Shape {
public:
    virtual ~Shape() = default;           // 仮想デストラクタ
    virtual string name() const = 0;      // 名前（純粋仮想関数）
    virtual double area() const = 0;      // 面積（純粋仮想関数）
};

// 円
class Circle : public Shape {
private:
    double radius;
public:
    Circle(double r) : radius(r) {}
    string name() const override { return "円"; }
    double area() const override { return M_PI * radius * radius; }
};

// 長方形
class Rectangle : public Shape {
private:
    double width, height;
public:
    Rectangle(double w, double h) : width(w), height(h) {}
    string name() const override { return "長方形"; }
    double area() const override { return width * height; }
};

// 三角形
class Triangle : public Shape {
private:
    double base, height;
public:
    Triangle(double b, double h) : base(b), height(h) {}
    string name() const override { return "三角形"; }
    double area() const override { return base * height / 2.0; }
};

int main() {
    // unique_ptrでメモリ管理
    vector<unique_ptr<Shape>> shapes;
    shapes.push_back(make_unique<Circle>(5.0));
    shapes.push_back(make_unique<Rectangle>(3.0, 4.0));
    shapes.push_back(make_unique<Triangle>(4.0, 5.0));

    double total = 0.0;
    for (const auto& s : shapes) {
        cout << s->name() << " の面積: " << s->area() << endl;
        total += s->area();
    }
    cout << "合計面積: " << total << endl;

    return 0;
}
```

## 問題2：動物クラスの階層（基本）

以下のクラス階層を作成してください。

- **基底クラス `Animal`**: 名前を持ち、`speak()` を仮想関数として定義
- **派生クラス `Dog`**: `speak()` で "ワンワン!" を返す
- **派生クラス `Cat`**: `speak()` で "ニャー!" を返す
- **派生クラス `Bird`**: `speak()` で "ピヨピヨ!" を返す

各動物を生成し、ポリモーフィズムで鳴き声を表示してください。

**期待される出力例：**
```
ポチ（犬）: ワンワン!
ミケ（猫）: ニャー!
ピーちゃん（鳥）: ピヨピヨ!
```

<details>
<summary>ヒント</summary>

- `Animal` クラスに名前（string）と種類名を返す仮想関数を持たせます
- `override` キーワードを付けると、正しくオーバーライドしているかコンパイラがチェックしてくれます
- 基底クラスにコンストラクタで名前を渡し、派生クラスのコンストラクタで基底クラスのコンストラクタを呼びます

</details>

<details>
<summary>解答例</summary>

```cpp
#include <iostream>
#include <string>
#include <vector>
#include <memory>
using namespace std;

// 基底クラス
class Animal {
protected:
    string name;
public:
    Animal(const string& n) : name(n) {}
    virtual ~Animal() = default;

    string getName() const { return name; }
    virtual string typeName() const = 0;  // 種類名
    virtual string speak() const = 0;     // 鳴き声
};

// 犬
class Dog : public Animal {
public:
    Dog(const string& n) : Animal(n) {}
    string typeName() const override { return "犬"; }
    string speak() const override { return "ワンワン!"; }
};

// 猫
class Cat : public Animal {
public:
    Cat(const string& n) : Animal(n) {}
    string typeName() const override { return "猫"; }
    string speak() const override { return "ニャー!"; }
};

// 鳥
class Bird : public Animal {
public:
    Bird(const string& n) : Animal(n) {}
    string typeName() const override { return "鳥"; }
    string speak() const override { return "ピヨピヨ!"; }
};

int main() {
    vector<unique_ptr<Animal>> animals;
    animals.push_back(make_unique<Dog>("ポチ"));
    animals.push_back(make_unique<Cat>("ミケ"));
    animals.push_back(make_unique<Bird>("ピーちゃん"));

    for (const auto& a : animals) {
        cout << a->getName() << "（" << a->typeName() << "）: "
             << a->speak() << endl;
    }

    return 0;
}
```

## 問題3：従業員管理システム（応用）

以下のクラス階層で従業員の給与計算システムを作成してください。

- **基底クラス `Employee`**: 名前と基本給を持ち、`calcSalary()` を仮想関数として定義
- **派生クラス `FullTimeEmployee`**: 月給制（基本給そのまま）
- **派生クラス `PartTimeEmployee`**: 時給制（時給 × 勤務時間）
- **派生クラス `Manager`**: 月給制 + 役職手当

全従業員の給与一覧と総支給額を表示してください。

**期待される出力例：**
```
=== 給与一覧 ===
[正社員] 田中一郎: 300,000円
[パート] 鈴木花子: 176,000円（時給1,100円 × 160時間）
[管理職] 佐藤部長: 450,000円（基本給350,000円 + 役職手当100,000円）

総支給額: 926,000円
```

<details>
<summary>ヒント</summary>

- `PartTimeEmployee` は時給と勤務時間をメンバ変数として追加します
- `Manager` は役職手当をメンバ変数として追加します
- 数値のカンマ区切り表示は `locale` を使うか、自作関数で実装します
- 表示形式を統一するため、`printInfo()` 仮想関数を定義するのも良い方法です

</details>

<details>
<summary>解答例</summary>

```cpp
#include <iostream>
#include <string>
#include <vector>
#include <memory>
using namespace std;

// カンマ区切りで数値を文字列にする簡易関数
string formatNumber(int n) {
    string s = to_string(n);
    int len = s.length();
    string result;
    for (int i = 0; i < len; i++) {
        if (i > 0 && (len - i) % 3 == 0) result += ',';
        result += s[i];
    }
    return result;
}

// 基底クラス
class Employee {
protected:
    string name;
    int baseSalary;
public:
    Employee(const string& n, int salary)
        : name(n), baseSalary(salary) {}
    virtual ~Employee() = default;

    virtual int calcSalary() const = 0;
    virtual void printInfo() const = 0;
};

// 正社員
class FullTimeEmployee : public Employee {
public:
    FullTimeEmployee(const string& n, int salary)
        : Employee(n, salary) {}

    int calcSalary() const override {
        return baseSalary;
    }

    void printInfo() const override {
        cout << "[正社員] " << name << ": "
             << formatNumber(calcSalary()) << "円" << endl;
    }
};

// パートタイム
class PartTimeEmployee : public Employee {
private:
    int hourlyRate;    // 時給
    int hoursWorked;   // 勤務時間
public:
    PartTimeEmployee(const string& n, int rate, int hours)
        : Employee(n, 0), hourlyRate(rate), hoursWorked(hours) {}

    int calcSalary() const override {
        return hourlyRate * hoursWorked;
    }

    void printInfo() const override {
        cout << "[パート] " << name << ": "
             << formatNumber(calcSalary()) << "円"
             << "（時給" << formatNumber(hourlyRate)
             << "円 × " << hoursWorked << "時間）" << endl;
    }
};

// 管理職
class Manager : public Employee {
private:
    int positionAllowance;  // 役職手当
public:
    Manager(const string& n, int salary, int allowance)
        : Employee(n, salary), positionAllowance(allowance) {}

    int calcSalary() const override {
        return baseSalary + positionAllowance;
    }

    void printInfo() const override {
        cout << "[管理職] " << name << ": "
             << formatNumber(calcSalary()) << "円"
             << "（基本給" << formatNumber(baseSalary)
             << "円 + 役職手当" << formatNumber(positionAllowance)
             << "円）" << endl;
    }
};

int main() {
    vector<unique_ptr<Employee>> employees;
    employees.push_back(make_unique<FullTimeEmployee>("田中一郎", 300000));
    employees.push_back(make_unique<PartTimeEmployee>("鈴木花子", 1100, 160));
    employees.push_back(make_unique<Manager>("佐藤部長", 350000, 100000));

    cout << "=== 給与一覧 ===" << endl;
    int total = 0;
    for (const auto& emp : employees) {
        emp->printInfo();
        total += emp->calcSalary();
    }

    cout << "\n総支給額: " << formatNumber(total) << "円" << endl;

    return 0;
}
```

</details>

## 問題4：RPGキャラクターシステム（応用）

RPGのキャラクターを表すクラス階層を作成してください。

- **基底クラス `Character`**: 名前、HP、攻撃力を持ち、`attack()` と `specialMove()` を仮想関数として定義
- **派生クラス `Warrior`**: 通常攻撃は攻撃力そのまま、必殺技は攻撃力×2
- **派生クラス `Mage`**: 通常攻撃は攻撃力×0.5、魔法攻撃（必殺技）は攻撃力×3
- **派生クラス `Archer`**: 通常攻撃は攻撃力×0.8、連射（必殺技）は攻撃力×1.5を3回

バトルのシミュレーションを表示してください。

**期待される出力例：**
```
=== パーティ紹介 ===
戦士アレス (HP:150, 攻撃力:30)
魔法使いメルリン (HP:80, 攻撃力:40)
弓使いロビン (HP:100, 攻撃力:25)

=== バトル開始 ===
戦士アレス の攻撃！ 30 ダメージ！
戦士アレス の必殺技「パワースラッシュ」！ 60 ダメージ！
魔法使いメルリン の攻撃！ 20 ダメージ！
魔法使いメルリン の必殺技「ファイアボール」！ 120 ダメージ！
弓使いロビン の攻撃！ 20 ダメージ！
弓使いロビン の必殺技「三連射」！ 37 + 37 + 37 = 111 ダメージ！
```

<details>
<summary>ヒント</summary>

- `Character` 基底クラスのコンストラクタで名前、HP、攻撃力を初期化します
- 各派生クラスの `attack()` は `int` を返し、ダメージ量を表します
- `specialMove()` は名前と効果を `cout` で表示しつつダメージも返す設計にします
- `int` へのキャストは `static_cast<int>()` で行います

</details>

<details>
<summary>解答例</summary>

```cpp
#include <iostream>
#include <string>
#include <vector>
#include <memory>
using namespace std;

// 基底クラス
class Character {
protected:
    string name;
    int hp;
    int attackPower;
public:
    Character(const string& n, int h, int atk)
        : name(n), hp(h), attackPower(atk) {}
    virtual ~Character() = default;

    string getName() const { return name; }
    int getHP() const { return hp; }
    int getAttackPower() const { return attackPower; }

    virtual string className() const = 0;

    // 通常攻撃（ダメージを返す）
    virtual int attack() const = 0;

    // 必殺技（表示込み、ダメージを返す）
    virtual int specialMove() const = 0;

    void introduce() const {
        cout << className() << name
             << " (HP:" << hp << ", 攻撃力:" << attackPower << ")" << endl;
    }
};

// 戦士
class Warrior : public Character {
public:
    Warrior(const string& n, int h, int atk) : Character(n, h, atk) {}
    string className() const override { return "戦士"; }

    int attack() const override {
        int dmg = attackPower;
        cout << "戦士" << name << " の攻撃！ " << dmg << " ダメージ！" << endl;
        return dmg;
    }

    int specialMove() const override {
        int dmg = attackPower * 2;
        cout << "戦士" << name << " の必殺技「パワースラッシュ」！ "
             << dmg << " ダメージ！" << endl;
        return dmg;
    }
};

// 魔法使い
class Mage : public Character {
public:
    Mage(const string& n, int h, int atk) : Character(n, h, atk) {}
    string className() const override { return "魔法使い"; }

    int attack() const override {
        int dmg = static_cast<int>(attackPower * 0.5);
        cout << "魔法使い" << name << " の攻撃！ " << dmg << " ダメージ！" << endl;
        return dmg;
    }

    int specialMove() const override {
        int dmg = attackPower * 3;
        cout << "魔法使い" << name << " の必殺技「ファイアボール」！ "
             << dmg << " ダメージ！" << endl;
        return dmg;
    }
};

// 弓使い
class Archer : public Character {
public:
    Archer(const string& n, int h, int atk) : Character(n, h, atk) {}
    string className() const override { return "弓使い"; }

    int attack() const override {
        int dmg = static_cast<int>(attackPower * 0.8);
        cout << "弓使い" << name << " の攻撃！ " << dmg << " ダメージ！" << endl;
        return dmg;
    }

    int specialMove() const override {
        int each = static_cast<int>(attackPower * 1.5);
        int total = each * 3;
        cout << "弓使い" << name << " の必殺技「三連射」！ "
             << each << " + " << each << " + " << each
             << " = " << total << " ダメージ！" << endl;
        return total;
    }
};

int main() {
    vector<unique_ptr<Character>> party;
    party.push_back(make_unique<Warrior>("アレス", 150, 30));
    party.push_back(make_unique<Mage>("メルリン", 80, 40));
    party.push_back(make_unique<Archer>("ロビン", 100, 25));

    cout << "=== パーティ紹介 ===" << endl;
    for (const auto& c : party) {
        c->introduce();
    }

    cout << "\n=== バトル開始 ===" << endl;
    for (const auto& c : party) {
        c->attack();
        c->specialMove();
    }

    return 0;
}
```

</details>

## 問題5：UIウィジェット階層（チャレンジ）

GUIのウィジェットを表すクラス階層を作成してください。
各ウィジェットはテキストベースで「描画」します。

- **基底クラス `Widget`**: `draw()` と `getType()` を純粋仮想関数として定義。位置（x, y）と幅、高さを持つ
- **派生クラス `Button`**: ラベルテキストを持ち、枠付きで描画
- **派生クラス `TextBox`**: 入力テキストを持ち、下線付きで描画
- **派生クラス `CheckBox`**: チェック状態を持ち、`[x]` or `[ ]` で描画
- **派生クラス `Panel`**: 子ウィジェットを複数持ち、まとめて描画

`Panel` に複数のウィジェットを追加して、一括描画してください。

**期待される出力例：**
```
=== UI描画 ===
Panel "メインパネル" を描画:
  +----------+
  | 送信     |  [Button] (10, 10)
  +----------+
  [入力してください___]  [TextBox] (10, 50)
  [x] 同意する  [CheckBox] (10, 90)
  [ ] 通知を受け取る  [CheckBox] (10, 120)
```

<details>
<summary>ヒント</summary>

- `Panel` クラスは `vector<unique_ptr<Widget>>` で子ウィジェットを管理します
- `addChild()` メンバ関数で子ウィジェットを追加します
- `draw()` 内で全子ウィジェットの `draw()` を呼びます
- テキストベースの描画なので、見た目のフォーマットは自由にアレンジしてかまいません

</details>

<details>
<summary>解答例</summary>

```cpp
#include <iostream>
#include <string>
#include <vector>
#include <memory>
using namespace std;

// 基底クラス
class Widget {
protected:
    int x, y;         // 位置
    int width, height; // サイズ
public:
    Widget(int x, int y, int w, int h)
        : x(x), y(y), width(w), height(h) {}
    virtual ~Widget() = default;

    virtual string getType() const = 0;
    virtual void draw(int indent = 0) const = 0;

    // インデント用ヘルパー
    string indentStr(int level) const {
        return string(level * 2, ' ');
    }
};

// ボタン
class Button : public Widget {
private:
    string label;
public:
    Button(int x, int y, int w, int h, const string& lbl)
        : Widget(x, y, w, h), label(lbl) {}

    string getType() const override { return "Button"; }

    void draw(int indent = 0) const override {
        string pad = indentStr(indent);
        string border(label.size() + 4, '-');
        cout << pad << "+" << border << "+" << endl;
        cout << pad << "| " << label << "   |  [" << getType()
             << "] (" << x << ", " << y << ")" << endl;
        cout << pad << "+" << border << "+" << endl;
    }
};

// テキストボックス
class TextBox : public Widget {
private:
    string placeholder;
public:
    TextBox(int x, int y, int w, int h, const string& ph)
        : Widget(x, y, w, h), placeholder(ph) {}

    string getType() const override { return "TextBox"; }

    void draw(int indent = 0) const override {
        string pad = indentStr(indent);
        cout << pad << "[" << placeholder << "___]  ["
             << getType() << "] (" << x << ", " << y << ")" << endl;
    }
};

// チェックボックス
class CheckBox : public Widget {
private:
    string label;
    bool checked;
public:
    CheckBox(int x, int y, int w, int h, const string& lbl, bool chk)
        : Widget(x, y, w, h), label(lbl), checked(chk) {}

    string getType() const override { return "CheckBox"; }

    void draw(int indent = 0) const override {
        string pad = indentStr(indent);
        cout << pad << (checked ? "[x] " : "[ ] ") << label
             << "  [" << getType() << "] ("
             << x << ", " << y << ")" << endl;
    }

    void toggle() { checked = !checked; }
};

// パネル（子ウィジェットを保持）
class Panel : public Widget {
private:
    string title;
    vector<unique_ptr<Widget>> children;
public:
    Panel(int x, int y, int w, int h, const string& t)
        : Widget(x, y, w, h), title(t) {}

    string getType() const override { return "Panel"; }

    // 子ウィジェットを追加
    void addChild(unique_ptr<Widget> child) {
        children.push_back(move(child));
    }

    void draw(int indent = 0) const override {
        string pad = indentStr(indent);
        cout << pad << "Panel \"" << title << "\" を描画:" << endl;
        for (const auto& child : children) {
            child->draw(indent + 1);
        }
    }
};

int main() {
    cout << "=== UI描画 ===" << endl;

    // パネルを作成
    auto panel = make_unique<Panel>(0, 0, 400, 300, "メインパネル");

    // 子ウィジェットを追加
    panel->addChild(make_unique<Button>(10, 10, 80, 30, "送信"));
    panel->addChild(make_unique<TextBox>(10, 50, 200, 30, "入力してください"));
    panel->addChild(make_unique<CheckBox>(10, 90, 150, 20, "同意する", true));
    panel->addChild(make_unique<CheckBox>(10, 120, 150, 20, "通知を受け取る", false));

    // 一括描画
    panel->draw();

    return 0;
}
```

</details>

## 問題6：デザインパターン - Strategyパターン（チャレンジ）

Strategyパターンを使って、異なるソートアルゴリズムを切り替え可能なシステムを作成してください。

- **インターフェース `SortStrategy`**: `sort()` と `name()` を純粋仮想関数として定義
- **実装クラス `BubbleSort`**: バブルソートを実装
- **実装クラス `SelectionSort`**: 選択ソートを実装
- **コンテキスト `Sorter`**: `SortStrategy` を保持し、`setStrategy()` で切り替え可能

**期待される出力例：**
```
元のデータ: 64 34 25 12 22 11 90

バブルソートで整列:
ソート結果: 11 12 22 25 34 64 90

選択ソートで整列:
ソート結果: 11 12 22 25 34 64 90
```

<details>
<summary>ヒント</summary>

- `SortStrategy` は `virtual void sort(vector<int>& data) = 0;` を持ちます
- `Sorter` クラスは `unique_ptr<SortStrategy>` をメンバとして持ちます
- `setStrategy()` で新しい戦略に切り替えます
- バブルソート: 隣接要素を比較して交換を繰り返す
- 選択ソート: 未ソート部分から最小値を見つけて先頭と交換

</details>

<details>
<summary>解答例</summary>

```cpp
#include <iostream>
#include <vector>
#include <memory>
using namespace std;

// ソート戦略インターフェース
class SortStrategy {
public:
    virtual ~SortStrategy() = default;
    virtual void sort(vector<int>& data) = 0;
    virtual string name() const = 0;
};

// バブルソート
class BubbleSort : public SortStrategy {
public:
    void sort(vector<int>& data) override {
        int n = data.size();
        for (int i = 0; i < n - 1; i++) {
            for (int j = 0; j < n - i - 1; j++) {
                if (data[j] > data[j + 1]) {
                    swap(data[j], data[j + 1]);
                }
            }
        }
    }
    string name() const override { return "バブルソート"; }
};

// 選択ソート
class SelectionSort : public SortStrategy {
public:
    void sort(vector<int>& data) override {
        int n = data.size();
        for (int i = 0; i < n - 1; i++) {
            int minIdx = i;
            for (int j = i + 1; j < n; j++) {
                if (data[j] < data[minIdx]) {
                    minIdx = j;
                }
            }
            swap(data[i], data[minIdx]);
        }
    }
    string name() const override { return "選択ソート"; }
};

// コンテキスト
class Sorter {
private:
    unique_ptr<SortStrategy> strategy;
public:
    void setStrategy(unique_ptr<SortStrategy> s) {
        strategy = move(s);
    }

    void performSort(vector<int>& data) {
        if (!strategy) {
            cout << "ソート戦略が設定されていません" << endl;
            return;
        }
        cout << strategy->name() << "で整列:" << endl;
        strategy->sort(data);
    }
};

// 配列を表示
void printData(const vector<int>& data) {
    for (int i = 0; i < (int)data.size(); i++) {
        if (i > 0) cout << " ";
        cout << data[i];
    }
    cout << endl;
}

int main() {
    vector<int> original = {64, 34, 25, 12, 22, 11, 90};
    cout << "元のデータ: ";
    printData(original);

    Sorter sorter;

    // バブルソートで整列
    cout << endl;
    vector<int> data1 = original;
    sorter.setStrategy(make_unique<BubbleSort>());
    sorter.performSort(data1);
    cout << "ソート結果: ";
    printData(data1);

    // 選択ソートで整列
    cout << endl;
    vector<int> data2 = original;
    sorter.setStrategy(make_unique<SelectionSort>());
    sorter.performSort(data2);
    cout << "ソート結果: ";
    printData(data2);

    return 0;
}
```
