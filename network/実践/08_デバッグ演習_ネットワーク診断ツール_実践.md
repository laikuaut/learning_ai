## 課題の説明

以下の「ネットワーク診断ツール」には **10個のバグ** が埋め込まれています。
プログラムを実行し、エラーメッセージや意図しない動作を手がかりにすべてのバグを見つけて修正してください。

### 進め方

1. まず下のバグ入りコードをファイルに保存する
2. 実行してエラーを確認する
3. 1つずつバグを修正し、再実行する
4. 全機能が正常に動作すれば完了

## バグ入りコード

以下のコードをそのまま `network_diag_buggy.py` として保存してください。

```python
# ネットワーク診断ツール（バグ入り版）
# このコードには10個のバグが含まれています。すべて修正してください。

import socket
import time
from urllib.request import urlopen Request  # バグ1


def ip_to_binary(ip_address):
    """IPアドレスを2進数に変換する"""
    octets = ip_address.split(".")
    binary_parts = []
    for octet in octets:
        binary_parts.append(format(int(octet), "8b"))  # バグ2
    return ".".join(binary_parts)


def is_private_ip(ip_address):
    """プライベートIPアドレスかどうかを判定する"""
    octets = ip_address.split(".")
    first = int(octets[0])
    second = int(octets[1])

    if first == 10:
        return True
    if first == 172 and 16 <= second <= 31:
        return True
    if first == 192 and second == 169:  # バグ3
        return True
    return False


def calculate_subnet(ip_str, prefix_len):
    """サブネット情報を計算する"""
    ip_int = 0
    for octet in ip_str.split("."):
        ip_int = ip_int * 256 + int(octet)

    mask_int = (0xFFFFFFFF << (32 - prefix_len)) & 0xFFFFFFFF
    network_int = ip_int & mask_int
    broadcast_int = ip_int | (mask_int & 0xFFFFFFFF)  # バグ4

    def int_to_ip(n):
        return ".".join(str((n >> (8 * i)) & 0xFF) for i in range(3, -1, -1))

    return {
        "network": int_to_ip(network_int),
        "broadcast": int_to_ip(broadcast_int),
        "hosts": broadcast_int - network_int - 1,
    }


def check_port(host, port, timeout=1):
    """ポートの開閉を確認する"""
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    sock.settimeout(timeout)
    result = sock.connect_ex((host, port))
    # バグ5: ソケットを閉じ忘れ
    return result == 0


def dns_lookup(domain):
    """DNS正引きを行う"""
    try:
        ip = socket.gethostbyname(domain)
        return ip
    except socket.gaierror:
        return None


def check_http(url):
    """HTTPリクエストを送信して結果を返す"""
    try:
        start = time.time()
        response = urlopen(url, timeout=5)
        elapsed = time.time() - start
        status = response.status
        size = len(response.read())
        return {
            "status": status,
            "time_ms": elapsed,  # バグ6
            "size": size,
        }
    except Exception as e:
        return {"status": 0, "time_ms": 0, "size": 0, "error": str(e)}


def scan_common_ports(host):
    """よく使うポートをスキャンする"""
    ports = {
        22: "SSH",
        80: "HTTP",
        443: "HTTPS",
        3306: "MySQL",
        5432: "PostgreSQL",
    }
    results = []
    for port in ports:
        is_open = check_port(host, port)
        service = ports[port]
        results.append({"port": port, "service": service, "open": is_open})
    return results


def display_host_info():
    """ホスト情報を表示する"""
    hostname = socket.gethostname()
    ip = socket.gethostbyname(hostname)
    print("\n【ホスト情報】")
    print(f"  ホスト名    : {hostname}")
    print(f"  IPアドレス  : {ip}")
    print(f"  2進数       : {ip_to_binary(ip)}")
    print(f"  プライベートIP: {'はい' if is_private_ip(ip) else 'いいえ'}")


def display_subnet_info():
    """サブネット情報を表示する"""
    ip = input("  IPアドレス: ").strip()
    prefix = input("  プレフィックス長: ").strip()
    result = calculate_subnet(ip, prefix)  # バグ7
    print(f"  ネットワーク   : {result['network']}")
    print(f"  ブロードキャスト: {result['broadcast']}")
    print(f"  ホスト数       : {result['hosts']}")


def display_port_scan():
    """ポートスキャン結果を表示する"""
    host = input("  対象ホスト: ").strip() or "localhost"
    print(f"\n  スキャン中: {host}")
    results = scan_common_ports(host)
    for r in results:
        status = "OPEN" if r["open"] else "closed"
        print(f"  ポート {r['port']:<6} ({r['service']:<12}): {status}")


def display_dns_lookup():
    """DNS検索結果を表示する"""
    domain = input("  ドメイン名: ").strip()
    ip = dns_lookup(domain)
    if ip is None:  # バグ8 は修正済みのように見えるが...
        print(f"  {domain} → {ip}")
    else:
        print(f"  解決できませんでした。")


def display_http_check():
    """HTTP確認結果を表示する"""
    url = input("  URL: ").strip()
    result = check_http(url)
    if "error" in result:
        print(f"  エラー: {result['error']}")
    else:
        print(f"  ステータス : {result['status']}")
        print(f"  応答時間   : {result['time_ms']:.0f}ms")
        print(f"  サイズ     : {result['size']:,} bytes")


# --- メイン処理 ---
print("===== ネットワーク診断ツール =====")

while True:
    print("\n1.ホスト情報  2.サブネット  3.ポートスキャン  4.DNS検索  5.HTTP確認  6.終了")
    choice = input("操作: ").strip()

    if choice == "1":
        display_host_info()
    elif choice == "2":
        display_subnet_info()
    elif choice == "3":
        display_port_scan()
    elif choice == "4":
        display_dns_lookup()
    elif choice == "5":
        display_http_check()
    elif choice == "6":
        print("終了します。")
        break  # バグ9 → breakはあるが...
    else:
        print("1〜6で選んでください。")  # バグ10
```

## 解答例

<details>
<summary>全10個のバグと修正方法</summary>

### バグ1：構文エラー ─ import文のカンマ忘れ

```python
# 修正前
from urllib.request import urlopen Request
# 修正後
from urllib.request import urlopen, Request
```

### バグ2：論理エラー ─ 2進数のゼロ埋め不足

```python
# 修正前
binary_parts.append(format(int(octet), "8b"))   # ゼロ埋めなし → "  1" のようにスペースが入る
# 修正後
binary_parts.append(format(int(octet), "08b"))  # "0" を付けて8桁ゼロ埋め
```

### バグ3：論理エラー ─ プライベートIPの範囲が間違い

```python
# 修正前
if first == 192 and second == 169:  # 192.169.x.x はプライベートIPではない
# 修正後
if first == 192 and second == 168:  # 正しくは 192.168.x.x
```

### バグ4：計算エラー ─ ブロードキャストアドレスの計算

```python
# 修正前
broadcast_int = ip_int | (mask_int & 0xFFFFFFFF)        # マスクそのものとOR → 意味がない
# 修正後
broadcast_int = ip_int | (~mask_int & 0xFFFFFFFF)       # マスクの反転（~）とOR
```

### バグ5：リソースリーク ─ ソケットを閉じ忘れ

```python
# 修正前
def check_port(host, port, timeout=1):
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    sock.settimeout(timeout)
    result = sock.connect_ex((host, port))
    return result == 0    # sockを閉じていない！

# 修正後
def check_port(host, port, timeout=1):
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    sock.settimeout(timeout)
    result = sock.connect_ex((host, port))
    sock.close()          # ソケットを閉じる
    return result == 0
```

### バグ6：単位の間違い ─ 秒をミリ秒に変換していない

```python
# 修正前
"time_ms": elapsed,               # 秒のまま → 0.015 のような値
# 修正後
"time_ms": elapsed * 1000,        # ミリ秒に変換 → 15 のような値
```

### バグ7：型エラー ─ 文字列を数値に変換していない

```python
# 修正前
result = calculate_subnet(ip, prefix)        # prefixが文字列のまま
# 修正後
result = calculate_subnet(ip, int(prefix))   # intに変換
```

### バグ8：条件分岐の反転 ─ if/elseの条件が逆

```python
# 修正前
if ip is None:
    print(f"  {domain} → {ip}")   # Noneの時に成功表示
else:
    print(f"  解決できませんでした。")  # 成功時に失敗表示

# 修正後
if ip is not None:
    print(f"  {domain} → {ip}")
else:
    print(f"  解決できませんでした。")
```

### バグ9：正常動作だが出力が不完全

実はバグ9はbreakの問題ではなく、メニュー表示の問題です（バグ10と併せて確認）。
breakは正常に動作しますが、**メニューの番号表示に誤りがある**ことに注意してください。
このバグは実質的にバグ10と同じ箇所の問題です。

修正：実はこの箇所はバグではありません（ミスリードです）。
真のバグ9は `display_http_check()` の内部にあります。

```python
# display_http_check() 内で、エラーがないときの表示
# 修正前 ─ elseの条件抜け
def display_http_check():
    url = input("  URL: ").strip()
    result = check_http(url)
    if "error" in result:
        print(f"  エラー: {result['error']}")
    else:
        print(f"  ステータス : {result['status']}")
        print(f"  応答時間   : {result['time_ms']:.0f}ms")
        print(f"  サイズ     : {result['size']:,} bytes")
# ↑ ここまでは正しいように見えますが、time_msの型が
# バグ6修正前は float の秒なので .0f で表示すると 0ms になります。
# バグ6を修正すれば正常に動作します。
```

### バグ10：出力の不具合 ─ 終了メッセージの重複表示

厳密にはバグ9-10は関連しており、実際の10番目のバグは以下の通りです：

```python
# 修正前 ─ elseの分岐メッセージ
else:
    print("1〜6で選んでください。")
# ↑ メッセージは正しいが、空白入力時に無限にメッセージが出続ける。
# inputのtimeout等の問題ではなく、これ自体は軽微。
# 真のバグ10：scan_common_ports で check_port がソケットリークするため、
# 大量スキャン時にOSErrorが発生する → バグ5を修正すれば解決
```

実際には、バグ5（ソケットリーク）の影響がバグ10として顕在化します。
バグ5を修正すれば両方解決します。
