## 読解対象コード

```typescript
// tests/online-shop.spec.ts
import { test, expect } from '@playwright/test';

// ==================== 設定 ====================

const BASE_URL = 'https://demo-shop.example.com';

// ==================== ヘルパー関数 ====================

async function login(page, email: string, password: string) {
  await page.goto(`${BASE_URL}/login`);
  await page.locator('#email').fill(email);
  await page.locator('#password').fill(password);
  await page.locator('button[type="submit"]').click();
  await page.waitForURL('**/dashboard');
}

async function addToCart(page, productName: string) {
  await page.goto(`${BASE_URL}/products`);
  const product = page.locator('.product-card').filter({ hasText: productName });
  await product.locator('.add-to-cart-btn').click();
  await page.waitForTimeout(2000);
}

async function getCartCount(page): Promise<number> {
  const text = await page.locator('.cart-badge').textContent();
  return parseInt(text, 10);
}

// ==================== テストデータ ====================

const validUser = {
  email: 'test@example.com',
  password: 'Test1234!',
};

const products = ['ワイヤレスイヤホン', 'USBケーブル', 'モバイルバッテリー'];

// ==================== テスト ====================

test.describe('ログイン機能', () => {

  test('正しい情報でログインできる', async ({ page }) => {
    await login(page, validUser.email, validUser.password);
    await expect(page).toHaveURL(/dashboard/);
    expect(await page.locator('.user-name').textContent()).toBe('テストユーザー');
  });

  test('間違ったパスワードでエラーになる', async ({ page }) => {
    await page.goto(`${BASE_URL}/login`);
    await page.locator('#email').fill(validUser.email);
    await page.locator('#password').fill('wrong');
    await page.locator('button[type="submit"]').click();
    await expect(page.locator('.error-message')).toBeVisible();
  });

  test('未入力で送信するとバリデーションエラー', async ({ page }) => {
    await page.goto(`${BASE_URL}/login`);
    await page.locator('button[type="submit"]').click();
    const errors = await page.locator('.field-error').count();
    expect(errors).toBe(2);
  });
});

test.describe('商品一覧', () => {

  test.beforeEach(async ({ page }) => {
    await login(page, validUser.email, validUser.password);
  });

  test('商品が表示される', async ({ page }) => {
    await page.goto(`${BASE_URL}/products`);
    const cards = page.locator('.product-card');
    const count = await cards.count();
    expect(count).toBeGreaterThan(0);
  });

  test('商品をカートに追加できる', async ({ page }) => {
    await addToCart(page, 'ワイヤレスイヤホン');
    const count = await getCartCount(page);
    expect(count).toBe(1);
  });

  test('複数の商品をカートに追加できる', async ({ page }) => {
    for (const product of products) {
      await addToCart(page, product);
    }
    const count = await getCartCount(page);
    expect(count).toBe(3);
  });
});

test.describe('カート機能', () => {

  test('カートに商品を追加して確認できる', async ({ page }) => {
    await login(page, validUser.email, validUser.password);
    await addToCart(page, 'ワイヤレスイヤホン');
    await page.goto(`${BASE_URL}/cart`);
    await expect(page.locator('.cart-item')).toHaveCount(1);
    expect(await page.locator('.cart-item .name').textContent()).toContain('ワイヤレスイヤホン');
  });

  test('カートの商品を削除できる', async ({ page }) => {
    await login(page, validUser.email, validUser.password);
    await addToCart(page, 'USBケーブル');
    await page.goto(`${BASE_URL}/cart`);
    await page.locator('.cart-item .delete-btn').click();
    await page.waitForTimeout(1000);
    expect(await page.locator('.cart-item').count()).toBe(0);
  });

  test('カートの合計金額が正しい', async ({ page }) => {
    await login(page, validUser.email, validUser.password);
    await addToCart(page, 'ワイヤレスイヤホン');
    await addToCart(page, 'USBケーブル');
    await page.goto(`${BASE_URL}/cart`);
    const total = await page.locator('.cart-total').textContent();
    expect(total).not.toBe('¥0');
  });
});

test.describe('購入フロー', () => {

  test('商品を購入できる', async ({ page }) => {
    await login(page, validUser.email, validUser.password);
    await addToCart(page, 'モバイルバッテリー');
    await page.goto(`${BASE_URL}/cart`);
    await page.locator('#checkout-btn').click();

    // 配送先入力
    await page.locator('#address').fill('東京都渋谷区1-1-1');
    await page.locator('#phone').fill('090-1234-5678');
    await page.locator('button:has-text("次へ")').click();

    // 支払い
    await page.locator('#card-number').fill('4111111111111111');
    await page.locator('#card-expiry').fill('12/25');
    await page.locator('#card-cvc').fill('123');
    await page.locator('button:has-text("注文確定")').click();

    await page.waitForURL('**/order/complete');
    await expect(page.locator('.order-success')).toBeVisible();
  });
});
```

## 設問

以下の10問に答えてください。

### 設問1（理解度：基本）
`login` ヘルパー関数で使われているロケーターの種類を列挙し、それぞれの問題点を指摘してください。Playwrightの推奨するロケーターは何ですか？

### 設問2（理解度：基本）
`addToCart` 関数の中に `page.waitForTimeout(2000)` があります。これはなぜ問題で、どう修正すべきですか？

### 設問3（処理追跡）
「複数の商品をカートに追加できる」テストで `addToCart` は何回呼ばれますか？各呼び出しでどのページに遷移しますか？

### 設問4（設計理解）
「商品一覧」テストグループの `test.beforeEach` でログインしています。この方法の利点と欠点は何ですか？

### 設問5（問題発見）
「正しい情報でログインできる」テストの中に、Webアサーション（自動リトライ）と汎用アサーション（即時判定）が混在しています。どの行が問題で、どう修正すべきですか？

### 設問6（問題発見）
「未入力で送信するとバリデーションエラー」テストで `.field-error` の数を `count()` で取得してから `expect(errors).toBe(2)` で検証しています。この方法にはどんなリスクがありますか？

### 設問7（設計理解）
「カート機能」のテスト3つでは、各テストの先頭で `login` と `addToCart` を呼んでいます。これは「テストの独立性」の観点では正しいですが、どんなデメリットがありますか？改善方法は？

### 設問8（問題発見）
「カートの合計金額が正しい」テストで `expect(total).not.toBe('¥0')` と検証しています。このアサーションの問題点は何ですか？

### 設問9（改善提案）
「購入フロー」テストでは `button:has-text("次へ")` というCSSセレクターが使われています。Playwrightの推奨するロケーターで書き換えてください。

### 設問10（拡張設計）
このテストスイート全体をPage Object Model（POM）で書き直すとしたら、どのようなクラスを作りますか？クラス名とメソッド名を設計してください。

## 解答例

<details>
<summary>設問1〜5の解答</summary>

### 設問1
使われているロケーター：
- `page.locator('#email')` — IDセレクター
- `page.locator('#password')` — IDセレクター
- `page.locator('button[type="submit"]')` — CSSセレクター

**問題点**: IDやCSSセレクターはHTMLの内部構造に依存しており、UIの変更で壊れやすいです。

**推奨される書き方:**
```typescript
await page.getByLabel('メールアドレス').fill(email);
await page.getByLabel('パスワード').fill(password);
await page.getByRole('button', { name: 'ログイン' }).click();
```

`getByRole`、`getByLabel` はアクセシビリティベースのロケーターで、UIの構造変更に強く、Playwrightが最も推奨する方法です。

### 設問2
**問題**: `waitForTimeout(2000)` は固定時間の待機（ハードコード待機）で、Playwrightのアンチパターンです。

- テストが不必要に遅くなる（常に2秒待つ）
- CI環境ではレスポンスが遅く2秒で足りないことがある
- ローカルでは一瞬で終わるのに2秒待つのは無駄

**修正案:**
```typescript
// カートバッジの数値が変わるのを待つ
await expect(page.locator('.cart-badge')).not.toHaveText('0');
// または
await page.waitForResponse('**/api/cart');
```

### 設問3
`products` 配列は3要素なので、`addToCart` は **3回** 呼ばれます。

各呼び出しの遷移：
1. `addToCart('ワイヤレスイヤホン')` → `/products` に遷移
2. `addToCart('USBケーブル')` → `/products` に遷移
3. `addToCart('モバイルバッテリー')` → `/products` に遷移

つまり、毎回商品一覧ページに遷移し直しています。すでに商品一覧ページにいる場合でも再遷移するのは非効率です。

### 設問4
**利点:**
- 各テストがログイン済みの状態から始まる（セットアップの重複排除）
- テストの本体がシンプルになる

**欠点:**
- テストごとにログインのネットワークリクエストが走るため遅い
- ログイン処理が失敗すると、全テストが失敗する（原因が分かりにくい）

**改善案**: `storageState` を使って認証状態を保存・再利用するか、セットアッププロジェクトで事前にログインを済ませるのが実務的です。

### 設問5
問題の行：
```typescript
expect(await page.locator('.user-name').textContent()).toBe('テストユーザー');
```

`textContent()` は即座に値を取得し、`expect().toBe()` は汎用アサーション（リトライなし）です。ページ遷移直後は `.user-name` の要素がまだ描画されていない可能性があり、テストが不安定になります。

**修正案:**
```typescript
await expect(page.locator('.user-name')).toHaveText('テストユーザー');
```

`toHaveText` はWebアサーション（自動リトライ）なので、要素が表示されるまで待ちます。

<summary>設問6〜10の解答</summary>

### 設問6
```typescript
const errors = await page.locator('.field-error').count();
expect(errors).toBe(2);
```

**リスク**: `count()` は呼び出し時点の要素数を即座に返します。バリデーションエラーの表示がまだ完了していない場合、`0` や `1` が返されてテストが失敗します。

**修正案:**
```typescript
await expect(page.locator('.field-error')).toHaveCount(2);
```

`toHaveCount` はWebアサーションなので、要素数が2になるまで自動リトライします。

### 設問7
**デメリット:**
- 各テストが `login` → `addToCart` → ページ遷移と多くのステップを踏むため、テスト全体の実行が遅い
- ログインやカート追加で失敗すると、テスト対象の「カート機能」の問題ではないのに失敗する

**改善方法:**
1. `storageState` で認証状態を保存し、ログインの重複を排除
2. APIを直接呼んでカートに商品を追加するヘルパーを作る（UIを経由しない）
3. `beforeEach` でログインとカート追加を共通化する

### 設問8
`expect(total).not.toBe('¥0')` は「0円でなければOK」というアサーションです。

**問題点:**
- 合計金額が正しいかを検証していない（例：本来 ¥5,000 のはずが ¥1 でもパスする）
- 表示が `¥NaN` や空文字でも、`¥0` でなければパスする
- テストが意味のある検証をしていない

**修正案:**
```typescript
await expect(page.locator('.cart-total')).toHaveText('¥4,580');
// または正規表現で
await expect(page.locator('.cart-total')).toHaveText(/¥[\d,]+/);
```

### 設問9
```typescript
// 修正前（CSSセレクター）
await page.locator('button:has-text("次へ")').click();
await page.locator('button:has-text("注文確定")').click();

// 修正後（推奨ロケーター）
await page.getByRole('button', { name: '次へ' }).click();
await page.getByRole('button', { name: '注文確定' }).click();
```
