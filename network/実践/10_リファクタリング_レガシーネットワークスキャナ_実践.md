## 課題の説明

以下の「レガシーネットワークスキャナ」は正しく動作しますが、コードの品質に多くの問題があります。
**動作を変えずに** コードを改善してください。

### 改善すべきポイント

1. **マジックナンバーの排除**: 意味のない数値をわかりやすい定数に置き換える
2. **関数への分割**: 1つの巨大な処理を適切な関数に分割する
3. **変数名の改善**: `x`, `a`, `tmp` などの不明瞭な変数名を意味のある名前に変える
4. **重複コードの排除**: 同じパターンのコードをまとめる
5. **エラーハンドリングの追加**: 例外処理を適切に行う
6. **ドキュメントの追加**: docstringやコメントを付ける

## リファクタリングの指針

<details>
<summary>指針1：定数の定義</summary>

マジックナンバーを定数に置き換えます。

```python
# Before
s.settimeout(0.5)
pts = [21, 22, 23, ...]

# After
SCAN_TIMEOUT = 0.5
COMMON_PORTS = [21, 22, 23, ...]
```

</details>

<details>
<summary>指針2：関数への分割</summary>

以下の関数に分割することを推奨します：
- `validate_ip(ip_str)` - IPアドレスの検証
- `classify_ip(ip_str)` - クラス判定
- `is_private_ip(ip_str)` - プライベートIP判定
- `ip_to_binary(ip_str)` - 2進数変換
- `calculate_subnet(ip_str, prefix_len)` - サブネット計算
- `scan_ports(host, ports)` - ポートスキャン
- `dns_lookup(ip_str)` - DNS逆引き
- `save_results(filename, results)` - 結果の保存
- `main()` - メイン処理

</details>

<details>
<summary>指針3：例外処理の改善</summary>

`except:` のような素の例外キャッチを具体的な例外クラスに変更します。

```python
# Before
except:
    sn = "?"

# After
except OSError:
    service_name = "不明"
```

<summary>解答例（初心者向け ─ 基本的なリファクタリング）</summary>

```python
# ネットワークスキャナ（リファクタリング版）
# 学べる内容：IPアドレス解析、サブネット計算、ポートスキャン、DNS逆引き
# 実行方法：python network_scanner.py

import socket
import time


# --- 定数定義 ---
SCAN_TIMEOUT = 0.5  # ポートスキャン時のタイムアウト（秒）
COMMON_PORTS = [21, 22, 23, 25, 53, 80, 110, 143, 443, 993, 995, 3306, 3389, 5432, 8080]


def validate_ip(ip_str):
    """IPアドレスの形式を検証する。有効ならTrue、無効ならFalseを返す。"""
    octets = ip_str.split(".")
    if len(octets) != 4:
        return False
    for octet in octets:
        if not octet.isdigit():
            return False
        value = int(octet)
        if value < 0 or value > 255:
            return False
    return True


def classify_ip(ip_str):
    """IPアドレスのクラスを判定する。"""
    first_octet = int(ip_str.split(".")[0])
    if first_octet <= 127:
        return "A"
    elif first_octet <= 191:
        return "B"
    elif first_octet <= 223:
        return "C"
    elif first_octet <= 239:
        return "D（マルチキャスト）"
    else:
        return "E（実験用）"


def is_private_ip(ip_str):
    """プライベートIPアドレスかどうかを判定する。"""
    octets = [int(o) for o in ip_str.split(".")]
    first, second = octets[0], octets[1]

    if first == 10:
        return True
    if first == 172 and 16 <= second <= 31:
        return True
    if first == 192 and second == 168:
        return True
    return False


def ip_to_binary(ip_str):
    """IPアドレスをドット区切りの2進数表記に変換する。"""
    octets = ip_str.split(".")
    binary_parts = [format(int(octet), "08b") for octet in octets]
    return ".".join(binary_parts)


def ip_to_int(ip_str):
    """IPアドレス文字列を32ビット整数に変換する。"""
    result = 0
    for octet in ip_str.split("."):
        result = result * 256 + int(octet)
    return result


def int_to_ip(ip_int):
    """32ビット整数をIPアドレス文字列に変換する。"""
    octets = []
    for i in range(3, -1, -1):
        octets.append(str((ip_int >> (8 * i)) & 0xFF))
    return ".".join(octets)


def calculate_subnet(ip_str, prefix_len):
    """サブネット情報を計算する。"""
    ip_int = ip_to_int(ip_str)
    mask_int = (0xFFFFFFFF << (32 - prefix_len)) & 0xFFFFFFFF
    network_int = ip_int & mask_int
    broadcast_int = ip_int | (~mask_int & 0xFFFFFFFF)

    return {
        "network": int_to_ip(network_int),
        "broadcast": int_to_ip(broadcast_int),
        "host_count": broadcast_int - network_int - 1,
        "mask": int_to_ip(mask_int),
    }


def scan_ports(host, ports):
    """指定ホストのポートをスキャンする。"""
    open_ports = []
    for port in ports:
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(SCAN_TIMEOUT)
        result = sock.connect_ex((host, port))
        sock.close()

        if result == 0:
            try:
                service_name = socket.getservbyport(port, "tcp")
            except OSError:
                service_name = "不明"
            open_ports.append({"port": port, "service": service_name})
            print(f"  ポート {port:<6} ({service_name:<12}) OPEN")

    return open_ports


def dns_reverse_lookup(ip_str):
    """DNS逆引きを行う。"""
    try:
        hostname, aliases, addresses = socket.gethostbyaddr(ip_str)
        return hostname
    except socket.herror:
        return None


def save_results(filename, ip_str, ip_class, is_private, subnet_info, open_ports):
    """スキャン結果をファイルに保存する。"""
    with open(filename, "w", encoding="utf-8") as f:
        f.write(f"スキャン結果: {ip_str}\n")
        f.write(f"クラス: {ip_class}\n")
        f.write(f"プライベートIP: {'はい' if is_private else 'いいえ'}\n")
        f.write(f"ネットワーク: {subnet_info['network']}\n")
        f.write(f"ブロードキャスト: {subnet_info['broadcast']}\n")
        f.write(f"開いているポート: {len(open_ports)}個\n")
        for port_info in open_ports:
            f.write(f"  {port_info['port']} ({port_info['service']})\n")


# --- メイン処理 ---
print("===== ネットワークスキャナ =====")
print()

# IPアドレスの入力と検証
ip_str = input("IPアドレスを入力してください: ").strip()
if not validate_ip(ip_str):
    print("エラー: 無効なIPアドレスです。")
    exit()

# IP情報の表示
ip_class = classify_ip(ip_str)
is_private = is_private_ip(ip_str)
binary_ip = ip_to_binary(ip_str)

print(f"\n【IP情報】")
print(f"  IPアドレス   : {ip_str}")
print(f"  クラス       : {ip_class}")
print(f"  プライベートIP: {'はい' if is_private else 'いいえ'}")
print(f"  2進数        : {binary_ip}")

# サブネット計算
prefix_str = input("\nプレフィックス長を入力してください: ").strip()
prefix_len = int(prefix_str)
subnet_info = calculate_subnet(ip_str, prefix_len)

print(f"\n【サブネット情報】")
print(f"  サブネットマスク : {subnet_info['mask']}")
print(f"  ネットワーク     : {subnet_info['network']}")
print(f"  ブロードキャスト : {subnet_info['broadcast']}")
print(f"  ホスト数         : {subnet_info['host_count']}")

# ポートスキャン
print(f"\n【ポートスキャン】")
start_time = time.time()
open_ports = scan_ports(ip_str, COMMON_PORTS)
elapsed = time.time() - start_time
print(f"  完了（{elapsed:.1f}秒）")
print(f"  開いているポート: {len(open_ports)}個")

# DNS逆引き
print(f"\n【DNS逆引き】")
hostname = dns_reverse_lookup(ip_str)
if hostname:
    print(f"  ホスト名: {hostname}")
else:
    print(f"  PTRレコードが見つかりません。")

# 結果の保存
octets = ip_str.split(".")
filename = f"scan_{'_'.join(octets)}.txt"
save_results(filename, ip_str, ip_class, is_private, subnet_info, open_ports)
print(f"\n結果を保存しました: {filename}")
```

<details>
<summary>解答例（改良版 ─ クラスで構造化）</summary>

クラスにまとめ、再利用性と拡張性を向上させたバージョンです。

```python
# ネットワークスキャナ（改良版）
# 学べる内容：リファクタリング、クラス設計、ネットワーク解析
# 実行方法：python network_scanner_v2.py

import socket
import time
from datetime import datetime


SCAN_TIMEOUT = 0.5
COMMON_PORTS = [21, 22, 23, 25, 53, 80, 110, 143, 443, 993, 995, 3306, 3389, 5432, 8080]


class IPAddress:
    """IPアドレスの解析・操作を行うクラス"""

    PRIVATE_RANGES = [
        (10, 0, 10, 255),       # 10.0.0.0/8
        (172, 16, 172, 31),     # 172.16.0.0/12
        (192, 168, 192, 168),   # 192.168.0.0/16
    ]

    def __init__(self, ip_str):
        self.ip_str = ip_str
        self.octets = [int(o) for o in ip_str.split(".")]
        self.int_value = sum(o << (8 * (3 - i)) for i, o in enumerate(self.octets))

    @staticmethod
    def validate(ip_str):
        """IPアドレスの形式を検証する"""
        parts = ip_str.split(".")
        if len(parts) != 4:
            return False
        return all(p.isdigit() and 0 <= int(p) <= 255 for p in parts)

    def to_binary(self):
        """2進数表記に変換する"""
        return ".".join(format(o, "08b") for o in self.octets)

    def get_class(self):
        """IPアドレスのクラスを判定する"""
        first = self.octets[0]
        if first <= 127:
            return "A"
        elif first <= 191:
            return "B"
        elif first <= 223:
            return "C"
        elif first <= 239:
            return "D"
        return "E"

    def is_private(self):
        """プライベートIPかどうか判定する"""
        f, s = self.octets[0], self.octets[1]
        return (f == 10 or
                (f == 172 and 16 <= s <= 31) or
                (f == 192 and s == 168))

    @staticmethod
    def from_int(ip_int):
        """整数からIPAddressオブジェクトを生成する"""
        ip_str = ".".join(str((ip_int >> (8 * i)) & 0xFF) for i in range(3, -1, -1))
        return IPAddress(ip_str)


class SubnetCalculator:
    """サブネット計算を行うクラス"""

    def __init__(self, ip, prefix_len):
        self.ip = ip
        self.prefix_len = prefix_len
        self.mask_int = (0xFFFFFFFF << (32 - prefix_len)) & 0xFFFFFFFF
        self.network_int = ip.int_value & self.mask_int
        self.broadcast_int = ip.int_value | (~self.mask_int & 0xFFFFFFFF)

    @property
    def mask(self):
        return IPAddress.from_int(self.mask_int).ip_str

    @property
    def network(self):
        return IPAddress.from_int(self.network_int).ip_str

    @property
    def broadcast(self):
        return IPAddress.from_int(self.broadcast_int).ip_str

    @property
    def host_count(self):
        return self.broadcast_int - self.network_int - 1


class PortScanner:
    """ポートスキャンを行うクラス"""

    def __init__(self, host, timeout=SCAN_TIMEOUT):
        self.host = host
        self.timeout = timeout

    def scan(self, ports):
        """指定ポートをスキャンする"""
        results = []
        for port in ports:
            is_open = self._check_port(port)
            service = self._get_service(port) if is_open else ""
            results.append({"port": port, "open": is_open, "service": service})
        return results

    def _check_port(self, port):
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(self.timeout)
        result = sock.connect_ex((self.host, port))
        sock.close()
        return result == 0

    @staticmethod
    def _get_service(port):
        try:
            return socket.getservbyport(port, "tcp")
        except OSError:
            return "不明"


def main():
    print("===== ネットワークスキャナ =====\n")

    ip_str = input("IPアドレス: ").strip()
    if not IPAddress.validate(ip_str):
        print("エラー: 無効なIPアドレスです。")
        return

    ip = IPAddress(ip_str)

    print(f"\n【IP情報】")
    print(f"  IPアドレス    : {ip.ip_str}")
    print(f"  クラス        : {ip.get_class()}")
    print(f"  プライベートIP : {'はい' if ip.is_private() else 'いいえ'}")
    print(f"  2進数         : {ip.to_binary()}")

    prefix = int(input("\nプレフィックス長: ").strip())
    subnet = SubnetCalculator(ip, prefix)

    print(f"\n【サブネット情報】")
    print(f"  マスク         : {subnet.mask} (/{prefix})")
    print(f"  ネットワーク   : {subnet.network}")
    print(f"  ブロードキャスト: {subnet.broadcast}")
    print(f"  ホスト数       : {subnet.host_count}")

    print(f"\n【ポートスキャン】")
    scanner = PortScanner(ip_str)
    start = time.time()
    results = scanner.scan(COMMON_PORTS)
    elapsed = time.time() - start

    open_ports = [r for r in results if r["open"]]
    for r in open_ports:
        print(f"  ポート {r['port']:<6} ({r['service']:<12}) OPEN")
    print(f"  完了（{elapsed:.1f}秒）開いているポート: {len(open_ports)}個")

    print(f"\n【DNS逆引き】")
    try:
        hostname, _, _ = socket.gethostbyaddr(ip_str)
        print(f"  ホスト名: {hostname}")
    except socket.herror:
        print(f"  PTRレコードが見つかりません。")

    filename = f"scan_{ip_str.replace('.', '_')}_{datetime.now():%Y%m%d_%H%M%S}.txt"
    with open(filename, "w", encoding="utf-8") as f:
        f.write(f"スキャン結果: {ip_str}\n")
        f.write(f"日時: {datetime.now()}\n")
        f.write(f"クラス: {ip.get_class()}\n")
        f.write(f"ネットワーク: {subnet.network}/{prefix}\n")
        f.write(f"開いているポート: {len(open_ports)}個\n")
        for r in open_ports:
            f.write(f"  {r['port']} ({r['service']})\n")
    print(f"\n保存しました: {filename}")


if __name__ == "__main__":
    main()
```

**初心者向けとの違い:**
- `IPAddress`, `SubnetCalculator`, `PortScanner` をクラスで実装し、責務を明確に分離
- `@property` デコレータで計算プロパティを実現（メソッド呼び出しに見えない自然なアクセス）
- `@staticmethod` でインスタンス不要なユーティリティメソッドを定義
- ファイル名にタイムスタンプを含めて上書き防止
