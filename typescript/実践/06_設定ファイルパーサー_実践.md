# 実践課題06：設定ファイルパーサー ★3

> **難易度**: ★★★☆☆（中級）
> **前提知識**: 第2章（インターフェースと型エイリアス）、第4章（ユニオン型と交差型）、第6章（列挙型とリテラル型）
> **課題の種類**: ミニプロジェクト
> **学習目標**: 列挙型（enum）、リテラル型（literal type）、ユニオン型を組み合わせて、設定データを型安全にパース・バリデーションする

---

## 完成イメージ

```
===== 設定ファイルパーサー =====

--- 設定の読み込み ---
✅ app.name = "MyApp" (string)
✅ app.port = 3000 (number)
✅ app.debug = true (boolean)
✅ app.logLevel = "info" (enum: LogLevel)
✅ database.host = "localhost" (string)
✅ database.port = 5432 (number)
✅ database.ssl = false (boolean)

--- バリデーション結果 ---
✅ すべての設定値が有効です

--- 無効な設定の検出 ---
❌ app.port: 数値が範囲外です（1〜65535）: 70000
❌ app.logLevel: 無効なログレベルです: "verbose"
❌ database.host: 空文字列は許可されていません

--- 設定のマージ（デフォルト + ユーザー） ---
app.name: "MyApp" → "CustomApp"（ユーザー設定で上書き）
app.port: 3000（デフォルト値を使用）
app.debug: true → false（ユーザー設定で上書き）
app.logLevel: "info"（デフォルト値を使用）
```

---

## 課題の要件

1. 設定値の型を列挙型とリテラル型で厳密に定義する
2. `LogLevel` 列挙型（`debug`, `info`, `warn`, `error`）を定義する
3. 各設定値にバリデーション（validation）ルールを設ける
4. 不正な設定値を検出してエラーメッセージを表示する
5. デフォルト設定とユーザー設定をマージする機能を作る
6. バリデーション結果をユニオン型（`成功 | 失敗`）で表現する

---

## 解答例

<details>
<summary>解答例（初心者向け）</summary>

```typescript
// 設定ファイルパーサー
// 学べる内容：enum、リテラル型、ユニオン型、バリデーション
// 実行方法：npx ts-node 06_config_parser.ts

// --- 列挙型 ---
enum LogLevel {
  Debug = "debug",
  Info = "info",
  Warn = "warn",
  Error = "error",
}

// --- 設定の型定義 ---
interface AppConfig {
  name: string;
  port: number;
  debug: boolean;
  logLevel: LogLevel;
}

interface DatabaseConfig {
  host: string;
  port: number;
  ssl: boolean;
}

interface Config {
  app: AppConfig;
  database: DatabaseConfig;
}

// --- バリデーション結果のユニオン型 ---
type ValidationSuccess = {
  valid: true;
  key: string;
  value: string | number | boolean;
};

type ValidationError = {
  valid: false;
  key: string;
  message: string;
};

type ValidationResult = ValidationSuccess | ValidationError;

// --- バリデーション関数群 ---
function validateNonEmptyString(key: string, value: string): ValidationResult {
  if (value.trim() === "") {
    return { valid: false, key, message: "空文字列は許可されていません" };
  }
  return { valid: true, key, value };
}

function validatePort(key: string, value: number): ValidationResult {
  if (value < 1 || value > 65535) {
    return {
      valid: false,
      key,
      message: `数値が範囲外です（1〜65535）: ${value}`,
    };
  }
  return { valid: true, key, value };
}

function validateLogLevel(key: string, value: string): ValidationResult {
  const validLevels: string[] = Object.values(LogLevel);
  if (!validLevels.includes(value)) {
    return {
      valid: false,
      key,
      message: `無効なログレベルです: "${value}"`,
    };
  }
  return { valid: true, key, value };
}

// --- 設定全体のバリデーション ---
function validateConfig(config: Config): ValidationResult[] {
  const results: ValidationResult[] = [];

  results.push(validateNonEmptyString("app.name", config.app.name));
  results.push(validatePort("app.port", config.app.port));
  results.push(validateLogLevel("app.logLevel", config.app.logLevel));
  results.push(validateNonEmptyString("database.host", config.database.host));
  results.push(validatePort("database.port", config.database.port));

  return results;
}

// --- 設定の表示 ---
function displayConfig(config: Config): void {
  console.log("--- 設定の読み込み ---");
  console.log(`✅ app.name = "${config.app.name}" (string)`);
  console.log(`✅ app.port = ${config.app.port} (number)`);
  console.log(`✅ app.debug = ${config.app.debug} (boolean)`);
  console.log(`✅ app.logLevel = "${config.app.logLevel}" (enum: LogLevel)`);
  console.log(`✅ database.host = "${config.database.host}" (string)`);
  console.log(`✅ database.port = ${config.database.port} (number)`);
  console.log(`✅ database.ssl = ${config.database.ssl} (boolean)`);
}

// --- バリデーション結果の表示 ---
function displayValidation(results: ValidationResult[]): void {
  const errors: ValidationError[] = [];
  for (const result of results) {
    if (!result.valid) {
      errors.push(result);
    }
  }

  if (errors.length === 0) {
    console.log("✅ すべての設定値が有効です");
  } else {
    for (const error of errors) {
      console.log(`❌ ${error.key}: ${error.message}`);
    }
  }
}

// --- デフォルト設定とユーザー設定のマージ ---
function mergeConfig(
  defaultConfig: Config,
  userConfig: Partial<Config>
): Config {
  const merged: Config = {
    app: { ...defaultConfig.app },
    database: { ...defaultConfig.database },
  };

  if (userConfig.app !== undefined) {
    if (userConfig.app.name !== undefined) {
      console.log(`app.name: "${defaultConfig.app.name}" → "${userConfig.app.name}"（ユーザー設定で上書き）`);
      merged.app.name = userConfig.app.name;
    } else {
      console.log(`app.name: "${defaultConfig.app.name}"（デフォルト値を使用）`);
    }
    if (userConfig.app.port !== undefined) {
      console.log(`app.port: ${defaultConfig.app.port} → ${userConfig.app.port}（ユーザー設定で上書き）`);
      merged.app.port = userConfig.app.port;
    } else {
      console.log(`app.port: ${defaultConfig.app.port}（デフォルト値を使用）`);
    }
    if (userConfig.app.debug !== undefined) {
      console.log(`app.debug: ${defaultConfig.app.debug} → ${userConfig.app.debug}（ユーザー設定で上書き）`);
      merged.app.debug = userConfig.app.debug;
    } else {
      console.log(`app.debug: ${defaultConfig.app.debug}（デフォルト値を使用）`);
    }
    if (userConfig.app.logLevel !== undefined) {
      console.log(`app.logLevel: "${defaultConfig.app.logLevel}" → "${userConfig.app.logLevel}"（ユーザー設定で上書き）`);
      merged.app.logLevel = userConfig.app.logLevel;
    } else {
      console.log(`app.logLevel: "${defaultConfig.app.logLevel}"（デフォルト値を使用）`);
    }
  }

  return merged;
}

// --- メイン処理 ---
console.log("===== 設定ファイルパーサー =====\n");

// 有効な設定
const validConfig: Config = {
  app: {
    name: "MyApp",
    port: 3000,
    debug: true,
    logLevel: LogLevel.Info,
  },
  database: {
    host: "localhost",
    port: 5432,
    ssl: false,
  },
};

displayConfig(validConfig);

console.log("\n--- バリデーション結果 ---");
const validResults: ValidationResult[] = validateConfig(validConfig);
displayValidation(validResults);

// 無効な設定
console.log("\n--- 無効な設定の検出 ---");
const invalidConfig: Config = {
  app: {
    name: "MyApp",
    port: 70000,
    debug: true,
    logLevel: "verbose" as LogLevel, // 無効なログレベル
  },
  database: {
    host: "",
    port: 5432,
    ssl: false,
  },
};

const invalidResults: ValidationResult[] = validateConfig(invalidConfig);
displayValidation(invalidResults);

// マージ
console.log("\n--- 設定のマージ（デフォルト + ユーザー） ---");
const userOverrides: Partial<Config> = {
  app: {
    name: "CustomApp",
    port: 3000,
    debug: false,
    logLevel: LogLevel.Info,
  },
};

mergeConfig(validConfig, userOverrides);
```

<details>
<summary>解答例（改良版 ─ const enumと高度な型パターン）</summary>

```typescript
// 設定ファイルパーサー（改良版）
// 学べる内容：const enum、テンプレートリテラル型、型の再利用パターン
// 実行方法：npx ts-node 06_config_parser_v2.ts

// --- 列挙型（const enum で最適化） ---
const enum LogLevel {
  Debug = "debug",
  Info = "info",
  Warn = "warn",
  Error = "error",
}

const enum Environment {
  Development = "development",
  Staging = "staging",
  Production = "production",
}

// --- リテラル型の活用 ---
type PortNumber = number; // 1〜65535 の意図を名前で示す
type NonEmptyString = string; // 空でない文字列の意図を名前で示す

// --- 設定の型定義 ---
interface AppConfig {
  readonly name: NonEmptyString;
  readonly port: PortNumber;
  readonly debug: boolean;
  readonly logLevel: LogLevel;
  readonly env: Environment;
}

interface DatabaseConfig {
  readonly host: NonEmptyString;
  readonly port: PortNumber;
  readonly ssl: boolean;
  readonly maxConnections: number;
}

interface Config {
  readonly app: AppConfig;
  readonly database: DatabaseConfig;
}

// --- バリデーション結果 ---
type ValidationSuccess = {
  readonly valid: true;
  readonly key: string;
};

type ValidationError = {
  readonly valid: false;
  readonly key: string;
  readonly message: string;
  readonly receivedValue: unknown;
};

type ValidationResult = ValidationSuccess | ValidationError;

// --- バリデーションルールの型 ---
type Validator<T> = (key: string, value: T) => ValidationResult;

// --- バリデータファクトリ関数 ---
function createRangeValidator(min: number, max: number): Validator<number> {
  return (key: string, value: number): ValidationResult => {
    if (value < min || value > max) {
      return {
        valid: false,
        key,
        message: `数値が範囲外です（${min}〜${max}）`,
        receivedValue: value,
      };
    }
    return { valid: true, key };
  };
}

function createNonEmptyValidator(): Validator<string> {
  return (key: string, value: string): ValidationResult => {
    if (value.trim() === "") {
      return {
        valid: false,
        key,
        message: "空文字列は許可されていません",
        receivedValue: value,
      };
    }
    return { valid: true, key };
  };
}

function createEnumValidator<T extends string>(validValues: readonly T[], label: string): Validator<string> {
  return (key: string, value: string): ValidationResult => {
    if (!(validValues as readonly string[]).includes(value)) {
      return {
        valid: false,
        key,
        message: `無効な${label}です: "${value}"（有効値: ${validValues.join(", ")})`,
        receivedValue: value,
      };
    }
    return { valid: true, key };
  };
}

// --- バリデータのインスタンス ---
const validatePort: Validator<number> = createRangeValidator(1, 65535);
const validateNonEmpty: Validator<string> = createNonEmptyValidator();
const validateLogLevel: Validator<string> = createEnumValidator(
  ["debug", "info", "warn", "error"] as const,
  "ログレベル"
);
const validateEnvironment: Validator<string> = createEnumValidator(
  ["development", "staging", "production"] as const,
  "環境"
);

// --- 設定全体のバリデーション ---
function validateConfig(config: Config): ValidationResult[] {
  return [
    validateNonEmpty("app.name", config.app.name),
    validatePort("app.port", config.app.port),
    validateLogLevel("app.logLevel", config.app.logLevel),
    validateEnvironment("app.env", config.app.env),
    validateNonEmpty("database.host", config.database.host),
    validatePort("database.port", config.database.port),
    createRangeValidator(1, 100)("database.maxConnections", config.database.maxConnections),
  ];
}

// --- 結果表示 ---
function displayResults(results: ValidationResult[]): void {
  const errors: ValidationError[] = results.filter(
    (r: ValidationResult): r is ValidationError => !r.valid
  );

  if (errors.length === 0) {
    console.log("✅ すべての設定値が有効です");
  } else {
    for (const error of errors) {
      console.log(`❌ ${error.key}: ${error.message} (受け取った値: ${JSON.stringify(error.receivedValue)})`);
    }
    console.log(`\n${errors.length}件のエラーが見つかりました`);
  }
}

// --- ディープマージ ---
type DeepPartial<T> = {
  [K in keyof T]?: T[K] extends object ? DeepPartial<T[K]> : T[K];
};

function deepMerge(defaultCfg: Config, overrides: DeepPartial<Config>): Config {
  const result: Config = {
    app: {
      ...defaultCfg.app,
      ...(overrides.app ?? {}),
    } as AppConfig,
    database: {
      ...defaultCfg.database,
      ...(overrides.database ?? {}),
    } as DatabaseConfig,
  };
  return result;
}

// --- メイン処理 ---
console.log("===== 設定ファイルパーサー =====\n");

const defaultConfig: Config = {
  app: { name: "MyApp", port: 3000, debug: true, logLevel: LogLevel.Info, env: Environment.Development },
  database: { host: "localhost", port: 5432, ssl: false, maxConnections: 10 },
};

console.log("--- 有効な設定のバリデーション ---");
displayResults(validateConfig(defaultConfig));

console.log("\n--- 無効な設定のバリデーション ---");
const invalidConfig: Config = {
  app: { name: "", port: 70000, debug: true, logLevel: "verbose" as LogLevel, env: "local" as Environment },
  database: { host: "", port: 5432, ssl: false, maxConnections: 200 },
};
displayResults(validateConfig(invalidConfig));

console.log("\n--- ディープマージ ---");
const userOverrides: DeepPartial<Config> = {
  app: { name: "CustomApp", debug: false },
  database: { ssl: true },
};

const merged: Config = deepMerge(defaultConfig, userOverrides);

const keys: string[] = ["name", "port", "debug", "logLevel", "env"];
for (const key of keys) {
  const defaultVal = defaultConfig.app[key as keyof AppConfig];
  const mergedVal = merged.app[key as keyof AppConfig];
  if (defaultVal !== mergedVal) {
    console.log(`app.${key}: ${JSON.stringify(defaultVal)} → ${JSON.stringify(mergedVal)}（上書き）`);
  } else {
    console.log(`app.${key}: ${JSON.stringify(mergedVal)}（デフォルト）`);
  }
}
```

**初心者向けとの違い:**
- `Validator<T>` 型で汎用的なバリデータをファクトリ関数で生成
- `DeepPartial<T>` ユーティリティ型を自前で定義（第8章の先取り）
- `result is ValidationError` の型ガード付き `filter`
- `as const` でリテラル配列の型を厳密に
- `const enum` でランタイムオーバーヘッドのないenum

</details>
