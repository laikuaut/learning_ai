### app/components/CartProvider.tsx

```tsx
'use client'

import { createContext, useContext, useReducer, ReactNode } from 'react'

type CartItem = { id: string; name: string; price: number; quantity: number }
type CartState = { items: CartItem[] }
type CartAction =
  | { type: 'ADD'; payload: Omit<CartItem, 'quantity'> }
  | { type: 'REMOVE'; payload: string }
  | { type: 'UPDATE_QUANTITY'; payload: { id: string; quantity: number } }
  | { type: 'CLEAR' }

function cartReducer(state: CartState, action: CartAction): CartState {
  switch (action.type) {
    case 'ADD': {
      const existing = state.items.find((item) => item.id === action.payload.id)
      if (existing) {
        return {
          items: state.items.map((item) =>
            item.id === action.payload.id
              ? { ...item, quantity: item.quantity + 1 }
              : item
          ),
        }
      }
      return {
        items: [...state.items, { ...action.payload, quantity: 1 }],
      }
    }
    case 'REMOVE':
      return {
        items: state.items.filter((item) => item.id !== action.payload),
      }
    case 'UPDATE_QUANTITY':
      return {
        items: state.items
          .map((item) =>
            item.id === action.payload.id
              ? { ...item, quantity: action.payload.quantity }
              : item
          )
          .filter((item) => item.quantity > 0),
      }
    case 'CLEAR':
      return { items: [] }
    default:
      return state
  }
}

type CartContextType = {
  items: CartItem[]
  dispatch: React.Dispatch<CartAction>
  totalPrice: number
  totalCount: number
}

const CartContext = createContext<CartContextType | null>(null)

export function CartProvider({ children }: { children: ReactNode }) {
  const [state, dispatch] = useReducer(cartReducer, { items: [] })

  const totalPrice = state.items.reduce(
    (sum, item) => sum + item.price * item.quantity, 0
  )
  const totalCount = state.items.reduce(
    (sum, item) => sum + item.quantity, 0
  )

  return (
    <CartContext.Provider value={{ items: state.items, dispatch, totalPrice, totalCount }}>
      {children}
    </CartContext.Provider>
  )
}

export function useCart() {
  const context = useContext(CartContext)
  if (!context) throw new Error('useCart must be used within CartProvider')
  return context
}
```

import type { Metadata } from 'next'
import { notFound } from 'next/navigation'
import AddToCartButton from '../../../components/AddToCartButton'

type Product = {
  id: string; name: string; price: number; description: string; category: string
}

async function getProduct(id: string): Promise<Product | null> {
  const res = await fetch(`https://api.example.com/products/${id}`, {
    next: { revalidate: 3600 },
  })
  if (res.status === 404) return null
  if (!res.ok) throw new Error('商品の取得に失敗しました')
  return res.json()
}

export async function generateMetadata({
  params,
}: {
  params: Promise<{ id: string }>
}): Promise<Metadata> {
  const { id } = await params
  const product = await getProduct(id)
  if (!product) return { title: '商品が見つかりません' }
  return {
    title: product.name,
    description: product.description,
    openGraph: { title: product.name, description: product.description },
  }
}

  params,
}: {
  params: Promise<{ id: string }>

      </p>
      <p>{product.description}</p>
      <AddToCartButton product={{ id: product.id, name: product.name, price: product.price }} />

import { revalidatePath } from 'next/cache'

type OrderData = {
  items: Array<{ id: string; quantity: number }>
  totalPrice: number
}

export async function submitOrder(data: OrderData) {
  // 注文処理（外部APIに送信）
  const res = await fetch('https://api.example.com/orders', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(data),
  })

  if (!res.ok) {
    return { success: false, error: '注文に失敗しました' }
  }

  const order = await res.json()
  revalidatePath('/cart')

  return { success: true, orderId: order.id }
}

import { NextResponse } from 'next/server'
import type { NextRequest } from 'next/server'

export function middleware(request: NextRequest) {
  const response = NextResponse.next()
  response.headers.set('x-pathname', request.nextUrl.pathname)

  if (request.nextUrl.pathname.startsWith('/admin')) {
    const token = request.cookies.get('admin-token')
    if (!token) {
      return NextResponse.redirect(new URL('/login', request.url))
    }
  }

  return response
}

## 設問

以下の10問に答えてください。

### 設問1
`CartProvider` はルートレイアウト（`app/layout.tsx`）で使われていますが、なぜ `'use client'` がファイルに宣言されているのに、ルートレイアウト自体はサーバーコンポーネントのままでいられるのですか？

### 設問2
`cartReducer` の `UPDATE_QUANTITY` アクションで `.filter((item) => item.quantity > 0)` が呼ばれています。この処理の目的は何ですか？

### 設問3
`(shop)` フォルダは Route Group です。このフォルダがURLに与える影響と、使う利点を説明してください。

### 設問4
`ProductPage` で `getProduct` 関数が2回呼ばれています（`generateMetadata` と `ProductPage` 内）。これはパフォーマンス上の問題になりますか？理由を説明してください。

### 設問5
`AddToCartButton` は `'use client'` が宣言されていますが、`ProductPage`（サーバーコンポーネント）から `product` props として `{ id, name, price }` が渡されています。この props の受け渡しにはどのような制約がありますか？

### 設問6
`CartIcon` コンポーネントは `(shop)/layout.tsx`（サーバーコンポーネント）からインポートされています。なぜサーバーコンポーネントからクライアントコンポーネントをインポートしても問題ないのですか？

### 設問7
`middleware.ts` の `config.matcher` に設定されている正規表現の意味を説明してください。なぜこのような除外パターンが必要なのですか？

### 設問8
`submitOrder` は Server Action ですが、カートの状態（`items`）はクライアント側の `useReducer` で管理されています。注文送信時、クライアントの状態をどのようにして Server Action に渡すと考えられますか？

### 設問9
`getProduct` の `fetch` に `{ next: { revalidate: 3600 } }` が設定されています。この設定の意味と、`cache: 'no-store'` や `cache: 'force-cache'` との違いを説明してください。

### 設問10
現在のアーキテクチャでカートの状態は `useReducer`（クライアントのみ）で管理されています。ページをリロードするとカートが空になります。この問題を解決するためのアプローチを2つ提案してください。

## 解答例

<details>
<summary>全設問の解答</summary>

### 設問1の解答
`CartProvider` は `'use client'` が宣言されたクライアントコンポーネントですが、ルートレイアウトから**インポートして使う**だけであれば問題ありません。`'use client'` の境界（Boundary）は、そのファイルとそこからインポートされるモジュールに適用されます。親のサーバーコンポーネントがクライアントコンポーネントを「子として配置する」のは正当な使い方です。ルートレイアウト自体は依然としてサーバーで実行され、`CartProvider` のクライアントバンドルは別途クライアントに送信されます。

### 設問2の解答
`UPDATE_QUANTITY` で `quantity` を0に更新した場合、そのアイテムを自動的にカートから削除するためです。`REMOVE` アクションを別途呼ばなくても、数量を減らしていって0になれば自然にアイテムが消えるという、ユーザーにとって直感的な動作を実現しています。

### 設問3の解答
`(shop)` フォルダはURLに**一切影響しません**。`app/(shop)/page.tsx` は `/` にマッピングされます（`/shop/` ではありません）。利点は、ルーティングに影響を与えずにレイアウトをグループ化できることです。この例では、ショップ用のヘッダー（ロゴ + カートアイコン付き）を `/` と `/products/[id]` に適用しつつ、`/cart` や `/admin` には別のレイアウトを使えます。

### 設問4の解答
パフォーマンス上の問題にはなりません。Next.js は同一リクエスト内で同じURLへの `fetch` 呼び出しを**自動的にメモ化（Request Memoization）** します。`generateMetadata` と `ProductPage` で同じURLに対して `fetch` を呼んでも、実際のHTTPリクエストは1回だけ発行されます。2回目はキャッシュから結果が返されます。

### ���問5の解答
サーバーコンポーネントからクライアントコンポーネントに渡す props は、**シリアライズ可能（Serializable）** でなければなりません。`{ id, name, price }` は文字列と数値のプレーンオブジェクトなので問題ありません。関数、クラスインスタンス、`Date` オブジェクト、`Map`/`Set` などのシリアライズ不可能な値は props として渡せません。

### 設問6の解答
サーバーコンポーネントからクライアントコンポーネントをインポートして**子として配置する**のは、Next.js の設計で意図された正当なパターンです。サーバーコンポーネントはサーバーで実行された後、クライアントコンポーネントの「プレースホルダー」を含むHTMLを生成します。クライアント側でハイドレーション（Hydration）が行われると、クライアントコンポーネントがインタラクティブになります。逆のパターン（クライアントコンポーネント内でサーバーコンポーネントを直接インポート）は禁止されています。

### 設問7の解答
`/((?!api|_next/static|_next/image|favicon.ico).*)` は**否定先読み（Negative Lookahead）** を使った正規表現で、「`api`、`_next/static`、`_next/image`、`favicon.ico` で始まる**以外**の全パス」にマッチします。これらを除外する理由は、APIルート、静的アセット、画像最適化、ファビコンにはミドルウェアの処理（認証チェックやヘッダー追加）が不要であり、パフォーマンスへの不要な影響を避けるためです。

### 設問8の解答
クライアントコンポーネントからServer Actionを呼ぶ際に、引数としてカートの状態を渡します。具体的には次のようになります。

```tsx
const { items, totalPrice, dispatch } = useCart()

async function handleSubmit() {
  const orderData = {
    items: items.map(item => ({ id: item.id, quantity: item.quantity })),
    totalPrice,
  }
  const result = await submitOrder(orderData)
  if (result.success) {
    dispatch({ type: 'CLEAR' })
  }
}
```

Server Actions は通常 `FormData` を受け取りますが、通常の引数として呼び出すことも可能です（`.bind()` や直接引数渡し）。

### 設問9の解答
`{ next: { revalidate: 3600 } }` は **ISR（Incremental Static Regeneration）** の設定です。
- 初回リクエスト時にデータを取得しキャッシュする
- 3600秒（1時間）以内のリクエストにはキャッシュから返す
- 3600秒経過後のリクエストでバックグラウンドで再取得し、キャッシュを更新する

`cache: 'force-cache'`（デフォルト）はビルド時に取得したデータを永続的にキャッシュし、`cache: 'no-store'` はリクエストごとに毎回データを取得します。`revalidate` はその中間で、適度な鮮度とパフォーマンスのバランスを取れます。

### 設問10の解答
**アプローチ1: localStorage を使った永続化**

```tsx
// CartProvider 内で localStorage と同期
useEffect(() => {
  const saved = localStorage.getItem('cart')
  if (saved) dispatch({ type: 'HYDRATE', payload: JSON.parse(saved) })
}, [])

useEffect(() => {
  localStorage.setItem('cart', JSON.stringify(state.items))
}, [state.items])
```

**アプローチ2: Cookie + Server Actions を使ったサーバー側永続化**

```tsx
// カート変更時に Cookie に保存
'use server'
import { cookies } from 'next/headers'

export async function saveCart(items: CartItem[]) {
  const cookieStore = await cookies()
  cookieStore.set('cart', JSON.stringify(items), { maxAge: 7 * 24 * 3600 })
}
```

アプローチ1はシンプルで実装が簡単ですが、デバイス間の同期ができません。アプローチ2はサーバー側で管理できるため、ログインユーザーのカートをデバイス間で共有できます。本格的なECサイトではデータベースに保存し、Cookie にはセッションIDのみを持たせるのが一般的です。
