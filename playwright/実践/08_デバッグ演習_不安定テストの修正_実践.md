  </style>
</head>
<body>
  <h1>タスク管理</h1>

  <div>
    <input id="task-input" type="text" placeholder="タスクを入力">
    <button id="add-btn" onclick="addTask()">追加</button>
  </div>

  <div style="margin: 12px 0;">
    <button class="filter-btn active" onclick="filterTasks('all')" data-filter="all">すべて</button>
    <button class="filter-btn" onclick="filterTasks('active')" data-filter="active">未完了</button>
    <button class="filter-btn" onclick="filterTasks('done')" data-filter="done">完了済み</button>
  </div>

  <div id="task-list"></div>
  <p class="counter" id="counter">0件のタスク</p>
  <div class="notification" id="notification"></div>

  <script>
    let tasks = [];
    let currentFilter = 'all';

    function addTask() {
      const input = document.getElementById('task-input');
      const text = input.value.trim();
      if (!text) return;

      tasks.push({ text, done: false, id: Date.now() });
      input.value = '';

      // 非同期で通知を表示（遅延あり）
      setTimeout(() => {
        showNotification(`「${text}」を追加しました`);
      }, 300);

      renderTasks();
    }

    function toggleTask(id) {
      const task = tasks.find(t => t.id === id);
      if (task) task.done = !task.done;

      // APIを模擬（非同期）
      setTimeout(() => renderTasks(), 200);
    }

    function deleteTask(id) {
      tasks = tasks.filter(t => t.id !== id);

      setTimeout(() => {
        showNotification('タスクを削除しました');
      }, 300);

      // 削除アニメーション（遅延後にレンダリング）
      setTimeout(() => renderTasks(), 500);
    }

    function filterTasks(filter) {
      currentFilter = filter;
      document.querySelectorAll('.filter-btn').forEach(btn => {
        btn.classList.toggle('active', btn.dataset.filter === filter);
      });
      renderTasks();
    }

    function renderTasks() {
      const list = document.getElementById('task-list');
      let filtered = tasks;
      if (currentFilter === 'active') filtered = tasks.filter(t => !t.done);
      if (currentFilter === 'done') filtered = tasks.filter(t => t.done);

      list.innerHTML = filtered.map(t => `
        <div class="task ${t.done ? 'done' : ''}" data-testid="task-item">
          <input type="checkbox" ${t.done ? 'checked' : ''} onchange="toggleTask(${t.id})">
          <span class="task-text">${t.text}</span>
          <button class="delete-btn" onclick="deleteTask(${t.id})">削除</button>
        </div>
      `).join('');

      const activeCount = tasks.filter(t => !t.done).length;
      document.getElementById('counter').textContent = `${activeCount}件のタスク`;
    }

    function showNotification(message) {
      const el = document.getElementById('notification');
      el.textContent = message;
      el.classList.add('show');
      setTimeout(() => el.classList.remove('show'), 2000);
    }
  </script>
</body>
</html>

## 解答例

<details>
<summary>問題1〜4の解答</summary>

### 問題1：固定時間待機 + 即時取得

**問題**: `waitForTimeout(500)` は固定待機で、通知の表示タイミングが 500ms 以内に収まらない場合に失敗します。また `textContent()` で即時取得しているため、表示完了前に空文字を取得する可能性があります。

**修正:**
```typescript
test('タスクを追加すると通知が表示される', async ({ page }) => {
  await page.getByPlaceholder('タスクを入力').fill('牛乳を買う');
  await page.getByRole('button', { name: '追加' }).click();

  // 通知が表示されるまで自動リトライ
  const notification = page.locator('.notification');
  await expect(notification).toBeVisible();
  await expect(notification).toContainText('牛乳を買う');
});
```

### 問題2：即時判定アサーション

**問題**: `count()` は呼び出し時点の要素数を返します。タスク追加の DOM 更新がまだ完了していない可能性があります。

**修正:**
```typescript
test('タスクが一覧に追加される', async ({ page }) => {
  await page.getByPlaceholder('タスクを入力').fill('レポートを書く');
  await page.getByRole('button', { name: '追加' }).click();

  // toHaveCount は自動リトライする
  await expect(page.getByTestId('task-item')).toHaveCount(1);
  await expect(page.getByTestId('task-item').locator('.task-text')).toHaveText('レポートを書く');
});
```

### 問題3：textContent()の即時取得

**問題**: 2つのタスクを連続追加した直後に `textContent()` で取得しているため、DOM更新が追いつかない場合に `"1件のタスク"` が返る可能性があります。

**修正:**
```typescript
test('タスクのカウンターが更新される', async ({ page }) => {
  await page.getByPlaceholder('タスクを入力').fill('タスクA');
  await page.getByRole('button', { name: '追加' }).click();
  await page.getByPlaceholder('タスクを入力').fill('タスクB');
  await page.getByRole('button', { name: '追加' }).click();

  // toHaveText は自動リトライ
  await expect(page.locator('#counter')).toHaveText('2件のタスク');
});
```

### 問題4：削除後の固定待機 + 即時判定

**問題**: 削除アニメーション（500ms 遅延）の後に `waitForTimeout(1000)` で待っていますが、環境によっては不安定です。さらに `count()` は即時判定です。

**修正:**
```typescript
test('タスクを削除できる', async ({ page }) => {
  await page.getByPlaceholder('タスクを入力').fill('削除するタスク');
  await page.getByRole('button', { name: '追加' }).click();
  await expect(page.getByTestId('task-item')).toHaveCount(1);

  await page.getByRole('button', { name: '削除' }).click();

  // 要素が0になるまで自動リトライ
  await expect(page.getByTestId('task-item')).toHaveCount(0);
});
```

### 問題5：チェック操作後の即時検証

**問題**: `check()` の後、`toggleTask` 内の `setTimeout(200ms)` でDOMが更新されるため、`getAttribute('class')` が即座に取得すると `"done"` クラスがまだ付いていない可能性があります。

**修正:**
```typescript
test('タスクを完了にできる', async ({ page }) => {
  await page.getByPlaceholder('タスクを入力').fill('完了するタスク');
  await page.getByRole('button', { name: '追加' }).click();

  await page.getByTestId('task-item').getByRole('checkbox').check();

  // toHaveClass は自動リトライ
  await expect(page.getByTestId('task-item')).toHaveClass(/done/);
});
```

### 問題6：フィルター切り替え後の即時判定

**問題**: `waitForTimeout(300)` でチェック後の更新を待ち、フィルタークリック後に `count()` で即時判定しています。2つの問題が重なっています。

**修正:**
```typescript
test('フィルターで未完了タスクだけ表示できる', async ({ page }) => {
  // タスクを2つ追加
  await page.getByPlaceholder('タスクを入力').fill('未完了タスク');
  await page.getByRole('button', { name: '追加' }).click();
  await page.getByPlaceholder('タスクを入力').fill('完了タスク');
  await page.getByRole('button', { name: '追加' }).click();
  await expect(page.getByTestId('task-item')).toHaveCount(2);

  // 2番目のタスクを完了にする
  await page.getByTestId('task-item').nth(1).getByRole('checkbox').check();
  await expect(page.getByTestId('task-item').nth(1)).toHaveClass(/done/);

  // 未完了フィルター
  await page.getByRole('button', { name: '未完了' }).click();
  await expect(page.getByTestId('task-item')).toHaveCount(1);
  await expect(page.getByTestId('task-item').locator('.task-text')).toHaveText('未完了タスク');
});
```

### 問題7：通知の消失を固定待機で確認

**問題**: `waitForTimeout(2000)` で通知が消えるのを待ち、`isVisible()` で即時判定しています。通知の表示時間（2000ms）と待機時間が同じなのでタイミングが微妙です。

**修正:**
```typescript
test('通知が自動で消える', async ({ page }) => {
  await page.getByPlaceholder('タスクを入力').fill('通知テスト');
  await page.getByRole('button', { name: '追加' }).click();

  // 通知が表示されるまで待つ
  await expect(page.locator('.notification')).toBeVisible();
  await expect(page.locator('.notification')).toContainText('通知テスト');

  // 通知が非表示になるまで待つ（自動リトライ）
  await expect(page.locator('.notification')).toBeHidden({ timeout: 5000 });
});
```

### 問題8：ロケーターがCSSセレクターに依存 + 即時取得

**問題**: `#add-btn` というIDセレクターはHTMLの内部構造に依存しており、`textContent()` による即時取得は非推奨です。

**修正:**
```typescript
test('追加ボタンのテキストが正しい', async ({ page }) => {
  const button = page.getByRole('button', { name: '追加' });
  await expect(button).toBeVisible();
  await expect(button).toHaveText('追加');
});
```
