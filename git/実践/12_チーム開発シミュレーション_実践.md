// タスク検索・フィルターモジュール
const { getAllTasks } = require('./task');

function searchTasks(keyword) {
  const tasks = getAllTasks();
  const lowerKeyword = keyword.toLowerCase();
  return tasks.filter(t =>
    t.title.toLowerCase().includes(lowerKeyword) ||
    t.description.toLowerCase().includes(lowerKeyword)
  );
}

function filterByStatus(status) {
  const tasks = getAllTasks();
  return tasks.filter(t => t.status === status);
}

function filterTasks({ keyword, status, sortBy = "createdAt" }) {
  let results = getAllTasks();

  if (keyword) {
    const lower = keyword.toLowerCase();
    results = results.filter(t =>
      t.title.toLowerCase().includes(lower) ||
      t.description.toLowerCase().includes(lower)
    );
  }

  if (status) {
    results = results.filter(t => t.status === status);
  }

  results.sort((a, b) => {
    if (sortBy === "title") return a.title.localeCompare(b.title);
    return new Date(a[sortBy]) - new Date(b[sortBy]);
  });

  return results;
}

module.exports = { searchTasks, filterByStatus, filterTasks };
EOF

const notifications = [];

function createNotification(userId, message, type = "info") {
  const notification = {

    userId,
    message,
    type,  // "info", "warning", "error"

  return notifications.filter(n => n.userId === userId && !n.read);
}

  const notification = notifications.find(n => n.id === notificationId);

  return true;
}

<summary>解答例（改良版 ─ 実務で押さえるべきポイント解説）</summary>

基本的な手順は上記と同じですが、実務で重要なポイントを補足します。

### 1. コミット署名（Signed Commits）

```bash
# GPG鍵でコミットに署名する（改ざん防止）
git commit -S -m "feat: 署名付きコミット"

# 署名の確認
git log --show-signature
```

### 2. プルリクエストのテンプレート

```markdown
## 概要
<!-- 何を変更したか、なぜ変更したか -->

## 変更内容
<!-- 具体的な変更点をリスト -->

## テスト方法
<!-- どうやって動作確認したか -->

## チェックリスト
- [ ] テストを追加した
- [ ] ドキュメントを更新した
- [ ] セルフレビューを完了した
```

### 3. ブランチ保護ルール

```
main ブランチの保護設定（GitHub / GitLab）：
- 直接 push 禁止
- PR のレビュー最低1人必須
- CI テストをパス必須
- force push 禁止
```

### 4. git bisect でバグの原因コミットを特定

```bash
# バグが入ったコミットを二分探索で特定
git bisect start
git bisect bad          # 現在のコミットはバグあり
git bisect good v1.0.0  # v1.0.0 はバグなし
# Git が中間のコミットをチェックアウトするので、バグの有無を報告
git bisect good  # または git bisect bad
# ...繰り返し...
git bisect reset  # 元に戻す
```

### 5. git blame で「誰がこの行を書いたか」を調査

```bash
# 各行の最終変更者とコミットを表示
git blame src/task.js

# 特定の行範囲だけ表示
git blame -L 10,20 src/task.js
```

**初心者向けとの違い:**
- コミット署名による信頼性の確保
