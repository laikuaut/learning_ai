## 課題の要件

1. ユーザー登録時、パスワードをソルト付きハッシュ（SHA-256）で保存する
2. ソルトはユーザーごとにランダムに生成する
3. ログイン時、入力されたパスワードに同じソルトを付加してハッシュ化し、保存値と比較する
4. 平文保存の危険性を示すデモ機能を含める
5. 登録済みユーザー一覧でハッシュ値が表示され、元のパスワードが推測できないことを確認する

## ステップガイド

<details>
<summary>ステップ1：ソルトの生成方法を理解する</summary>

ソルトとは、パスワードに付加するランダムな文字列です。同じパスワードでもソルトが異なればハッシュ値が変わるため、レインボーテーブル攻撃（rainbow table attack）を防げます。

```python
import os

# 暗号学的に安全な乱数でソルトを生成
salt = os.urandom(16)  # 16バイト = 128ビット
print(salt.hex())       # 16進数で表示
```

import hashlib

def hash_password(password, salt):

## 解答例

<details>
<summary>解答例（初心者向け）</summary>

```python
# パスワードハッシュと安全な保存
# 学べる内容：ソルト付きハッシュ、パスワード保存のベストプラクティス
# 実行方法：python password_hash.py

import hashlib
import os

# ユーザーデータを格納する辞書
# キー: ユーザー名、値: {"hash": ハッシュ値, "salt": ソルト}
users = {}


def hash_password(password, salt):
    """パスワードをソルト付きSHA-256でハッシュ化します。"""
    data = salt + password.encode("utf-8")
    return hashlib.sha256(data).hexdigest()


def register_user(username, password):
    """ユーザーを登録します。"""
    if username in users:
        print(f"エラー: '{username}' は既に登録されています。")
        return False

    # ランダムなソルトを生成
    salt = os.urandom(16)
    pw_hash = hash_password(password, salt)

    users[username] = {"hash": pw_hash, "salt": salt}

    print(f"\n登録完了！")
    print(f"保存データ: {username}:{pw_hash[:16]}...:{salt.hex()[:16]}...")
    print("※ 元のパスワードは保存されていません")
    return True


def login(username, password):
    """ログイン認証を行います。"""
    if username not in users:
        print("認証失敗。ユーザーが見つかりません。")
        return False

    user_data = users[username]
    # 保存されているソルトで入力パスワードをハッシュ化
    input_hash = hash_password(password, user_data["salt"])

    if input_hash == user_data["hash"]:
        print(f"\n認証成功！ようこそ、{username}さん。")
        return True
    else:
        print("\n認証失敗。パスワードが正しくありません。")
        return False


def show_users():
    """登録済みユーザーの一覧を表示します。"""
    if not users:
        print("登録されているユーザーはいません。")
        return

    print("\n--- 登録済みユーザー一覧 ---")
    print(f"{'ユーザー名':<12} {'ハッシュ値（先頭16文字）':<20} {'ソルト（先頭16文字）'}")
    print("-" * 70)
    for username, data in users.items():
        print(f"{username:<12} {data['hash'][:16]}...   {data['salt'].hex()[:16]}...")
    print("\n※ ハッシュ値から元のパスワードは復元できません。")


def plaintext_vs_hash_demo():
    """平文保存とハッシュ保存の危険性を比較するデモです。"""
    print("\n--- 平文保存 vs ハッシュ保存 比較デモ ---")

    demo_password = "Secret123!"

    # 危険な例：平文保存
    print("\n【危険な例】平文保存:")
    print(f"  保存データ: alice:{demo_password}")
    print("  → データベースが漏洩すると、パスワードがそのまま読める！")

    # 不十分な例：ソルトなしハッシュ
    print("\n【不十分な例】ソルトなしハッシュ:")
    no_salt_hash = hashlib.sha256(demo_password.encode("utf-8")).hexdigest()
    print(f"  保存データ: alice:{no_salt_hash[:32]}...")
    print("  → 同じパスワードは同じハッシュ値になるため、")
    print("    レインボーテーブル攻撃で解読される可能性がある")

    # 安全な例：ソルト付きハッシュ
    print("\n【安全な例】ソルト付きハッシュ:")
    salt1 = os.urandom(16)
    salt2 = os.urandom(16)
    hash1 = hash_password(demo_password, salt1)
    hash2 = hash_password(demo_password, salt2)
    print(f"  alice: {hash1[:32]}... (ソルト: {salt1.hex()[:16]}...)")
    print(f"  bob:   {hash2[:32]}... (ソルト: {salt2.hex()[:16]}...)")
    print("  → 同じパスワードでもソルトが異なるため、ハッシュ値が全く異なる")
    print("  → レインボーテーブル攻撃が無効化される")


# --- メイン処理 ---
print("===== パスワード管理デモ =====")

while True:
    print("\n1. ユーザー登録")
    print("2. ログイン")
    print("3. 登録済みユーザー一覧")
    print("4. 平文保存 vs ハッシュ保存の比較デモ")
    print("5. 終了")

    choice = input("\n選択 (1-5): ")

    if choice == "1":
        username = input("ユーザー名: ")
        password = input("パスワード: ")
        register_user(username, password)

    elif choice == "2":
        username = input("ユーザー名: ")
        password = input("パスワード: ")
        login(username, password)

    elif choice == "3":
        show_users()

    elif choice == "4":
        plaintext_vs_hash_demo()

    elif choice == "5":
        print("終了します。")
        break

    else:
        print("1〜5の番号を入力してください。")
```

<summary>解答例（改良版 ─ PBKDF2とストレッチングを使用）</summary>

```python
# パスワードハッシュと安全な保存（改良版）
# 学べる内容：PBKDF2、ストレッチング、hmacによる安全な比較
# 実行方法：python password_hash_v2.py

import hashlib
import hmac
import os
import time

# ストレッチング回数（実務では100,000回以上が推奨されます）
ITERATIONS = 100_000
HASH_ALGORITHM = "sha256"
SALT_LENGTH = 32  # 256ビット

users = {}


def hash_password_pbkdf2(password, salt):
    """PBKDF2（Password-Based Key Derivation Function 2）でハッシュ化します。
    ストレッチング（stretching）により、ブルートフォース攻撃を大幅に遅くします。
    """
    dk = hashlib.pbkdf2_hmac(
        HASH_ALGORITHM,
        password.encode("utf-8"),
        salt,
        ITERATIONS,
    )
    return dk.hex()


def register_user(username, password):
    """ユーザーを登録します。"""
    if username in users:
        print(f"エラー: '{username}' は既に登録されています。")
        return False

    salt = os.urandom(SALT_LENGTH)

    start = time.time()
    pw_hash = hash_password_pbkdf2(password, salt)
    elapsed = time.time() - start

    users[username] = {
        "hash": pw_hash,
        "salt": salt,
        "iterations": ITERATIONS,
    }

    print(f"\n登録完了！（ハッシュ計算時間: {elapsed:.3f}秒）")
    print(f"保存データ: {username}:{pw_hash[:16]}...:{salt.hex()[:16]}...")
    print(f"ストレッチング回数: {ITERATIONS:,}回")
    return True


def login(username, password):
    """ログイン認証を行います。"""
    if username not in users:
        # タイミング攻撃（timing attack）対策として、
        # ユーザーが存在しない場合もハッシュ計算を行います
        dummy_salt = os.urandom(SALT_LENGTH)
        hash_password_pbkdf2(password, dummy_salt)
        print("\n認証失敗。ユーザー名またはパスワードが正しくありません。")
        return False

    user_data = users[username]
    input_hash = hash_password_pbkdf2(password, user_data["salt"])

    # hmac.compare_digest で定時間比較（timing-safe comparison）
    if hmac.compare_digest(input_hash, user_data["hash"]):
        print(f"\n認証成功！ようこそ、{username}さん。")
        return True
    else:
        print("\n認証失敗。ユーザー名またはパスワードが正しくありません。")
        return False


def show_users():
    """登録済みユーザーの一覧を表示します。"""
    if not users:
        print("登録されているユーザーはいません。")
        return

    print("\n--- 登録済みユーザー一覧 ---")
    for username, data in users.items():
        print(f"  {username}:")
        print(f"    ハッシュ: {data['hash'][:32]}...")
        print(f"    ソルト:   {data['salt'].hex()[:32]}...")
        print(f"    反復回数: {data['iterations']:,}")


def stretching_demo():
    """ストレッチングの効果を体感するデモです。"""
    print("\n--- ストレッチング効果デモ ---")
    password = "test"
    salt = os.urandom(16)

    for iters in [1, 1000, 10_000, 100_000]:
        start = time.time()
        hashlib.pbkdf2_hmac("sha256", password.encode(), salt, iters)
        elapsed = time.time() - start
        print(f"  反復 {iters:>7,}回: {elapsed:.4f}秒")

    print()
    print("ポイント:")
    print("  - 反復回数を増やすとハッシュ計算に時間がかかります")
    print("  - 正規ユーザーは1回しか計算しないので影響は小さい")
    print("  - 攻撃者は何億回も試す必要があるため、大きな障壁になります")


def main():
    print("===== パスワード管理デモ（PBKDF2版） =====")

    while True:
        print("\n1. ユーザー登録")
        print("2. ログイン")
        print("3. 登録済みユーザー一覧")
        print("4. ストレッチング効果デモ")
        print("5. 終了")

        choice = input("\n選択 (1-5): ")

        if choice == "1":
            username = input("ユーザー名: ")
            password = input("パスワード: ")
            register_user(username, password)
        elif choice == "2":
            username = input("ユーザー名: ")
            password = input("パスワード: ")
            login(username, password)
        elif choice == "3":
            show_users()
        elif choice == "4":
            stretching_demo()
        elif choice == "5":
            print("終了します。")
            break
        else:
            print("1〜5の番号を入力してください。")


if __name__ == "__main__":
    main()
```

**初心者向けとの違い:**
- `hashlib.pbkdf2_hmac()` で PBKDF2（鍵導出関数）を使用 → 実務標準のハッシュ方式
- ストレッチング（10万回反復）でブルートフォース攻撃を大幅に遅延 → 実際の認証基盤と同じ考え方
- `hmac.compare_digest()` でタイミング攻撃（timing attack）を防止 → 定時間比較
- ユーザーが存在しない場合もダミーのハッシュ計算を実行 → 応答時間からユーザーの存在を推測されない
- ストレッチングの効果を体感できるデモ機能を追加
