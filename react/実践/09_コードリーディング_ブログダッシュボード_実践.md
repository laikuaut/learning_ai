## 読解対象コード

```jsx
// ==================== 型定義（コメント） ====================
// State: { posts: Post[], filter: string, sortBy: string, editingId: number|null }
// Post: { id: number, title: string, body: string, status: "draft"|"published", createdAt: string, tags: string[] }

// ==================== Reducer ====================
const initialState = {
  posts: [
    { id: 1, title: "React入門", body: "Reactの基本を学びます...", status: "published", createdAt: "2024-03-01", tags: ["React", "入門"] },
    { id: 2, title: "useState解説", body: "stateの管理方法...", status: "published", createdAt: "2024-03-05", tags: ["React", "hooks"] },
    { id: 3, title: "CSS設計", body: "BEM記法について...", status: "draft", createdAt: "2024-03-10", tags: ["CSS"] },
    { id: 4, title: "TypeScript入門", body: "型安全なコードを...", status: "draft", createdAt: "2024-03-12", tags: ["TypeScript", "入門"] },
    { id: 5, title: "useEffect詳解", body: "副作用の管理...", status: "published", createdAt: "2024-03-15", tags: ["React", "hooks"] },
  ],
  filter: "all",
  sortBy: "newest",
  editingId: null,
};

function reducer(state, action) {
  switch (action.type) {
    case "ADD_POST":
      return {
        ...state,
        posts: [...state.posts, { ...action.payload, id: Date.now() }],
      };

    case "DELETE_POST":
      return {
        ...state,
        posts: state.posts.filter(p => p.id !== action.payload),
        editingId: state.editingId === action.payload ? null : state.editingId,
      };

    case "UPDATE_POST":
      return {
        ...state,
        posts: state.posts.map(p =>
          p.id === action.payload.id ? { ...p, ...action.payload.changes } : p
        ),
        editingId: null,
      };

    case "TOGGLE_STATUS":
      return {
        ...state,
        posts: state.posts.map(p =>
          p.id === action.payload
            ? { ...p, status: p.status === "draft" ? "published" : "draft" }
            : p
        ),
      };

    case "SET_FILTER":
      return { ...state, filter: action.payload };

    case "SET_SORT":
      return { ...state, sortBy: action.payload };

    case "SET_EDITING":
      return { ...state, editingId: action.payload };

    default:
      return state;
  }
}

// ==================== セレクタ関数 ====================
function getFilteredPosts(state) {
  let posts = state.posts;

  // フィルタ
  if (state.filter === "published") {
    posts = posts.filter(p => p.status === "published");
  } else if (state.filter === "draft") {
    posts = posts.filter(p => p.status === "draft");
  }

  // ソート
  posts = [...posts].sort((a, b) => {
    if (state.sortBy === "newest") return b.createdAt.localeCompare(a.createdAt);
    if (state.sortBy === "oldest") return a.createdAt.localeCompare(b.createdAt);
    if (state.sortBy === "title") return a.title.localeCompare(b.title);
    return 0;
  });

  return posts;
}

function getStats(state) {
  const total = state.posts.length;
  const published = state.posts.filter(p => p.status === "published").length;
  const draft = total - published;
  const allTags = state.posts.flatMap(p => p.tags);
  const uniqueTags = [...new Set(allTags)];
  return { total, published, draft, uniqueTags };
}

// ==================== コンポーネント ====================
function StatsBar({ stats }) {
  return (
    <div className="stats">
      <span>全{stats.total}件</span>
      <span>公開: {stats.published}</span>
      <span>下書き: {stats.draft}</span>
      <span>タグ: {stats.uniqueTags.join(", ")}</span>
    </div>
  );
}

function FilterBar({ filter, sortBy, dispatch }) {
  return (
    <div className="filter-bar">
      <select value={filter} onChange={e => dispatch({ type: "SET_FILTER", payload: e.target.value })}>
        <option value="all">すべて</option>
        <option value="published">公開済み</option>
        <option value="draft">下書き</option>
      </select>
      <select value={sortBy} onChange={e => dispatch({ type: "SET_SORT", payload: e.target.value })}>
        <option value="newest">新しい順</option>
        <option value="oldest">古い順</option>
        <option value="title">タイトル順</option>
      </select>
    </div>
  );
}

function PostCard({ post, isEditing, dispatch }) {
  const [editTitle, setEditTitle] = React.useState(post.title);
  const [editBody, setEditBody] = React.useState(post.body);

  if (isEditing) {
    return (
      <div className="post-card editing">
        <input value={editTitle} onChange={e => setEditTitle(e.target.value)} />
        <textarea value={editBody} onChange={e => setEditBody(e.target.value)} />
        <button onClick={() => dispatch({
          type: "UPDATE_POST",
          payload: { id: post.id, changes: { title: editTitle, body: editBody } }
        })}>保存</button>
        <button onClick={() => dispatch({ type: "SET_EDITING", payload: null })}>キャンセル</button>
      </div>
    );
  }

  return (
    <div className="post-card">
      <h3>{post.title}</h3>
      <p>{post.body.substring(0, 50)}...</p>
      <div className="meta">
        <span className={"status " + post.status}>{post.status === "published" ? "公開" : "下書き"}</span>
        <span>{post.createdAt}</span>
        <span>{post.tags.map(t => `#${t}`).join(" ")}</span>
      </div>
      <div className="actions">
        <button onClick={() => dispatch({ type: "SET_EDITING", payload: post.id })}>編集</button>
        <button onClick={() => dispatch({ type: "TOGGLE_STATUS", payload: post.id })}>
          {post.status === "draft" ? "公開する" : "下書きに戻す"}
        </button>
        <button onClick={() => dispatch({ type: "DELETE_POST", payload: post.id })}>削除</button>
      </div>
    </div>
  );
}

function App() {
  const [state, dispatch] = React.useReducer(reducer, initialState);
  const filteredPosts = getFilteredPosts(state);
  const stats = getStats(state);

  return (
    <div>
      <h1>ブログダッシュボード</h1>
      <StatsBar stats={stats} />
      <FilterBar filter={state.filter} sortBy={state.sortBy} dispatch={dispatch} />
      {filteredPosts.map(post => (
        <PostCard
          key={post.id}
          post={post}
          isEditing={state.editingId === post.id}
          dispatch={dispatch}
        />
      ))}
    </div>
  );
}
```

### 設問1
編集中の記事が削除された場合、`editingId` が存在しないIDを指したままになります。その結果、画面上で編集フォームが空の状態で残ったり、エラーが発生する可能性があります。削除された記事のIDと一致する場合に `null` にリセットすることで、整合性を保っています。

### 設問2
`Array.sort()` は**破壊的メソッド**（元の配列を直接変更する）です。`state.posts` を直接ソートすると、Reactのstateを間接的に変更することになり、**不変性の原則**に違反します。スプレッドで新しい配列を作ってからソートすることで、元のstateを変更しません。

### 設問3

タイトル順（`localeCompare`）でソートすると：

2. **TypeScript入門**

### 設問5
- **再レンダリング時に関数が再生成されない**: コンポーネント内に書くと毎レンダリングで新しい関数が作られる

### 設問6
**dispatch直渡しのメリット:**
- コードが簡潔（親コンポーネントにコールバック定義不要）
- 新しいactionを追加しても親の修正不要

**dispatch直渡しのデメリット:**
- PostCardがreducerのaction type を知っている必要がある（密結合）
- PostCardの再利用性が低下（このreducer専用になる）
- テスト時にdispatchをモック化する必要がある

**個別コールバックのメリット:**
- PostCardが汎用的になる（どんな状態管理にも使える）
- インターフェースが明確

**個別コールバックのデメリット:**
- 親にラッパー関数が増える

一般的に小規模アプリではdispatch直渡し、大規模では個別コールバックが推奨されます。

### 設問7
**バグです。** `PostCard` 内の `editTitle` と `editBody` は `useState(post.title)` で初期化されますが、`isEditing` が `false → true` に変わっても `useState` の初期値は再評価されません（Reactの仕様）。

結果として、前に編集していた値がそのまま残る可能性があります。

修正方法: `useEffect` で `isEditing` が `true` になったとき値をリセットする。
```jsx
React.useEffect(() => {
  if (isEditing) {
    setEditTitle(post.title);
    setEditBody(post.body);
  }
}, [isEditing, post.title, post.body]);
```

### 設問8
どちらにもメリットがあります。

**`return state` のメリット:** 未知のactionを無視するため、複数のreducerを組み合わせるときに柔軟です。

**`throw new Error()` のメリット:** タイプミス（例: `"DELET_POST"`）をすぐに検出できます。

**判断基準:** 単一のreducerで全actionを処理する設計なら `throw` で早期検出がベター。複数のreducerやミドルウェアを組み合わせるなら `return state` が安全です。
