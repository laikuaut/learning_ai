# 実践課題07：フォームバリデーションエンジン ★3

> **難易度**: ★★★☆☆（中級）
> **前提知識**: 第3章（関数の型）、第4章（ユニオン型と交差型）、第5章（ジェネリクス）、第7章（型ガードと型の絞り込み）
> **課題の種類**: ミニプロジェクト
> **学習目標**: 型ガード（type guard）と型の絞り込み（narrowing）を駆使して、堅牢なバリデーション処理を構築する

---

## 完成イメージ

```
===== フォームバリデーションエンジン =====

--- ユーザー登録フォームのバリデーション ---

入力データ:
  username: "tanaka_taro"
  email: "tanaka@example.com"
  age: 28
  password: "Str0ngP@ss!"
  confirmPassword: "Str0ngP@ss!"

バリデーション結果: ✅ 成功
  すべてのフィールドが有効です。

--- 不正データのバリデーション ---

入力データ:
  username: "ab"
  email: "invalid-email"
  age: -5
  password: "123"
  confirmPassword: "456"

バリデーション結果: ❌ 失敗（5件のエラー）
  ❌ username: 3文字以上で入力してください（現在: 2文字）
  ❌ email: 有効なメールアドレス形式ではありません
  ❌ age: 0以上の値を入力してください
  ❌ password: 8文字以上で、大文字・小文字・数字・記号を含めてください
  ❌ confirmPassword: パスワードが一致しません

--- 型ガードの活用 ---
値 "hello" はstring型です → 長さ: 5
値 42 はnumber型です → 偶数: true
値 null はnull型です → デフォルト値を使用します
値 [1,2,3] はarray型です → 要素数: 3
```

---

## 課題の要件

1. フォーム入力値の型を定義する（ユニオン型で「未入力 / 文字列 / 数値」を表現）
2. 各フィールドにバリデーションルールを適用する
3. バリデーション結果をユニオン型（成功 | 失敗）で返す
4. 型ガード関数を使って、入力値の型に応じた処理を行う
5. カスタム型ガードを定義する（`value is SomeType` 構文）
6. `typeof`, `instanceof`, `in` 演算子による型の絞り込みを実践する

## 解答例

<details>
<summary>解答例（初心者向け）</summary>

```typescript
// フォームバリデーションエンジン
// 学べる内容：型ガード、型の絞り込み、カスタム型ガード、ユニオン型
// 実行方法：npx ts-node 07_form_validator.ts

// --- 型定義 ---
type FormValue = string | number | boolean | null | undefined;

type FieldSuccess = {
  valid: true;
  fieldName: string;
};

type FieldError = {
  valid: false;
  fieldName: string;
  message: string;
};

type FieldResult = FieldSuccess | FieldError;

type FormResult =
  | { success: true; data: Record<string, FormValue> }
  | { success: false; errors: FieldError[] };

// --- 型ガード関数 ---
function isString(value: FormValue): value is string {
  return typeof value === "string";
}

function isNumber(value: FormValue): value is number {
  return typeof value === "number";
}

function isBoolean(value: FormValue): value is boolean {
  return typeof value === "boolean";
}

function isNonNullable(value: FormValue): value is string | number | boolean {
  return value !== null && value !== undefined;
}

// --- バリデーション関数 ---
function validateRequired(fieldName: string, value: FormValue): FieldResult {
  if (!isNonNullable(value)) {
    return { valid: false, fieldName, message: "必須項目です" };
  }
  return { valid: true, fieldName };
}

function validateMinLength(
  fieldName: string,
  value: FormValue,
  min: number
): FieldResult {
  if (!isString(value)) {
    return { valid: false, fieldName, message: "文字列を入力してください" };
  }
  if (value.length < min) {
    return {
      valid: false,
      fieldName,
      message: `${min}文字以上で入力してください（現在: ${value.length}文字）`,
    };
  }
  return { valid: true, fieldName };
}

function validateEmail(fieldName: string, value: FormValue): FieldResult {
  if (!isString(value)) {
    return { valid: false, fieldName, message: "文字列を入力してください" };
  }
  // 簡易的なメールアドレスチェック
  const hasAtSign: boolean = value.includes("@");
  const hasDot: boolean = value.includes(".");
  if (!hasAtSign || !hasDot) {
    return {
      valid: false,
      fieldName,
      message: "有効なメールアドレス形式ではありません",
    };
  }
  return { valid: true, fieldName };
}

function validatePositiveNumber(
  fieldName: string,
  value: FormValue
): FieldResult {
  if (!isNumber(value)) {
    return { valid: false, fieldName, message: "数値を入力してください" };
  }
  if (value < 0) {
    return {
      valid: false,
      fieldName,
      message: "0以上の値を入力してください",
    };
  }
  return { valid: true, fieldName };
}

function validatePassword(fieldName: string, value: FormValue): FieldResult {
  if (!isString(value)) {
    return { valid: false, fieldName, message: "文字列を入力してください" };
  }
  const hasUpperCase: boolean = /[A-Z]/.test(value);
  const hasLowerCase: boolean = /[a-z]/.test(value);
  const hasDigit: boolean = /[0-9]/.test(value);
  const hasSpecial: boolean = /[^A-Za-z0-9]/.test(value);
  const longEnough: boolean = value.length >= 8;

  if (!longEnough || !hasUpperCase || !hasLowerCase || !hasDigit || !hasSpecial) {
    return {
      valid: false,
      fieldName,
      message:
        "8文字以上で、大文字・小文字・数字・記号を含めてください",
    };
  }
  return { valid: true, fieldName };
}

function validateMatch(
  fieldName: string,
  value: FormValue,
  otherValue: FormValue
): FieldResult {
  if (value !== otherValue) {
    return {
      valid: false,
      fieldName,
      message: "パスワードが一致しません",
    };
  }
  return { valid: true, fieldName };
}

// --- フォーム全体のバリデーション ---
interface RegistrationForm {
  username: FormValue;
  email: FormValue;
  age: FormValue;
  password: FormValue;
  confirmPassword: FormValue;
}

function validateRegistration(form: RegistrationForm): FormResult {
  const results: FieldResult[] = [
    validateMinLength("username", form.username, 3),
    validateEmail("email", form.email),
    validatePositiveNumber("age", form.age),
    validatePassword("password", form.password),
    validateMatch("confirmPassword", form.confirmPassword, form.password),
  ];

  const errors: FieldError[] = [];
  for (const result of results) {
    if (!result.valid) {
      errors.push(result);
    }
  }

  if (errors.length === 0) {
    return {
      success: true,
      data: {
        username: form.username,
        email: form.email,
        age: form.age,
        password: form.password,
      },
    };
  }
  return { success: false, errors };
}

// --- 結果表示 ---
function displayFormData(form: RegistrationForm): void {
  console.log("入力データ:");
  console.log(`  username: ${JSON.stringify(form.username)}`);
  console.log(`  email: ${JSON.stringify(form.email)}`);
  console.log(`  age: ${JSON.stringify(form.age)}`);
  console.log(`  password: ${JSON.stringify(form.password)}`);
  console.log(`  confirmPassword: ${JSON.stringify(form.confirmPassword)}`);
}

function displayResult(result: FormResult): void {
  if (result.success) {
    console.log("\nバリデーション結果: ✅ 成功");
    console.log("  すべてのフィールドが有効です。");
  } else {
    console.log(
      `\nバリデーション結果: ❌ 失敗（${result.errors.length}件のエラー）`
    );
    for (const error of result.errors) {
      console.log(`  ❌ ${error.fieldName}: ${error.message}`);
    }
  }
}

// --- 型ガードのデモ ---
function describeValue(value: unknown): string {
  if (value === null) {
    return "null型です → デフォルト値を使用します";
  }
  if (value === undefined) {
    return "undefined型です → デフォルト値を使用します";
  }
  if (typeof value === "string") {
    return `string型です → 長さ: ${value.length}`;
  }
  if (typeof value === "number") {
    return `number型です → 偶数: ${value % 2 === 0}`;
  }
  if (typeof value === "boolean") {
    return `boolean型です → 値: ${value}`;
  }
  if (Array.isArray(value)) {
    return `array型です → 要素数: ${value.length}`;
  }
  return `その他の型です: ${typeof value}`;
}

// --- メイン処理 ---
console.log("===== フォームバリデーションエンジン =====\n");

// 有効なフォーム
console.log("--- ユーザー登録フォームのバリデーション ---\n");
const validForm: RegistrationForm = {
  username: "tanaka_taro",
  email: "tanaka@example.com",
  age: 28,
  password: "Str0ngP@ss!",
  confirmPassword: "Str0ngP@ss!",
};
displayFormData(validForm);
displayResult(validateRegistration(validForm));

// 不正なフォーム
console.log("\n--- 不正データのバリデーション ---\n");
const invalidForm: RegistrationForm = {
  username: "ab",
  email: "invalid-email",
  age: -5,
  password: "123",
  confirmPassword: "456",
};
displayFormData(invalidForm);
displayResult(validateRegistration(invalidForm));

// 型ガードのデモ
console.log("\n--- 型ガードの活用 ---");
const testValues: unknown[] = ["hello", 42, null, [1, 2, 3]];
for (const val of testValues) {
  console.log(`値 ${JSON.stringify(val)} は${describeValue(val)}`);
}
```

<summary>解答例（改良版 ─ ジェネリック型ガードとチェイン可能なバリデーション）</summary>

```typescript
// フォームバリデーションエンジン（改良版）
// 学べる内容：ジェネリック型ガード、バリデーションの合成、in演算子による型の絞り込み
// 実行方法：npx ts-node 07_form_validator_v2.ts

// --- 型定義 ---
type FormValue = string | number | boolean | null | undefined;

type FieldSuccess = { readonly valid: true; readonly fieldName: string };
type FieldError = { readonly valid: false; readonly fieldName: string; readonly message: string };
type FieldResult = FieldSuccess | FieldError;

// バリデーション関数の型
type Validator = (fieldName: string, value: FormValue) => FieldResult;

// --- 型ガード ---
function isString(value: unknown): value is string {
  return typeof value === "string";
}

function isNumber(value: unknown): value is number {
  return typeof value === "number" && !Number.isNaN(value);
}

function isNonNullable<T>(value: T): value is NonNullable<T> {
  return value !== null && value !== undefined;
}

// in 演算子による型の絞り込み
interface ApiError {
  code: number;
  message: string;
}

interface NetworkError {
  type: "timeout" | "connection";
  retryable: boolean;
}

type AppError = ApiError | NetworkError;

function handleError(error: AppError): string {
  if ("code" in error) {
    // TypeScriptはここでerrorがApiError型であることを認識する
    return `APIエラー (${error.code}): ${error.message}`;
  } else {
    // ここではNetworkError型
    return `ネットワークエラー (${error.type}): リトライ${error.retryable ? "可" : "不可"}`;
  }
}

// --- バリデータビルダー（合成パターン） ---
function required(): Validator {
  return (fieldName: string, value: FormValue): FieldResult => {
    if (!isNonNullable(value) || (isString(value) && value.trim() === "")) {
      return { valid: false, fieldName, message: "必須項目です" };
    }
    return { valid: true, fieldName };
  };
}

function minLength(min: number): Validator {
  return (fieldName: string, value: FormValue): FieldResult => {
    if (!isString(value)) return { valid: true, fieldName }; // 他のバリデータに委ねる
    if (value.length < min) {
      return { valid: false, fieldName, message: `${min}文字以上で入力してください（現在: ${value.length}文字）` };
    }
    return { valid: true, fieldName };
  };
}

function emailFormat(): Validator {
  return (fieldName: string, value: FormValue): FieldResult => {
    if (!isString(value)) return { valid: true, fieldName };
    if (!value.includes("@") || !value.includes(".")) {
      return { valid: false, fieldName, message: "有効なメールアドレス形式ではありません" };
    }
    return { valid: true, fieldName };
  };
}

function positiveNumber(): Validator {
  return (fieldName: string, value: FormValue): FieldResult => {
    if (!isNumber(value)) return { valid: false, fieldName, message: "数値を入力してください" };
    if (value < 0) return { valid: false, fieldName, message: "0以上の値を入力してください" };
    return { valid: true, fieldName };
  };
}

function strongPassword(): Validator {
  return (fieldName: string, value: FormValue): FieldResult => {
    if (!isString(value)) return { valid: true, fieldName };
    const checks: boolean[] = [
      value.length >= 8,
      /[A-Z]/.test(value),
      /[a-z]/.test(value),
      /[0-9]/.test(value),
      /[^A-Za-z0-9]/.test(value),
    ];
    if (checks.some((ok: boolean): boolean => !ok)) {
      return { valid: false, fieldName, message: "8文字以上で、大文字・小文字・数字・記号を含めてください" };
    }
    return { valid: true, fieldName };
  };
}

// --- バリデータの合成 ---
function compose(...validators: Validator[]): Validator {
  return (fieldName: string, value: FormValue): FieldResult => {
    for (const validator of validators) {
      const result: FieldResult = validator(fieldName, value);
      if (!result.valid) return result; // 最初のエラーで停止
    }
    return { valid: true, fieldName };
  };
}

// --- フォームスキーマ ---
type FormSchema = Record<string, Validator>;

function validateForm(
  schema: FormSchema,
  data: Record<string, FormValue>
): FieldResult[] {
  const results: FieldResult[] = [];
  for (const [fieldName, validator] of Object.entries(schema)) {
    results.push(validator(fieldName, data[fieldName]));
  }
  return results;
}

// --- メイン処理 ---
console.log("===== フォームバリデーションエンジン =====\n");

// バリデーションスキーマを宣言的に定義
const registrationSchema: FormSchema = {
  username: compose(required(), minLength(3)),
  email: compose(required(), emailFormat()),
  age: compose(required(), positiveNumber()),
  password: compose(required(), strongPassword()),
};

// 有効なデータ
console.log("--- 有効なデータ ---");
const validData: Record<string, FormValue> = {
  username: "tanaka_taro",
  email: "tanaka@example.com",
  age: 28,
  password: "Str0ngP@ss!",
};

const validResults: FieldResult[] = validateForm(registrationSchema, validData);
const validErrors: FieldError[] = validResults.filter(
  (r: FieldResult): r is FieldError => !r.valid
);
console.log(validErrors.length === 0 ? "✅ すべて有効です" : `❌ ${validErrors.length}件のエラー`);

// 無効なデータ
console.log("\n--- 無効なデータ ---");
const invalidData: Record<string, FormValue> = {
  username: "ab",
  email: "bad-email",
  age: -5,
  password: "123",
};

const invalidResults: FieldResult[] = validateForm(registrationSchema, invalidData);
for (const r of invalidResults) {
  if (!r.valid) {
    console.log(`  ❌ ${r.fieldName}: ${r.message}`);
  }
}

// in 演算子による型の絞り込みのデモ
console.log("\n--- in演算子による型の絞り込み ---");
const errors: AppError[] = [
  { code: 404, message: "Not Found" },
  { type: "timeout", retryable: true },
  { code: 500, message: "Internal Server Error" },
  { type: "connection", retryable: false },
];
for (const error of errors) {
  console.log(`  ${handleError(error)}`);
}

// typeof による型の絞り込み
console.log("\n--- typeofによる絞り込み ---");
const mixed: unknown[] = ["hello", 42, true, null, undefined, [1, 2, 3]];
for (const val of mixed) {
  if (typeof val === "string") {
    console.log(`  "${val}" → 大文字: ${val.toUpperCase()}`);
  } else if (typeof val === "number") {
    console.log(`  ${val} → 2倍: ${val * 2}`);
  } else if (typeof val === "boolean") {
    console.log(`  ${val} → 反転: ${!val}`);
  } else if (val === null) {
    console.log(`  null → スキップ`);
  } else if (Array.isArray(val)) {
    console.log(`  [${val}] → 要素数: ${val.length}`);
  }
}
```

**初心者向けとの違い:**
- `compose()` でバリデータを合成し、宣言的にスキーマを定義
- `FormSchema` でフォーム全体のバリデーションルールを1箇所にまとめる
- `in` 演算子による型の絞り込み（`"code" in error` で `ApiError` と `NetworkError` を判別）
- `NonNullable<T>` ジェネリック型ガード
- 関数型プログラミング的なパターン（バリデータを値として扱う）
