```python
# git_server.py の実装例（抜粋）
import subprocess
from pathlib import Path
from mcp.server.fastmcp import FastMCP

mcp = FastMCP("git-server")
REPO_PATH = Path(os.environ.get("REPO_PATH", ".")).resolve()

def run_git(args: list[str]) -> str:
    """Gitコマンドを安全に実行します。"""
    result = subprocess.run(
        ["git", "-C", str(REPO_PATH)] + args,
        capture_output=True, text=True, timeout=30,
    )
    if result.returncode != 0:
        raise RuntimeError(f"Git error: {result.stderr.strip()}")
    return result.stdout.strip()

@mcp.tool()
def git_log(count: int = 10, branch: str = "HEAD") -> str:
    """コミット履歴を表示します。

    Args:
        count: 表示するコミット数（デフォルト10）
        branch: ブランチ名（デフォルトはHEAD）
    """
    # ブランチ名のバリデーション（コマンドインジェクション防止）
    if not all(c.isalnum() or c in "-_/." for c in branch):
        raise ValueError("不正なブランチ名です")
    if count < 1 or count > 100:
        raise ValueError("��示数��1〜100の範囲で指定してください")

    return run_git([
        "log", f"--max-count={count}",
        "--format=%h %s (%an, %ar)", branch,
    ])

@mcp.tool()
def git_status() -> str:
    """リポジトリのステータスを表示します。"""
    return run_git(["status", "--short"])

@mcp.tool()
def git_diff(target: str = "HEAD") -> str:
    """変更差分を表示します。

    Args:
        target: 比較対象（デフォルトはHEAD）
    """
    if not all(c.isalnum() or c in "-_/.~^" for c in target):
        raise ValueError("不正な対象指定です")
    return run_git(["diff", target])
```

### マルチサーバー統合の設計課題と解決策

#### 課題1：ツール名の衝突

```
問題: 4つのサーバーすべてに "search" ツールがあると、LLMが混乱する

解決策1: プレフィックスによる名前空間
  git_search, task_search, doc_search, monitor_search

解決策2: 説明文での差別化
  各ツールのdocstringに「どのデータを検索するか」を明記

推奨: 両方を併用する
```

#### 課題2：サーバーの起動失敗

```
問題: 4つのうち1つのサーバーが起動に失敗した場合

設計パターン: グレースフルデグラデーション（graceful degradation）
  ・起動に失敗したサーバーのツールは利用不可として扱う
  ・残りのサーバーは正常に動作を継続する
  ・ユーザーに失敗したサーバーを通知する

実装例:
  ホスト側で各サーバーの接続状態を管理
  ┌──────────────────────────────────────┐
  │ Server Status                        │
  │  git:     ✓ Connected (5 tools)      │
  │  tasks:   ✓ Connected (6 tools)      │
  │  docs:    ✗ Failed (timeout)         │
  │  monitor: ✓ Connected (3 tools)      │
  │                                      │
  │  利用可能ツール: 14/19               │
  └──────────────────────────────────────┘
```

#### 課���3：ツール呼び出しのタイムアウト

```
問題: 外部API連携するサーバーが応答しない場合

設計パターン: タイムアウト + フォールバック
  ┌──────────────────────────────────────────┐
  │  タイムアウト戦略                          │
  │                                          │
  │  Layer 1: HTTPリクエスト → 10秒           │
  │  Layer 2: ツール呼び出し → 30秒           │
  │  Layer 3: ワークフロー全体 → 120秒        │
  │                                          │
  │  タ��ムアウト時の動作:                     │
  │  ・エラーメッセージをLLMに返す             │
  │  ・LLMが代替手段を判断                    │
  │  ・ユーザーに状況を報告                    │
  └──────────────────────────────────────────┘
```

#### 課題4：サーバー間のデータ整合性

```
問題: タスクを完了にしたが、Gitのブランチはマージされていない

設計原則:
  ���MCPサーバー間は直接通信しない（疎結合）
  ・整合性の担保はLLMの判断に委ねる
  ・チェックリスト型のプロンプトで手順を標準化

例: task_close プロンプト
  1. git_status で未コミットの変更がないか確認
  2. git_log でPRがマージ済みか確認
  3. monitor_get_logs でエラーがないか確認
  4. すべてOKなら task_update_status で完了に
```

### 監視・ログ設計

```python
# 各サーバー共通のログ設定
import logging
import json
from datetime import datetime, timezone

class MCPJsonLogger:
    """構造化ログを出力するロガーです。"""

    def __init__(self, server_name: str):
        self.server_name = server_name
        self.logger = logging.getLogger(server_name)

    def log_tool_call(self, tool_name: str, args: dict, result: str, duration_ms: float):
        """ツール呼び出しのログを記録します。"""
        self.logger.info(json.dumps({
            "event": "tool_call",
            "server": self.server_name,
            "tool": tool_name,
            "args": args,
            "result_length": len(result),
            "duration_ms": duration_ms,
            "timestamp": datetime.now(timezone.utc).isoformat(),
        }, ensure_ascii=False))

    def log_error(self, tool_name: str, error: Exception):
        """エラーログを記録します。"""
        self.logger.error(json.dumps({
            "event": "tool_error",
            "server": self.server_name,
            "tool": tool_name,
            "error_type": type(error).__name__,
            "error_message": str(error),
            "timestamp": datetime.now(timezone.utc).isoformat(),
        }, ensure_ascii=False))
```

## よくある間違い

| 間違い | 正しい理解 |
|--------|-----------|
| 全機能を1��のサーバーに詰め込む | 単一責任の原則に従い、領域ごとにサーバーを分離しましょう。保守性・テスト性・デプロイの独立性が向上します |
| サーバー間で直接通信させようとする | MCPサーバー同士は直接通信しません。連携はLLMの判断による「ツールの連鎖」で実現します |
| ツール名にプレフィックスをつけない | 複数サーバーを統合すると名前が衝突します。`git_log`, `task_search` のようにプレフィックスで区別しましょう |
| 1つのサーバー障害で全体が止まる設計 | 各サーバーは独立して動作し、1つの障害が他に波及しない設計にしましょう |
| エラー時にリトライを無制限にする | リトライ回数を制限し（2〜3回）、失敗時はLLMにエラーを返して代替手段を判断させましょう |
