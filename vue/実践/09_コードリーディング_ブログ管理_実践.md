// ==================== 型定義（コメント） ====================
// Post: { id: number, title: string, body: string, status: "draft"|"published", createdAt: string, tags: string[] }
// State: posts配列, filter文字列, sortBy文字列, editingId（number|null）

// ==================== composable: usePostStore ====================
function usePostStore() {
  const posts = ref([
    { id: 1, title: "Vue 3入門", body: "Vue 3の基本を学びます...", status: "published", createdAt: "2024-03-01", tags: ["Vue", "入門"] },
    { id: 2, title: "Composition API解説", body: "setup関数とref/reactive...", status: "published", createdAt: "2024-03-05", tags: ["Vue", "Composition API"] },
    { id: 3, title: "CSS設計", body: "BEM記法について...", status: "draft", createdAt: "2024-03-10", tags: ["CSS"] },
    { id: 4, title: "TypeScript入門", body: "型安全なコードを...", status: "draft", createdAt: "2024-03-12", tags: ["TypeScript", "入門"] },
    { id: 5, title: "Pinia入門", body: "Vue 3の状態管理...", status: "published", createdAt: "2024-03-15", tags: ["Vue", "Pinia"] },
  ])

  const filter = ref("all")        // "all" | "draft" | "published"
  const sortBy = ref("newest")     // "newest" | "oldest" | "title"
  const editingId = ref(null)

  // (A)
  const filteredPosts = computed(() => {
    let result = posts.value
    if (filter.value !== "all") {
      result = result.filter(p => p.status === filter.value)
    }
    return result
  })

  // (B)
  const sortedPosts = computed(() => {
    const arr = [...filteredPosts.value]
    switch (sortBy.value) {
      case "newest": return arr.sort((a, b) => b.createdAt.localeCompare(a.createdAt))
      case "oldest": return arr.sort((a, b) => a.createdAt.localeCompare(b.createdAt))
      case "title":  return arr.sort((a, b) => a.title.localeCompare(b.title))
      default: return arr
    }
  })

  // (C)
  const stats = computed(() => ({
    total: posts.value.length,
    published: posts.value.filter(p => p.status === "published").length,
    draft: posts.value.filter(p => p.status === "draft").length,
    // (D)
    allTags: [...new Set(posts.value.flatMap(p => p.tags))],
  }))

  // (E)
  function addPost(postData) {
    posts.value.push({
      ...postData,
      id: Date.now(),
      createdAt: new Date().toISOString().slice(0, 10),
    })
  }

  function deletePost(id) {
    posts.value = posts.value.filter(p => p.id !== id)
    if (editingId.value === id) editingId.value = null
  }

  // (F)
  function updatePost(id, changes) {
    const post = posts.value.find(p => p.id === id)
    if (post) Object.assign(post, changes)
    editingId.value = null
  }

  function toggleStatus(id) {
    const post = posts.value.find(p => p.id === id)
    if (post) post.status = post.status === "draft" ? "published" : "draft"
  }

  return {
    posts, filter, sortBy, editingId,
    filteredPosts, sortedPosts, stats,
    addPost, deletePost, updatePost, toggleStatus
  }
}

## 設問

各設問に答えてください。

### Q1. `(A)` の `filteredPosts` は何をしていますか？また、`filter` が `"all"` のとき何を返しますか？

### Q2. `(B)` で `[...filteredPosts.value]` としてコピーを作っている理由は何ですか？

### Q3. `(D)` の `[...new Set(posts.value.flatMap(p => p.tags))]` は何を生成しますか？初期データでの結果を具体的に答えてください。

### Q4. `(E)` の `addPost` で `Date.now()` をIDに使っています。この方法のメリットとデメリットを1つずつ挙げてください。

### Q5. `(F)` の `updatePost` で `Object.assign(post, changes)` を使っています。Vue のリアクティビティの観点で、この書き方は正しく動作しますか？理由も説明してください。

### Q6. `(G)` の `provide('postStore', store)` で渡される `store` はどのようなオブジェクトですか？`reactive` でラップする必要はありますか？

### Q7. `(H)` の `watch` は `store.posts.value.length` を監視しています。タイトルを編集したとき（`updatePost`呼び出し時）、このwatchは発火しますか？

### Q8. `(I)` で `inject` したstoreを使って `store.sortedPosts.value` と `.value` でアクセスしています。もし `store` 全体を `reactive()` でラップしていた場合、この `.value` は必要ですか？

### Q9. `deletePost` を呼んだあと、`filteredPosts` → `sortedPosts` → テンプレートの表示がどのような順序で更新されますか？

### Q10. このコードをPinia（第8章）で書き直す場合、`usePostStore` をPiniaストアにするには何を変更すべきですか？主な違いを3つ挙げてください。

## 解答例

<details>
<summary>Q1 の解答</summary>

`filteredPosts` は `posts` 配列を `filter`（`"all"`, `"draft"`, `"published"`）に応じて絞り込む **computed プロパティ** です。

`filter` が `"all"` のとき、フィルタ条件の `if` ブロックに入らないため、`posts.value`（全記事）がそのまま返されます。

**ポイント**: `computed` なので、`posts` または `filter` が変更されたときだけ再計算されます。

</details>

<details>
<summary>Q2 の解答</summary>

`Array.prototype.sort()` は元の配列を **破壊的に変更する（in-place sort）** ためです。

コピーを作らず `filteredPosts.value.sort(...)` とすると、`filteredPosts` が参照している元の配列が直接ソートされてしまい、`posts` のデータ順序にまで影響を与える可能性があります。スプレッド構文 `[...arr]` で浅いコピー（shallow copy）を作ることで、元のデータを保護しています。

</details>

<details>
<summary>Q3 の解答</summary>

全記事のタグ配列をフラット化し、重複を除去したユニークなタグ一覧を生成しています。

処理の流れ:
1. `posts.value.flatMap(p => p.tags)` → `["Vue", "入門", "Vue", "Composition API", "CSS", "TypeScript", "入門", "Vue", "Pinia"]`
2. `new Set(...)` → 重複を除去: `{"Vue", "入門", "Composition API", "CSS", "TypeScript", "Pinia"}`
3. `[...Set]` → 配列に変換: `["Vue", "入門", "Composition API", "CSS", "TypeScript", "Pinia"]`

</details>

<details>
<summary>Q4 の解答</summary>

**メリット**: ユニークなIDを簡単に生成でき、外部ライブラリが不要です。タイムスタンプなので生成順序も自然に保たれます。

**デメリット**: 同一ミリ秒以内に複数回呼ばれると重複する可能性があります。また、IDから生成日時が推測できてしまうため、セキュリティが求められる場面には不向きです。本番ではUUID（`crypto.randomUUID()`）やサーバー側でのID採番が推奨されます。

</details>

<details>
<summary>Q5 の解答</summary>

**正しく動作します**。

Vue 3のリアクティビティシステムは、`ref` で管理されている配列内のオブジェクトのプロパティ変更もProxy経由で追跡できます。`Object.assign(post, changes)` は既存のプロパティを上書きするため、Vueはこの変更を検知してUIを更新します。

ただし、`changes` に**新しいプロパティ**（元のオブジェクトに存在しないキー）が含まれる場合も、Vue 3のProxyベースのリアクティビティでは正しく検知されます（Vue 2の `Object.defineProperty` 方式では問題になるケースでした）。

</details>

<details>
<summary>Q6 の解答</summary>

`store` は `usePostStore()` が返すプレーンなオブジェクトで、中に `ref` と `computed` と関数が含まれています。

**`reactive` でラップする必要はありません**。各プロパティ（`posts`, `filter` など）がすでに `ref` / `computed` としてリアクティブであるため、オブジェクト自体を `reactive` にする必要はありません。ただし、`inject` 先で `store.posts.value` のように `.value` アクセスが必要になります。

もし `reactive` でラップすると `.value` が不要になりますが、`ref` のアンラップが自動で行われるため挙動が変わる可能性があり、一般的には推奨されません。

</details>

<details>
<summary>Q7 の解答</summary>

**発火しません**。

`watch(() => store.posts.value.length, ...)` は配列の**長さ**だけを監視しています。`updatePost` はタイトルなどのプロパティを変更するだけで、配列の要素数は変わりません。したがって `length` は変化せず、watchコールバックは実行されません。

</details>

<details>
<summary>Q8 の解答</summary>

**不要になります**。

`reactive()` でオブジェクトをラップすると、ネストされた `ref` が**自動的にアンラップ**されます。そのため `store.sortedPosts`（`.value` なし）で直接アクセスできるようになります。

```js
// reactive でラップした場合
const store = reactive(usePostStore())
// store.posts ← .value 不要
// store.sortedPosts ← .value 不要
```

</details>

<details>
<summary>Q9 の解答</summary>

更新の流れ:

1. `deletePost(id)` で `posts.value` に新しいフィルタ済み配列が代入される
2. `posts` の変更をVueのリアクティビティシステムが検知
3. `filteredPosts`（computed）が再計算される（`posts` に依存しているため）
4. `sortedPosts`（computed）が再計算される（`filteredPosts` に依存しているため）
5. `sortedPosts` を使っているテンプレート（PostList）が再レンダリングされる
6. `stats`（computed）も再計算される（`posts` に依存しているため）
7. `stats` を使っているStatsBarも再レンダリングされる

Vue は同期的にリアクティブな依存グラフを辿り、必要な部分だけを効率的に更新します。

</details>

<details>
<summary>Q10 の解答</summary>

Piniaストアに変換する場合の主な違い:

**1. ストアの定義方法**
```js
// 現在: 普通のcomposable関数
function usePostStore() { ... }

// Pinia: defineStore で定義
const usePostStore = defineStore('posts', () => {
  // 同じComposition APIスタイルだが、defineStoreでラップ
  // ...
  return { posts, filter, ... }
})
```

**2. コンポーネントでの利用方法**
```js
// 現在: provide/inject で手動共有
provide('postStore', store)
const store = inject('postStore')

// Pinia: どこからでも usePostStore() で直接取得
const store = usePostStore()
// provide/inject 不要、自動的にシングルトン
```

**3. DevToolsとの統合**
```js
// 現在: DevToolsで状態を確認できない
// Pinia: Vue DevToolsで状態の変更履歴をタイムトラベルデバッグ可能
// また、$reset() で初期状態に戻す機能も自動提供される
```

</details>
