## 課題の要件

### 必須機能

1. **型安全なイベントバス**: イベント名とペイロードの型が厳密に紐づいている
2. **リスナー登録/解除**: `on` / `off` / `once` をサポートする
3. **ミドルウェア**: イベント処理の前後に割り込む処理を追加できる
4. **イベント履歴**: 発火されたイベントの履歴を保持する
5. **統計情報**: イベントの発火回数やリスナー数を集計する

### 設計要件

- **イベントマップ型**: イベント名とペイロード型のマッピングをインターフェースで定義
- **ジェネリクス**: イベントバスをジェネリックにし、異なるイベントマップに対応
- **ユーティリティ型**: `Readonly`, `Partial`, `Pick`, `Record`, `Required` を活用
- **クラスのアクセス修飾子**: `private`, `protected`, `public` を適切に使い分ける
- **抽象クラスまたはインターフェース**: 拡張ポイントを定義する
- **条件付き型**: 少なくとも1箇所で使用する

```typescript
// ===== イベントバスインターフェース =====

interface IEventBus<EventMap extends Record<string, unknown>> {
  on<K extends keyof EventMap>(event: K, listener: EventListener<EventMap[K]>): () => void;
  off<K extends keyof EventMap>(event: K, listener: EventListener<EventMap[K]>): void;
  once<K extends keyof EventMap>(event: K, listener: EventListener<EventMap[K]>): () => void;
  emit<K extends keyof EventMap>(event: K, payload: EventMap[K]): void;
}

// ===== イベントバス実装 =====

class EventBus<EventMap extends Record<string, unknown>> implements IEventBus<EventMap> {
  private listeners: Map<keyof EventMap, Set<EventListener<any>>> = new Map();
  private history: EventRecord[] = [];
  private eventCounts: Map<keyof EventMap, number> = new Map();

  on<K extends keyof EventMap>(event: K, listener: EventListener<EventMap[K]>): () => void {
    if (!this.listeners.has(event)) {
      this.listeners.set(event, new Set());
    }
    this.listeners.get(event)!.add(listener);
    return () => this.off(event, listener);
  }

  off<K extends keyof EventMap>(event: K, listener: EventListener<EventMap[K]>): void {
    const set: Set<EventListener<any>> | undefined = this.listeners.get(event);
    if (set) {
      set.delete(listener);
    }
  }

  once<K extends keyof EventMap>(event: K, listener: EventListener<EventMap[K]>): () => void {
    const wrapper: EventListener<EventMap[K]> = (payload: EventMap[K]): void => {
      listener(payload);
      this.off(event, wrapper);
    };
    return this.on(event, wrapper);
  }

  emit<K extends keyof EventMap>(event: K, payload: EventMap[K]): void {
    // 履歴に記録
    this.history.push({
      event: event as string,
      payload,
      timestamp: new Date(),
    });

    // カウントを更新
    this.eventCounts.set(event, (this.eventCounts.get(event) ?? 0) + 1);

    // リスナーを呼び出す
    const set: Set<EventListener<any>> | undefined = this.listeners.get(event);
    if (set) {
      for (const listener of set) {
        listener(payload);
      }
    }
  }

  getHistory(): readonly EventRecord[] {
    return this.history;
  }

  getListenerCount(): number {
    let count: number = 0;
    for (const set of this.listeners.values()) {
      count += set.size;
    }
    return count;
  }

  getEmitCount(): number {
    return this.history.length;
  }

  getEventCounts(): ReadonlyMap<keyof EventMap, number> {
    return this.eventCounts;
  }
}

// ===== ミドルウェア付きイベントバス =====

type Middleware<EventMap extends Record<string, unknown>> = <K extends keyof EventMap>(
  event: K,
  payload: EventMap[K],
  next: () => void
) => void;

class MiddlewareEventBus<EventMap extends Record<string, unknown>> extends EventBus<EventMap> {
  private middlewares: Middleware<EventMap>[] = [];

  use(middleware: Middleware<EventMap>): void {
    this.middlewares.push(middleware);
  }

  emit<K extends keyof EventMap>(event: K, payload: EventMap[K]): void {
    if (this.middlewares.length === 0) {
      super.emit(event, payload);
      return;
    }

    let index: number = 0;
    const middlewares: Middleware<EventMap>[] = this.middlewares;
    const bus: EventBus<EventMap> = this;

    const next = (): void => {
      if (index < middlewares.length) {
        const mw: Middleware<EventMap> = middlewares[index++];
        mw(event, payload, next);
      } else {
        // すべてのミドルウェアを通過したら、実際にイベントを発火
        EventBus.prototype.emit.call(bus, event, payload);
      }
    };

    next();
  }
}
```

```typescript
// ミニフレームワーク設計（改良版）
// 学べる内容：抽象クラス、条件付き型、Mapped Type、デコレータパターン
// 実行方法：npx ts-node 12_event_system_v2.ts

// ===== 基盤型 =====

// イベントマップの制約型
type EventMap = Record<string, unknown>;

// リスナーの型
type Listener<T> = (payload: Readonly<T>) => void;

// イベントが特定のプロパティを持つかチェックする条件付き型
type HasProperty<T, K extends string> = T extends Record<K, unknown> ? true : false;

// イベントマップからペイロードが特定の型のイベントだけを抽出
type EventsWithPayload<M extends EventMap, PayloadType> = {
  [K in keyof M]: M[K] extends PayloadType ? K : never;
}[keyof M];

// ===== イベント履歴の型 =====

interface EventRecord<M extends EventMap, K extends keyof M = keyof M> {
  readonly id: number;
  readonly event: K;
  readonly payload: Readonly<M[K]>;
  readonly timestamp: Date;
}

// ===== イベントバスの抽象クラス =====

abstract class AbstractEventBus<M extends EventMap> {
  protected listeners: Map<keyof M, Set<Listener<any>>> = new Map();
  protected history: EventRecord<M>[] = [];
  private nextId: number = 1;

  // サブクラスで実装するフック
  protected abstract onBeforeEmit<K extends keyof M>(event: K, payload: M[K]): boolean;
  protected abstract onAfterEmit<K extends keyof M>(event: K, payload: M[K]): void;

  on<K extends keyof M>(event: K, listener: Listener<M[K]>): () => void {
    if (!this.listeners.has(event)) {
      this.listeners.set(event, new Set());
    }
    this.listeners.get(event)!.add(listener);
    return () => this.off(event, listener);
  }

  off<K extends keyof M>(event: K, listener: Listener<M[K]>): void {
    this.listeners.get(event)?.delete(listener);
  }

  once<K extends keyof M>(event: K, listener: Listener<M[K]>): () => void {
    const wrapper: Listener<M[K]> = (payload: Readonly<M[K]>): void => {
      listener(payload);
      this.off(event, wrapper);
    };
    return this.on(event, wrapper);
  }

  emit<K extends keyof M>(event: K, payload: M[K]): void {
    // フック: 発火前
    if (!this.onBeforeEmit(event, payload)) {
      return; // キャンセルされた
    }

    // 履歴記録
    this.history.push({
      id: this.nextId++,
      event,
      payload: Object.freeze({ ...payload as any }) as Readonly<M[K]>,
      timestamp: new Date(),
    });

    // リスナー呼び出し
    const set: Set<Listener<any>> | undefined = this.listeners.get(event);
    if (set) {
      for (const listener of Array.from(set)) {
        try {
          listener(payload);
        } catch (error) {
          this.handleError(event, error);
        }
      }
    }

    // フック: 発火後
    this.onAfterEmit(event, payload);
  }

  protected handleError<K extends keyof M>(event: K, error: unknown): void {
    console.error(`[Error] ${String(event)}:`, error instanceof Error ? error.message : error);
  }

  getHistory(): readonly EventRecord<M>[] {
    return this.history;
  }

  getListenerCount<K extends keyof M>(event?: K): number {
    if (event !== undefined) {
      return this.listeners.get(event)?.size ?? 0;
    }
    let count: number = 0;
    for (const set of this.listeners.values()) {
      count += set.size;
    }
    return count;
  }

  getStats(): Readonly<Record<string, number>> {
    const stats: Record<string, number> = {};
    for (const record of this.history) {
      const key: string = String(record.event);
      stats[key] = (stats[key] ?? 0) + 1;
    }
    return Object.freeze(stats);
  }
}

// ===== 具象イベントバス =====

class SimpleEventBus<M extends EventMap> extends AbstractEventBus<M> {
  protected onBeforeEmit<K extends keyof M>(_event: K, _payload: M[K]): boolean {
    return true; // 常に許可
  }

  protected onAfterEmit<K extends keyof M>(_event: K, _payload: M[K]): void {
    // 何もしない
  }
}

// ===== ロギング付きイベントバス =====

class LoggingEventBus<M extends EventMap> extends AbstractEventBus<M> {
  protected onBeforeEmit<K extends keyof M>(event: K, _payload: M[K]): boolean {
    console.log(`  [LOG] 発火: ${String(event)}`);
    return true;
  }

  protected onAfterEmit<K extends keyof M>(event: K, _payload: M[K]): void {
    const count: number = this.getListenerCount(event);
    console.log(`  [LOG] ${String(event)} → ${count}個のリスナーに配信完了`);
  }
}

for (const [event, count] of Object.entries(stats)) {

// ===== 型安全性のデモ =====
// 以下のコードはすべてコンパイルエラーになります（コメントを外すと確認できます）

// bus.emit("user:login", { user: "太郎" }); 
//   → エラー: email と role が不足
```

**初心者向けとの違い:**
- `AbstractEventBus` 抽象クラスで拡張ポイント（`onBeforeEmit`, `onAfterEmit`）を定義
- `SimpleEventBus` と `LoggingEventBus` で継承による振る舞いの変更
- `EventsWithPayload<M, PayloadType>` Mapped Type + 条件付き型でイベントのフィルタリング
- `Readonly<T>` でペイロードの不変性を保証
- `Object.freeze` と `Readonly` の組み合わせでランタイムとコンパイルタイム両方で保護
- エラーハンドリング（リスナーの例外がバス全体を止めない）
- `protected` アクセス修飾子によるサブクラスへのAPI公開制御
