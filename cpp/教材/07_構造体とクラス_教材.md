## 7.1 構造体（struct）

構造体（struct）は、複数のデータをひとまとめにする仕組みです。

### なぜ構造体が必要か

```
学生の情報を管理したい場合:

【構造体なし】                  【構造体あり】
string name1, name2, name3;    Student s1, s2, s3;
int age1, age2, age3;          ← すべての情報が1つに
double gpa1, gpa2, gpa3;        まとまっている
→ 変数がバラバラで管理が大変   → スッキリ管理できる
```

### 構造体の定義と使用

```cpp
#include <iostream>
#include <string>
using namespace std;

// 構造体の定義
struct Student {
    string name;
    int age;
    double gpa;
};

int main() {
    // 構造体変数の作成と初期化
    Student s1;
    s1.name = "田中太郎";
    s1.age = 20;
    s1.gpa = 3.5;

    // 集約初期化（aggregate initialization）
    Student s2 = {"佐藤花子", 21, 3.8};

    // C++20 指示付き初期化（designated initializers）
    // Student s3 = {.name = "鈴木", .age = 19, .gpa = 3.2};

    // メンバへのアクセスは . （ドット演算子）
    cout << s1.name << "（" << s1.age << "歳）GPA: " << s1.gpa << endl;
    cout << s2.name << "（" << s2.age << "歳）GPA: " << s2.gpa << endl;

    return 0;
}
```

### 構造体にメンバ関数を追加

C++の構造体はC言語と違い、関数を持つことができます。

```cpp
#include <iostream>
#include <string>
using namespace std;

struct Rectangle {
    double width;
    double height;

    // メンバ関数
    double area() const {
        return width * height;
    }

    double perimeter() const {
        return 2 * (width + height);
    }

    void display() const {
        cout << "幅: " << width << ", 高さ: " << height
             << ", 面積: " << area()
             << ", 周囲: " << perimeter() << endl;
    }
};

int main() {
    Rectangle rect = {5.0, 3.0};
    rect.display();
    // 幅: 5, 高さ: 3, 面積: 15, 周囲: 16

    return 0;
}
```

## 7.2 クラス（class）の定義

クラス（class）は構造体をさらに発展させたもので、オブジェクト指向プログラミング（Object-Oriented Programming, OOP）の中核です。

### structとclassの違い

```
┌───────────────┬──────────────────────────┬──────────────────────────┐
│               │  struct                  │  class                   │
├───────────────┼──────────────────────────┼──────────────────────────┤
│ デフォルトの   │  public                  │  private                 │
│ アクセス指定   │                          │                          │
├───────────────┼──────────────────────────┼──────────────────────────┤
│ 一般的な用途   │  単純なデータの集まり     │  データ + ロジックの       │
│               │  （POD型など）            │  カプセル化               │
└───────────────┴──────────────────────────┴──────────────────────────┘
```

### 最初のクラス

```cpp
#include <iostream>
#include <string>
using namespace std;

class Dog {
private:  // 外部からアクセス不可
    string name;
    int age;

public:   // 外部からアクセス可能
    // コンストラクタ（constructor）: オブジェクト生成時に呼ばれる
    Dog(const string& n, int a) : name(n), age(a) {
        cout << name << " が生まれました！" << endl;
    }

    // メンバ関数
    void bark() const {
        cout << name << "「ワンワン！」" << endl;
    }

    void introduce() const {
        cout << "名前: " << name << ", 年齢: " << age << "歳" << endl;
    }
};

int main() {
    Dog dog1("ポチ", 3);
    Dog dog2("ハチ", 5);

    dog1.bark();        // ポチ「ワンワン！」
    dog2.introduce();   // 名前: ハチ, 年齢: 5歳

    return 0;
}
```

## 7.3 アクセス指定子（access specifier）

クラスのメンバに対するアクセス制御を行います。

```
┌──────────────┬──────────────┬──────────────┬──────────────┐
│ アクセス指定  │ クラス内部   │ 派生クラス    │ クラス外部    │
├──────────────┼──────────────┼──────────────┼──────────────┤
│ public       │    ○        │    ○         │    ○         │
│ protected    │    ○        │    ○         │    ×         │
│ private      │    ○        │    ×         │    ×         │
└──────────────┴──────────────┴──────────────┴──────────────┘
```

```cpp
#include <iostream>
#include <string>
using namespace std;

class BankAccount {
private:
    string owner;
    double balance;  // 外部から直接変更させない

public:
    BankAccount(const string& name, double initial)
        : owner(name), balance(initial) {}

    // ゲッター（getter）: 値を取得する
    string getOwner() const { return owner; }
    double getBalance() const { return balance; }

    // セッター的な操作: 検証ロジック付きで値を変更する
    bool deposit(double amount) {
        if (amount <= 0) {
            cout << "エラー: 入金額は正の値にしてください" << endl;
            return false;
        }
        balance += amount;
        cout << amount << "円を入金しました（残高: " << balance << "円）" << endl;
        return true;
    }

    bool withdraw(double amount) {
        if (amount <= 0) {
            cout << "エラー: 出金額は正の値にしてください" << endl;
            return false;
        }
        if (amount > balance) {
            cout << "エラー: 残高不足です" << endl;
            return false;
        }
        balance -= amount;
        cout << amount << "円を出金しました（残高: " << balance << "円）" << endl;
        return true;
    }
};

int main() {
    BankAccount account("田中太郎", 10000);

    cout << "口座名義: " << account.getOwner() << endl;
    cout << "残高: " << account.getBalance() << "円" << endl;

    account.deposit(5000);
    account.withdraw(3000);
    account.withdraw(20000);  // 残高不足

    // account.balance = 1000000;  // コンパイルエラー！privateなので直接変更できない

    return 0;
}
```

### カプセル化（encapsulation）の重要性

```
┌─────────────────────────────────────────────────────┐
│  カプセル化とは                                      │
├─────────────────────────────────────────────────────┤
│  データ（メンバ変数）を private にして直接触らせず、  │
│  public なメンバ関数を通じてのみ操作させること。     │
│                                                     │
│  メリット:                                           │
│  ・不正な値の設定を防げる（バリデーション）            │
│  ・内部実装を変更しても外部に影響しない               │
│  ・バグの原因を特定しやすくなる                       │
└─────────────────────────────────────────────────────┘
```

## 7.4 コンストラクタとデストラクタ

### コンストラクタ（constructor）

オブジェクト生成時に自動で呼ばれる特別な関数です。

```cpp
#include <iostream>
#include <string>
using namespace std;

class Product {
private:
    string name;
    int price;
    int stock;

public:
    // デフォルトコンストラクタ
    Product() : name("未設定"), price(0), stock(0) {
        cout << "デフォルトコンストラクタ" << endl;
    }

    // 引数付きコンストラクタ
    Product(const string& n, int p, int s) : name(n), price(p), stock(s) {
        cout << name << " のコンストラクタ" << endl;
    }

    // コピーコンストラクタ
    Product(const Product& other)
        : name(other.name), price(other.price), stock(other.stock) {
        cout << name << " のコピーコンストラクタ" << endl;
    }

    void display() const {
        cout << name << " - " << price << "円（在庫: " << stock << "）" << endl;
    }
};

int main() {
    Product p1;                          // デフォルトコンストラクタ
    Product p2("ノートPC", 98000, 50);   // 引数付きコンストラクタ
    Product p3 = p2;                     // コピーコンストラクタ

    p1.display();  // 未設定 - 0円（在庫: 0）
    p2.display();  // ノートPC - 98000円（在庫: 50）
    p3.display();  // ノートPC - 98000円（在庫: 50）

    return 0;
}
```

### メンバ初期化リスト（member initializer list）

コンストラクタではメンバ初期化リスト `: メンバ(値)` を使うのがベストプラクティスです。

```cpp
class Point {
    double x, y;
public:
    // 推奨: メンバ初期化リストを使う
    Point(double x, double y) : x(x), y(y) {}

    // 非推奨: 代入で初期化（一度デフォルト構築してから代入するので非効率）
    // Point(double x, double y) {
    //     this->x = x;
    //     this->y = y;
    // }
};
```

### デストラクタ（destructor）

オブジェクトが破棄されるときに自動で呼ばれる関数です。

```cpp
#include <iostream>
#include <string>
using namespace std;

class Logger {
private:
    string filename;

public:
    Logger(const string& fname) : filename(fname) {
        cout << "[" << filename << "] ログファイルを開きました" << endl;
    }

    ~Logger() {  // デストラクタ: ~ + クラス名
        cout << "[" << filename << "] ログファイルを閉じました" << endl;
    }

    void log(const string& message) const {
        cout << "[" << filename << "] " << message << endl;
    }
};

int main() {
    cout << "=== プログラム開始 ===" << endl;

    {
        Logger logger("app.log");
        logger.log("処理を開始します");
        logger.log("データを処理中...");
        logger.log("処理完了");
    }  // スコープを抜けるとデストラクタが呼ばれる

    cout << "=== プログラム終了 ===" << endl;

    return 0;
}
```

出力:
```
=== プログラム開始 ===
[app.log] ログファイルを開きました
[app.log] 処理を開始します
[app.log] データを処理中...
[app.log] 処理完了
[app.log] ログファイルを閉じました
=== プログラム終了 ===
```

### コンストラクタ/デストラクタの呼び出し順序

```
オブジェクト生成: コンストラクタが呼ばれる
     │
     ▼
オブジェクトを使用（メンバ関数呼び出しなど）
     │
     ▼
スコープを抜ける: デストラクタが呼ばれる

※ 複数のオブジェクトがある場合:
   生成: A → B → C（宣言順）
   破棄: C → B → A（逆順）
```

## 7.5 thisポインタ

`this` は、メンバ関数内で「自分自身のオブジェクト」を指すポインタです。

```cpp
#include <iostream>
#include <string>
using namespace std;

class Counter {
private:
    int count;
    string name;

public:
    Counter(const string& name, int count = 0) : name(name), count(count) {}

    // メソッドチェーンを実現するために *this を返す
    Counter& increment() {
        count++;
        return *this;
    }

    Counter& add(int n) {
        count += n;
        return *this;
    }

    void display() const {
        cout << name << ": " << count << endl;
    }
};

int main() {
    Counter c("カウンタA");

    // メソッドチェーン（method chaining）
    c.increment().increment().add(5).increment();
    c.display();  // カウンタA: 8

    return 0;
}
```

```
メソッドチェーンの流れ:

c.increment()     → count=1, *thisを返す
 .increment()     → count=2, *thisを返す
 .add(5)          → count=7, *thisを返す
 .increment()     → count=8, *thisを返す
```

## 7.6 static メンバ

`static` メンバは、個々のオブジェクトではなくクラス全体で共有されるメンバです。

```cpp
#include <iostream>
#include <string>
using namespace std;

class Employee {
private:
    string name;
    int id;

    static int nextId;       // 静的メンバ変数（全オブジェクトで共有）
    static int totalCount;

public:
    Employee(const string& n) : name(n), id(nextId++) {
        totalCount++;
    }

    ~Employee() {
        totalCount--;
    }

    void display() const {
        cout << "ID: " << id << ", 名前: " << name << endl;
    }

    // 静的メンバ関数（オブジェクトなしで呼べる）
    static int getTotalCount() {
        return totalCount;
    }
};

// 静的メンバ変数の定義（クラス外で行う）
int Employee::nextId = 1001;
int Employee::totalCount = 0;

int main() {
    cout << "社員数: " << Employee::getTotalCount() << endl;  // 0

    Employee e1("田中");
    Employee e2("佐藤");
    Employee e3("鈴木");

    e1.display();  // ID: 1001, 名前: 田中
    e2.display();  // ID: 1002, 名前: 佐藤
    e3.display();  // ID: 1003, 名前: 鈴木

    cout << "社員数: " << Employee::getTotalCount() << endl;  // 3

    return 0;
}
```

### static メンバの図解

```
┌─── Employee クラス ─────────────────────────────┐
│                                                 │
│  static nextId = 1001  ← クラス全体で1つ        │
│  static totalCount = 3 ← クラス全体で1つ        │
│                                                 │
│  ┌── e1 ──┐  ┌── e2 ──┐  ┌── e3 ──┐           │
│  │name:田中│  │name:佐藤│  │name:鈴木│           │
│  │id: 1001 │  │id: 1002 │  │id: 1003 │           │
│  └────────┘  └────────┘  └────────┘           │
│   各オブジェクトは独自のメンバ変数を持つ          │
└─────────────────────────────────────────────────┘
```

## 7.7 const メンバ関数

`const` メンバ関数は、オブジェクトの状態を変更しないことを保証する関数です。

```cpp
#include <iostream>
#include <string>
using namespace std;

class Temperature {
private:
    double celsius;

public:
    Temperature(double c) : celsius(c) {}

    // const メンバ関数: メンバ変数を変更しない
    double getCelsius() const { return celsius; }
    double getFahrenheit() const { return celsius * 9.0 / 5.0 + 32.0; }
    double getKelvin() const { return celsius + 273.15; }

    void display() const {
        cout << getCelsius() << "°C = "
             << getFahrenheit() << "°F = "
             << getKelvin() << "K" << endl;
    }

    // 非constメンバ関数: メンバ変数を変更する
    void setCelsius(double c) { celsius = c; }
};

// const オブジェクトにはconstメンバ関数しか呼べない
void printTemp(const Temperature& temp) {
    temp.display();       // OK: constメンバ関数
    // temp.setCelsius(0); // NG: 非constメンバ関数は呼べない
}

int main() {
    Temperature t(100.0);
    t.display();       // 100°C = 212°F = 373.15K
    printTemp(t);

    return 0;
}
```

### よくある間違い

```cpp
class MyClass {
    int value;
public:
    // NG: constが付いていないため、const参照経由で呼べない
    int getValue() { return value; }

    // OK: 状態を変更しないなら必ずconstを付ける
    int getValue() const { return value; }
};
```

### ポイントまとめ
- オブジェクトの状態を変更しないメンバ関数には必ず `const` を付ける
- `const` オブジェクトや `const` 参照からは `const` メンバ関数しか呼べない
- ゲッターには必ず `const` を付けるのが鉄則

## 7.8 演算子オーバーロード（operator overloading）

C++では、クラスに対して `+` や `<<` などの演算子の動作を定義できます。

```cpp
#include <iostream>
using namespace std;

class Vector2D {
private:
    double x, y;

public:
    Vector2D(double x = 0, double y = 0) : x(x), y(y) {}

    // + 演算子のオーバーロード
    Vector2D operator+(const Vector2D& other) const {
        return Vector2D(x + other.x, y + other.y);
    }

    // - 演算子
    Vector2D operator-(const Vector2D& other) const {
        return Vector2D(x - other.x, y - other.y);
    }

    // * 演算子（スカラー倍）
    Vector2D operator*(double scalar) const {
        return Vector2D(x * scalar, y * scalar);
    }

    // == 演算子
    bool operator==(const Vector2D& other) const {
        return x == other.x && y == other.y;
    }

    // << 演算子（フレンド関数として定義）
    friend ostream& operator<<(ostream& os, const Vector2D& v) {
        os << "(" << v.x << ", " << v.y << ")";
        return os;
    }

    double getX() const { return x; }
    double getY() const { return y; }
};

int main() {
    Vector2D a(3.0, 4.0);
    Vector2D b(1.0, 2.0);

    Vector2D c = a + b;
    Vector2D d = a - b;
    Vector2D e = a * 2.0;

    cout << "a = " << a << endl;       // (3, 4)
    cout << "b = " << b << endl;       // (1, 2)
    cout << "a + b = " << c << endl;   // (4, 6)
    cout << "a - b = " << d << endl;   // (2, 2)
    cout << "a * 2 = " << e << endl;   // (6, 8)

    if (a == a) {
        cout << "a == a は true" << endl;
    }

    return 0;
}
```

### friend 関数

`friend` を使うと、クラス外の関数からプライベートメンバにアクセスできます。`operator<<` を定義するときによく使います。

```
┌─────────────────────────────────────────────────┐
│  friend の使いどころ                             │
├─────────────────────────────────────────────────┤
│  ・ operator<< （ストリーム出力）                 │
│  ・ operator>> （ストリーム入力）                 │
│  ・ 左辺がクラス型でない二項演算子               │
│                                                 │
│  注意: friend の多用はカプセル化を壊すので注意   │
└─────────────────────────────────────────────────┘
```

## 7.9 実践例：日付クラス

これまでの知識を総合した実践的なクラス設計の例です。

```cpp
#include <iostream>
#include <string>
#include <iomanip>
using namespace std;

class Date {
private:
    int year, month, day;

    // 月の日数を返すヘルパー関数
    static int daysInMonth(int y, int m) {
        static const int days[] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
        if (m == 2 && isLeapYear(y)) return 29;
        return days[m];
    }

    // うるう年の判定
    static bool isLeapYear(int y) {
        return (y % 4 == 0 && y % 100 != 0) || (y % 400 == 0);
    }

    // 有効な日付かチェック
    bool isValid() const {
        if (month < 1 || month > 12) return false;
        if (day < 1 || day > daysInMonth(year, month)) return false;
        return true;
    }

public:
    Date(int y, int m, int d) : year(y), month(m), day(d) {
        if (!isValid()) {
            cout << "警告: 無効な日付です。1970/1/1に設定します。" << endl;
            year = 1970; month = 1; day = 1;
        }
    }

    // ゲッター
    int getYear() const { return year; }
    int getMonth() const { return month; }
    int getDay() const { return day; }

    // 曜日を返す（ツェラーの公式）
    string getDayOfWeek() const {
        static const string days[] = {"土", "日", "月", "火", "水", "木", "金"};
        int y = year, m = month;
        if (m <= 2) { m += 12; y--; }
        int h = (day + 26 * (m + 1) / 10 + y + y / 4 - y / 100 + y / 400) % 7;
        return days[h];
    }

    // 比較演算子
    bool operator==(const Date& other) const {
        return year == other.year && month == other.month && day == other.day;
    }

    bool operator<(const Date& other) const {
        if (year != other.year) return year < other.year;
        if (month != other.month) return month < other.month;
        return day < other.day;
    }

    // ストリーム出力
    friend ostream& operator<<(ostream& os, const Date& d) {
        os << d.year << "/"
           << setfill('0') << setw(2) << d.month << "/"
           << setfill('0') << setw(2) << d.day
           << "（" << d.getDayOfWeek() << "）";
        return os;
    }
};

int main() {
    Date today(2026, 3, 28);
    Date newYear(2026, 1, 1);
    Date invalid(2026, 13, 40);  // 無効な日付

    cout << "今日: " << today << endl;
    cout << "元旦: " << newYear << endl;
    cout << "無効: " << invalid << endl;

    if (newYear < today) {
        cout << newYear << " は " << today << " より前です" << endl;
    }

    return 0;
}
```

## 7.10 よくある間違いまとめ

```cpp
// 1. メンバ変数をpublicにしてしまう
class Bad {
public:
    int data;  // 外部から自由に変更できてしまう
};

// 2. constを付け忘れる
class Bad2 {
    int value;
public:
    int getValue() { return value; }  // constが付いていない
};

// 3. コンストラクタで代入を使う（初期化リストを使うべき）
class Bad3 {
    string name;
public:
    Bad3(const string& n) { name = n; }  // 初期化リスト未使用
};

// 4. デストラクタでリソース解放を忘れる
class Bad4 {
    int* data;
public:
    Bad4() : data(new int[100]) {}
    // ~Bad4() がないのでメモリリーク！
};
```
