## 課題の要件

1. ログインページ（`/login`）を作成する
2. ダッシュボード（`/dashboard`）は認証済みユーザーのみアクセス可能にする
3. `middleware.ts` で `/dashboard` へのアクセスを認証チェックする
4. ログイン処理を Server Action で実装する（Cookie にセッショントークンを保存）
5. ログアウト処理も Server Action で実装する
6. ユーザー情報はハードコードでOK（データベース不要）

<details>
<summary>ステップ1：認証フローを理解する</summary>

```
ログインフロー:
1. ユーザーがメール・パスワードを入力して送信
2. Server Action でメール・パスワードを検証
3. 正しければセッショントークンを生成し Cookie に保存
4. /dashboard にリダイレクト

認証チェックフロー:
1. ユーザーが /dashboard にアクセス
2. middleware.ts が Cookie のセッショントークンを確認
3. トークンがなければ /login にリダイレクト
4. トークンがあればアクセスを許可

ログアウトフロー:
1. ユーザーがログアウトボタンをクリック
2. Server Action で Cookie を削除
3. /login にリダイレクト
```

### lib/auth.ts

```tsx
// 簡易的なユーザーデータ（実際のアプリではデータベースを使用）
type User = {
  id: string
  email: string
  password: string
  name: string
  role: string
}

const users: User[] = [
  { id: '1', email: 'admin@example.com', password: 'admin123', name: '田中太郎', role: '管理者' },
  { id: '2', email: 'user@example.com', password: 'user123', name: '鈴木花子', role: '一般ユーザー' },
]

export function authenticateUser(
  email: string,
  password: string
): Omit<User, 'password'> | null {
  const user = users.find(function (u) {
    return u.email === email && u.password === password
  })
  if (!user) return null
  // パスワードを除外して返す
  return { id: user.id, email: user.email, name: user.name, role: user.role }
}

export function getUserById(id: string): Omit<User, 'password'> | null {
  const user = users.find(function (u) {
    return u.id === id
  })
  if (!user) return null
  return { id: user.id, email: user.email, name: user.name, role: user.role }
}

// 簡易的なトークン生成（実際のアプリではJWTやセキュアなトークンを使用）
export function generateToken(userId: string): string {
  return Buffer.from(`${userId}:${Date.now()}`).toString('base64')
}

export function parseToken(token: string): string | null {
  try {
    const decoded = Buffer.from(token, 'base64').toString()
    const userId = decoded.split(':')[0]
    return userId
  } catch {
    return null
  }
}
```

### middleware.ts

```tsx
import { NextResponse } from 'next/server'
import type { NextRequest } from 'next/server'

export function middleware(request: NextRequest) {
  const sessionToken = request.cookies.get('session-token')?.value

  // /dashboard 以下のページは認証が必要
  if (request.nextUrl.pathname.startsWith('/dashboard')) {
    if (!sessionToken) {
      // 未認証の場合はログインページにリダイレクト
      const loginUrl = new URL('/login', request.url)
      // リダイレクト先を保持（ログイン後に戻れるように）
      loginUrl.searchParams.set('redirect', request.nextUrl.pathname)
      return NextResponse.redirect(loginUrl)
    }
  }

  // ログイン済みの場合、/login にアクセスしたら /dashboard にリダイレクト
  if (request.nextUrl.pathname === '/login') {
    if (sessionToken) {
      return NextResponse.redirect(new URL('/dashboard', request.url))
    }
  }

  return NextResponse.next()
}

export const config = {
  matcher: ['/dashboard/:path*', '/login'],
}
```

### app/login/actions.ts

```tsx
'use server'

import { cookies } from 'next/headers'
import { redirect } from 'next/navigation'
import { authenticateUser, generateToken } from '@/lib/auth'

export async function loginAction(
  prevState: { error?: string } | null,
  formData: FormData
) {
  const email = formData.get('email') as string
  const password = formData.get('password') as string
  const redirectTo = formData.get('redirect') as string

  // バリデーション
  if (!email || !password) {
    return { error: 'メールアドレスとパスワードを入力してください' }
  }

  // ユーザー認証
  const user = authenticateUser(email, password)

  if (!user) {
    return { error: 'メールアドレスまたはパスワードが間違っています' }
  }

  // セッショントークンを Cookie に保存
  const token = generateToken(user.id)
  const cookieStore = await cookies()
  cookieStore.set('session-token', token, {
    httpOnly: true,     // JavaScriptからアクセス不可
    secure: process.env.NODE_ENV === 'production',
    sameSite: 'lax',
    maxAge: 60 * 60 * 24, // 24時間
    path: '/',
  })

  // リダイレクト
  redirect(redirectTo || '/dashboard')
}

export async function logoutAction() {
  const cookieStore = await cookies()
  cookieStore.delete('session-token')
  redirect('/login')
}
```

import { cookies } from 'next/headers'
import { parseToken, getUserById } from './auth'

export type SessionUser = {
  id: string
  email: string
  name: string
  role: string
}

// サーバーコンポーネントからセッション情報を取得するユーティリティ
export async function getSession(): Promise<SessionUser | null> {

  const token = cookieStore.get('session-token')?.value

  const userId = parseToken(token)

  return getUserById(userId)
