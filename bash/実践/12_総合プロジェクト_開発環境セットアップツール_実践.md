# 実践課題12：総合プロジェクト ─ 開発環境セットアップツール ★5

> **難易度**: ★★★★★（総合）
> **前提知識**: 第1章〜第8章（全章）
> **課題の種類**: ミニプロジェクト
> **学習目標**: 全章の知識を統合し、実務で使える開発環境セットアップツールを設計・実装する

---

## 課題の説明

新しいプロジェクトを始める際に、開発環境を自動でセットアップするツールを作成してください。
プロジェクトのテンプレート生成、必要なツールのチェック、設定ファイルの配置、Gitリポジトリの初期化をワンコマンドで行えるようにします。

## 課題の要件

### 必須要件

1. **サブコマンド**: `init`, `check`, `list` の3コマンド
2. **プロジェクトテンプレート**: 最低3種類（node, python, bash）
3. **前提条件チェック**: 必要なコマンドのバージョン確認
4. **テンプレート生成**: ディレクトリ構造とファイルを自動作成
5. **Git初期化**: `--git` オプションでリポジトリ初期化と初期コミット
6. **エラーハンドリング**: 堅牢なスクリプト（`set -euo pipefail`, `trap`）
7. **ドライラン**: `-n` オプションで実行せずに確認

### 各テンプレートの生成内容

**Node.js テンプレート:**

# 開発環境セットアップツール
# 学べる内容：全章の総合（変数、条件分岐、ループ、関数、配列、リダイレクト、実践テクニック）
# 実行方法：chmod +x devsetup.sh && ./devsetup.sh init <プロジェクト名> --type <タイプ> [--git]

# ============================================================
# 設定
# ============================================================

readonly VERSION="1.0"

DRY_RUN=false
INIT_GIT=false
PROJECT_TYPE="node"

# ユーティリティ
# ============================================================
log_step() {
    local step="$1"; shift
    echo ""
    echo "[STEP ${step}] $*"
}

log_ok()    { printf "  ✓ %s\n" "$*"; }
log_fail()  { printf "  ✗ %s\n" "$*"; }
log_info()  { printf "  %s\n" "$*"; }
log_error() { echo "[ERROR] $*" >&2; }

run() {
    if $DRY_RUN; then
        echo "  [DRY] $*"
        return 0
    fi
    "$@"
}

# 前提条件チェック
# ============================================================
check_tool() {
    local tool="$1"
    local required="${2:-false}"

    if command -v "$tool" > /dev/null 2>&1; then
        local version
        version=$("$tool" --version 2>&1 | head -1 | grep -oE '[0-9]+\.[0-9]+[.0-9]*' | head -1)
        printf "  ✓ %-10s %s\n" "$tool" "${version:-（バージョン不明）}"
        return 0
    else
        if [ "$required" = "true" ]; then
            printf "  ✗ %-10s 未インストール（必須）\n" "$tool"
        else
            printf "  ✗ %-10s 未インストール（オプショナル）\n" "$tool"
        fi
        return 1
    fi
}

check_prerequisites() {
    local project_type="$1"
    local errors=0

    log_step "1/5" "前提条件チェック"

    # 共通ツール
    check_tool "git" "true" || errors=$((errors + 1))

    # タイプ別ツール
    case "$project_type" in
        node)
            check_tool "node" "true" || errors=$((errors + 1))
            check_tool "npm" "true" || errors=$((errors + 1))
            ;;
        python)
            check_tool "python3" "true" || errors=$((errors + 1))
            check_tool "pip3" "false" || true
            ;;
        bash)
            check_tool "bash" "true" || errors=$((errors + 1))
            check_tool "shellcheck" "false" || true
            ;;
        html)
            # 特別な必須ツールなし
            ;;
    esac

    check_tool "docker" "false" || true

    return "$errors"
}

create_gitignore() {
    local project_dir="$1"
    local project_type="$2"

    cat > "${project_dir}/.gitignore" <<EOF
# OS
.DS_Store
Thumbs.db

# Editor
*.swp
*.swo
*~
.vscode/
.idea/
EOF

    case "$project_type" in
        node)
            cat >> "${project_dir}/.gitignore" <<EOF

# Node.js
node_modules/
dist/
*.log
.env
EOF
            ;;
        python)
            cat >> "${project_dir}/.gitignore" <<EOF

# Python
__pycache__/
*.pyc
*.pyo
.env
venv/
*.egg-info/
dist/
build/
EOF
            ;;
        bash)
            cat >> "${project_dir}/.gitignore" <<EOF

# Bash
*.log
tmp/
EOF
            ;;
    esac
}

indent_size = 2
end_of_line = lf
charset = utf-8
trim_trailing_whitespace = true
insert_final_newline = true

[*.md]
trim_trailing_whitespace = false

[*.py]

[*.sh]
indent_size = 4

## 概要

generate_python_template() {
    local project_dir="$1"
    local project_name
    project_name=$(basename "$project_dir")

    # pyproject.toml
    cat > "${project_dir}/pyproject.toml" <<EOF
[project]
name = "${project_name}"
version = "1.0.0"
description = ""
requires-python = ">=3.8"

[project.scripts]
${project_name} = "src.main:main"
EOF

    # requirements.txt
    touch "${project_dir}/requirements.txt"

    # src/
    run mkdir -p "${project_dir}/src"
    cat > "${project_dir}/src/__init__.py" <<'EOF'
EOF

    cat > "${project_dir}/src/main.py" <<'EOF'
"""メインモジュール"""


def main():
    """エントリポイント"""
    print("Hello, World!")


if __name__ == "__main__":
    main()
EOF

    # tests/
    run mkdir -p "${project_dir}/tests"
    cat > "${project_dir}/tests/__init__.py" <<'EOF'
EOF

    log_ok "pyproject.toml"
    log_ok "requirements.txt"
    log_ok "src/__init__.py"
    log_ok "src/main.py"
    log_ok "tests/"
}

generate_bash_template() {
    local project_dir="$1"

    # bin/main.sh
    run mkdir -p "${project_dir}/bin"
    cat > "${project_dir}/bin/main.sh" <<'MAIN_EOF'
#!/usr/bin/env bash
set -euo pipefail

readonly SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
readonly PROJECT_DIR="$(dirname "$SCRIPT_DIR")"

# 共通ライブラリの読み込み
source "${PROJECT_DIR}/lib/common.sh"

main() {
    log_info "Hello, World!"
}

main "$@"
MAIN_EOF
    chmod +x "${project_dir}/bin/main.sh"

    # lib/common.sh
    run mkdir -p "${project_dir}/lib"
    cat > "${project_dir}/lib/common.sh" <<'LIB_EOF'
#!/usr/bin/env bash
# 共通関数ライブラリ

log_info()  { echo "[INFO] $*" >&2; }
log_warn()  { echo "[WARN] $*" >&2; }
log_error() { echo "[ERROR] $*" >&2; }
LIB_EOF

    # test/
    run mkdir -p "${project_dir}/test"

    log_ok "bin/main.sh"
    log_ok "lib/common.sh"
    log_ok "test/"
}

    <meta charset="UTF-8">

# --- init コマンド ---
cmd_init() {
    local project_name=""

    # オプション解析
    while [ $# -gt 0 ]; do
        case "$1" in
            --type)    PROJECT_TYPE="$2"; shift 2 ;;
            --git)     INIT_GIT=true; shift ;;
            -n)        DRY_RUN=true; shift ;;
            --dry-run) DRY_RUN=true; shift ;;
            -*)        log_error "不明なオプション: $1"; exit 1 ;;
            *)         project_name="$1"; shift ;;
        esac
    done

    if [ -z "$project_name" ]; then
        log_error "プロジェクト名を指定してください"
        echo "使い方: $TOOL_NAME init <プロジェクト名> --type <node|python|bash|html> [--git]"
        exit 1
    fi

    local project_dir
    project_dir="$(pwd)/${project_name}"

    # ヘッダー
    echo ""
    echo "===================================="
    echo "  開発環境セットアップ v${VERSION}"
    echo "===================================="
    if $DRY_RUN; then
        echo "  (ドライランモード)"
    fi

    # STEP 1: 前提条件チェック
    local check_failed=false
    check_prerequisites "$PROJECT_TYPE" || check_failed=true

    if $check_failed && ! $DRY_RUN; then
        log_error "必須ツールが不足しています。インストール後に再実行してください。"
        exit 1
    fi

    # STEP 2: ディレクトリ作成
    log_step "2/5" "プロジェクトディレクトリ作成"
    log_info "ディレクトリ: $project_dir"

    if [ -d "$project_dir" ]; then
        log_error "ディレクトリが既に存在します: $project_dir"
        exit 1
    fi

    run mkdir -p "$project_dir"
    log_info "作成しました"

    # STEP 3: テンプレート生成
    log_step "3/5" "テンプレート生成"

    if ! $DRY_RUN; then
        # 共通ファイル
        create_gitignore "$project_dir" "$PROJECT_TYPE"
        log_ok ".gitignore"

        create_editorconfig "$project_dir"
        log_ok ".editorconfig"

        create_readme "$project_dir" "$project_name" "$PROJECT_TYPE"
        log_ok "README.md"

        # タイプ別ファイル
        case "$PROJECT_TYPE" in
            node)   generate_node_template "$project_dir" ;;
            python) generate_python_template "$project_dir" ;;
            bash)   generate_bash_template "$project_dir" ;;
            html)   generate_html_template "$project_dir" ;;
            *)
                log_error "未対応のテンプレートタイプ: $PROJECT_TYPE"
                exit 1
                ;;
        esac
    else
        log_info "[DRY] テンプレートファイルを生成"
    fi

    # STEP 4: Git初期化
    log_step "4/5" "Git初期化"
    if $INIT_GIT; then
        if ! $DRY_RUN; then
            (
                cd "$project_dir"
                git init --quiet
                log_ok "git init"
                git add -A
                git commit --quiet -m "Initial commit by ${TOOL_NAME}"
                log_ok "初期コミット作成"
            )
        else
            log_info "[DRY] git init && git commit"
        fi
    else
        log_info "スキップ（--git オプションなし）"
    fi

    # STEP 5: サマリー
    log_step "5/5" "完了サマリー"
    local file_count=0
    if [ -d "$project_dir" ] && ! $DRY_RUN; then
        file_count=$(find "$project_dir" -type f -not -path '*/.git/*' | wc -l | tr -d '[:space:]')
    fi

    log_info "プロジェクト: $project_name"
    log_info "タイプ:       $PROJECT_TYPE"
    log_info "パス:         $project_dir"
    log_info "ファイル数:   $file_count"

    echo ""
    echo "次のステップ:"
    echo "  cd ${project_name}"
    case "$PROJECT_TYPE" in
        node)
            echo "  npm install"
            echo "  npm start"
            ;;
        python)
            echo "  python3 -m venv venv"
            echo "  source venv/bin/activate"
            echo "  python src/main.py"
            ;;
        bash)
            echo "  ./bin/main.sh"
            ;;
        html)
            echo "  open index.html  # macOS"
            echo "  xdg-open index.html  # Linux"
            ;;
    esac

    echo ""
    echo "===================================="
    echo "  セットアップ完了！"
    echo "===================================="
}

# --- check コマンド ---
cmd_check() {
    echo "必要なツールの状態:"
    check_tool "git" "true" || true
    check_tool "node" "false" || true
    check_tool "npm" "false" || true
    check_tool "python3" "false" || true
    check_tool "pip3" "false" || true
    check_tool "docker" "false" || true
    check_tool "shellcheck" "false" || true
}

# --- list コマンド ---
cmd_list() {
    echo "利用可能なテンプレート:"
    printf "  %-10s %s\n" "node" "Node.js プロジェクト"
    printf "  %-10s %s\n" "python" "Python プロジェクト"
    printf "  %-10s %s\n" "bash" "Bashスクリプトプロジェクト"
    printf "  %-10s %s\n" "html" "静的Webサイトプロジェクト"
}

# --- help コマンド ---
cmd_help() {
    cat <<EOF
${TOOL_NAME} - 開発環境セットアップツール (v${VERSION})

使い方: ${TOOL_NAME} <コマンド> [オプション]

コマンド:
  init <名前> --type <タイプ> [--git] [-n]
              プロジェクトを初期化する
  check       必要なツールの状態を確認する
  list        利用可能なテンプレートを一覧する
  help        このヘルプを表示する

テンプレートタイプ: node, python, bash, html

例:
  ${TOOL_NAME} init myapp --type node --git
  ${TOOL_NAME} init mylib --type python
  ${TOOL_NAME} check
  ${TOOL_NAME} list
EOF
}

# ============================================================
# メイン
# ============================================================
main() {
    local command="${1:-help}"
    shift || true

    case "$command" in
        init)  cmd_init "$@" ;;
        check) cmd_check "$@" ;;
        list)  cmd_list "$@" ;;
        help)  cmd_help ;;
        -h|--help) cmd_help ;;
        *)
            log_error "不明なコマンド: $command"
            echo ""
            cmd_help
            exit 1
            ;;
    esac
}

main "$@"

```bash
#!/usr/bin/env bash
# 開発環境セットアップツール（改良版）
# カスタムテンプレート読み込み、フック、設定ファイル対応
# 実行方法：chmod +x devsetup.sh && ./devsetup.sh init <名前> --type <タイプ> [--git]

set -euo pipefail

readonly TOOL_NAME="devsetup"
readonly VERSION="1.0"
readonly TOOL_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
readonly TEMPLATES_DIR="${TOOL_DIR}/templates"
readonly CONFIG_FILE="${HOME}/.devsetup.conf"

DRY_RUN=false
INIT_GIT=false
PROJECT_TYPE="node"
VERBOSE=false

# ============================================================
# ユーティリティ
# ============================================================
if [ -t 1 ]; then
    readonly C_GREEN='\033[32m' C_RED='\033[31m' C_YELLOW='\033[33m'
    readonly C_CYAN='\033[36m' C_BOLD='\033[1m' C_RESET='\033[0m'
else
    readonly C_GREEN="" C_RED="" C_YELLOW="" C_CYAN="" C_BOLD="" C_RESET=""
fi

log_step()  { echo -e "\n${C_BOLD}[STEP ${1}]${C_RESET} ${2}"; }
log_ok()    { echo -e "  ${C_GREEN}✓${C_RESET} $*"; }
log_fail()  { echo -e "  ${C_RED}✗${C_RESET} $*"; }
log_info()  { echo -e "  $*"; }
log_error() { echo -e "${C_RED}[ERROR]${C_RESET} $*" >&2; }
log_debug() { $VERBOSE && echo -e "${C_CYAN}[DEBUG]${C_RESET} $*" >&2 || true; }

run() {
    if $DRY_RUN; then
        echo -e "  ${C_YELLOW}[DRY]${C_RESET} $*"
        return 0
    fi
    log_debug "実行: $*"
    "$@"
}

die() {
    log_error "$@"
    exit 1
}

# ============================================================
# 設定ファイル読み込み
# ============================================================
load_config() {
    [ -f "$CONFIG_FILE" ] || return 0
    log_debug "設定読み込み: $CONFIG_FILE"
    while IFS='=' read -r key value; do
        [[ "$key" =~ ^[[:space:]]*# ]] && continue
        [[ -z "$key" ]] && continue
        key=$(echo "$key" | sed 's/^[[:space:]]*//;s/[[:space:]]*$//')
        value=$(echo "$value" | sed 's/^[[:space:]]*//;s/[[:space:]]*$//')
        case "$key" in
            default_type) PROJECT_TYPE="$value" ;;
            author)       AUTHOR="$value" ;;
            license)      LICENSE="$value" ;;
        esac
    done < "$CONFIG_FILE"
}

AUTHOR="${AUTHOR:-}"
LICENSE="${LICENSE:-MIT}"
load_config

# ============================================================
# テンプレート検出
# ============================================================
# ビルトインテンプレート + templates/ ディレクトリのカスタムテンプレート
declare -A TEMPLATE_DESCRIPTIONS=(
    [node]="Node.js プロジェクト"
    [python]="Python プロジェクト"
    [bash]="Bashスクリプトプロジェクト"
    [html]="静的Webサイトプロジェクト"
)

discover_custom_templates() {
    [ -d "$TEMPLATES_DIR" ] || return 0
    for tpl_dir in "$TEMPLATES_DIR"/*/; do
        [ -d "$tpl_dir" ] || continue
        local tpl_name
        tpl_name=$(basename "$tpl_dir")
        if [ -f "${tpl_dir}/description.txt" ]; then
            TEMPLATE_DESCRIPTIONS[$tpl_name]=$(cat "${tpl_dir}/description.txt")
        else
            TEMPLATE_DESCRIPTIONS[$tpl_name]="カスタムテンプレート"
        fi
    done
}

discover_custom_templates

# ============================================================
# ツールチェック
# ============================================================
check_tool() {
    local tool="$1"
    local required="${2:-false}"

    if command -v "$tool" > /dev/null 2>&1; then
        local version
        version=$("$tool" --version 2>&1 | head -1 | grep -oE '[0-9]+\.[0-9]+[.0-9]*' | head -1)
        printf "  ${C_GREEN}✓${C_RESET} %-10s %s\n" "$tool" "${version:-不明}"
        return 0
    else
        if [ "$required" = "true" ]; then
            printf "  ${C_RED}✗${C_RESET} %-10s 未インストール（必須）\n" "$tool"
        else
            printf "  ${C_YELLOW}✗${C_RESET} %-10s 未インストール（オプション）\n" "$tool"
        fi
        return 1
    fi
}

check_prerequisites() {
    local project_type="$1"
    local errors=0

    check_tool "git" "true" || errors=$((errors + 1))

    case "$project_type" in
        node)   check_tool "node" "true" || errors=$((errors + 1))
                check_tool "npm" "true"  || errors=$((errors + 1)) ;;
        python) check_tool "python3" "true" || errors=$((errors + 1))
                check_tool "pip3" "false" || true ;;
        bash)   check_tool "bash" "true" || errors=$((errors + 1))
                check_tool "shellcheck" "false" || true ;;
        html)   ;;
    esac
    check_tool "docker" "false" || true
    return "$errors"
}

# ============================================================
# フック実行
# ============================================================
run_hook() {
    local hook_name="$1"
    local project_dir="$2"
    local hook_file="${TEMPLATES_DIR}/${PROJECT_TYPE}/hooks/${hook_name}.sh"

    if [ -f "$hook_file" ] && [ -x "$hook_file" ]; then
        log_debug "フック実行: $hook_name"
        (cd "$project_dir" && bash "$hook_file")
    fi
}

# ============================================================
# テンプレート生成（ビルトイン）
# ============================================================
create_common_files() {
    local project_dir="$1"
    local project_name="$2"
    local project_type="$3"

    # .editorconfig
    cat > "${project_dir}/.editorconfig" <<'EOF'
root = true

[*]
indent_style = space
indent_size = 2
end_of_line = lf
charset = utf-8
trim_trailing_whitespace = true
insert_final_newline = true

[*.py]
indent_size = 4

[*.sh]
indent_size = 4
EOF
    log_ok ".editorconfig"

    # .gitignore（タイプ別の内容は省略、初心者版と同様）
    cat > "${project_dir}/.gitignore" <<EOF
.DS_Store
Thumbs.db
*.swp
*~
.vscode/
.idea/
EOF
    log_ok ".gitignore"

    # README.md
    cat > "${project_dir}/README.md" <<EOF
# ${project_name}

> Generated by ${TOOL_NAME} v${VERSION}

## セットアップ

\`\`\`bash
# ここにセットアップ手順
\`\`\`

## ライセンス

${LICENSE}
EOF
    log_ok "README.md"
}

generate_template() {
    local project_dir="$1"
    local project_type="$2"

    # カスタムテンプレートが存在すればそれを使う
    local custom_dir="${TEMPLATES_DIR}/${project_type}/files"
    if [ -d "$custom_dir" ]; then
        log_debug "カスタムテンプレート使用: $custom_dir"
        cp -r "$custom_dir"/. "$project_dir/"
        # ファイル一覧を表示
        find "$project_dir" -type f -not -path '*/.git/*' | while read -r f; do
            log_ok "$(basename "$f")"
        done
        return 0
    fi

    # ビルトインテンプレート
    case "$project_type" in
        node)
            run mkdir -p "${project_dir}/src" "${project_dir}/test"
            local pname
            pname=$(basename "$project_dir")
            cat > "${project_dir}/package.json" <<EOF
{
  "name": "${pname}",
  "version": "1.0.0",
  "main": "src/index.js",
  "scripts": { "start": "node src/index.js", "test": "echo 'no tests' && exit 1" },
  "license": "${LICENSE}"
}
EOF
            echo "'use strict';\nconsole.log('Hello!');" > "${project_dir}/src/index.js"
            log_ok "package.json" && log_ok "src/index.js" && log_ok "test/"
            ;;
        python)
            run mkdir -p "${project_dir}/src" "${project_dir}/tests"
            echo "[project]\nname = \"$(basename "$project_dir")\"\nversion = \"1.0.0\"" > "${project_dir}/pyproject.toml"
            touch "${project_dir}/src/__init__.py" "${project_dir}/requirements.txt"
            cat > "${project_dir}/src/main.py" <<'EOF'
def main():
    print("Hello, World!")

if __name__ == "__main__":
    main()
EOF
            log_ok "pyproject.toml" && log_ok "src/main.py" && log_ok "tests/"
            ;;
        bash)
            run mkdir -p "${project_dir}/bin" "${project_dir}/lib" "${project_dir}/test"
            cat > "${project_dir}/bin/main.sh" <<'BASH_EOF'
#!/usr/bin/env bash
set -euo pipefail
echo "Hello, World!"
BASH_EOF
            chmod +x "${project_dir}/bin/main.sh"
            log_ok "bin/main.sh" && log_ok "lib/" && log_ok "test/"
            ;;
        html)
            run mkdir -p "${project_dir}/css" "${project_dir}/js"
            echo '<!DOCTYPE html><html><head><meta charset="UTF-8"><title>App</title></head><body><h1>Hello!</h1></body></html>' > "${project_dir}/index.html"
            touch "${project_dir}/css/style.css" "${project_dir}/js/main.js"
            log_ok "index.html" && log_ok "css/style.css" && log_ok "js/main.js"
            ;;
        *)
            die "未対応のテンプレート: $project_type"
            ;;
    esac
}

# ============================================================
# サブコマンド
# ============================================================
cmd_init() {
    local project_name=""
    while [ $# -gt 0 ]; do
        case "$1" in
            --type)    PROJECT_TYPE="$2"; shift 2 ;;
            --git)     INIT_GIT=true; shift ;;
            -n|--dry-run) DRY_RUN=true; shift ;;
            -v|--verbose) VERBOSE=true; shift ;;
            -*)        die "不明なオプション: $1" ;;
            *)         project_name="$1"; shift ;;
        esac
    done

    [ -z "$project_name" ] && die "プロジェクト名を指定してください"

    local project_dir
    project_dir="$(pwd)/${project_name}"

    echo -e "\n${C_BOLD}====================================${C_RESET}"
    echo -e "  開発環境セットアップ v${VERSION}"
    echo -e "${C_BOLD}====================================${C_RESET}"

    # Step 1
    log_step "1/5" "前提条件チェック"
    check_prerequisites "$PROJECT_TYPE" || {
        $DRY_RUN || die "必須ツール不足"
    }

    # Step 2
    log_step "2/5" "ディレクトリ作成"
    [ -d "$project_dir" ] && die "既に存在: $project_dir"
    run mkdir -p "$project_dir"
    log_ok "$project_dir"

    # Step 3
    log_step "3/5" "テンプレート生成"
    if ! $DRY_RUN; then
        run_hook "pre-generate" "$project_dir"
        create_common_files "$project_dir" "$project_name" "$PROJECT_TYPE"
        generate_template "$project_dir" "$PROJECT_TYPE"
        run_hook "post-generate" "$project_dir"
    fi

    # Step 4
    log_step "4/5" "Git初期化"
    if $INIT_GIT && ! $DRY_RUN; then
        (cd "$project_dir" && git init -q && git add -A && git commit -q -m "Initial commit")
        log_ok "初期コミット作成"
    else
        log_info "スキップ"
    fi

    # Step 5
    log_step "5/5" "完了"
    local fc=0
    [ -d "$project_dir" ] && ! $DRY_RUN && fc=$(find "$project_dir" -type f -not -path '*/.git/*' | wc -l | tr -d '[:space:]')
    log_info "プロジェクト: $project_name | タイプ: $PROJECT_TYPE | ファイル: $fc"

    echo -e "\n${C_BOLD}===== セットアップ完了！ =====${C_RESET}"
}

cmd_check() {
    echo "必要なツールの状態:"
    for tool in git node npm python3 pip3 docker shellcheck; do
        check_tool "$tool" "false" || true
    done
}

cmd_list() {
    echo "利用可能なテンプレート:"
    for tpl in $(echo "${!TEMPLATE_DESCRIPTIONS[@]}" | tr ' ' '\n' | sort); do
        printf "  %-10s %s\n" "$tpl" "${TEMPLATE_DESCRIPTIONS[$tpl]}"
    done
}

cmd_help() {
    cat <<EOF
${TOOL_NAME} v${VERSION} - 開発環境セットアップツール

使い方: ${TOOL_NAME} <コマンド> [オプション]

コマンド:
  init <名前> --type <タイプ> [--git] [-n] [-v]
  check       ツール状態確認
  list        テンプレート一覧
  help        ヘルプ

設定ファイル: ~/.devsetup.conf
カスタムテンプレート: ${TEMPLATES_DIR}/<名前>/files/
EOF
}

# ============================================================
# メイン
# ============================================================
main() {
    local cmd="${1:-help}"; shift || true
    case "$cmd" in
        init)    cmd_init "$@" ;;
        check)   cmd_check ;;
        list)    cmd_list ;;
        help|-h|--help) cmd_help ;;
        *)       die "不明なコマンド: $cmd" ;;
    esac
}

main "$@"
```

- 設定ファイル (`~/.devsetup.conf`) → デフォルト設定をユーザーごとにカスタマイズ
- カラー出力とverboseモード → 視認性とデバッグ性の向上
- 連想配列でテンプレート情報を管理 → カスタムテンプレートの動的検出が可能
- `die` 関数でエラー終了を統一
