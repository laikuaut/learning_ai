> **難易度**: ★★★☆☆（中級）
> **前提知識**: 第3章（テンプレート構文）、第4章（コンポーネント基礎）、第5章（Composition API）、第6章（フォームとバリデーション）

ユーザー登録フォームで、各フィールドのバリデーションをリアルタイムに行います。

## 課題の要件

1. 名前、メール、パスワード、パスワード確認、年齢、利用規約同意のフィールドを含む
2. 各フィールドにバリデーションルールを適用する
   - 名前: 必須、2文字以上
   - メール: 必須、メール形式
   - パスワード: 必須、8文字以上、英字+数字を含む
   - パスワード確認: パスワードと一致
   - 年齢: 任意、入力する場合は18~120
   - 利用規約: 必須チェック
3. パスワード強度メーター（弱・まあまあ・強い）を表示する
4. フォーム入力の進捗率をプログレスバーで表示する
5. バリデーションエラーはフィールドの下にリアルタイム表示する
6. すべてのバリデーションが通ったときのみ送信ボタンを有効にする

```js
function validateEmail(email) {
  const re = /^[^\s@]+@[^\s@]+\.[^\s@]+$/
  return re.test(email)
}

function getPasswordStrength(password) {
  let score = 0
  if (password.length >= 8) score++
  if (/[a-z]/.test(password) && /[A-Z]/.test(password)) score++
  if (/\d/.test(password)) score++
  if (/[^a-zA-Z0-9]/.test(password)) score++
  return score // 0~4
}
```

```html
    const { createApp, ref, reactive, computed } = Vue

    // --- composable: useFormValidation ---
    function useFormValidation(form) {
      const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/

      const fieldErrors = computed(() => {
        const e = {}
        if (!form.name) e.name = "必須です"
        else if (form.name.length < 2) e.name = "2文字以上で入力してください"

        if (!form.email) e.email = "必須です"
        else if (!emailRegex.test(form.email)) e.email = "有効なメールアドレスを入力してください"

        if (!form.password) e.password = "必須です"
        else if (form.password.length < 8) e.password = "8文字以上にしてください"
        else if (!/[a-zA-Z]/.test(form.password) || !/\d/.test(form.password)) e.password = "英字と数字を含めてください"

        if (!form.confirmPassword) e.confirm = "再入力してください"
        else if (form.confirmPassword !== form.password) e.confirm = "パスワードが一致しません"

        if (form.age !== null && form.age !== "" && (form.age < 18 || form.age > 120)) e.age = "18~120の範囲で入力してください"
        return e

      const isValid = computed(() =>

        if (/\d/.test(p)) s++
        if (/[^a-zA-Z0-9]/.test(p)) s++
        return s
      })

      const pwLabel = computed(() => ["", "弱い", "まあまあ", "強い", "とても強い"][pwStrength.value])
      const pwColor = computed(() => ["#e0e0e0", "#f44336", "#ff9800", "#4caf50", "#2e7d32"][pwStrength.value])

      return { fieldErrors, isValid, progress, pwStrength, pwLabel, pwColor }
    }

    const app = createApp({
      setup() {
        const form = reactive({ name: "", email: "", password: "", confirmPassword: "", age: null, agreed: false })
        const fieldTouched = reactive({ name: false, email: false, password: false, confirm: false, age: false, agreed: false })
        const submitted = ref(false)

        const { fieldErrors, isValid, progress, pwStrength, pwLabel, pwColor } = useFormValidation(form)

        function submit() {
          Object.keys(fieldTouched).forEach(k => { fieldTouched[k] = true })
          if (isValid.value) submitted.value = true
        }

        function reset() {
          Object.assign(form, { name: "", email: "", password: "", confirmPassword: "", age: null, agreed: false })
          Object.keys(fieldTouched).forEach(k => { fieldTouched[k] = false })
          submitted.value = false
        }

        return { form, fieldTouched, submitted, fieldErrors, isValid, progress, pwStrength, pwLabel, pwColor, submit, reset }
      }
    })

    // 汎用フォーム入力コンポーネント（スロット付き）
    app.component('form-input', {
      props: {
        label: String,
        modelValue: [String, Number],
        type: { type: String, default: 'text' },
        required: Boolean,
        error: String,
        touched: Boolean,
        placeholder: String
      },
      emits: ['update:modelValue', 'blur'],
      template: `
        <div class="form-field">
          <label>{{ label }} <span v-if="required" class="required">*</span></label>
          <input
            :type="type"
            :value="modelValue"
            @input="$emit('update:modelValue', type === 'number' ? Number($event.target.value) : $event.target.value.trim())"
            @blur="$emit('blur')"
            :class="{ 'has-error': touched && error, 'is-valid': touched && !error && modelValue }"
            :placeholder="placeholder"
          />
          <div v-if="touched && error" class="field-error">{{ error }}</div>
          <div v-else-if="touched && !error && modelValue" class="field-ok">✅ OK</div>
          <slot name="extra"></slot>
        </div>
      `
    })

    app.mount('#app')
  </script>
</body>
</html>
```

**初心者向けとの違い:**
- `useFormValidation` composable にバリデーションロジックを集約
- `form-input` コンポーネントで `v-model` カスタム実装（`modelValue` + `update:modelValue`）
- `<slot name="extra">` でパスワード強度バーを挿入する名前付きスロット（named slot）
- `reactive` でフォームデータをオブジェクト1つにまとめて管理
- `Object.assign` で一括リセット
