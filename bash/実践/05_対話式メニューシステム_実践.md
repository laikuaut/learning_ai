## 課題の要件

1. タスクの追加・一覧表示・完了マーク・削除ができる
2. メニューは繰り返し表示し、「終了」を選ぶまで動作する
3. 各操作を関数として分離する
4. タスク番号が範囲外の場合はエラーメッセージを表示する
5. 終了時にファイルにタスクを保存し、次回起動時に復元する

## 解答例

<details>
<summary>解答例（初心者向け）</summary>

```bash
#!/usr/bin/env bash
# TODOリスト管理スクリプト
# 学べる内容：配列、関数、whileループ、case文、ファイル保存・復元
# 実行方法：chmod +x todo.sh && ./todo.sh

SAVE_FILE="$HOME/.todo_list.txt"

# --- タスクデータ ---
declare -a tasks=()
declare -a done_flags=()

# --- ファイルから復元 ---
load_tasks() {
    if [ -f "$SAVE_FILE" ]; then
        while IFS='|' read -r flag task; do
            tasks+=("$task")
            done_flags+=("$flag")
        done < "$SAVE_FILE"
        echo "${#tasks[@]} 件のタスクを読み込みました。"
    fi
}

# --- ファイルに保存 ---
save_tasks() {
    > "$SAVE_FILE"  # ファイルを空にする
    for i in "${!tasks[@]}"; do
        echo "${done_flags[$i]}|${tasks[$i]}" >> "$SAVE_FILE"
    done
}

# --- 一覧表示 ---
show_list() {
    if [ ${#tasks[@]} -eq 0 ]; then
        echo "タスクがありません。"
        return
    fi

    echo ""
    echo "--- TODOリスト ---"

    local done_count=0
    for i in "${!tasks[@]}"; do
        local num=$((i + 1))
        if [ "${done_flags[$i]}" -eq 1 ]; then
            echo "  ${num}. [x] ${tasks[$i]}"
            done_count=$((done_count + 1))
        else
            echo "  ${num}. [ ] ${tasks[$i]}"
        fi
    done

    local total=${#tasks[@]}
    local undone=$((total - done_count))
    echo "タスク数: ${total} (未完了: ${undone} / 完了: ${done_count})"
}

# --- タスク追加 ---
add_task() {
    read -rp "タスクを入力: " task
    if [ -z "$task" ]; then
        echo "タスクが空です。"
        return
    fi
    tasks+=("$task")
    done_flags+=(0)
    echo "追加しました: $task"
}

# --- タスク完了 ---
complete_task() {
    if [ ${#tasks[@]} -eq 0 ]; then
        echo "タスクがありません。"
        return
    fi

    show_list
    read -rp "完了にするタスク番号: " num

    if [ "$num" -lt 1 ] 2>/dev/null || [ "$num" -gt "${#tasks[@]}" ] 2>/dev/null; then
        echo "無効な番号です。"
        return
    fi

    local index=$((num - 1))
    done_flags[$index]=1
    echo "完了にしました: ${tasks[$index]}"
}

# --- タスク削除 ---
delete_task() {
    if [ ${#tasks[@]} -eq 0 ]; then
        echo "タスクがありません。"
        return
    fi

    show_list
    read -rp "削除するタスク番号: " num

    if [ "$num" -lt 1 ] 2>/dev/null || [ "$num" -gt "${#tasks[@]}" ] 2>/dev/null; then
        echo "無効な番号です。"
        return
    fi

    local index=$((num - 1))
    echo "削除しました: ${tasks[$index]}"

    # 配列から要素を削除して詰める
    unset 'tasks[$index]'
    unset 'done_flags[$index]'
    tasks=("${tasks[@]}")
    done_flags=("${done_flags[@]}")
}

# --- メイン処理 ---
echo ""
echo "===== TODOリスト管理 ====="
load_tasks

while true; do
    echo ""
    echo "1) 一覧表示  2) タスク追加  3) タスク完了  4) タスク削除  5) 終了"
    read -rp "選択してください: " choice

    case "$choice" in
        1) show_list ;;
        2) add_task ;;
        3) complete_task ;;
        4) delete_task ;;
        5)
            save_tasks
            echo "保存しました。さようなら！"
            break
            ;;
        *)
            echo "1〜5の番号を選んでください。"
            ;;
    esac
done
```

<summary>解答例（改良版 ─ selectメニューと優先度付き）</summary>

```bash
#!/usr/bin/env bash
# TODOリスト管理スクリプト（改良版）
# selectメニュー、優先度、期限、検索機能を追加
# 実行方法：chmod +x todo.sh && ./todo.sh

set -euo pipefail

SAVE_FILE="${TODO_FILE:-$HOME/.todo_list.txt}"

declare -a tasks=()
declare -a done_flags=()
declare -a priorities=()

# --- ファイル入出力 ---
load_tasks() {
    [ -f "$SAVE_FILE" ] || return 0
    while IFS='|' read -r flag priority task; do
        tasks+=("$task")
        done_flags+=("$flag")
        priorities+=("$priority")
    done < "$SAVE_FILE"
    if [ ${#tasks[@]} -gt 0 ]; then
        echo "${#tasks[@]} 件のタスクを読み込みました。"
    fi
}

save_tasks() {
    : > "$SAVE_FILE"
    for i in "${!tasks[@]}"; do
        printf '%s|%s|%s\n' "${done_flags[$i]}" "${priorities[$i]}" "${tasks[$i]}" >> "$SAVE_FILE"
    done
}

# --- 優先度の表示ラベル ---
priority_label() {
    case "$1" in
        H) echo "[高]" ;;
        M) echo "[中]" ;;
        L) echo "[低]" ;;
        *) echo "[--]" ;;
    esac
}

# --- 一覧表示 ---
show_list() {
    if [ ${#tasks[@]} -eq 0 ]; then
        echo "タスクがありません。"
        return
    fi

    echo ""
    echo "--- TODOリスト ---"
    local done_count=0
    for i in "${!tasks[@]}"; do
        local num=$((i + 1))
        local mark="[ ]"
        if [ "${done_flags[$i]}" -eq 1 ]; then
            mark="[x]"
            done_count=$((done_count + 1))
        fi
        local plabel
        plabel=$(priority_label "${priorities[$i]}")
        printf "  %2d. %s %s %s\n" "$num" "$mark" "$plabel" "${tasks[$i]}"
    done

    local total=${#tasks[@]}
    echo "タスク数: ${total} (未完了: $((total - done_count)) / 完了: ${done_count})"
}

# --- タスク追加 ---
add_task() {
    read -rp "タスクを入力: " task
    [ -z "$task" ] && { echo "タスクが空です。"; return; }

    echo "優先度を選択: H=高 M=中 L=低"
    read -rp "優先度 [M]: " pri
    pri="${pri:-M}"
    pri="${pri^^}"  # 大文字に変換
    [[ "$pri" =~ ^[HML]$ ]] || pri="M"

    tasks+=("$task")
    done_flags+=(0)
    priorities+=("$pri")
    echo "追加しました: $(priority_label "$pri") $task"
}

# --- タスク完了/復帰 ---
toggle_task() {
    [ ${#tasks[@]} -eq 0 ] && { echo "タスクがありません。"; return; }
    show_list
    read -rp "完了/未完了を切り替える番号: " num

    if ! [[ "$num" =~ ^[0-9]+$ ]] || [ "$num" -lt 1 ] || [ "$num" -gt ${#tasks[@]} ]; then
        echo "無効な番号です。"
        return
    fi

    local index=$((num - 1))
    if [ "${done_flags[$index]}" -eq 0 ]; then
        done_flags[$index]=1
        echo "完了にしました: ${tasks[$index]}"
    else
        done_flags[$index]=0
        echo "未完了に戻しました: ${tasks[$index]}"
    fi
}

# --- タスク削除 ---
delete_task() {
    [ ${#tasks[@]} -eq 0 ] && { echo "タスクがありません。"; return; }
    show_list
    read -rp "削除するタスク番号: " num

    if ! [[ "$num" =~ ^[0-9]+$ ]] || [ "$num" -lt 1 ] || [ "$num" -gt ${#tasks[@]} ]; then
        echo "無効な番号です。"
        return
    fi

    local index=$((num - 1))
    read -rp "「${tasks[$index]}」を削除しますか？ (y/N): " confirm
    if [[ "${confirm,,}" != "y" ]]; then
        echo "キャンセルしました。"
        return
    fi

    echo "削除しました: ${tasks[$index]}"
    unset 'tasks[$index]' 'done_flags[$index]' 'priorities[$index]'
    tasks=("${tasks[@]}")
    done_flags=("${done_flags[@]}")
    priorities=("${priorities[@]}")
}

# --- 検索 ---
search_task() {
    read -rp "検索キーワード: " keyword
    [ -z "$keyword" ] && { echo "キーワードが空です。"; return; }

    echo ""
    local found=0
    for i in "${!tasks[@]}"; do
        if [[ "${tasks[$i]}" == *"$keyword"* ]]; then
            local num=$((i + 1))
            local mark="[ ]"
            [ "${done_flags[$i]}" -eq 1 ] && mark="[x]"
            local plabel
            plabel=$(priority_label "${priorities[$i]}")
            printf "  %2d. %s %s %s\n" "$num" "$mark" "$plabel" "${tasks[$i]}"
            found=$((found + 1))
        fi
    done

    if [ "$found" -eq 0 ]; then
        echo "見つかりませんでした。"
    else
        echo "${found} 件見つかりました。"
    fi
}

# --- メイン処理 ---
echo ""
echo "===== TODOリスト管理 ====="
load_tasks

PS3="選択してください: "
select action in "一覧表示" "タスク追加" "完了切替" "タスク削除" "検索" "終了"; do
    case "$action" in
        "一覧表示")  show_list ;;
        "タスク追加") add_task ;;
        "完了切替")  toggle_task ;;
        "タスク削除") delete_task ;;
        "検索")     search_task ;;
        "終了")
            save_tasks
            echo "保存しました。さようなら！"
            break
            ;;
        *)
            echo "有効な番号を選んでください。"
            ;;
    esac
done
```

**初心者向けとの違い:**
- `select` 文でメニュー表示を自動化 → メニュー表示のコードが不要に
- 優先度 (H/M/L) を追加 → タスクの重要度を管理
- 完了/未完了のトグル → 一度完了にしたタスクを未完了に戻せる
- 削除時に確認プロンプト → 誤削除の防止
- キーワード検索機能 → タスクが多い場合に便利
- `${pri^^}` で大文字変換 → 入力が `h` でも `H` として扱う
- 環境変数 `TODO_FILE` で保存先変更可能 → テスト時に便利
