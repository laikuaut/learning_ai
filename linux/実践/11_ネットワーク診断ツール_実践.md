> **難易度**: ★★★☆☆（中級）
> **前提知識**: 第1章（Linuxの基本）、第5章（シェルスクリプト基礎）、第6章（ネットワーク基本コマンド）
> **課題の種類**: ミニプロジェクト
> **学習目標**: ネットワーク関連コマンドを組み合わせて、実務で使える診断ツールを作成できるようになる

## 完成イメージ

ネットワークの状態を総合的に診断し、レポートを出力するツールを作成します。


---

## 課題の要件

1. ネットワークインターフェースの情報を表示する
2. デフォルトゲートウェイとDNS設定を確認する
3. ゲートウェイ、DNS、外部ホストへの疎通確認（ping）を行う
4. DNS名前解決のテストを行う
5. リッスンしているポートの一��を表示する
6. アクティブなネットワーク接続の統計を表示する
7. 結果をレポートファイルに保存する

<summary>解答例（初心者向け）</summary>

```bash
#!/bin/bash
# ��ットワーク診断ツール
# 学べる内容：ip, ping, ss, DNS, ネットワークコマンド
# 実行方法：bash network_diag.sh

REPORT="network_report_$(date '+%Y%m%d_%H%M%S').txt"

# 画面とファイルの両方に出力する関数
out() {
    echo "$1" | tee -a "$REPORT"
}

# ヘッダー
out "===== ネットワーク診断ツール ====="
out "実行日時: $(date '+%Y-%m-%d %H:%M:%S')"
out "ホスト名: $(hostname)"
out ""

# --- ネットワークインターフェース ---
out "【ネットワークインターフェー��】"
ip -4 addr show | awk '/inet / {
    split($NF, iface, " ")
    status = "UP"
    printf "  %-8s %s  %s\n", $NF":", $2, status
}' | tee -a "$REPORT"
out ""

# --- デフォルトゲートウェイ ---
out "【���フォルトゲー��ウェイ】"
gateway=$(ip route show default | awk '{print $3}')
gateway_dev=$(ip route show default | awk '{print $5}')
if [ -n "$gateway" ]; then
    out "  $gateway  via $gateway_dev"
else
    out "  デフォル��ゲートウェイが設定されていません"
fi
out ""

# --- DNS設定 ---
out "【DNS設定】"
if [ -f /etc/resolv.conf ]; then
    grep "^nameserver" /etc/resolv.conf | while read -r line; do
        out "  $line"
    done
else
    out "  /etc/resolv.conf が見つかりませ���"
fi
out ""

# --- 接続テスト ---
out "【��続テスト】"

ping_test() {
    local name="$1"
    local target="$2"
    local result

    if result=$(ping -c 1 -W 3 "$target" 2>/dev/null); then
        local time_ms
        time_ms=$(echo "$result" | grep "time=" | sed 's/.*time=\([^ ]*\).*/\1/')
        printf "  %-30s: OK (%s)\n" "$name ($target)" "${time_ms}ms"
    else
        printf "  %-30s: NG (到達不可)\n" "$name ($target)"
    fi
}

# ゲートウェイへのping
if [ -n "$gateway" ]; then
    ping_test "ゲートウェイ" "$gateway" | tee -a "$REPORT"
fi

# DNSへのping
dns_server=$(grep "^nameserver" /etc/resolv.conf 2>/dev/null | head -1 | awk '{print $2}')
if [ -n "$dns_server" ]; then
    ping_test "DNS" "$dns_server" | tee -a "$REPORT"
fi

# 外部ホストへのping
ping_test "外部" "google.com" | tee -a "$REPORT"
out ""

# --- DNS解決テスト ---
out "【DNS解決テスト】"
for domain in google.com github.com; do
    resolved=$(dig +short "$domain" 2>/dev/null | head -1)
    if [ -z "$resolved" ]; then
        # dig がなければ host コマンドを試す
        resolved=$(host "$domain" 2>/dev/null | awk '/has address/ {print $4; exit}')
    fi
    if [ -n "$resolved" ]; then
        printf "  %-15s → %-20s OK\n" "$domain" "$resolved" | tee -a "$REPORT"
    else
        printf "  %-15s → %-20s NG\n" "$domain" "(解決失敗)" | tee -a "$REPORT"
    fi
done
out ""

# --- リッスン���ート ---
out "【リッスンポート】"
ss -tlnp 2>/dev/null | awk 'NR>1 {
    split($4, addr, ":")
    port = addr[length(addr)]
    proc = $6
    gsub(/.*"/, "", proc)
    gsub(/".*/, "", proc)
    if (port != "") printf "  %-6s tcp  %s\n", port, proc
}' | sort -t' ' -k1 -n | uniq | tee -a "$REPORT"
out ""

# --- アクティブ接続数 ---
out "【アクティブ接続数】"
ss -tan 2>/dev/null | awk 'NR>1 {state[$1]++} END {
    for (s in state) printf "  %-15s: %d\n", s, state[s]
}' | sort | tee -a "$REPORT"
out ""

out "レポート保存: $REPORT"
```

<summary>解答例（改良版 ─ 総合判定・カラー出力・詳細診断）</summary>

```bash
#!/bin/bash
# ネットワーク診断ツール（改良版）
# 学べる内容：ネットワーク診断、カラー出力、終了コード
# 実行方法：bash network_diag_v2.sh [--no-color] [--verbose]

set -uo pipefail

# --- 設定 ---
REPORT="network_report_$(date '+%Y%m%d_%H%M%S').txt"
USE_COLOR=true
VERBOSE=false
TOTAL_TESTS=0
PASSED_TESTS=0
FAILED_TESTS=0

# 引数解析
for arg in "$@"; do
    case "$arg" in
        --no-color) USE_COLOR=false ;;
        --verbose)  VERBOSE=true ;;
    esac
done

# --- カラー出力 ---
if [ "$USE_COLOR" = true ] && [ -t 1 ]; then
    RED='\033[0;31m'
    GREEN='\033[0;32m'
    YELLOW='\033[0;33m'
    BLUE='\033[0;34m'
    BOLD='\033[1m'
    NC='\033[0m'
else
    RED='' GREEN='' YELLOW='' BLUE='' BOLD='' NC=''
fi

out() { echo -e "$1" | tee -a "$REPORT"; }
section() { out "\n${BOLD}${BLUE}$1${NC}"; }

record_test() {
    ((TOTAL_TESTS++))
    if [ "$1" = "pass" ]; then
        ((PASSED_TESTS++))
    else
        ((FAILED_TESTS++))
    fi
}

# --- ヘッダー ---
out "${BOLD}===== ネットワーク診断ツール =====${NC}"
out "実行日時: $(date '+%Y-%m-%d %H:%M:%S')"
out "ホスト名: $(hostname)"
out "カーネル:  $(uname -r)"

# --- インターフェース情報 ---
section "【ネットワ���クインターフェース】"
ip -o -4 addr show 2>/dev/null | while read -r _ iface _ addr _rest; do
    state=$(ip link show "$iface" | grep -oP '(?<=state )\w+')
    if [ "$state" = "UP" ]; then
        out "  ${GREEN}${iface}${NC}: $addr  ${GREEN}UP${NC}"
    else
        out "  ${RED}${iface}${NC}: $addr  ${RED}${state}${NC}"
    fi
done

# --- ルーティング ---
section "【デフォルトゲートウェイ】"
gateway=$(ip route show default 2>/dev/null | awk '{print $3}')
gateway_dev=$(ip route show default 2>/dev/null | awk '{print $5}')
if [ -n "$gateway" ]; then
    out "  $gateway  via $gateway_dev"
else
    out "  ${RED}デフォルトゲートウェイなし${NC}"
    record_test "fail"
fi

# --- DNS ---
section "【DNS設定】"
if [ -f /etc/resolv.conf ]; then
    grep "^nameserver" /etc/resolv.conf | while read -r _ server; do
        out "  nameserver $server"
    done
else
    out "  ${RED}/etc/resolv.conf なし${NC}"
fi

# --- 接続テスト ---
section "【接続テスト】"

ping_check() {
    local label="$1" target="$2"
    local result rtt

    if result=$(ping -c 1 -W 3 "$target" 2>/dev/null); then
        rtt=$(echo "$result" | grep "time=" | sed 's/.*time=\([^ ]*\).*/\1/')
        out "  $(printf '%-35s' "$label ($target)"): ${GREEN}OK${NC} (${rtt}ms)"
        record_test "pass"
    else
        out "  $(printf '%-35s' "$label ($target)"): ${RED}NG${NC}"
        record_test "fail"
    fi
}

[ -n "$gateway" ] && ping_check "ゲートウェイ" "$gateway"
dns_server=$(grep "^nameserver" /etc/resolv.conf 2>/dev/null | head -1 | awk '{print $2}')
[ -n "$dns_server" ] && ping_check "プライマリDNS" "$dns_server"
ping_check "外部ホスト" "8.8.8.8"
ping_check "外部ドメイン" "google.com"

# --- DNS解決テスト ---
section "【DNS解���テスト】"
dns_resolve_check() {
    local domain="$1"
    local resolved

    # dig → host → nslookup の順に試す
    resolved=$(dig +short "$domain" A 2>/dev/null | head -1)
    [ -z "$resolved" ] && resolved=$(host -t A "$domain" 2>/dev/null | awk '/has address/ {print $4; exit}')
    [ -z "$resolved" ] && resolved=$(nslookup "$domain" 2>/dev/null | awk '/^Address: / {print $2; exit}')

    if [ -n "$resolved" ]; then
        out "  $(printf '%-20s' "$domain") → $(printf '%-20s' "$resolved") ${GREEN}OK${NC}"
        record_test "pass"
    else
        out "  $(printf '%-20s' "$domain") → $(printf '%-20s' "(失敗)")    ${RED}NG${NC}"
        record_test "fail"
    fi
}

for domain in google.com github.com example.com; do
    dns_resolve_check "$domain"
done

# --- リッスンポート ---
section "【リッスンポート】"
out "  $(printf '%-8s %-8s %s' 'PORT' 'PROTO' 'PROCESS')"
out "  $(printf '%s' '------------------------------')"
ss -tlnp 2>/dev/null | awk 'NR>1' | while read -r state recv send local_addr _ process; do
    port=$(echo "$local_addr" | awk -F: '{print $NF}')
    proc_name=$(echo "$process" | grep -oP '(?<=")\w+' | head -1)
    [ -z "$proc_name" ] && proc_name="-"
    printf "  %-8s %-8s %s\n" "$port" "tcp" "$proc_name"
done | sort -t' ' -k1 -n | uniq | tee -a "$REPORT"

# --- 接続状態 ---
section "【アクティブ接続統計】"
ss -tan 2>/dev/null | awk '
NR>1 {state[$1]++; total++}
END {
    important["ESTAB"] = 1
    important["TIME-WAIT"] = 1
    important["CLOSE-WAIT"] = 1
    important["FIN-WAIT-1"] = 1
    important["FIN-WAIT-2"] = 1
    important["SYN-SENT"] = 1
    important["SYN-RECV"] = 1
    important["LISTEN"] = 1

    for (s in state) {
        if (s in important) printf "  %-15s: %d\n", s, state[s]
    }
    printf "  %-15s: %d\n", "TOTAL", total
}' | sort | tee -a "$REPORT"

# --- 追加情報（verboseモード） ---
if [ "$VERBOSE" = true ]; then
    section "【ルーティングテーブル】"
    ip route show | while read -r line; do
        out "  $line"
    done

    section "【ARP テーブル】"
    ip neigh show | while read -r line; do
        out "  $line"
    done
fi

# --- 総合判定 ---
section "【診断結果】"
if [ "$FAILED_TESTS" -eq 0 ]; then
    out "  ${GREEN}${BOLD}全テスト合格 ($PASSED_TESTS/$TOTAL_TESTS): ネットワーク正常${NC}"
    EXIT_CODE=0
elif [ "$FAILED_TESTS" -le 1 ]; then
    out "  ${YELLOW}${BOLD}一部テスト失敗 ($FAILED_TESTS/$TOTAL_TESTS): 軽微な問題あり${NC}"
    EXIT_CODE=1
else
    out "  ${RED}${BOLD}複数テスト失敗 ($FAILED_TESTS/$TOTAL_TESTS): 要確認${NC}"
    EXIT_CODE=2
fi

out ""
out "レポ���ト保存: $REPORT"
exit "$EXIT_CODE"
```

**初心者向けとの違い:**
- カラー出力対応（`--no-color` で無効化可能、パイプ時は自動無効化）
- 合格/不合格の自動カウントと総合判定
- DNS解決で複数コマンドをフォールバック（`dig` → `host` → `nslookup`）
- `--verbose` で追加情報（ルーティングテーブル、ARPテーブル）を表示
- 終了コードで結果を返す → 他のスクリプトから呼び出して判定可能
