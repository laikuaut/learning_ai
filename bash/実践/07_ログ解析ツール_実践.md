## 課題の要件

1. コマンドライン引数でログファイルを受け取る
2. ログ形式は「日時 メソッド パス ステータスコード」を想定する
3. ステータスコード別の集計と割合を表示する
4. 時間帯別（0〜23時）のアクセス数を表示する
5. アクセス数の多いパスを上位10件表示する
6. 4xx/5xx のエラーリクエストを詳細表示する
7. サンプルログ生成機能を備える（テスト用）

<summary>ステップ1：サンプルログを生成する関数を作る</summary>

テスト用のログファイルを生成する関数を作りましょう。

```bash
generate_sample_log() {
    local filename=$1
    local paths=("/index.html" "/about.html" "/api/users" "/api/login" "/contact.html")
    local methods=("GET" "GET" "GET" "POST" "GET")
    local statuses=(200 200 200 301 404 500)

    for i in $(seq 1 150); do
        local hour=$(printf "%02d" $((RANDOM % 24)))
        local min=$(printf "%02d" $((RANDOM % 60)))
        local sec=$(printf "%02d" $((RANDOM % 60)))
        local path_idx=$((RANDOM % ${#paths[@]}))
        local status_idx=$((RANDOM % ${#statuses[@]}))
        echo "2024-03-15 ${hour}:${min}:${sec} ${methods[$path_idx]} ${paths[$path_idx]} ${statuses[$status_idx]}"
    done > "$filename"
}
```

## 解答例

<details>
<summary>解答例（初心者向け）</summary>

```bash
#!/usr/bin/env bash
# ログ解析ツール
# 学べる内容：連想配列、whileループ、パイプライン、テキスト処理
# 実行方法：chmod +x loganalyzer.sh && ./loganalyzer.sh <ログファイル>
# テスト：./loganalyzer.sh --generate sample.log && ./loganalyzer.sh sample.log

# --- サンプルログ生成 ---
generate_sample_log() {
    local filename=$1
    local -a paths=("/index.html" "/about.html" "/api/users" "/api/login" "/contact.html" "/old-page.html" "/missing.css" "/api/submit" "/dashboard" "/products")
    local -a methods=("GET" "GET" "GET" "POST" "GET" "GET" "GET" "POST" "GET" "GET")
    local -a statuses=(200 200 200 200 200 200 200 200 301 301 404 404 500)

    for i in $(seq 1 150); do
        local hour=$(printf "%02d" $((RANDOM % 24)))
        local min=$(printf "%02d" $((RANDOM % 60)))
        local sec=$(printf "%02d" $((RANDOM % 60)))
        local path_idx=$((RANDOM % ${#paths[@]}))
        local status_idx=$((RANDOM % ${#statuses[@]}))
        echo "2024-03-15 ${hour}:${min}:${sec} ${methods[$path_idx]} ${paths[$path_idx]} ${statuses[$status_idx]}"
    done | sort > "$filename"

    echo "サンプルログを ${filename} に生成しました（150件）。"
}

# --- 引数チェック ---
if [ "${1:-}" = "--generate" ]; then
    if [ -z "${2:-}" ]; then
        echo "使い方: $0 --generate <出力ファイル名>"
        exit 1
    fi
    generate_sample_log "$2"
    exit 0
fi

logfile="${1:-}"
if [ -z "$logfile" ]; then
    echo "使い方: $0 <ログファイル>"
    echo "        $0 --generate <出力ファイル名>"
    exit 1
fi

if [ ! -f "$logfile" ]; then
    echo "エラー: '$logfile' が見つかりません" >&2
    exit 1
fi

# --- 集計 ---
declare -A status_counts
declare -A hour_counts
declare -A path_counts
declare -A error_details
total=0

while IFS=' ' read -r date time method path status; do
    # ステータス別
    status_counts[$status]=$(( ${status_counts[$status]:-0} + 1 ))

    # 時間帯別
    hour="${time%%:*}"
    hour_counts[$hour]=$(( ${hour_counts[$hour]:-0} + 1 ))

    # パス別
    path_counts[$path]=$(( ${path_counts[$path]:-0} + 1 ))

    # エラー詳細
    if [ "$status" -ge 400 ] 2>/dev/null; then
        local key="${status} ${method} ${path}"
        error_details[$key]=$(( ${error_details[$key]:-0} + 1 ))
    fi

    total=$((total + 1))
done < "$logfile"

# --- 表示 ---
echo ""
echo "===== ログ解析レポート ====="
echo "対象ファイル: $logfile"
echo "総リクエスト数: $total"

# ステータスコード別
echo ""
echo "■ ステータスコード別集計"
for status in $(echo "${!status_counts[@]}" | tr ' ' '\n' | sort); do
    count=${status_counts[$status]}
    if [ "$total" -gt 0 ]; then
        percent=$((count * 1000 / total))
        percent_int=$((percent / 10))
        percent_dec=$((percent % 10))
    else
        percent_int=0
        percent_dec=0
    fi
    # バー生成
    bar_len=$((count * 30 / total))
    bar=""
    for ((i = 0; i < bar_len; i++)); do bar+="█"; done
    printf "  %s  %3d件 (%2d.%d%%) %s\n" "$status" "$count" "$percent_int" "$percent_dec" "$bar"
done

# 時間帯別
echo ""
echo "■ 時間帯別アクセス数"
for h in $(seq -w 0 23); do
    count=${hour_counts[$h]:-0}
    bar=""
    for ((i = 0; i < count; i++)); do bar+="█"; done
    printf "  %s時: %3d件 %s\n" "$h" "$count" "$bar"
done

# パス別 TOP10
echo ""
echo "■ アクセス数 TOP10 パス"
rank=1
for path in $(for k in "${!path_counts[@]}"; do echo "$k ${path_counts[$k]}"; done | sort -k2 -rn | head -10 | awk '{print $1}'); do
    count=${path_counts[$path]}
    printf "  %2d. %-20s %3d件\n" "$rank" "$path" "$count"
    rank=$((rank + 1))
done

# エラー詳細
echo ""
echo "■ エラーリクエスト（4xx/5xx）"
if [ ${#error_details[@]} -eq 0 ]; then
    echo "  エラーなし"
else
    for key in $(for k in "${!error_details[@]}"; do echo "${error_details[$k]} $k"; done | sort -rn | awk '{$1=""; print substr($0,2)}'); do
        count=${error_details[$key]}
        printf "  %-30s %3d件\n" "$key" "$count"
    done
fi
```

<summary>解答例（改良版 ─ awkベースの高速処理）</summary>

```bash
#!/usr/bin/env bash
# ログ解析ツール（改良版）
# awk を活用した高速処理版
# 実行方法：chmod +x loganalyzer.sh && ./loganalyzer.sh <ログファイル>

set -euo pipefail

# --- サンプルログ生成 ---
generate_sample_log() {
    local filename=$1
    awk 'BEGIN {
        srand()
        split("/index.html /about.html /api/users /api/login /contact.html /old-page.html /api/submit /dashboard /products /search", paths, " ")
        split("GET GET GET POST GET GET POST GET GET GET", methods, " ")
        split("200 200 200 200 200 200 200 301 301 404 404 500", statuses, " ")
        for (i = 1; i <= 150; i++) {
            h = int(rand() * 24)
            m = int(rand() * 60)
            s = int(rand() * 60)
            pi = int(rand() * 10) + 1
            si = int(rand() * 12) + 1
            printf "2024-03-15 %02d:%02d:%02d %s %s %s\n", h, m, s, methods[pi], paths[pi], statuses[si]
        }
    }' | sort > "$filename"
    echo "サンプルログを ${filename} に生成しました（150件）。"
}

# --- 引数チェック ---
if [ "${1:-}" = "--generate" ]; then
    [ -z "${2:-}" ] && { echo "使い方: $0 --generate <出力ファイル名>" >&2; exit 1; }
    generate_sample_log "$2"
    exit 0
fi

logfile="${1:-}"
[ -z "$logfile" ] && { echo "使い方: $0 <ログファイル>"; exit 1; }
[ -f "$logfile" ] || { echo "エラー: '$logfile' が見つかりません" >&2; exit 1; }

# --- awk で全集計を一括処理 ---
awk '
{
    total++

    # ステータスコード別
    status_counts[$5]++

    # 時間帯別
    split($2, t, ":")
    hour = t[1]
    hour_counts[hour]++

    # パス別
    path_counts[$4]++

    # エラー詳細
    if ($5 >= 400) {
        error_key = $5 " " $3 " " $4
        error_counts[error_key]++
    }
}

function bar(count, max,    s, i, len) {
    if (max == 0) return ""
    len = int(count * 30 / max)
    s = ""
    for (i = 0; i < len; i++) s = s "█"
    return s
}

END {
    print ""
    print "===== ログ解析レポート ====="
    print "対象ファイル: " FILENAME
    print "総リクエスト数: " total

    # ステータスコード別
    print ""
    print "■ ステータスコード別集計"
    n = asorti(status_counts, sorted_status)
    for (i = 1; i <= n; i++) {
        s = sorted_status[i]
        c = status_counts[s]
        pct = (total > 0) ? c * 100.0 / total : 0
        printf "  %s  %3d件 (%5.1f%%) %s\n", s, c, pct, bar(c, total)
    }

    # 時間帯別
    print ""
    print "■ 時間帯別アクセス数"
    max_hour = 0
    for (h in hour_counts) if (hour_counts[h] > max_hour) max_hour = hour_counts[h]
    for (h = 0; h < 24; h++) {
        hh = sprintf("%02d", h)
        c = (hh in hour_counts) ? hour_counts[hh] : 0
        b = ""
        len = (max_hour > 0) ? int(c * 30 / max_hour) : 0
        for (j = 0; j < len; j++) b = b "█"
        printf "  %s時: %3d件 %s\n", hh, c, b
    }

    # パス別 TOP10
    print ""
    print "■ アクセス数 TOP10 パス"
    # ソートのためにカウントとパスを配列に格納
    n = 0
    for (p in path_counts) {
        n++
        path_list[n] = path_counts[p] "\t" p
    }
    # バブルソート（降順）
    for (i = 1; i <= n; i++) {
        for (j = i + 1; j <= n; j++) {
            split(path_list[i], a, "\t")
            split(path_list[j], b, "\t")
            if (a[1]+0 < b[1]+0) {
                tmp = path_list[i]
                path_list[i] = path_list[j]
                path_list[j] = tmp
            }
        }
    }
    top = (n < 10) ? n : 10
    for (i = 1; i <= top; i++) {
        split(path_list[i], parts, "\t")
        printf "  %2d. %-20s %3d件\n", i, parts[2], parts[1]
    }

    # エラー詳細
    print ""
    print "■ エラーリクエスト（4xx/5xx）"
    has_error = 0
    for (e in error_counts) has_error = 1
    if (!has_error) {
        print "  エラーなし"
    } else {
        n = 0
        for (e in error_counts) {
            n++
            err_list[n] = error_counts[e] "\t" e
        }
        for (i = 1; i <= n; i++) {
            for (j = i + 1; j <= n; j++) {
                split(err_list[i], a, "\t")
                split(err_list[j], b, "\t")
                if (a[1]+0 < b[1]+0) {
                    tmp = err_list[i]
                    err_list[i] = err_list[j]
                    err_list[j] = tmp
                }
            }
        }
        for (i = 1; i <= n; i++) {
            split(err_list[i], parts, "\t")
            printf "  %-30s %3d件\n", parts[2], parts[1]
        }
    }
}
' "$logfile"
```

**初心者向けとの違い:**
- 全集計を `awk` 一括処理 → シェルループと比べて数十〜数百倍高速
- サンプルログ生成も `awk` → 外部コマンド呼び出しの最小化
- バーグラフを最大値基準でスケーリング → 時間帯別が見やすい
- `asorti` でキーをソート → ステータスコードが数値順に並ぶ
- 大量のログ（数万〜数十万行）でも実用的な速度で動作
- 実務ではさらに `sort | uniq -c` パイプラインと比較して使い分ける
