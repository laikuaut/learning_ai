## 1. カスタムフックとは？

**カスタムフック**は、Reactのフック（useState、useEffectなど）を使ったロジックを再利用可能な関数として切り出したものです。

### なぜカスタムフックが必要？

```jsx
// 問題：同じロジックが複数のコンポーネントに重複している
function UserProfile() {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    fetch('/api/user')
      .then(res => res.json())
      .then(data => { setData(data); setLoading(false); })
      .catch(err => { setError(err.message); setLoading(false); });
  }, []);

  // ...
}

function ProductList() {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    fetch('/api/products')
      .then(res => res.json())
      .then(data => { setData(data); setLoading(false); })
      .catch(err => { setError(err.message); setLoading(false); });
  }, []);

  // ...
}
```

このような重複したロジックをカスタムフックに切り出せます。

## 2. カスタムフックの基本

### 命名ルール

カスタムフックの関数名は必ず **`use`で始める** 必要があります。

```jsx
// OK
function useCounter() { ... }
function useFetch() { ... }
function useLocalStorage() { ... }

// NG：useで始まっていないとフックとして認識されない
function getCounter() { ... }
function fetchData() { ... }
```

### 最初のカスタムフック：useCounter

```jsx
import { useState } from 'react';

// カスタムフック定義
function useCounter(initialValue = 0, step = 1) {
  const [count, setCount] = useState(initialValue);

  const increment = () => setCount(prev => prev + step);
  const decrement = () => setCount(prev => prev - step);
  const reset = () => setCount(initialValue);

  return { count, increment, decrement, reset };
}

// 使用するコンポーネント
function CounterA() {
  const { count, increment, decrement, reset } = useCounter(0, 1);

  return (
    <div>
      <h3>カウンターA: {count}</h3>
      <button onClick={decrement}>-1</button>
      <button onClick={reset}>リセット</button>
      <button onClick={increment}>+1</button>
    </div>
  );
}

function CounterB() {
  const { count, increment, reset } = useCounter(100, 10);

  return (
    <div>
      <h3>カウンターB: {count}</h3>
      <button onClick={reset}>リセット</button>
      <button onClick={increment}>+10</button>
    </div>
  );
}
```

> **ポイント**：各コンポーネントでフックを呼ぶと、それぞれ独立したstateが作られます。stateが共有されるわけではありません。

## 3. useToggle：真偽値の切り替え

```jsx
import { useState, useCallback } from 'react';

function useToggle(initialValue = false) {
  const [value, setValue] = useState(initialValue);

  const toggle = useCallback(() => {
    setValue(prev => !prev);
  }, []);

  const setTrue = useCallback(() => setValue(true), []);
  const setFalse = useCallback(() => setValue(false), []);

  return { value, toggle, setTrue, setFalse };
}

// 使用例
function App() {
  const modal = useToggle(false);
  const darkMode = useToggle(false);

  return (
    <div style={{ backgroundColor: darkMode.value ? '#333' : '#fff', color: darkMode.value ? '#fff' : '#333' }}>
      <button onClick={darkMode.toggle}>
        {darkMode.value ? 'ライトモード' : 'ダークモード'}
      </button>

      <button onClick={modal.setTrue}>モーダルを開く</button>

      {modal.value && (
        <div style={{
          position: 'fixed', top: 0, left: 0, right: 0, bottom: 0,
          backgroundColor: 'rgba(0,0,0,0.5)',
          display: 'flex', alignItems: 'center', justifyContent: 'center',
        }}>
          <div style={{ backgroundColor: 'white', padding: '24px', borderRadius: '8px', color: '#333' }}>
            <h2>モーダルウィンドウ</h2>
            <p>これはモーダルです。</p>
            <button onClick={modal.setFalse}>閉じる</button>
          </div>
        </div>
      )}
    </div>
  );
}
```

## 4. useLocalStorage：ローカルストレージとの連携

```jsx
import { useState, useEffect } from 'react';

function useLocalStorage(key, initialValue) {
  // ローカルストレージから初期値を読み込む
  const [storedValue, setStoredValue] = useState(() => {
    try {
      const item = localStorage.getItem(key);
      return item ? JSON.parse(item) : initialValue;
    } catch (error) {
      console.error(`useLocalStorage: "${key}" の読み込みに失敗`, error);
      return initialValue;
    }
  });

  // 値が変わったらローカルストレージに保存
  useEffect(() => {
    try {
      localStorage.setItem(key, JSON.stringify(storedValue));
    } catch (error) {
      console.error(`useLocalStorage: "${key}" の保存に失敗`, error);
    }
  }, [key, storedValue]);

  return [storedValue, setStoredValue];
}

// 使用例：設定画面
function Settings() {
  const [theme, setTheme] = useLocalStorage('theme', 'light');
  const [fontSize, setFontSize] = useLocalStorage('fontSize', 16);
  const [username, setUsername] = useLocalStorage('username', '');

  return (
    <div style={{ fontSize: `${fontSize}px` }}>
      <h2>設定</h2>

      <div>
        <label>ユーザー名：</label>
        <input
          value={username}
          onChange={(e) => setUsername(e.target.value)}
        />
      </div>

      <div>
        <label>テーマ：</label>
        <select value={theme} onChange={(e) => setTheme(e.target.value)}>
          <option value="light">ライト</option>
          <option value="dark">ダーク</option>
        </select>
      </div>

      <div>
        <label>フォントサイズ：{fontSize}px</label>
        <input
          type="range"
          min="12"
          max="24"
          value={fontSize}
          onChange={(e) => setFontSize(Number(e.target.value))}
        />
      </div>

      <p>※ 設定はブラウザに自動保存されます</p>
    </div>
  );
}
```

## 5. useFetch：データ取得の汎用フック

```jsx
import { useState, useEffect } from 'react';

function useFetch(url) {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    // URLが空の場合は何もしない
    if (!url) {
      setLoading(false);
      return;
    }

    const controller = new AbortController();

    const fetchData = async () => {
      setLoading(true);
      setError(null);

      try {
        const response = await fetch(url, { signal: controller.signal });
        if (!response.ok) {
          throw new Error(`HTTP Error: ${response.status}`);
        }
        const json = await response.json();
        setData(json);
      } catch (err) {
        if (err.name !== 'AbortError') {
          setError(err.message);
        }
      } finally {
        setLoading(false);
      }
    };

    fetchData();

    return () => {
      controller.abort();
    };
  }, [url]);

  return { data, loading, error };
}

// 使用例：ユーザー一覧
function UserList() {
  const { data: users, loading, error } = useFetch(
    'https://jsonplaceholder.typicode.com/users'
  );

  if (loading) return <p>読み込み中...</p>;
  if (error) return <p style={{ color: 'red' }}>エラー：{error}</p>;

  return (
    <ul>
      {users.map(user => (
        <li key={user.id}>{user.name} ({user.email})</li>
      ))}
    </ul>
  );
}

// 使用例：動的URL
function UserDetail() {
  const [userId, setUserId] = useState(1);
  const { data: user, loading, error } = useFetch(
    `https://jsonplaceholder.typicode.com/users/${userId}`
  );

  return (
    <div>
      <select value={userId} onChange={(e) => setUserId(Number(e.target.value))}>
        {[1, 2, 3, 4, 5].map(id => (
          <option key={id} value={id}>ユーザー {id}</option>
        ))}
      </select>

      {loading && <p>読み込み中...</p>}
      {error && <p style={{ color: 'red' }}>{error}</p>}
      {user && (
        <div>
          <h3>{user.name}</h3>
          <p>{user.email}</p>
          <p>{user.phone}</p>
        </div>
      )}
    </div>
  );
}
```

## 6. useForm：フォーム管理フック

```jsx
import { useState } from 'react';

function useForm(initialValues, validate) {
  const [values, setValues] = useState(initialValues);
  const [errors, setErrors] = useState({});
  const [touched, setTouched] = useState({});

  const handleChange = (e) => {
    const { name, value, type, checked } = e.target;
    setValues(prev => ({
      ...prev,
      [name]: type === 'checkbox' ? checked : value,
    }));
  };

  const handleBlur = (e) => {
    const { name } = e.target;
    setTouched(prev => ({ ...prev, [name]: true }));

    // バリデーション実行
    if (validate) {
      const validationErrors = validate(values);
      setErrors(validationErrors);
    }
  };

  const handleSubmit = (onSubmit) => (e) => {
    e.preventDefault();

    // 全フィールドをtouchedにする
    const allTouched = {};
    Object.keys(values).forEach(key => { allTouched[key] = true; });
    setTouched(allTouched);

    // バリデーション
    if (validate) {
      const validationErrors = validate(values);
      setErrors(validationErrors);
      if (Object.keys(validationErrors).length > 0) return;
    }

    onSubmit(values);
  };

  const reset = () => {
    setValues(initialValues);
    setErrors({});
    setTouched({});
  };

  return { values, errors, touched, handleChange, handleBlur, handleSubmit, reset };
}

// 使用例
function ContactForm() {
  const validate = (values) => {
    const errors = {};
    if (!values.name) errors.name = '名前は必須です';
    if (!values.email) errors.email = 'メールは必須です';
    else if (!/\S+@\S+\.\S+/.test(values.email)) errors.email = 'メール形式が正しくありません';
    if (!values.message) errors.message = 'メッセージは必須です';
    return errors;
  };

  const { values, errors, touched, handleChange, handleBlur, handleSubmit, reset } = useForm(
    { name: '', email: '', message: '' },
    validate
  );

  const onSubmit = (data) => {
    alert(`送信完了！\n${JSON.stringify(data, null, 2)}`);
    reset();
  };

  return (
    <form onSubmit={handleSubmit(onSubmit)}>
      <div>
        <label>名前：</label>
        <input name="name" value={values.name} onChange={handleChange} onBlur={handleBlur} />
        {touched.name && errors.name && <p style={{ color: 'red' }}>{errors.name}</p>}
      </div>
      <div>
        <label>メール：</label>
        <input name="email" value={values.email} onChange={handleChange} onBlur={handleBlur} />
        {touched.email && errors.email && <p style={{ color: 'red' }}>{errors.email}</p>}
      </div>
      <div>
        <label>メッセージ：</label>
        <textarea name="message" value={values.message} onChange={handleChange} onBlur={handleBlur} />
        {touched.message && errors.message && <p style={{ color: 'red' }}>{errors.message}</p>}
      </div>
      <button type="submit">送信</button>
      <button type="button" onClick={reset}>リセット</button>
    </form>
  );
}
```

## 7. フックのルール

Reactのフック（useState、useEffect、カスタムフック）には厳格なルールがあります。

### ルール1：トップレベルでのみ呼び出す

```jsx
// NG：条件分岐の中でフックを呼ぶ
function Bad({ isLoggedIn }) {
  if (isLoggedIn) {
    const [name, setName] = useState(""); // NG！
  }

  // NG：ループの中
  for (let i = 0; i < 3; i++) {
    useEffect(() => { ... }); // NG！
  }

  // NG：早期リターンの後
  if (!isLoggedIn) return <p>ログインしてください</p>;
  const [data, setData] = useState(null); // NG！
}

// OK：常にトップレベルで呼ぶ
function Good({ isLoggedIn }) {
  const [name, setName] = useState("");
  const [data, setData] = useState(null);

  useEffect(() => {
    if (isLoggedIn) {
      // 条件分岐はフックの中で行う
    }
  }, [isLoggedIn]);

  if (!isLoggedIn) return <p>ログインしてください</p>;

  return <div>{name}</div>;
}
```

### ルール2：React関数内でのみ呼び出す

```jsx
// OK：関数コンポーネント内
function MyComponent() {
  const [count, setCount] = useState(0); // OK
  return <p>{count}</p>;
}

// OK：カスタムフック内
function useMyHook() {
  const [value, setValue] = useState(""); // OK
  return [value, setValue];
}

// NG：通常の関数内
function regularFunction() {
  const [count, setCount] = useState(0); // NG！
}
```

### なぜルールが必要？

Reactはフックの呼び出し順序に依存してstateを管理しています。条件分岐やループでフックの呼び出し回数が変わると、stateの対応がずれてバグが発生します。

| 1回目のレンダリング | 2回目のレンダリング（条件が変わった場合） |
|----|----|
| useState("太郎") → name | useState("太郎") → name |
| useState(0) → count | ~~スキップされた~~ → ずれが発生！ |
| useEffect → ... | useState(0) → 間違った対応 |

## 8. カスタムフック設計のコツ

### 単一責任の原則

```jsx
// NG：1つのフックに詰め込みすぎ
function useEverything() {
  const [user, setUser] = useState(null);
  const [theme, setTheme] = useState('light');
  const [notifications, setNotifications] = useState([]);
  // ... 大量のロジック
}

// OK：役割ごとに分ける
function useAuth() { ... }
function useTheme() { ... }
function useNotifications() { ... }
```

### 適切な返り値の設計

```jsx
// 値が1つの場合：配列で返す（useStateと同じ形式）
function useToggle(initial) {
  const [value, setValue] = useState(initial);
  const toggle = () => setValue(prev => !prev);
  return [value, toggle];
}
const [isOpen, toggleOpen] = useToggle(false);

// 値が多い場合：オブジェクトで返す
function useFetch(url) {
  // ...
  return { data, loading, error, refetch };
}
const { data, loading, error } = useFetch(url);
```

## まとめ

| 項目 | ポイント |
|------|---------|
| カスタムフック | ロジックを再利用可能な関数に切り出す仕組み |
| 命名規則 | 必ず `use` で始める |
| state独立性 | 各コンポーネントで独立したstateが作られる |
| useLocalStorage | ローカルストレージとstateを同期する |
| useFetch | API取得のロジックを汎用化する |
| フックのルール | トップレベルでのみ、React関数内でのみ呼び出す |
| 設計のコツ | 単一責任、適切な返り値の形式 |
