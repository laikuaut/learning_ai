```html
<!DOCTYPE html>
<html lang="ja">
<head>
  <meta charset="UTF-8">
  <title>アサーション練習</title>
  <style>
    body { font-family: sans-serif; max-width: 600px; margin: 40px auto; }
    .card { border: 1px solid #ccc; padding: 16px; margin: 12px 0; border-radius: 8px; }
    .hidden { display: none; }
    .active { background-color: #e6f3ff; border-color: #0066cc; }
    .error { color: red; font-weight: bold; }
    .success { color: green; font-weight: bold; }
    button { padding: 8px 16px; margin: 4px; cursor: pointer; }
    #counter-value { font-size: 2em; font-weight: bold; }
  </style>
</head>
<body>
  <h1>アサーション練習ページ</h1>

  <!-- トグル表示 -->
  <div class="card">
    <h2>表示/非表示の切り替え</h2>
    <button id="toggle-btn" onclick="toggleMessage()">メッセージを表示</button>
    <p id="toggle-message" class="hidden">こんにちは！これは切り替え可能なメッセージです。</p>
  </div>

  <!-- テキスト変化 -->
  <div class="card">
    <h2>ステータス表示</h2>
    <p id="status" class="success">接続済み</p>
    <button onclick="changeStatus('error')">エラーにする</button>
    <button onclick="changeStatus('success')">成功にする</button>
  </div>

  <!-- カウンター -->
  <div class="card" id="counter-card">
    <h2>カウンター</h2>
    <p id="counter-value" data-count="0">0</p>
    <button onclick="increment()">+1</button>
    <button onclick="decrement()">-1</button>
    <button onclick="resetCounter()">リセット</button>
  </div>

  <!-- リスト -->
  <div class="card">
    <h2>タスクリスト</h2>
    <ul id="task-list">
      <li>牛乳を買う</li>
      <li>メールを返す</li>
      <li>レポートを書く</li>
    </ul>
    <button onclick="addTask()">タスク追加</button>
  </div>

  <script>
    function toggleMessage() {
      const msg = document.getElementById('toggle-message');
      const btn = document.getElementById('toggle-btn');
      if (msg.classList.contains('hidden')) {
        msg.classList.remove('hidden');
        btn.textContent = 'メッセージを隠す';
      } else {
        msg.classList.add('hidden');
        btn.textContent = 'メッセージを表示';
      }
    }

    function changeStatus(type) {
      const status = document.getElementById('status');
      if (type === 'error') {
        status.textContent = 'エラー発生';
        status.className = 'error';
      } else {
        status.textContent = '接続済み';
        status.className = 'success';
      }
    }

    let count = 0;
    function increment() {
      count++;
      updateCounter();
    }
    function decrement() {
      count--;
      updateCounter();
    }
    function resetCounter() {
      count = 0;
      updateCounter();
    }
    function updateCounter() {
      const el = document.getElementById('counter-value');
      el.textContent = count;
      el.setAttribute('data-count', count);
    }

    let taskCount = 3;
    function addTask() {
      taskCount++;
      const li = document.createElement('li');
      li.textContent = '新しいタスク' + taskCount;
      document.getElementById('task-list').appendChild(li);
    }
  </script>
</body>
</html>
```

## 課題の要件

1. 初期状態でh1見出し「アサーション練習ページ」が表示されていることを検証する
2. トグルボタンをクリックしてメッセージが表示/非表示になることを検証する（`toBeVisible` / `toBeHidden`）
3. ステータスのテキストを `toHaveText` で完全一致検証する
4. ステータスのテキストを正規表現で部分一致検証する
5. ステータス変更後のクラス属性を `toHaveClass` で検証する
6. カウンターの増減を検証する（`data-count` 属性の検証も含む）
7. タスク追加ボタンで `<li>` の数が増えることを `toHaveCount` で検証する
8. 否定アサーション（`.not`）を使って非表示要素が見えていないことを検証する

## ステップガイド

<details>
<summary>ステップ1：Webアサーションと汎用アサーションの違いを意識する</summary>

```typescript
// Webアサーション（自動リトライ、await必須）
await expect(page.locator('#status')).toHaveText('接続済み');

// 汎用アサーション（即時判定、await不要）
const count = await page.locator('#counter-value').textContent();
expect(count).toBe('0');
```

基本的にWebアサーション（`await expect(locator).toXxx()`）を優先してください。UIの変化にはリトライが必要なためです。

// 実行方法：npx playwright test tests/assertion-drill.spec.ts --project=chromium

import { test, expect } from '@playwright/test';
import path from 'path';

test.describe('要素の表示状態', () => {

  test.beforeEach(async ({ page }) => {
    const filePath = path.resolve(__dirname, '../assertion-page.html');

  test('初期表示で見出しが見えている', async ({ page }) => {
    const heading = page.getByRole('heading', { level: 1 });

    await expect(heading).toHaveText('アサーション練習ページ');
  });

  test('ボタンクリックで要素が表示/非表示になる', async ({ page }) => {
    const message = page.locator('#toggle-message');
    const button = page.locator('#toggle-btn');

    // 初期状態：非表示
    await expect(message).toBeHidden();

    // ボタンクリック：表示になる
    await button.click();
    await expect(message).toBeVisible();
    await expect(message).toHaveText('こんにちは！これは切り替え可能なメッセージです。');

    // もう一度クリック：非表示になる
    await button.click();
    await expect(message).toBeHidden();
  });
});

test.describe('テキストの検証', () => {

  test.beforeEach(async ({ page }) => {
    const filePath = path.resolve(__dirname, '../assertion-page.html');
    await page.goto(`file://${filePath}`);
  });

  test('完全一致で検証できる', async ({ page }) => {
    await expect(page.locator('#status')).toHaveText('接続済み');
  });

  test('正規表現で検証できる', async ({ page }) => {
    await expect(page.locator('#status')).toHaveText(/接続/);

<summary>解答例（改良版 ─ テスト構造の最適化）</summary>

```typescript
// tests/assertion-drill-advanced.spec.ts
// 改良版：beforeEachの共通化とソフトアサーション
// 実行方法：npx playwright test tests/assertion-drill-advanced.spec.ts --project=chromium

import { test, expect } from '@playwright/test';
import path from 'path';

// テストファイルのパスを定数化
const TEST_PAGE = path.resolve(__dirname, '../assertion-page.html');

// 全テスト共通のbeforeEach
test.beforeEach(async ({ page }) => {
  await page.goto(`file://${TEST_PAGE}`);
});

test.describe('トグル機能', () => {

  test('表示/非表示を連続で切り替えられる', async ({ page }) => {
    const message = page.locator('#toggle-message');
    const button = page.locator('#toggle-btn');

    // 3回連続トグル
    for (let i = 0; i < 3; i++) {
      await button.click();
      if (i % 2 === 0) {
        await expect(message).toBeVisible();
        await expect(button).toHaveText('メッセージを隠す');
      } else {
        await expect(message).toBeHidden();
        await expect(button).toHaveText('メッセージを表示');
      }
    }
  });
});

test.describe('ステータス変更', () => {

  test('ステータスの変更と復帰を検証する', async ({ page }) => {
    const status = page.locator('#status');

    // 初期状態
    await expect(status).toHaveText('接続済み');
    await expect(status).toHaveClass('success');

    // エラーに変更
    await page.getByRole('button', { name: 'エラーにする' }).click();
    await expect(status).toHaveText('エラー発生');
    await expect(status).toHaveClass('error');

    // 成功に戻す
    await page.getByRole('button', { name: '成功にする' }).click();
    await expect(status).toHaveText('接続済み');
    await expect(status).toHaveClass('success');
  });
});

test.describe('カウンター', () => {

  test('複数回の操作で正しい値になる', async ({ page }) => {
    const counter = page.locator('#counter-value');
    const plusBtn = page.getByRole('button', { name: '+1' });
    const minusBtn = page.getByRole('button', { name: '-1' });

    // 5回加算
    for (let i = 0; i < 5; i++) {
      await plusBtn.click();
    }
    await expect(counter).toHaveText('5');

    // 2回減算
    for (let i = 0; i < 2; i++) {
      await minusBtn.click();
    }
    await expect(counter).toHaveText('3');

    // テキストとdata属性が一致することを検証
    const text = await counter.textContent();
    await expect(counter).toHaveAttribute('data-count', text!);
  });
});

test.describe('ソフトアサーション', () => {

  test('複数の検証をまとめて実行する', async ({ page }) => {
    // ソフトアサーション: 1つが失敗しても残りの検証を続行する
    await expect.soft(page.getByRole('heading', { level: 1 })).toHaveText('アサーション練習ページ');
    await expect.soft(page.locator('#status')).toHaveText('接続済み');
    await expect.soft(page.locator('#counter-value')).toHaveText('0');
    await expect.soft(page.locator('#task-list li')).toHaveCount(3);
  });
});
```

**初心者向けとの違い:**
- `test.beforeEach` をファイルレベルで1箇所にまとめ、各 `describe` での重複を排除
- ループを使って繰り返し操作をテスト（トグル、カウンター）
- `expect.soft()` （ソフトアサーション）を使い、1つの失敗で全体が止まらないテストを紹介
- テキストと属性の一致を動的に検証するパターンを使用
