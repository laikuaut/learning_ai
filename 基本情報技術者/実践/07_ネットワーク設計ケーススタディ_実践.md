## ステップガイド

<details>
<summary>ステップ1：IPアドレスの2進数表現を理解する</summary>

IPアドレスは32ビットの数値を8ビットずつ4つに区切ったものです。

```
192.168.1.100
= 11000000.10101000.00000001.01100100

サブネットマスク 255.255.255.0
= 11111111.11111111.11111111.00000000

ネットワークアドレス = IPアドレス AND サブネットマスク
= 11000000.10101000.00000001.00000000
= 192.168.1.0
```

</details>

<details>
<summary>ステップ2：Pythonでビット演算を行う</summary>

```python
def ip_to_int(ip_str):
    """IPアドレス文字列を32ビット整数に変換"""
    parts = ip_str.split(".")
    return (int(parts[0]) << 24) | (int(parts[1]) << 16) | \
           (int(parts[2]) << 8) | int(parts[3])

def int_to_ip(num):
    """32ビット整数をIPアドレス文字列に変換"""
    return f"{(num >> 24) & 0xFF}.{(num >> 16) & 0xFF}.{(num >> 8) & 0xFF}.{num & 0xFF}"

def cidr_to_mask(prefix):
    """CIDR表記からサブネットマスクを計算"""
    return (0xFFFFFFFF << (32 - prefix)) & 0xFFFFFFFF
```

</details>

<details>
<summary>ステップ3：ネットワーク計算の実装</summary>

```python
# ネットワークアドレス = IP AND マスク
network = ip_int & mask_int

# ブロードキャスト = ネットワーク OR (NOT マスク)
broadcast = network | (~mask_int & 0xFFFFFFFF)

# ホスト数 = ブロードキャスト - ネットワーク - 1
host_count = broadcast - network - 1
```

</details>

<summary>ネットワーク計算問題の解答</summary>

**Q1.** IP: 192.168.1.100 / マスク: 255.255.255.0 (/24)
- ネットワークアドレス: **192.168.1.0**
- ブロードキャストアドレス: **192.168.1.255**
- 使用可能ホスト数: **254台**（256 - 2）

**Q2.** IP: 172.16.50.200/20
- サブネットマスク: **255.255.240.0**（/20 → 上位20ビットが1）
- ネットワークアドレス: 172.16.50.200 AND 255.255.240.0 = **172.16.48.0**
- ブロードキャストアドレス: **172.16.63.255**（172.16.48.0 + 2^12 - 1）

**Q3.** 最低30台 → 30 + 2（ネットワーク＋ブロードキャスト）= 32 → 2^5 = 32
- サブネットマスク: **255.255.255.224** (/27)
- ホスト部5ビット → 各サブネット30台
- 最大サブネット数: 2^(27-24) = **8個**

```python
# IPアドレス・サブネット計算ツール
# 学べる内容：IPアドレス、サブネットマスク、ビット演算
# 実行方法：python 07_ip_calculator.py

def ip_to_int(ip_str):
    """IPアドレス文字列を整数に変換"""
    parts = ip_str.split(".")
    result = 0
    for part in parts:
        result = result * 256 + int(part)
    return result

def int_to_ip(num):
    """整数をIPアドレス文字列に変換"""
    parts = []
    for i in range(4):
        parts.append(str(num & 255))
        num = num >> 8
    parts.reverse()
    return ".".join(parts)

def int_to_binary(num):
    """整数を2進数表記に変換（8ビットずつ区切り）"""
    binary = format(num, '032b')
    return f"{binary[0:8]}.{binary[8:16]}.{binary[16:24]}.{binary[24:32]}"

# --- メイン ---
print("===== IPアドレス・サブネット計算ツール =====\n")

ip_str = input("IPアドレス (例: 192.168.10.130): ")

mask_input = input("サブネットマスク (例: 255.255.255.192 または /26): ")

if mask_input.startswith("/"):
    # CIDR表記
    prefix = int(mask_input[1:])
    mask_int = (0xFFFFFFFF << (32 - prefix)) & 0xFFFFFFFF
    mask_str = int_to_ip(mask_int)
else:
    mask_str = mask_input
    mask_int = ip_to_int(mask_str)
    # プレフィックス長を計算
    prefix = bin(mask_int).count("1")

ip_int = ip_to_int(ip_str)

# 計算
network_int = ip_int & mask_int
broadcast_int = network_int | (~mask_int & 0xFFFFFFFF)
first_host = network_int + 1
last_host = broadcast_int - 1
host_count = broadcast_int - network_int - 1

# 結果表示
print(f"\n--- 計算結果 ---")
print(f"IPアドレス:           {ip_str}")
print(f"  (2進数)             {int_to_binary(ip_int)}")
print(f"サブネットマスク:     {mask_str} (/{prefix})")
print(f"  (2進数)             {int_to_binary(mask_int)}")
print(f"ネットワークアドレス: {int_to_ip(network_int)}")
print(f"ブロードキャスト:     {int_to_ip(broadcast_int)}")
print(f"ホスト範囲:           {int_to_ip(first_host)} ～ {int_to_ip(last_host)}")
print(f"使用可能ホスト数:     {host_count}台")
```

<summary>解答例（改良版 ─ 同一サブネット判定・VLSM設計支援）</summary>

```python
# IPアドレス計算ツール（改良版）
# 同一サブネット判定・VLSM設計支援機能付き

import math


def ip_to_int(ip_str):
    parts = [int(p) for p in ip_str.split(".")]
    return (parts[0] << 24) | (parts[1] << 16) | (parts[2] << 8) | parts[3]


def int_to_ip(num):
    return f"{(num>>24)&0xFF}.{(num>>16)&0xFF}.{(num>>8)&0xFF}.{num&0xFF}"


def cidr_to_mask(prefix):
    return (0xFFFFFFFF << (32 - prefix)) & 0xFFFFFFFF


def calc_subnet(ip_str, prefix):
    """サブネット情報を計算して辞書で返す"""
    ip_int = ip_to_int(ip_str)
    mask = cidr_to_mask(prefix)
    network = ip_int & mask
    broadcast = network | (~mask & 0xFFFFFFFF)
    return {
        "network": int_to_ip(network),
        "broadcast": int_to_ip(broadcast),
        "first": int_to_ip(network + 1),
        "last": int_to_ip(broadcast - 1),
        "hosts": broadcast - network - 1,
        "prefix": prefix,
        "mask": int_to_ip(mask),
    }


def subnet_calculator():
    """基本的なサブネット計算"""
    print("\n--- サブネット計算 ---")
    ip = input("IPアドレス: ")
    mask_input = input("マスク (/CIDR または 10進数): ")

    if mask_input.startswith("/"):
        prefix = int(mask_input[1:])
    else:
        prefix = bin(ip_to_int(mask_input)).count("1")

    info = calc_subnet(ip, prefix)
    print(f"\n  ネットワーク:  {info['network']}/{info['prefix']}")
    print(f"  マスク:        {info['mask']}")
    print(f"  ブロードキャスト: {info['broadcast']}")
    print(f"  ホスト範囲:    {info['first']} ～ {info['last']}")
    print(f"  ホスト数:      {info['hosts']}台")


def same_subnet_check():
    """同一サブネット判定"""
    print("\n--- 同一サブネット判定 ---")
    ip1 = input("IPアドレス1: ")
    ip2 = input("IPアドレス2: ")
    prefix = int(input("プレフィックス長 (/XX): "))

    mask = cidr_to_mask(prefix)
    net1 = ip_to_int(ip1) & mask
    net2 = ip_to_int(ip2) & mask

    if net1 == net2:
        print(f"\n  ✓ 同一サブネット（{int_to_ip(net1)}/{prefix}）")
    else:
        print(f"\n  ✗ 異なるサブネット")
        print(f"    {ip1} → {int_to_ip(net1)}/{prefix}")
        print(f"    {ip2} → {int_to_ip(net2)}/{prefix}")


def vlsm_design():
    """VLSM設計支援"""
    print("\n--- VLSM サブネット設計 ---")
    base_ip = input("ベースネットワーク (例: 10.0.0.0): ")
    base_prefix = int(input("プレフィックス長 (例: 24): "))

    print("各部署の必要ホスト数を入力してください（空行で終了）:")
    departments = []
    while True:
        name = input("  部署名 (空でEnd): ").strip()
        if not name:
            break
        hosts = int(input(f"  {name}の必要ホスト数: "))
        departments.append((name, hosts))

    # 大きい順にソート
    departments.sort(key=lambda x: x[1], reverse=True)

    # VLSM割り当て
    current = ip_to_int(base_ip)
    max_addr = current + (1 << (32 - base_prefix))

    print(f"\n--- VLSM設計結果 ---")
    print(f"{'部署':<12} {'サブネット':<22} {'マスク':<8} {'ホスト数':<10}")
    print("-" * 55)

    for name, needed in departments:
        # 必要なホスト部のビット数
        host_bits = math.ceil(math.log2(needed + 2))  # +2はネットワーク＋ブロードキャスト
        subnet_prefix = 32 - host_bits
        subnet_size = 1 << host_bits

        if current + subnet_size > max_addr:
            print(f"  ※ {name}: アドレス空間が不足しています")
            continue

        network = int_to_ip(current)
        usable = subnet_size - 2
        print(f"{name:<12} {network + '/' + str(subnet_prefix):<22} /{subnet_prefix:<6} {usable}台")

        current += subnet_size

    remaining = max_addr - current
    print(f"\n未使用アドレス: {remaining}個")


def main():
    print("===== ネットワーク計算ツール =====")
    print("  1. サブネット計算")
    print("  2. 同一サブネット判定")
    print("  3. VLSM設計支援")

    choice = input("\n選択 (1-3): ").strip()
    if choice == "1":
        subnet_calculator()
    elif choice == "2":
        same_subnet_check()
    elif choice == "3":
        vlsm_design()
    else:
        print("無効な選択です。")


if __name__ == "__main__":
    main()
```

**初心者向けとの違い:**
- 同一サブネット判定機能 → 試験でよく出るパターンを実践
- VLSM設計支援 → 部署名とホスト数を入力するだけで自動設計
- 大きいサブネットから順に割り当て → 試験の解法と同じ手順
- 関数分離で各機能が独立 → 保守性が高い
