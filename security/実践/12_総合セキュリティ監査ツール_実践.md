## 課題の要件

### 必須機能

1. **パスワードセキュリティ監査**: 登録ユーザーのパスワード強度チェック、辞書攻撃耐性テスト、有効期限チェック
2. **暗号化・ハッシュ検証**: ハッシュアルゴリズムの安全性チェック、データの完全性検証
3. **アクセスログ監視**: ブルートフォース検知、攻撃パターン検知、異常アクセスの集計
4. **システム設定チェック**: TLS設定、開放ポート、ファイルパーミッションの検査
5. **脆弱性スキャン（シミュレーション）**: XSS、SQLインジェクション、CSRF等の脆弱性チェック
6. **総合監査**: 上記5つを一括実行し、総合スコアとレポートを生成
7. **レポート出力**: テキスト形式の監査レポートをファイルに出力

### 設計要件

- 各監査モジュールが独立しており、個別にも一括でも実行できる
- 監査結果を統一的なデータ構造で管理する
- 監査履歴を保持し、過去の結果と比較できる

COMMON_PASSWORDS = [

def audit_passwords(users):
    """パスワードセキュリティ監査を実行します。"""
    results = []

    for username, data in users.items():
        password = data["plain_for_audit"]

        # 強度チェック
        checks = {
            "長さ8以上": len(password) >= 8,
            "大文字": any(c.isupper() for c in password),
            "小文字": any(c.islower() for c in password),
            "数字": any(c.isdigit() for c in password),
            "特殊文字": any(c in "!@#$%^&*()-_=+" for c in password),
        }
        score = sum(checks.values())

        if score <= 2:
            results.append(AuditResult(
                "パスワード", username, AuditResult.NG,
                f"弱いパスワード（スコア: {score}/5）"
            ))
        elif score <= 3:
            results.append(AuditResult(
                "パスワード", username, AuditResult.WARN,
                f"パスワード強度が不十分（スコア: {score}/5）"
            ))
        else:
            results.append(AuditResult(
                "パスワード", username, AuditResult.OK,
                f"パスワード強度は十分です（スコア: {score}/5）"
            ))

        # 辞書攻撃チェック
        if password.lower() in COMMON_PASSWORDS:
            results.append(AuditResult(
                "パスワード", f"{username}（辞書攻撃）", AuditResult.NG,
                "辞書に含まれるパスワードです（即座に解読可能）"

        # 有効期限チェック
        age = data["password_age_days"]
        if age > MAX_PASSWORD_AGE_DAYS:
            results.append(AuditResult(
                "パスワード", f"{username}（有効期限）", AuditResult.WARN,
                f"パスワード変更から{age}日経過（推奨: {MAX_PASSWORD_AGE_DAYS}日以内）"
            ))

    return results


def audit_crypto():
    """暗号化・ハッシュ検証を実行します。"""
    results = []

    # ハッシュアルゴリズムの安全性チェック
    test_data = b"test_data_for_audit"

    safe_algos = {"sha256": "SHA-256", "sha384": "SHA-384", "sha512": "SHA-512"}
    unsafe_algos = {"md5": "MD5", "sha1": "SHA-1"}

    # 安全なアルゴリズムのテスト
    for algo, name in safe_algos.items():
        try:
            h = hashlib.new(algo, test_data)
            results.append(AuditResult(
                "暗号化", f"{name}ハッシュ", AuditResult.OK,
                f"利用可能（{len(h.hexdigest())}文字）"
            ))
        except Exception as e:
            results.append(AuditResult(
                "暗号化", f"{name}ハッシュ", AuditResult.NG,
                f"エラー: {e}"
            ))

    # 非推奨アルゴリズムの検出（使用されていたら警告）
    for algo, name in unsafe_algos.items():
        results.append(AuditResult(
            "暗号化", f"{name}の使用確認", AuditResult.WARN,
            f"{name}は非推奨です。SHA-256以上を使用してください"
        ))

    # データ完全性検証のデモ
    original = "重要なデータ"
    original_hash = hashlib.sha256(original.encode()).hexdigest()
    verify_hash = hashlib.sha256(original.encode()).hexdigest()

    if original_hash == verify_hash:
        results.append(AuditResult(
            "暗号化", "データ完全性", AuditResult.OK,
            "SHA-256ハッシュによる完全性検証: 正常"
        ))

    return results


def audit_logs(logs):
    """アクセスログ監視を実行します。"""
    results = []

    # ブルートフォース検知
    failures_by_ip = {}
    for log in logs:
        if log["status"] in [401, 403]:
            ip = log["ip"]
            failures_by_ip[ip] = failures_by_ip.get(ip, 0) + 1

    for ip, count in failures_by_ip.items():
        if count >= 10:
            results.append(AuditResult(
                "ログ監視", f"ブルートフォース({ip})", AuditResult.NG,
                f"{count}回の認証失敗（ブルートフォース攻撃の兆候）"
            ))
        elif count >= 5:
            results.append(AuditResult(
                "ログ監視", f"認証失敗({ip})", AuditResult.WARN,
                f"{count}回の認証失敗（要監視）"
            ))

    # 攻撃パターン検知
    sqli_keywords = ["UNION", "SELECT", "DROP", "OR 1=1", "'--"]
    xss_keywords = ["<script", "javascript:", "onerror="]

    sqli_count = 0
    xss_count = 0
    for log in logs:
        path = log["path"].upper()
        if any(kw.upper() in path for kw in sqli_keywords):
            sqli_count += 1
        path_lower = log["path"].lower()
        if any(kw.lower() in path_lower for kw in xss_keywords):
            xss_count += 1

    if sqli_count > 0:
        results.append(AuditResult(
            "ログ監視", "SQLインジェクション", AuditResult.NG,
            f"{sqli_count}件のSQLインジェクション攻撃パターンを検知"
        ))
    else:
        results.append(AuditResult(
            "ログ監視", "SQLインジェクション", AuditResult.OK,
            "SQLインジェクション攻撃パターンは検知されませんでした"
        ))

    if xss_count > 0:
        results.append(AuditResult(
            "ログ監視", "XSS攻撃", AuditResult.WARN,
            f"{xss_count}件のXSS攻撃パターンを検知"
        ))
    else:
        results.append(AuditResult(
            "ログ監視", "XSS攻撃", AuditResult.OK,
            "XSS攻撃パターンは検知されませんでした"
        ))

    # エラー率
    error_count = sum(1 for log in logs if log["status"] >= 400)
    error_rate = error_count / len(logs) * 100 if logs else 0
    if error_rate > 20:
        results.append(AuditResult(
            "ログ監視", "エラー率", AuditResult.NG,
            f"エラー率: {error_rate:.1f}%（異常に高い）"
        ))
    elif error_rate > 10:
        results.append(AuditResult(
            "ログ監視", "エラー率", AuditResult.WARN,
            f"エラー率: {error_rate:.1f}%（やや高い）"
        ))
    else:
        results.append(AuditResult(
            "ログ監視", "エラー率", AuditResult.OK,
            f"エラー率: {error_rate:.1f}%（正常範囲）"
        ))

    return results


def audit_system():
    """システム設定チェックを実行します。"""
    results = []

    # TLS設定（シミュレーション）
    tls_version = "TLS 1.2"
    tls_levels = {
        "TLS 1.3": (AuditResult.OK, "最新バージョン"),
        "TLS 1.2": (AuditResult.OK, "推奨バージョン"),
        "TLS 1.1": (AuditResult.WARN, "非推奨（2021年に廃止）"),
        "TLS 1.0": (AuditResult.NG, "脆弱（使用禁止）"),
    }
    level, msg = tls_levels.get(tls_version, (AuditResult.NG, "不明"))
    results.append(AuditResult("システム設定", f"TLS({tls_version})", level, msg))

    # 開放ポート（シミュレーション）
    open_ports = [22, 80, 443, 3306, 23]
    dangerous = {
        21: ("FTP", AuditResult.NG),
        23: ("Telnet", AuditResult.NG),
        3306: ("MySQL", AuditResult.WARN),
        3389: ("RDP", AuditResult.WARN),
    }
    for port in open_ports:
        if port in dangerous:
            service, level = dangerous[port]
            results.append(AuditResult(
                "システム設定", f"ポート{port}({service})", level,
                f"{service}が外部に公開されています"
            ))
        else:
            results.append(AuditResult(
                "システム設定", f"ポート{port}", AuditResult.OK, "問題なし"
            ))

    # ファイルパーミッション（シミュレーション）
    file_checks = [
        ("/etc/shadow", 0o640, 0o666),      # 不正
        ("/etc/ssh/sshd_config", 0o600, 0o600),  # 正常
    ]
    for filepath, expected, actual in file_checks:
        if actual == expected:
            results.append(AuditResult(
                "システム設定", filepath, AuditResult.OK,
                f"パーミッション正常 ({oct(actual)})"
            ))
        else:
            results.append(AuditResult(
                "システム設定", filepath, AuditResult.WARN,
                f"パーミッション不正（期待: {oct(expected)}, 実際: {oct(actual)}）"
            ))

    return results

# --- スコア計算とレポート ---

def calculate_score(results):
    """0〜100の総合スコアを算出します。"""
    if not results:
        return 0
    total = len(results)
    ok = sum(1 for r in results if r.level == AuditResult.OK)
    warn = sum(1 for r in results if r.level == AuditResult.WARN)
    ng = sum(1 for r in results if r.level == AuditResult.NG)
    # OK=100点, WARN=50点, NG=0点 の重み付き平均
    score = (ok * 100 + warn * 50) / total
    # NG1件あたり5点のペナルティ
    score = max(0, score - ng * 5)
    return int(score)


def get_verdict(score):
    """スコアから総合判定を返します。"""
    if score >= 80:
        return "合格（良好な状態）"
    elif score >= 60:
        return "条件付き合格（軽微な改善が必要）"
    elif score >= 40:
        return "要改善（重大なリスクあり）"
    else:
        return "不合格（即座に対応が必要）"


def print_results(results, module_name=None):
    """監査結果を表示します。"""
    if module_name:
        filtered = [r for r in results if r.module == module_name]
    else:
        filtered = results

    for r in filtered:
        print(r)

class AuditResult:
    """1件の監査結果を表すクラスです。"""
    OK = "OK"
    WARN = "WARN"
    NG = "NG"

    WEIGHTS = {OK: 100, WARN: 50, NG: 0}

    def __init__(self, module, item, level, message):
        self.module = module
        self.item = item
        self.level = level
        self.message = message
        self.timestamp = datetime.now()

    def __str__(self):
        return f"  [{self.level:<4}] {self.item}: {self.message}"

    def to_dict(self):
        return {
            "module": self.module,
            "item": self.item,
            "level": self.level,
            "message": self.message,
            "timestamp": self.timestamp.isoformat(),
        }


class AuditModule:
    """監査モジュールの基底クラスです。
    新しい監査項目を追加するには、このクラスを継承して
    name と run() メソッドを実装します。
    """

    name = "基底モジュール"

    def __init__(self, context):
        self.context = context  # 共有データ（ユーザー情報、ログ等）

    def run(self):
        """監査を実行し、AuditResult のリストを返します。"""
        raise NotImplementedError

    def result(self, item, level, message):
        """AuditResult を簡潔に生成するヘルパーです。"""
        return AuditResult(self.name, item, level, message)


class PasswordAuditModule(AuditModule):
    """パスワードセキュリティ監査モジュールです。"""

    name = "パスワード監査"

    COMMON_PASSWORDS = {
        "password", "123456", "admin", "letmein", "welcome",
        "monkey", "dragon", "master", "qwerty", "admin123",
        "password1", "abc123", "iloveyou", "trustno1",
    }
    MAX_AGE_DAYS = 90

    def run(self):
        results = []
        users = self.context.get("users", {})

        for username, data in users.items():
            pw = data.get("plain_for_audit", "")
            score = self._strength_score(pw)

            if score <= 2:
                results.append(self.result(username, AuditResult.NG,
                    f"弱いパスワード（{score}/5）"))
            elif score <= 3:
                results.append(self.result(username, AuditResult.WARN,
                    f"強度不十分（{score}/5）"))
            else:
                results.append(self.result(username, AuditResult.OK,
                    f"強度十分（{score}/5）"))

            if pw.lower() in self.COMMON_PASSWORDS:
                results.append(self.result(
                    f"{username}(辞書)", AuditResult.NG, "辞書攻撃で解読可能"))

            age = data.get("password_age_days", 0)
            if age > self.MAX_AGE_DAYS:
                results.append(self.result(
                    f"{username}(期限)", AuditResult.WARN,
                    f"{age}日経過（推奨: {self.MAX_AGE_DAYS}日以内）"))

        return results

    def _strength_score(self, password):
        checks = [
            len(password) >= 8,
            any(c.isupper() for c in password),
            any(c.islower() for c in password),
            any(c.isdigit() for c in password),
            any(c in "!@#$%^&*()-_=+" for c in password),
        ]
        return sum(checks)


class CryptoAuditModule(AuditModule):
    """暗号化・ハッシュ検証モジュールです。"""

    name = "暗号化検証"

    SAFE = {"sha256": "SHA-256", "sha384": "SHA-384", "sha512": "SHA-512"}
    UNSAFE = {"md5": "MD5", "sha1": "SHA-1"}

    def run(self):
        results = []
        test_data = b"audit_test_data"

        for algo, label in self.SAFE.items():
            try:
                hashlib.new(algo, test_data)
                results.append(self.result(label, AuditResult.OK, "利用可能"))
            except Exception as e:
                results.append(self.result(label, AuditResult.NG, f"エラー: {e}"))

        for algo, label in self.UNSAFE.items():
            results.append(self.result(
                f"{label}使用確認", AuditResult.WARN, f"{label}は非推奨です"))

        # 完全性検証
        data = "テストデータ"
        h1 = hashlib.sha256(data.encode()).hexdigest()
        h2 = hashlib.sha256(data.encode()).hexdigest()
        level = AuditResult.OK if h1 == h2 else AuditResult.NG
        results.append(self.result("完全性検証", level, "SHA-256検証"))

        return results


class LogAuditModule(AuditModule):
    """アクセスログ監視モジュールです。"""

    name = "ログ監視"
    BF_THRESHOLD = 10
    SQLI_PATTERNS = ["UNION", "SELECT", "DROP", "OR 1=1", "'--"]
    XSS_PATTERNS = ["<script", "javascript:", "onerror="]

    def run(self):
        results = []
        logs = self.context.get("logs", [])
        if not logs:
            results.append(self.result("ログ", AuditResult.WARN, "ログデータなし"))
            return results

        # ブルートフォース検知
        fail_by_ip = Counter(
            log["ip"] for log in logs if log["status"] in (401, 403))
        for ip, count in fail_by_ip.items():
            if count >= self.BF_THRESHOLD:
                results.append(self.result(
                    f"BF({ip})", AuditResult.NG, f"{count}回の認証失敗"))
            elif count >= 5:
                results.append(self.result(
                    f"認証失敗({ip})", AuditResult.WARN, f"{count}回"))

        # 攻撃パターン
        sqli = sum(1 for l in logs if any(
            k.upper() in l["path"].upper() for k in self.SQLI_PATTERNS))
        xss = sum(1 for l in logs if any(
            k.lower() in l["path"].lower() for k in self.XSS_PATTERNS))

        level = AuditResult.NG if sqli > 0 else AuditResult.OK
        results.append(self.result("SQLi検知", level, f"{sqli}件"))
        level = AuditResult.WARN if xss > 0 else AuditResult.OK
        results.append(self.result("XSS検知", level, f"{xss}件"))

        # エラー率
        errors = sum(1 for l in logs if l["status"] >= 400)
        rate = errors / len(logs) * 100
        if rate > 20:
            results.append(self.result("エラー率", AuditResult.NG, f"{rate:.1f}%"))
        elif rate > 10:
            results.append(self.result("エラー率", AuditResult.WARN, f"{rate:.1f}%"))
        else:
            results.append(self.result("エラー率", AuditResult.OK, f"{rate:.1f}%"))

        return results


class SystemAuditModule(AuditModule):
    """システム設定チェックモジュールです。"""

    name = "システム設定"

    DANGEROUS_PORTS = {
        21: ("FTP", AuditResult.NG), 23: ("Telnet", AuditResult.NG),

    }
    TLS_LEVELS = {
        "TLS 1.3": (AuditResult.OK, "最新"), "TLS 1.2": (AuditResult.OK, "推奨"),
        "TLS 1.1": (AuditResult.WARN, "非推奨"), "TLS 1.0": (AuditResult.NG, "脆弱"),
    }

    def run(self):
        results = []
        config = self.context.get("system_config", {})

        tls = config.get("tls_version", "TLS 1.2")
        level, msg = self.TLS_LEVELS.get(tls, (AuditResult.NG, "不明"))

        for port in config.get("open_ports", []):
            if port in self.DANGEROUS_PORTS:

                results.append(self.result(f"ポート{port}({svc})", lvl, "外部公開"))
            else:
                results.append(self.result(f"ポート{port}", AuditResult.OK, "問題なし"))

        for path, expected, actual in config.get("file_perms", []):
            if actual == expected:
                results.append(self.result(path, AuditResult.OK, f"正常({oct(actual)})"))
            else:
                results.append(self.result(path, AuditResult.WARN,
                    f"不正(期待:{oct(expected)}, 実際:{oct(actual)})"))

class SecurityAuditor:
    """総合セキュリティ監査を管理するクラスです。"""

    def __init__(self):
        self.modules = []
        self.results = []
        self.history = []
        self.context = {}

    def register_module(self, module_class):
        """監査モジュールを登録します。"""
        self.modules.append(module_class)

    def set_context(self, key, value):
        """共有データを設定します。"""
        self.context[key] = value

    def run_module(self, module_class):
        """単一のモジュールを実行します。"""
        module = module_class(self.context)
        return module.run()

    def run_all(self):
        """全モジュールを実行します。"""
        self.results = []
        for i, mod_class in enumerate(self.modules, 1):
            module = mod_class(self.context)
            print(f"\n[{i}/{len(self.modules)}] {module.name}...")
            try:
                module_results = module.run()
                self.results.extend(module_results)
                for r in module_results:
                    print(r)
            except Exception as e:
                self.results.append(AuditResult(
                    module.name, "エラー", AuditResult.NG, str(e)))
        return self.results

    def score(self):
        if not self.results:
            return 0
        total = len(self.results)
        weighted = sum(AuditResult.WEIGHTS.get(r.level, 0) for r in self.results)
        base = weighted / total
        ng_count = sum(1 for r in self.results if r.level == AuditResult.NG)
        return max(0, int(base - ng_count * 5))

    def verdict(self):
        s = self.score()
        if s >= 80: return "合格"
        elif s >= 60: return "条件付き合格"
        elif s >= 40: return "要改善"
        else: return "不合格"

    def print_summary(self):
        ok = sum(1 for r in self.results if r.level == AuditResult.OK)
        warn = sum(1 for r in self.results if r.level == AuditResult.WARN)
        ng = sum(1 for r in self.results if r.level == AuditResult.NG)
        print(f"\n{'=' * 50}")
        print(f"  合格: {ok}  警告: {warn}  不合格: {ng}  合計: {len(self.results)}")
        print(f"  スコア: {self.score()}/100  判定: {self.verdict()}")
        print(f"{'=' * 50}")

    def save_to_history(self):
        self.history.append({
            "timestamp": datetime.now().isoformat(),
            "score": self.score(),
            "verdict": self.verdict(),
            "results": [r.to_dict() for r in self.results],
        })

    def export_json(self, filename):
        report = {
            "generated_at": datetime.now().isoformat(),
            "score": self.score(),
            "verdict": self.verdict(),
            "results": [r.to_dict() for r in self.results],
            "history_count": len(self.history),
        }
        with open(filename, "w", encoding="utf-8") as f:
            json.dump(report, f, ensure_ascii=False, indent=2)
        print(f"JSONレポートを {filename} に出力しました。")

- `AuditModule` 基底クラスによるプラグインアーキテクチャ → 新しい監査項目の追加が容易
- `SecurityAuditor` が監査の実行、スコアリング、レポート生成を統括 → 責務の分離
- `context` 辞書で共有データを管理 → モジュール間の依存を最小化
- JSON形式のレポート出力 → 他ツールとの連携や経年比較が可能
- `try-except` で各モジュールの障害を隔離 → 1モジュールの失敗が全体に影響しない
- 監査履歴の保持 → セキュリティ状態の推移を追跡可能
