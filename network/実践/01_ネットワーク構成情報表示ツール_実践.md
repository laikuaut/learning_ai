---

## 課題の要件

1. `socket` モジュールを使ってホスト名とIPアドレスを取得する
2. IPアドレスを2進数表記に変換して表示する
3. IPアドレスのクラス（A/B/C）を判定する
4. IPアドレスがプライベートIPかどうかを判定する
5. よく使うポート番号の一覧を表示する

---

## ステップガイド

<details>
<summary>ステップ1：ホスト名とIPアドレスを取得する</summary>

`socket` モジュールの関数を使います。

```python
import socket

hostname = socket.gethostname()
fqdn = socket.getfqdn()
ip_address = socket.gethostbyname(hostname)

print(f"ホスト名: {hostname}")
print(f"FQDN: {fqdn}")
print(f"IPアドレス: {ip_address}")
```

</details>

<details>
<summary>ステップ2：IPアドレスを2進数に変換する</summary>

IPアドレスの各オクテットを `int()` で数値にし、`format()` で2進数に変換します。

```python
octets = ip_address.split(".")
binary_parts = []
for octet in octets:
    binary_parts.append(format(int(octet), "08b"))
binary_ip = ".".join(binary_parts)
```

</details>

<details>
<summary>ステップ3：クラス判定とプライベートIP判定</summary>

先頭オクテットの値でクラスを判定し、プライベートIPの範囲を条件分岐で判定します。

```python
first_octet = int(octets[0])

# クラス判定
if first_octet <= 127:
    ip_class = "クラスA"
elif first_octet <= 191:
    ip_class = "クラスB"
elif first_octet <= 223:
    ip_class = "クラスC"
```

</details>

```python
# ネットワーク構成情報表示ツール
# 学べる内容：socketモジュール、IPアドレスの基礎、2進数変換
# 実行方法：python network_info.py

import socket

# --- ホスト情報の取得 ---
hostname = socket.gethostname()
fqdn = socket.getfqdn()
ip_address = socket.gethostbyname(hostname)

print("===== ネットワーク構成情報 =====")
print()
print("【ホスト情報】")

print(f"  FQDN           : {fqdn}")

print()

# --- IPアドレスを2進数に変換 ---
octets = ip_address.split(".")
binary_parts = []
for octet in octets:
    binary_parts.append(format(int(octet), "08b"))
binary_ip = ".".join(binary_parts)

# --- クラス判定 ---
first_octet = int(octets[0])
if first_octet <= 127:
    ip_class = "クラスA（1.0.0.0 〜 127.255.255.255）"
elif first_octet <= 191:
    ip_class = "クラスB（128.0.0.0 〜 191.255.255.255）"
elif first_octet <= 223:
    ip_class = "クラスC（192.0.0.0 〜 223.255.255.255）"
elif first_octet <= 239:
    ip_class = "クラスD（224.0.0.0 〜 239.255.255.255）マルチキャスト"
else:
    ip_class = "クラスE（240.0.0.0 〜 255.255.255.255）実験用"

print("【IPアドレスの詳細】")
print(f"  IPアドレス（10進数）: {ip_address}")
print(f"  IPアドレス（2進数） : {binary_ip}")
print(f"  クラス              : {ip_class}")
print()

# --- プライベートIP判定 ---
first = int(octets[0])
second = int(octets[1])

is_private = False
private_range = ""

if first == 10:
    is_private = True
    private_range = "10.0.0.0 〜 10.255.255.255"
elif first == 172 and 16 <= second <= 31:
    is_private = True
    private_range = "172.16.0.0 〜 172.31.255.255"
elif first == 192 and second == 168:
    is_private = True
    private_range = "192.168.0.0 〜 192.168.255.255"

print("【プライベートIPアドレス判定】")
if is_private:
    print(f"  {ip_address} はプライベートIPアドレスです。")
    print(f"  （範囲: {private_range}）")
else:
    print(f"  {ip_address} はグローバルIPアドレスです。")
print()

# --- よく使うポート番号一覧 ---
ports = [
    (80, "HTTP", "Web閲覧"),
    (443, "HTTPS", "暗号化Web閲覧"),
    (22, "SSH", "リモート接続"),
    (53, "DNS", "名前解決"),
    (25, "SMTP", "メール送信"),
    (110, "POP3", "メール受信"),
    (143, "IMAP", "メール受信（同期型）"),
    (21, "FTP", "ファイル転送"),
    (23, "Telnet", "リモート接続（非暗号化）"),
    (3389, "RDP", "リモートデスクトップ"),
]

print("【よく使うポート番号一覧】")
print(f"  {'ポート番号':<12}{'サービス名':<12}{'用途'}")
print(f"  {'─' * 40}")
for port, service, description in ports:
    print(f"  {port:<12}{service:<12}{description}")
```

<details>
<summary>解答例（改良版 ─ 関数で整理）</summary>

関数に分割し、再利用しやすい構造にしたバージョンです。

```python
# ネットワーク構成情報表示ツール（改良版）
# 学べる内容：socketモジュール、IPアドレスの基礎、関数設計
# 実行方法：python network_info_v2.py

import socket


def get_host_info():
    """ホスト名・FQDN・IPアドレスを取得する"""
    hostname = socket.gethostname()
    return {
        "hostname": hostname,
        "fqdn": socket.getfqdn(),
        "ip": socket.gethostbyname(hostname),
    }


def ip_to_binary(ip_address):
    """IPアドレスを2進数表記に変換する"""
    return ".".join(format(int(o), "08b") for o in ip_address.split("."))


def classify_ip(ip_address):
    """IPアドレスのクラスを判定する"""
    first = int(ip_address.split(".")[0])
    classes = [
        (127, "クラスA", "1.0.0.0 〜 127.255.255.255"),
        (191, "クラスB", "128.0.0.0 〜 191.255.255.255"),
        (223, "クラスC", "192.0.0.0 〜 223.255.255.255"),
        (239, "クラスD（マルチキャスト）", "224.0.0.0 〜 239.255.255.255"),
        (255, "クラスE（実験用）", "240.0.0.0 〜 255.255.255.255"),
    ]
    for upper, name, range_str in classes:
        if first <= upper:
            return name, range_str
    return "不明", ""


def is_private_ip(ip_address):
    """プライベートIPアドレスかどうか判定する"""
    octets = [int(o) for o in ip_address.split(".")]
    private_ranges = [
        (octets[0] == 10, "10.0.0.0 〜 10.255.255.255"),
        (octets[0] == 172 and 16 <= octets[1] <= 31, "172.16.0.0 〜 172.31.255.255"),
        (octets[0] == 192 and octets[1] == 168, "192.168.0.0 〜 192.168.255.255"),
    ]
    for condition, range_str in private_ranges:
        if condition:
            return True, range_str
    return False, ""


def main():
    info = get_host_info()

    print("===== ネットワーク構成情報 =====\n")

    print("【ホスト情報】")
    print(f"  ホスト名       : {info['hostname']}")
    print(f"  FQDN           : {info['fqdn']}")
    print(f"  IPアドレス     : {info['ip']}\n")

    print("【IPアドレスの詳細】")
    print(f"  IPアドレス（10進数）: {info['ip']}")
    print(f"  IPアドレス（2進数） : {ip_to_binary(info['ip'])}")
    ip_class, ip_range = classify_ip(info['ip'])
    print(f"  クラス              : {ip_class}（{ip_range}）\n")

    print("【プライベートIPアドレス判定】")
    private, p_range = is_private_ip(info['ip'])
    if private:
        print(f"  {info['ip']} はプライベートIPアドレスです。")
        print(f"  （範囲: {p_range}）")
    else:
        print(f"  {info['ip']} はグローバルIPアドレスです。")
    print()

    ports = [
        (80, "HTTP", "Web閲覧"),
        (443, "HTTPS", "暗号化Web閲覧"),
        (22, "SSH", "リモート接続"),
        (53, "DNS", "名前解決"),
        (25, "SMTP", "メール送信"),
    ]
    print("【よく使うポート番号一覧】")
    print(f"  {'ポート番号':<12}{'サービス名':<12}{'用途'}")
    print(f"  {'─' * 40}")
    for port, service, desc in ports:
        print(f"  {port:<12}{service:<12}{desc}")


if __name__ == "__main__":
    main()
```

**初心者向けとの違い:**
- 機能ごとに関数に分割し、`main()` で組み立てる構成
- 各関数が単一の責務を持ち、単体テストしやすい
- 辞書やタプルのリストでデータを管理し、拡張性を向上
- `if __name__ == "__main__"` でモジュールとしても利用可能
