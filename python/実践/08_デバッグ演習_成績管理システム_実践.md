## 課題の説明

以下の「成績管理システム」には **10個のバグ** が埋め込まれています。
プログラムを実行し、エラーメッセージや意図しない動作を手がかりにすべてのバグを見つけて修正してください。

### 進め方

1. まず下のバグ入りコードをファイルに保存する
2. 実行してエラーを確認する
3. 1つずつバグを修正し、再実行する
4. 全機能が正常に動作すれば完了

## バグ入りコード

以下のコードをそのまま `grade_manager_buggy.py` として保存してください。

```python
# 成績管理システム（バグ入り版）
# このコードには10個のバグが含まれています。すべて修正してください。

def add_student(students, name score):  # バグ1
    """生徒と点数を追加する"""
    student = {"name": name, "score": score}
    students.append(student)
    print(f"{name}さん（{score}点）を追加しました。")


def show_all(students):
    """全生徒の成績を表示する"""
    if len(students) == 0:
        print("生徒が登録されていません。")
        return

    print("\n--- 成績一覧 ---")
    for i in range(len(students)):
        student = students[i]
        grade = get_grade(student["score"])
        print(f" {i}. {student['name']:<8} {student['score']:>3}点  {grade}")  # バグ2


def get_grade(score):
    """点数から評価を返す"""
    if score >= 90:
        return "A（優秀）"
    if score >= 70:
        return "B（良好）"
    if score >= 50:
        return "C（合格）"
    if score > 0:  # バグ3
        return "D（要努力）"
    return "F（不合格）"


def calculate_average(students):
    """平均点を計算する"""
    total = 0
    for student in students:
        total += student["score"]
    average = total / len(students)  # バグ4
    return average


def find_top_student(students):
    """最高得点の生徒を返す"""
    if not students:
        return None
    top = students[0]
    for student in students:
        if student["score"] > top:  # バグ5
            top = student
    return top


def find_by_name(students, name):
    """名前で生徒を検索する"""
    results = []
    for student in students:
        if name in student["name"]:
            results.append(student)
    return results


def remove_student(students, index):
    """指定番号の生徒を削除する"""
    if index < 0 or index >= len(students):
        print("無効な番号です。")
        return
    removed = students.pop(index)
    print(f"{removed['name']}さんを削除しました")  # バグ6: 文末のピリオドではなくプログラム動作のバグ


def save_to_file(students, filename):
    """成績をファイルに保存する"""
    with open(filename, "w", encoding="utf-8") as f:  # バグ7
        for student in students:
            f.write(f"{student['name']},{student['score']}")
    print(f"{filename} に保存しました。")


def load_from_file(students, filename):
    """ファイルから成績を読み込む"""
    try:
        with open(filename, "r", encoding="utf-8") as f:
            for line in f:
                parts = line.strip().split(",")
                name = parts[0]
                score = parts[1]  # バグ8
                students.append({"name": name, "score": score})
        print(f"{filename} から読み込みました。")
    except FileNotFoundError:
        print(f"{filename} が見つかりません。")


# --- メイン処理 ---
students = []
print("===== 成績管理システム =====")

while True:
    print("\n1.追加  2.一覧  3.平均  4.最高点  5.検索  6.削除  7.保存  8.読込  9.終了")
    choice = input("操作: ")

    if choice == "1":
        name = input("名前: ")
        score = input("点数: ")  # バグ9
        add_student(students, name, score)

    elif choice == "2":
        show_all(students)

    elif choice == "3":
        avg = calculate_average(students)
        print(f"平均点: {avg:.1f}点")

    elif choice == "4":
        top = find_top_student(students)
        if top:
            print(f"最高得点: {top['name']}さん（{top['score']}点）")

    elif choice == "5":
        keyword = input("検索する名前: ")
        results = find_by_name(students, keyword)
        if results:
            for s in results:
                print(f"  {s['name']} {s['score']}点")
        else:
            print("見つかりませんでした。")

    elif choice == "6":
        num = int(input("削除する番号: "))
        remove_student(students, num)

    elif choice == "7":
        save_to_file(students, "grades.txt")

    elif choice == "8":
        load_from_file(students, "grades.txt")

    elif choice == "9":
        print("終了します。")  # バグ10
```

<details>
<summary>全10個のバグと修正方法</summary>

### バグ1：構文エラー ─ 引数のカンマ忘れ

```python
# 修正前
def add_student(students, name score):
# 修正後
def add_student(students, name, score):
```

### バグ2：論理エラー ─ 表示番号が0始まり

```python
# 修正前
print(f" {i}. {student['name']:<8} ...")
# 修正後
print(f" {i + 1}. {student['name']:<8} ...")
```

### バグ3：論理エラー ─ 0点がDになる

```python
# 修正前
if score > 0:  # 0点はFに落ちるべきだが、ここでは0を含めていない
    return "D（要努力）"
# 修正後
if score >= 0:  # 0点以上（0〜49）はD
    return "D（要努力）"
```

### バグ4：実行時エラー ─ 空リストでのゼロ除算

```python
# 修正前
average = total / len(students)
# 修正後
if len(students) == 0:
    return 0
average = total / len(students)
```

### バグ5：型エラー ─ 辞書と数値を比較

```python
# 修正前
if student["score"] > top:       # topは辞書なのに数値と比較
# 修正後
if student["score"] > top["score"]:  # 辞書のscoreキー同士を比較
```

### バグ6：論理エラー ─ 表示番号とインデックスのずれ

`show_all` で番号を1始まりで表示しているが（バグ2を修正後）、削除時にその番号をそのまま使うとインデックスが1ずれる。

```python
# メイン処理の削除部分を修正
num = int(input("削除する番号: "))
remove_student(students, num - 1)  # 表示番号→インデックスに変換
```

### バグ7：機能不全 ─ 改行なしで保存

```python
# 修正前
f.write(f"{student['name']},{student['score']}")
# 修正後
f.write(f"{student['name']},{student['score']}\n")  # 改行を追加
```

### バグ8：型エラー ─ 文字列のまま格納

```python
# 修正前
score = parts[1]         # 文字列 "85" のまま
# 修正後
score = int(parts[1])    # 数値に変換
```

### バグ9：型エラー ─ 点数が文字列

```python
# 修正前
score = input("点数: ")           # 文字列が返る
# 修正後
score = int(input("点数: "))      # 数値に変換
```

### バグ10：機能不全 ─ ループが終了しない

```python
# 修正前
print("終了します。")    # breakがない → ループ継続
# 修正後
print("終了します。")
break                    # ループを抜ける
```
