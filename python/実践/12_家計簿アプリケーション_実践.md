## 課題の要件

### 必須機能

1. **収入記録**: 日付・金額・カテゴリ（給与/副業/その他）・メモ
2. **支出記録**: 日付・金額・カテゴリ（食費/交通費/日用品/娯楽/その他）・メモ
3. **履歴一覧**: 全取引を日付順で表示（収入はプラス、支出はマイナス表記）
4. **残高確認**: 総収入 − 総支出を表示
5. **月別レポート**: 指定月の収支合計・支出カテゴリ内訳・日別推移
6. **カテゴリ別分析**: 全期間のカテゴリ別支出割合と棒グラフ
7. **保存/読込**: JSON形式で永続化

### 設計要件

- **3つのクラス** を使うこと
  - `Transaction`: 1件の取引（収入または支出）
  - `Ledger`: 取引の集合を管理（追加・検索・集計）
  - `ReportGenerator`: レポート生成と表示
- **内包表記/ジェネレータ** を少なくとも3箇所で使うこと
- **エラー処理**: 金額に文字列を入れてもクラッシュしないこと
- **日付処理**: `datetime` モジュールで日付を管理する

```python
# 家計簿アプリケーション
# 学べる内容：複数クラスの連携、JSON永続化、日付処理、データ分析、内包表記

import json
from datetime import date


# ==================== データクラス ====================

class Transaction:
    """1件の取引を表すクラス"""

    INCOME_CATEGORIES = {"1": "給与", "2": "副業", "3": "その他"}
    EXPENSE_CATEGORIES = {"1": "食費", "2": "交通費", "3": "日用品", "4": "娯楽", "5": "その他"}

    def __init__(self, tx_type, amount, category, memo="", tx_date=None):
        self.tx_type = tx_type
        self.amount = amount
        self.category = category
        self.memo = memo
        self.date = tx_date if tx_date else date.today()

    def signed_amount(self):
        """収入は正、支出は負で返す"""
        if self.tx_type == "income":
            return self.amount
        else:
            return -self.amount

    def format_row(self):
        """一覧表示用の整形文字列"""
        type_label = "収入" if self.tx_type == "income" else "支出"
        sign = "+" if self.tx_type == "income" else "-"
        return (f"  {self.date}  {type_label}  {sign}{self.amount:>8,}円  "
                f"{self.category:<8} {self.memo}")

    def to_dict(self):
        """辞書に変換（保存用）"""
        return {
            "tx_type": self.tx_type,
            "amount": self.amount,
            "category": self.category,
            "memo": self.memo,
            "date": str(self.date),
        }

    @classmethod
    def from_dict(cls, data):
        """辞書から生成（読込用）"""
        return cls(
            tx_type=data["tx_type"],
            amount=data["amount"],
            category=data["category"],
            memo=data.get("memo", ""),
            tx_date=date.fromisoformat(data["date"]),
        )


# ==================== 帳簿クラス ====================

class Ledger:
    """取引の集合を管理するクラス"""

    def __init__(self):
        self.transactions = []

    def add(self, transaction):
        self.transactions.append(transaction)

    def get_balance(self):
        """残高を計算する"""
        total = 0
        for t in self.transactions:
            total += t.signed_amount()
        return total

    def get_sorted(self):
        """日付順に並べて返す"""
        return sorted(self.transactions, key=lambda t: t.date)

    def filter_by_month(self, year, month):
        """指定月の取引だけ返す"""
        results = []
        for t in self.transactions:
            if t.date.year == year and t.date.month == month:
                results.append(t)
        return results

    def filter_by_type(self, tx_type):
        """種別でフィルタする"""
        return [t for t in self.transactions if t.tx_type == tx_type]

    def save(self, filename):
        """JSONファイルに保存する"""
        data = [t.to_dict() for t in self.transactions]
        with open(filename, "w", encoding="utf-8") as f:
            json.dump(data, f, ensure_ascii=False, indent=2)

    def load(self, filename):
        """JSONファイルから読み込む"""
        with open(filename, "r", encoding="utf-8") as f:
            data = json.load(f)
        self.transactions = [Transaction.from_dict(d) for d in data]


# ==================== レポートクラス ====================

class ReportGenerator:
    """レポート生成と表示を担当するクラス"""

    @staticmethod
    def category_breakdown(transactions, tx_type="expense"):
        """カテゴリ別の集計を返す"""
        totals = {}
        for t in transactions:
            if t.tx_type == tx_type:
                totals[t.category] = totals.get(t.category, 0) + t.amount
        return totals

    @staticmethod
    def daily_totals(transactions, tx_type="expense"):
        """日別の合計を返す"""
        totals = {}
        for t in transactions:
            if t.tx_type == tx_type:
                day_str = str(t.date)
                totals[day_str] = totals.get(day_str, 0) + t.amount
        return dict(sorted(totals.items()))

    @staticmethod
    def print_bar(label, value, max_value, max_width=30):
        """棒グラフの1行を表示する"""
        if max_value > 0:
            bar_len = int(value / max_value * max_width)
        else:
            bar_len = 0
        print(f"  {label:<8}: {value:>8,}円 {'█' * bar_len}")

    @classmethod
    def monthly_report(cls, ledger, year, month):
        """月別レポートを表示する"""
        monthly = ledger.filter_by_month(year, month)
        if not monthly:
            print(f"{year}年{month}月のデータがありません。")
            return

        income = sum(t.amount for t in monthly if t.tx_type == "income")
        expense = sum(t.amount for t in monthly if t.tx_type == "expense")
        balance = income - expense

        print(f"\n===== {year}年{month}月 レポート =====")
        print(f"収入合計: {income:>10,}円")
        print(f"支出合計: {expense:>10,}円")
        sign = "+" if balance >= 0 else ""
        print(f"収支:     {sign}{balance:>9,}円")

        # カテゴリ別支出内訳
        cat_totals = cls.category_breakdown(monthly, "expense")
        if cat_totals:
            print("\n支出内訳:")
            max_val = max(cat_totals.values())
            total_expense = sum(cat_totals.values())
            for cat, amount in cat_totals.items():
                percent = round(amount / total_expense * 100, 1) if total_expense > 0 else 0
                bar_len = int(amount / max_val * 25) if max_val > 0 else 0
                print(f"  {cat:<8}: {amount:>8,}円 {'█' * bar_len} ({percent}%)")

        # 日別支出推移
        daily = cls.daily_totals(monthly, "expense")
        if daily:
            print("\n日別支出推移:")
            max_daily = max(daily.values())
            for day, amount in daily.items():
                short_day = day[5:]  # "YYYY-MM-DD" → "MM-DD"
                bar_len = int(amount / max_daily * 20) if max_daily > 0 else 0
                print(f"  {short_day} : {amount:>8,}円 {'█' * bar_len}")

    @classmethod
    def category_analysis(cls, ledger):
        """全期間のカテゴリ別分析を表示する"""
        expenses = ledger.filter_by_type("expense")
        if not expenses:
            print("支出データがありません。")
            return

        cat_totals = cls.category_breakdown(ledger.transactions, "expense")
        total = sum(cat_totals.values())

        print("\n===== カテゴリ別分析（全期間） =====")
        print(f"支出合計: {total:,}円\n")

        # 金額降順でソート
        sorted_cats = sorted(cat_totals.items(), key=lambda x: x[1], reverse=True)
        max_val = sorted_cats[0][1] if sorted_cats else 1

        for cat, amount in sorted_cats:
            percent = round(amount / total * 100, 1)
            bar_len = int(amount / max_val * 30)
            print(f"  {cat:<8}: {amount:>8,}円 {'█' * bar_len} ({percent}%)")


# ==================== 入力ヘルパー ====================

def safe_int_input(prompt):
    """安全に整数を入力させる"""
    while True:
        try:
            return int(input(prompt))
        except ValueError:
            print("数字を入力してください。")


def safe_positive_int(prompt):
    """正の整数を入力させる"""
    while True:
        value = safe_int_input(prompt)
        if value > 0:
            return value
        print("1以上の数字を入力してください。")


def input_date(prompt):
    """日付を入力させる（空欄で今日）"""
    while True:
        s = input(prompt)
        if s == "":
            return date.today()
        try:
            return date.fromisoformat(s)
        except ValueError:
            print("YYYY-MM-DD 形式で入力してください（空欄で今日）。")


def input_category(categories):
    """カテゴリ番号を入力させる"""
    options = " ".join(f"{k}:{v}" for k, v in categories.items())
    while True:
        choice = input(f"カテゴリ ({options}): ")
        if choice in categories:
            return categories[choice]
        print("有効な番号を入力してください。")


# ==================== メニュー操作 ====================

def menu_income(ledger):
    """収入を記録する"""
    tx_date = input_date("日付 (YYYY-MM-DD, 空欄で今日): ")
    amount = safe_positive_int("金額: ")
    category = input_category(Transaction.INCOME_CATEGORIES)
    memo = input("メモ: ")

    tx = Transaction("income", amount, category, memo, tx_date)
    ledger.add(tx)
    print(f"収入 {amount:,}円（{category}）を記録しました。")


def menu_expense(ledger):
    """支出を記録する"""
    tx_date = input_date("日付 (YYYY-MM-DD, 空欄で今日): ")
    amount = safe_positive_int("金額: ")
    category = input_category(Transaction.EXPENSE_CATEGORIES)
    memo = input("メモ: ")

    tx = Transaction("expense", amount, category, memo, tx_date)
    ledger.add(tx)
    print(f"支出 {amount:,}円（{category}）を記録しました。")


def menu_history(ledger):
    """履歴一覧を表示する"""
    sorted_tx = ledger.get_sorted()
    if not sorted_tx:
        print("取引データがありません。")
        return

    print("\n--- 取引履歴 ---")
    print("  日付        種別    金額       カテゴリ   メモ")
    for tx in sorted_tx:
        print(tx.format_row())


def menu_balance(ledger):
    """残高を確認する"""
    incomes = sum(t.amount for t in ledger.filter_by_type("income"))
    expenses = sum(t.amount for t in ledger.filter_by_type("expense"))
    balance = ledger.get_balance()

    print(f"\n--- 残高 ---")
    print(f"総収入:  {incomes:>10,}円")
    print(f"総支出:  {expenses:>10,}円")
    print(f"残高:    {balance:>+10,}円")


def menu_monthly(ledger):
    """月別レポートを表示する"""
    year = safe_int_input("年 (例: 2024): ")
    month = safe_int_input("月 (1-12): ")
    if not (1 <= month <= 12):
        print("1〜12で入力してください。")
        return
    ReportGenerator.monthly_report(ledger, year, month)


def menu_category(ledger):
    """カテゴリ別分析を表示する"""
    ReportGenerator.category_analysis(ledger)


def menu_save(ledger):
    """保存する"""
    filename = "kakeibo.json"
    try:
        ledger.save(filename)
        print(f"{filename} に保存しました。({len(ledger.transactions)}件)")
    except OSError as e:
        print(f"保存に失敗しました: {e}")


def menu_load(ledger):
    """読み込む"""
    filename = "kakeibo.json"
    try:
        ledger.load(filename)
        print(f"{filename} から読み込みました。({len(ledger.transactions)}件)")
    except FileNotFoundError:
        print(f"{filename} が見つかりません。")
    except (json.JSONDecodeError, KeyError) as e:
        print(f"ファイルの形式が不正です: {e}")


# ==================== メインループ ====================

def main():
    ledger = Ledger()
    print("===== 家計簿アプリ =====")

    menu_items = {
        "1": ("収入を記録", menu_income),
        "2": ("支出を記録", menu_expense),
        "3": ("履歴一覧", menu_history),
        "4": ("残高確認", menu_balance),
        "5": ("月別レポート", menu_monthly),
        "6": ("カテゴリ別分析", menu_category),
        "7": ("保存", menu_save),
        "8": ("読込", menu_load),
    }

    while True:
        print("\n[メニュー]")
        keys = list(menu_items.keys())
        for i in range(0, len(keys), 2):
            left = f" {keys[i]}. {menu_items[keys[i]][0]}"
            if i + 1 < len(keys):
                right = f" {keys[i+1]}. {menu_items[keys[i+1]][0]}"
            else:
                right = ""
            print(f"{left:<20}{right}")
        print(f"{'':<20} 9. 終了")

        choice = input("\n操作: ")

        if choice == "9":
            print("お疲れさまでした！")
            break
        elif choice in menu_items:
            menu_items[choice][1](ledger)
        else:
            print("1〜9の番号を入力してください。")


if __name__ == "__main__":
    main()
```

```python
class Ledger:
    def iter_by_type(self, tx_type):
```

### テクニック3：汎用集計関数

```python
def group_sum(items, key_func, value_func):
    """任意のキーと値で集計する汎用関数"""
    result = {}
    for item in items:
        key = key_func(item)
        result[key] = result.get(key, 0) + value_func(item)
    return result

# 使用例
cat_totals = group_sum(
    expenses,
    key_func=lambda t: t.category,
    value_func=lambda t: t.amount,
)

daily_totals = group_sum(
    expenses,
    key_func=lambda t: str(t.date),
    value_func=lambda t: t.amount,
)
```

- ジェネレータで不要な中間リストを排除
- `classmethod` で入力処理をデータクラスに集約
- 汎用 `group_sum` で重複する集計ロジックを統一
