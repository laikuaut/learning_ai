```html
      { name: 'コットンTシャツ', price: 1980, category: 'clothing', inStock: true },
      { name: 'デニムジャケット', price: 8900, category: 'clothing', inStock: true },

    function search(page = 1) {
      const query = document.getElementById('search-input').value.trim().toLowerCase();
      const category = document.getElementById('category-filter').value;
      const sort = document.getElementById('sort').value;
      const inStockOnly = document.getElementById('in-stock').checked;

          if (category && p.category !== category) return false;
          if (inStockOnly && !p.inStock) return false;
          return true;

          document.getElementById('no-result').style.display = 'block';
          document.getElementById('result-info').textContent = '';
          document.getElementById('pagination').innerHTML = '';
          return;

        // ページネーション
        const totalPages = Math.ceil(results.length / PAGE_SIZE);
        const start = (page - 1) * PAGE_SIZE;
        const pageResults = results.slice(start, start + PAGE_SIZE);

        document.getElementById('product-list').innerHTML = pageResults.map(p => `

            <div>
              <h3>${p.name}</h3>
              <span class="category">${categoryNames[p.category]}</span>
              ${p.inStock ? '' : '<span style="color:red; margin-left:8px;">品切れ</span>'}
            </div>
            <span class="price">¥${p.price.toLocaleString()}</span>
          </div>
        `).join('');

        // ページネーションボタン
        let paginationHtml = '';
        if (totalPages > 1) {
          paginationHtml += `<button ${page === 1 ? 'disabled' : ''} onclick="search(${page - 1})">前へ</button>`;
          for (let i = 1; i <= totalPages; i++) {
            paginationHtml += `<button class="${i === page ? 'active' : ''}" onclick="search(${i})">${i}</button>`;
          }
          paginationHtml += `<button ${page === totalPages ? 'disabled' : ''} onclick="search(${page + 1})">次へ</button>`;
        }
        document.getElementById('pagination').innerHTML = paginationHtml;
      }, 300);
    }

    // 初期表示
    search();
  </script>
```

### POMクラス

```typescript
// pages/search-page.ts
import { type Locator, type Page, expect } from '@playwright/test';

export class SearchPage {
  readonly page: Page;
  readonly searchInput: Locator;
  readonly searchButton: Locator;
  readonly categoryFilter: Locator;
  readonly sortSelect: Locator;
  readonly inStockCheckbox: Locator;
  readonly productCards: Locator;
  readonly resultInfo: Locator;
  readonly noResult: Locator;
  readonly loading: Locator;

  constructor(page: Page) {
    this.page = page;
    this.searchInput = page.getByLabel('商品検索');
    this.searchButton = page.getByRole('button', { name: '検索' });
    this.categoryFilter = page.getByLabel('カテゴリ：');
    this.sortSelect = page.getByLabel('並び替え：');
    this.inStockCheckbox = page.getByLabel('在庫ありのみ');
    this.productCards = page.getByTestId('product-card');
    this.resultInfo = page.locator('#result-info');
    this.noResult = page.locator('#no-result');
    this.loading = page.locator('#loading');
  }

  async goto(filePath: string) {
    await this.page.goto(`file://${filePath}`);
    // 初期ロードの完了を待つ
    await expect(this.loading).toBeHidden();
  }

  async searchByKeyword(keyword: string) {
    await this.searchInput.fill(keyword);
    await this.searchButton.click();
    await expect(this.loading).toBeHidden();
  }

  async filterByCategory(category: string) {
    await this.categoryFilter.selectOption(category);
    await expect(this.loading).toBeHidden();
  }

  async sortBy(value: string) {
    await this.sortSelect.selectOption(value);
    await expect(this.loading).toBeHidden();
  }

  async toggleInStock() {
    await this.inStockCheckbox.check();
    await expect(this.loading).toBeHidden();
  }

  async expectProductCount(count: number) {
    await expect(this.productCards).toHaveCount(count);
  }

  async expectNoResult() {
    await expect(this.noResult).toBeVisible();
  }

  async expectResultInfo(text: string | RegExp) {
    await expect(this.resultInfo).toHaveText(text);
  }

  async expectFirstProductName(name: string) {
    await expect(this.productCards.first().locator('h3')).toHaveText(name);
  }

  async expectFirstProductPrice(price: string) {
    await expect(this.productCards.first().locator('.price')).toHaveText(price);
  }

  async goToPage(pageNum: number) {
    await this.page.locator('.pagination').getByRole('button', { name: String(pageNum) }).click();
    await expect(this.loading).toBeHidden();
  }
}
```

```typescript
// 学べる内容：テスト設計、POM、フィルター・ソートのテスト

import { test, expect } from '@playwright/test';
import path from 'path';
import { SearchPage } from '../pages/search-page';

const TEST_PAGE = path.resolve(__dirname, '../search-page.html');

test.describe('商品検索', () => {
  let searchPage: SearchPage;

    searchPage = new SearchPage(page);
    await searchPage.goto(TEST_PAGE);
  });

  test('初期表示で商品が表示される', async () => {
    // 全12商品のうち最初の5件が表示（ページネーション）

  });

  test('キーワードで検索できる', async () => {
    await searchPage.searchByKeyword('イヤホン');
    await searchPage.expectProductCount(1);
    await searchPage.expectFirstProductName('ワイヤレスイヤホン');
  });

  test('カテゴリでフィルタできる', async () => {
    await searchPage.filterByCategory('books');
    await searchPage.expectProductCount(4);

    await searchPage.sortBy('price-asc');
    await searchPage.expectFirstProductPrice('¥980');

  test('在庫ありのみ表示できる', async () => {
    await searchPage.toggleInStock();
    // 在庫あり9商品のうち最初の5件
    await searchPage.expectProductCount(5);
    await searchPage.expectResultInfo('9件中 1〜5件を表示');
  });

  test('検索結果が0件のときメッセージが表示される', async () => {
    await searchPage.searchByKeyword('存在しない商品XYZ');
    await searchPage.expectNoResult();
  });

  test('空文字で検索すると全商品が表示される', async () => {
    await searchPage.searchByKeyword('');
    await searchPage.expectProductCount(5);
  });

  test('キーワードとカテゴリの組み合わせ検索', async () => {
    await searchPage.searchByKeyword('JavaScript');
    await searchPage.filterByCategory('books');
    await searchPage.expectProductCount(1);
    await searchPage.expectFirstProductName('JavaScript入門');

  test('2ページ目に遷移できる', async () => {
```

```typescript
// tests/search-advanced.spec.ts
// 改良版：データ駆動テストで網羅性を向上
// 実行方法：npx playwright test tests/search-advanced.spec.ts --project=chromium

import { test, expect } from '@playwright/test';
import path from 'path';
import { SearchPage } from '../pages/search-page';

const TEST_PAGE = path.resolve(__dirname, '../search-page.html');

let searchPage: SearchPage;

test.beforeEach(async ({ page }) => {
  searchPage = new SearchPage(page);
  await searchPage.goto(TEST_PAGE);
});

// --- データ駆動テスト：キーワード検索 ---
const keywordTests = [
  { keyword: 'イヤホン', expectedCount: 1, description: '完全な商品名の一部' },
  { keyword: 'ケーブル', expectedCount: 1, description: '部分一致' },
  { keyword: 'javascript', expectedCount: 1, description: '小文字でも検索可能' },
  { keyword: '', expectedCount: 5, description: '空文字は全件（1ページ分）' },
  { keyword: 'ZZZZZ', expectedCount: 0, description: '存在しないキーワード' },
];

for (const tc of keywordTests) {
  test(`キーワード検索：${tc.description}（"${tc.keyword}"）`, async () => {
    await searchPage.searchByKeyword(tc.keyword);
    if (tc.expectedCount === 0) {
      await searchPage.expectNoResult();
    } else {
      await searchPage.expectProductCount(tc.expectedCount);
    }
  });
}

// --- データ駆動テスト：カテゴリフィルター ---
const categoryTests = [
  { category: 'electronics', label: '家電', expectedCount: 4 },
  { category: 'books', label: '書籍', expectedCount: 4 },
  { category: 'clothing', label: '衣類', expectedCount: 4 },
  { category: '', label: 'すべて', expectedCount: 5 },
];

for (const tc of categoryTests) {
  test(`カテゴリフィルター：${tc.label}`, async () => {
    await searchPage.filterByCategory(tc.category);
    await searchPage.expectProductCount(tc.expectedCount);
  });
}

// --- ソート検証 ---
const sortTests = [
  { sort: 'price-asc', firstProduct: 'USB-C ケーブル', firstPrice: '¥980' },
  { sort: 'price-desc', firstProduct: 'デニムジャケット', firstPrice: '¥8,900' },
  { sort: 'name-asc', firstProduct: 'Bluetooth スピーカー', firstPrice: '¥5,480' },
];

for (const tc of sortTests) {
  test(`ソート：${tc.sort}`, async () => {
    await searchPage.sortBy(tc.sort);
    await searchPage.expectFirstProductName(tc.firstProduct);
    await searchPage.expectFirstProductPrice(tc.firstPrice);
  });
}

// --- 組み合わせテスト ---
test('家電 + 在庫あり + 価格安い順', async () => {
  await searchPage.filterByCategory('electronics');
  await searchPage.toggleInStock();
  await searchPage.sortBy('price-asc');

  // 家電で在庫あり: USB-Cケーブル(980), ワイヤレスイヤホン(3980), Bluetoothスピーカー(5480)
  await searchPage.expectProductCount(3);
  await searchPage.expectFirstProductName('USB-C ケーブル');
  await searchPage.expectFirstProductPrice('¥980');
});

test('書籍 + キーワード「Script」', async () => {
  await searchPage.filterByCategory('books');
  await searchPage.searchByKeyword('Script');

  // JavaScript入門, TypeScript実践ガイド
  await searchPage.expectProductCount(2);
});
```

**初心者向けとの違い:**
- パラメータ化テストでキーワード・カテゴリ・ソートを網羅的にテスト
- テストデータをオブジェクト配列で定義し、`for` ループで自動展開
- 組み合わせテストで複数フィルターの同時適用を検証
- テストケースの追加が配列への要素追加だけで済む
