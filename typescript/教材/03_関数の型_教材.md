# 第3章: 関数の型

## 3.1 基本的な関数の型注釈

TypeScriptでは、関数のパラメータと戻り値に型を付けることができます。

```typescript
// 関数宣言
function add(a: number, b: number): number {
  return a + b;
}

// アロー関数
const multiply = (a: number, b: number): number => {
  return a * b;
};

// 関数式
const divide = function (a: number, b: number): number {
  return a / b;
};

// 戻り値の型推論（明示しなくても推論される）
function subtract(a: number, b: number) {
  return a - b; // 戻り値は number と推論される
}
```

### 戻り値の型を明示すべき場面

```typescript
// 複雑なロジックで意図を明確にしたい場合
function processData(input: string): string[] {
  // 実装が複雑でも、戻り値の型が明確
  return input.split(",").map((s) => s.trim());
}

// 非同期関数
async function fetchUser(id: number): Promise<User> {
  const response = await fetch(`/api/users/${id}`);
  return response.json();
}

// コールバックを含む関数
function debounce(fn: () => void, delay: number): () => void {
  let timeoutId: ReturnType<typeof setTimeout>;
  return () => {
    clearTimeout(timeoutId);
    timeoutId = setTimeout(fn, delay);
  };
}
```

## 3.2 オプショナルパラメータ

`?` を付けるとパラメータを省略可能にできます。オプショナルパラメータは必ず最後に置きます。

```typescript
function greet(name: string, honorific?: string): string {
  if (honorific) {
    return `こんにちは、${name}${honorific}`;
  }
  return `こんにちは、${name}さん`;
}

greet("田中");        // "こんにちは、田中さん"
greet("田中", "先生"); // "こんにちは、田中先生"
```

### デフォルトパラメータ

```typescript
function createUser(
  name: string,
  age: number,
  role: string = "member",     // デフォルト値
  isActive: boolean = true     // デフォルト値
) {
  return { name, age, role, isActive };
}

createUser("太郎", 25);                    // role: "member", isActive: true
createUser("花子", 30, "admin");           // role: "admin", isActive: true
createUser("次郎", 28, "editor", false);   // role: "editor", isActive: false

// デフォルト値があるパラメータの型は自動推論される
// role は string 型、isActive は boolean 型
```

## 3.3 残余パラメータ（Rest Parameters）

```typescript
// 可変長引数
function sum(...numbers: number[]): number {
  return numbers.reduce((total, n) => total + n, 0);
}

sum(1, 2, 3);       // 6
sum(10, 20, 30, 40); // 100

// 固定パラメータとの併用
function log(level: string, ...messages: string[]): void {
  console.log(`[${level}]`, ...messages);
}

log("INFO", "サーバー起動", "ポート3000");
log("ERROR", "接続失敗");

// タプル型での残余パラメータ
function create(...args: [string, number, boolean]): void {
  const [name, age, active] = args;
  console.log(name, age, active);
}

create("太郎", 25, true);
```

## 3.4 関数の型定義

### 型エイリアスで関数型を定義

```typescript
// コールバック関数の型
type Callback = (error: Error | null, data: string) => void;

// イベントハンドラの型
type EventHandler = (event: MouseEvent) => void;

// 高階関数の型
type Transformer<T, U> = (input: T) => U;

// 使用例
const toUpper: Transformer<string, string> = (s) => s.toUpperCase();
const toLength: Transformer<string, number> = (s) => s.length;

function fetchData(url: string, callback: Callback): void {
  try {
    // データ取得処理
    callback(null, "取得したデータ");
  } catch (e) {
    callback(e as Error, "");
  }
}
```

### インターフェースで関数型を定義

```typescript
interface MathOperation {
  (a: number, b: number): number;
}

const add: MathOperation = (a, b) => a + b;
const subtract: MathOperation = (a, b) => a - b;

// ジェネリックな関数型インターフェース
interface Converter<Input, Output> {
  (input: Input): Output;
}

const stringify: Converter<number, string> = (n) => n.toString();
const parse: Converter<string, number> = (s) => parseInt(s, 10);
```

## 3.5 関数オーバーロード

同じ関数名で、異なる引数パターンに応じた型を定義できます。

```typescript
// オーバーロードシグネチャ（型定義）
function format(value: string): string;
function format(value: number): string;
function format(value: Date): string;

// 実装シグネチャ（実際の処理）
function format(value: string | number | Date): string {
  if (typeof value === "string") {
    return value.trim();
  } else if (typeof value === "number") {
    return value.toFixed(2);
  } else {
    return value.toISOString();
  }
}

format("  hello  "); // "hello"
format(3.14159);       // "3.14"
format(new Date());    // "2025-01-01T00:00:00.000Z"

// より実践的な例
function createElement(tag: "a"): HTMLAnchorElement;
function createElement(tag: "canvas"): HTMLCanvasElement;
function createElement(tag: "input"): HTMLInputElement;
function createElement(tag: string): HTMLElement;
function createElement(tag: string): HTMLElement {
  return document.createElement(tag);
}

const link = createElement("a");       // HTMLAnchorElement
const canvas = createElement("canvas"); // HTMLCanvasElement
const input = createElement("input");   // HTMLInputElement
const div = createElement("div");       // HTMLElement
```

### オーバーロード vs ユニオン型

```typescript
// オーバーロードが適切な場面：
// 引数の型によって戻り値の型が変わる場合
function process(input: string): string;
function process(input: number): number;
function process(input: string | number): string | number {
  if (typeof input === "string") return input.toUpperCase();
  return input * 2;
}

const r1 = process("hello"); // string型と推論される
const r2 = process(42);       // number型と推論される

// ユニオン型で十分な場面：
// 戻り値の型が同じ場合
function toString(value: string | number | boolean): string {
  return String(value);
}
```

## 3.6 void型 と undefined型 の違い

```typescript
// void: 戻り値がないことを示す
function logMessage(msg: string): void {
  console.log(msg);
  // return; は書いても書かなくてもOK
}

// undefined: undefinedを明示的に返す
function findItem(id: number): string | undefined {
  const items = ["a", "b", "c"];
  return items[id]; // 見つからない場合はundefined
}

// 重要な違い：コールバックの戻り値としてのvoid
type VoidCallback = () => void;

// voidの戻り値型は、実際に値を返しても型チェックを通る
const cb: VoidCallback = () => {
  return 42; // エラーにならない！voidは戻り値を無視する
};

// これはArray.forEachなどで便利
// forEach((item) => void) なので、中でpushなどの値を返す式を書ける
const numbers: number[] = [];
[1, 2, 3].forEach((n) => numbers.push(n)); // push()はnumberを返すがvoidなのでOK
```

## 3.7 this パラメータ

TypeScriptでは、関数の `this` の型を明示的に宣言できます。

```typescript
interface User {
  name: string;
  greet(this: User): string;
}

const user: User = {
  name: "太郎",
  greet() {
    return `こんにちは、${this.name}です`;
  },
};

user.greet(); // OK: "こんにちは、太郎です"

// thisの型が合わない呼び出しはエラー
// const greetFn = user.greet;
// greetFn(); // エラー！thisの型がUserではない

// コールバックでのthis
interface EventTarget {
  addEventListener(
    type: string,
    listener: (this: EventTarget, event: Event) => void
  ): void;
}
```

## 3.8 ジェネリック関数（入門）

関数にジェネリック型パラメータを追加できます（詳しくは第5章で解説）。

```typescript
// 型パラメータ T を使った汎用関数
function identity<T>(value: T): T {
  return value;
}

identity<string>("hello"); // string
identity<number>(42);       // number
identity("hello");          // 型推論で string

// 配列の最初の要素を返す
function first<T>(arr: T[]): T | undefined {
  return arr[0];
}

first([1, 2, 3]);          // number | undefined
first(["a", "b", "c"]);   // string | undefined

// 2つの型パラメータ
function pair<A, B>(a: A, b: B): [A, B] {
  return [a, b];
}

pair("name", 42);  // [string, number]
pair(true, "yes"); // [boolean, string]
```

## 3.9 アサーション関数

`asserts` キーワードを使って、条件を保証する関数を定義できます。

```typescript
// assertion関数：条件を満たさなければ例外をスロー
function assertIsString(value: unknown): asserts value is string {
  if (typeof value !== "string") {
    throw new Error(`Expected string, got ${typeof value}`);
  }
}

function processInput(input: unknown) {
  assertIsString(input);
  // ここではinputはstring型として扱える
  console.log(input.toUpperCase());
}

// assert関数
function assert(condition: unknown, message?: string): asserts condition {
  if (!condition) {
    throw new Error(message ?? "Assertion failed");
  }
}

function divide(a: number, b: number): number {
  assert(b !== 0, "ゼロで割ることはできません");
  return a / b; // bは0ではないことが保証される
}
```

## 3.10 実践パターン：ビルダーパターンの型付け

```typescript
interface QueryBuilder {
  select(fields: string[]): QueryBuilder;
  from(table: string): QueryBuilder;
  where(condition: string): QueryBuilder;
  orderBy(field: string, direction?: "ASC" | "DESC"): QueryBuilder;
  limit(count: number): QueryBuilder;
  build(): string;
}

function createQueryBuilder(): QueryBuilder {
  const parts: {
    fields: string[];
    table: string;
    conditions: string[];
    order: string | null;
    limitCount: number | null;
  } = {
    fields: [],
    table: "",
    conditions: [],
    order: null,
    limitCount: null,
  };

  const builder: QueryBuilder = {
    select(fields) {
      parts.fields = fields;
      return builder;
    },
    from(table) {
      parts.table = table;
      return builder;
    },
    where(condition) {
      parts.conditions.push(condition);
      return builder;
    },
    orderBy(field, direction = "ASC") {
      parts.order = `${field} ${direction}`;
      return builder;
    },
    limit(count) {
      parts.limitCount = count;
      return builder;
    },
    build() {
      let query = `SELECT ${parts.fields.join(", ")} FROM ${parts.table}`;
      if (parts.conditions.length > 0) {
        query += ` WHERE ${parts.conditions.join(" AND ")}`;
      }
      if (parts.order) {
        query += ` ORDER BY ${parts.order}`;
      }
      if (parts.limitCount !== null) {
        query += ` LIMIT ${parts.limitCount}`;
      }
      return query;
    },
  };

  return builder;
}

// メソッドチェーンで型安全なクエリ構築
const query = createQueryBuilder()
  .select(["name", "age", "email"])
  .from("users")
  .where("age > 18")
  .where("active = true")
  .orderBy("name", "ASC")
  .limit(10)
  .build();

console.log(query);
// SELECT name, age, email FROM users WHERE age > 18 AND active = true ORDER BY name ASC LIMIT 10
```
