## 完成イメージ

```
分析するテキストを入力してください（空行で終了）:
Python is a great programming language.
Python is easy to learn.
I love Python programming.

===== テキスト分析結果 =====
文字数（空白含む）: 82
文字数（空白除く）: 68
単語数: 14
行数: 3
文の数: 3

--- 単語出現頻度 TOP5 ---
 1. python      : 3回
 2. is          : 2回
 3. programming : 2回
 4. a           : 1回
 5. great       : 1回
```

---

## 課題の要件

1. 複数行のテキストを入力として受け取る（空行で入力終了）
2. 以下の基本統計を計算する
   - 文字数（空白含む / 空白除く）
   - 単語数
   - 行数
   - 文の数（ピリオド `.` の数で判定）
3. 単語の出現頻度を辞書で集計し、上位5件を表示する
4. 大文字・小文字を区別しない（すべて小文字に統一して集計）

## 解答例

<details>
<summary>解答例（初心者向け）</summary>

```python
# テキスト分析ツール
# 学べる内容：文字列メソッド、辞書による集計、ソート

# --- 入力 ---
lines = []
print("分析するテキストを入力してください（空行で終了）:")
while True:
    line = input()
    if line == "":
        break
    lines.append(line)

if len(lines) == 0:
    print("テキストが入力されませんでした。")
else:
    text = "\n".join(lines)

    # --- 基本統計 ---
    char_count = len(text)
    char_no_space = len(text.replace(" ", "").replace("\n", ""))
    words = text.split()
    word_count = len(words)
    line_count = len(lines)
    sentence_count = text.count(".")

    print("\n===== テキスト分析結果 =====")
    print(f"文字数（空白含む）: {char_count}")
    print(f"文字数（空白除く）: {char_no_space}")
    print(f"単語数: {word_count}")
    print(f"行数: {line_count}")
    print(f"文の数: {sentence_count}")

    # --- 単語頻度集計 ---
    freq = {}
    for word in words:
        # 小文字にして、前後の記号を除去
        cleaned = word.lower().strip(".,!?;:\"'()")
        if cleaned == "":
            continue
        if cleaned in freq:
            freq[cleaned] = freq[cleaned] + 1
        else:
            freq[cleaned] = 1

    # 頻度が高い順にソート
    sorted_words = sorted(freq.items(), key=lambda x: x[1], reverse=True)

    print("\n--- 単語出現頻度 TOP5 ---")
    for i in range(min(5, len(sorted_words))):
        word, count = sorted_words[i]
        print(f" {i + 1}. {word:<12}: {count}回")
```

</details>

<summary>解答例（改良版 ─ 関数分離＋詳細分析）</summary>

```python
# テキスト分析ツール（改良版）
# 関数分離＋追加分析機能付き

def read_multiline_input():
    """複数行の入力を受け取る"""
    lines = []
    print("分析するテキストを入力してください（空行で終了）:")
    while True:
        line = input()
        if line == "":
            break
        lines.append(line)
    return lines


def count_stats(text, lines):
    """基本統計を計算して辞書で返す"""
    words = text.split()
    return {
        "chars_with_space": len(text),
        "chars_no_space": len(text.replace(" ", "").replace("\n", "")),
        "words": len(words),
        "lines": len(lines),
        "sentences": text.count(".") + text.count("!") + text.count("?"),
        "avg_word_length": sum(len(w.strip(".,!?;:")) for w in words) / max(len(words), 1),
    }


def count_word_frequency(text):
    """単語頻度を集計して降順ソートで返す"""
    freq = {}
    for word in text.split():
        cleaned = word.lower().strip(".,!?;:\"'()")
        if cleaned:
            freq[cleaned] = freq.get(cleaned, 0) + 1
    return sorted(freq.items(), key=lambda x: x[1], reverse=True)


def display_results(stats, sorted_words, top_n=5):
    """分析結果を表示する"""
    print("\n===== テキスト分析結果 =====")
    print(f"文字数（空白含む）: {stats['chars_with_space']}")
    print(f"文字数（空白除く）: {stats['chars_no_space']}")
    print(f"単語数: {stats['words']}")
    print(f"行数: {stats['lines']}")
    print(f"文の数: {stats['sentences']}")
    print(f"平均単語長: {stats['avg_word_length']:.1f}文字")

    print(f"\n--- 単語出現頻度 TOP{top_n} ---")
    for i, (word, count) in enumerate(sorted_words[:top_n], 1):
        bar = "█" * count
        print(f" {i}. {word:<12}: {count}回 {bar}")


# --- メイン処理 ---
lines = read_multiline_input()
if not lines:
    print("テキストが入力されませんでした。")
else:
    text = "\n".join(lines)
    stats = count_stats(text, lines)
    sorted_words = count_word_frequency(text)
    display_results(stats, sorted_words)
```

**初心者向けとの違い:**
- 関数分離で見通しが良い
- `dict.get(key, default)` で存在チェックを簡潔に
- 平均単語長の追加分析
- 棒グラフ風の視覚表示
