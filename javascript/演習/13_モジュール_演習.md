### 問題1：名前付きエクスポートとインポート

以下の関数を `mathUtils.js` からエクスポートし、`app.js` でインポートして使用するコードを書いてください。

```javascript
// mathUtils.js に定義する関数
function add(a, b) { return a + b; }
function subtract(a, b) { return a - b; }
function multiply(a, b) { return a * b; }
const PI = 3.14159;
```

**期待されるコード構成と出力：**
```
add(3, 5) = 8
subtract(10, 4) = 6
PI = 3.14159
```

<details>
<summary>解答例</summary>

```javascript
// --- mathUtils.js ---
export function add(a, b) {
  return a + b;
}

export function subtract(a, b) {
  return a - b;
}

export function multiply(a, b) {
  return a * b;
}

export const PI = 3.14159;

// --- app.js ---
import { add, subtract, PI } from './mathUtils.js';

console.log(`add(3, 5) = ${add(3, 5)}`);
console.log(`subtract(10, 4) = ${subtract(10, 4)}`);
console.log(`PI = ${PI}`);
```

</details>

### 問題2：デフォルトエクスポート

`User` クラスをデフォルトエクスポートし、別ファイルでインポートして使用するコードを書いてください。

```javascript
// User クラスの仕様
// - name, email を受け取るコンストラクタ
// - greet() メソッド: "こんにちは、{name}です" を返す
// - toJSON() メソッド: { name, email } のオブジェクトを返す
```

**期待される出力：**
```
こんにちは、太郎です
{ name: "太郎", email: "taro@example.com" }
```

<details>
<summary>解答例</summary>

```javascript
// --- User.js ---
export default class User {
  constructor(name, email) {
    this.name = name;
    this.email = email;
  }

  greet() {
    return `こんにちは、${this.name}です`;
  }

  toJSON() {
    return { name: this.name, email: this.email };
  }
}

// --- app.js ---
import User from './User.js';

const user = new User("太郎", "taro@example.com");
console.log(user.greet());
console.log(JSON.stringify(user.toJSON()));
```

</details>

### 問題3：名前の衝突解決

2つのモジュールから同じ名前の関数をインポートする場合の解決方法を示してください。

```javascript
// stringUtils.js
export function format(str) {
  return str.trim().toLowerCase();
}

// dateUtils.js
export function format(date) {
  return `${date.getFullYear()}/${date.getMonth() + 1}/${date.getDate()}`;
}
```

3つの方法で名前の衝突を解決してください。

<details>
<summary>解答例</summary>

```javascript
// 方法1: エイリアス（as）を使う
import { format as formatString } from './stringUtils.js';
import { format as formatDate } from './dateUtils.js';

console.log(formatString("  Hello  "));     // "hello"
console.log(formatDate(new Date()));         // "2026/3/25"

// 方法2: 名前空間インポート
import * as stringUtils from './stringUtils.js';
import * as dateUtils from './dateUtils.js';

console.log(stringUtils.format("  Hello  "));
console.log(dateUtils.format(new Date()));

// 方法3: 片方だけ別名をつける
import { format } from './stringUtils.js';
import { format as formatDate2 } from './dateUtils.js';

console.log(format("  Hello  "));
console.log(formatDate2(new Date()));
```

</details>

<details>

### 問題5：再エクスポートとバレルファイル

以下のモジュール構成で、`index.js`（バレルファイル）を作成してください。

```
utils/
  ├── index.js      ← これを作成
  ├── math.js       ← add, subtract, PI をエクスポート
  ├── string.js     ← capitalize, truncate をエクスポート
  └── array.js      ← unique, chunk をデフォルトエクスポート & 名前付きエクスポート
```

`index.js` からすべてを再エクスポートし、使う側が1つのパスからインポートできるようにしてください。

<details>
<summary>解答例</summary>

```javascript
// --- utils/math.js ---
export function add(a, b) { return a + b; }
export function subtract(a, b) { return a - b; }
export const PI = 3.14159;

// --- utils/string.js ---
export function capitalize(str) {
  return str.charAt(0).toUpperCase() + str.slice(1);
}
export function truncate(str, maxLen, suffix = "...") {
  if (str.length <= maxLen) return str;
  return str.slice(0, maxLen - suffix.length) + suffix;
}

// --- utils/array.js ---
export function unique(arr) {
  return [...new Set(arr)];
}
export function chunk(arr, size) {
  const chunks = [];
  for (let i = 0; i < arr.length; i += size) {
    chunks.push(arr.slice(i, i + size));
  }
  return chunks;
}

// --- utils/index.js（バレルファイル）---
export { add, subtract, PI } from './math.js';
export { capitalize, truncate } from './string.js';
export { unique, chunk } from './array.js';

// --- app.js（使う側）---
import { add, capitalize, unique, PI } from './utils/index.js';

console.log(add(3, 5));                    // 8
console.log(capitalize("hello"));          // "Hello"
console.log(unique([1, 2, 2, 3, 3]));     // [1, 2, 3]
console.log(PI);                            // 3.14159
```

</details>

## チャレンジ問題

### 問題7：モジュール設計の実践

以下の要件を満たすモジュール構成を設計・実装してください。

**要件:** 簡易的なタスク管理システム

- `Task` クラス（id, title, completed を持つ）
- `TaskManager` クラス（タスクの追加・完了・削除・一覧取得）
- `taskStore` モジュール（データの永続化を模擬するストレージ）

各モジュールのコードと、それらを使用する `app.js` を書いてください。

<details>
<summary>解答例</summary>

```javascript
// --- models/Task.js ---
let nextId = 1;

export default class Task {
  constructor(title) {
    this.id = nextId++;
    this.title = title;
    this.completed = false;
    this.createdAt = new Date().toISOString();
  }

  complete() {
    this.completed = true;
  }

  toString() {
    const status = this.completed ? "[完了]" : "[未完了]";
    return `${status} #${this.id}: ${this.title}`;
  }
}

// --- services/TaskManager.js ---
import Task from '../models/Task.js';

export default class TaskManager {
  constructor() {
    this.tasks = [];
  }

  add(title) {
    const task = new Task(title);
    this.tasks.push(task);
    return task;
  }

  complete(id) {
    const task = this.tasks.find(t => t.id === id);
    if (!task) throw new Error(`タスク #${id} が見つかりません`);
    task.complete();
    return task;
  }

  remove(id) {
    const index = this.tasks.findIndex(t => t.id === id);
    if (index === -1) throw new Error(`タスク #${id} が見つかりません`);
    return this.tasks.splice(index, 1)[0];
  }

  getAll() {
    return [...this.tasks];
  }

  getPending() {
    return this.tasks.filter(t => !t.completed);
  }

  getCompleted() {
    return this.tasks.filter(t => t.completed);
  }
}

// --- app.js ---
import TaskManager from './services/TaskManager.js';

const manager = new TaskManager();

// タスクの追加
manager.add("JavaScriptの教材を読む");
manager.add("演習問題を解く");
manager.add("サンプルコードを実行する");
manager.add("オリジナルアプリを作る");

// タスクの完了
manager.complete(1);
manager.complete(2);

// タスクの削除
manager.remove(4);

// 一覧表示
console.log("--- 全タスク ---");
manager.getAll().forEach(task => console.log(task.toString()));

console.log("--- 未完了 ---");
manager.getPending().forEach(task => console.log(task.toString()));

console.log("--- 完了済み ---");
manager.getCompleted().forEach(task => console.log(task.toString()));
```

**期待される出力：**
```
--- 全タスク ---
[完了] #1: JavaScriptの教材を読む
[完了] #2: 演習問題を解く
[未完了] #3: サンプルコードを実行する
--- 未完了 ---
[未完了] #3: サンプルコードを実行する
--- 完了済み ---
[完了] #1: JavaScriptの教材を読む
[完了] #2: 演習問題を解く
```

</details>
