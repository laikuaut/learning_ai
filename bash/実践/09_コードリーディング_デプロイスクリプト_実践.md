## 読解対象コード

```bash
#!/usr/bin/env bash
set -euo pipefail

# ==================== 設定 ====================

readonly SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
readonly CONFIG_FILE="${SCRIPT_DIR}/deploy.conf"
readonly LOCK_FILE="/tmp/deploy.lock"

# デフォルト設定
DEPLOY_USER="${DEPLOY_USER:-deploy}"
DEPLOY_HOST="${DEPLOY_HOST:-}"
DEPLOY_PATH="${DEPLOY_PATH:-/var/www/app}"
KEEP_RELEASES=5
ROLLBACK=false
DRY_RUN=false
VERBOSE=false

# ==================== ユーティリティ ====================

log() {
    local level="$1"
    shift
    local message="$*"
    local timestamp
    timestamp=$(date "+%Y-%m-%d %H:%M:%S")

    case "$level" in
        INFO)  local color='\033[32m' ;;
        WARN)  local color='\033[33m' ;;
        ERROR) local color='\033[31m' ;;
        *)     local color='\033[0m' ;;
    esac

    if [ -t 2 ]; then
        echo -e "${color}[${timestamp}] [${level}] ${message}\033[0m" >&2
    else
        echo "[${timestamp}] [${level}] ${message}" >&2
    fi
}

verbose() {
    if $VERBOSE; then
        log "DEBUG" "$@"
    fi
}

die() {
    log "ERROR" "$@"
    exit 1
}

run() {
    if $DRY_RUN; then
        log "DRY" "$*"
        return 0
    fi
    verbose "実行: $*"
    "$@"
}

# ==================== ロック管理 ====================

acquire_lock() {
    if [ -f "$LOCK_FILE" ]; then
        local pid
        pid=$(cat "$LOCK_FILE")
        if kill -0 "$pid" 2>/dev/null; then
            die "別のデプロイプロセスが実行中です (PID: $pid)"
        else
            log "WARN" "古いロックファイルを削除します"
            rm -f "$LOCK_FILE"
        fi
    fi
    echo $$ > "$LOCK_FILE"
    verbose "ロック取得: PID $$"
}

release_lock() {
    rm -f "$LOCK_FILE"
    verbose "ロック解放"
}

# ==================== 設定読み込み ====================

load_config() {
    if [ -f "$CONFIG_FILE" ]; then
        verbose "設定ファイル読み込み: $CONFIG_FILE"
        while IFS='=' read -r key value; do
            [[ "$key" =~ ^#.*$ ]] && continue
            [[ -z "$key" ]] && continue
            key=$(echo "$key" | tr -d '[:space:]')
            value=$(echo "$value" | tr -d '[:space:]')
            case "$key" in
                deploy_host) DEPLOY_HOST="$value" ;;
                deploy_user) DEPLOY_USER="$value" ;;
                deploy_path) DEPLOY_PATH="$value" ;;
                keep_releases) KEEP_RELEASES="$value" ;;
            esac
        done < "$CONFIG_FILE"
    fi
}

# ==================== デプロイ処理 ====================

prepare_release() {
    local release_dir="$1"
    local source_dir="${SCRIPT_DIR}"

    log "INFO" "リリース準備: $release_dir"

    run mkdir -p "$release_dir"

    # ソースコードをコピー（.gitと node_modules を除外）
    run rsync -a \
        --exclude='.git' \
        --exclude='node_modules' \
        --exclude='*.log' \
        --exclude='deploy.conf' \
        "$source_dir/" "$release_dir/"

    # ビルド処理
    if [ -f "${release_dir}/package.json" ]; then
        log "INFO" "npm install 実行中..."
        run bash -c "cd '$release_dir' && npm install --production 2>&1 | tail -5"
    fi

    # 設定ファイルのシンボリックリンク
    local shared_dir
    shared_dir="$(dirname "$release_dir")/shared"
    if [ -d "$shared_dir" ]; then
        for conf in "$shared_dir"/config/*; do
            [ -f "$conf" ] || continue
            local name
            name=$(basename "$conf")
            run ln -sf "$conf" "${release_dir}/config/${name}"
            verbose "シンボリックリンク: $name"
        done
    fi
}

switch_release() {
    local release_dir="$1"
    local current_link
    current_link="$(dirname "$release_dir")/current"

    log "INFO" "リリース切り替え: $(basename "$release_dir")"

    # 現在のリリースを記録（ロールバック用）
    if [ -L "$current_link" ]; then
        local previous
        previous=$(readlink -f "$current_link")
        echo "$previous" > "$(dirname "$release_dir")/.previous_release"
    fi

    run ln -sfn "$release_dir" "$current_link"
}

cleanup_releases() {
    local releases_dir="$1"
    local keep=$2

    local -a releases=()
    mapfile -t releases < <(ls -dt "${releases_dir}"/20* 2>/dev/null)

    local total=${#releases[@]}
    if [ "$total" -gt "$keep" ]; then
        log "INFO" "古いリリースを削除: $((total - keep))件"
        for ((i = keep; i < total; i++)); do
            run rm -rf "${releases[$i]}"
            verbose "削除: $(basename "${releases[$i]}")"
        done
    fi
}

rollback() {
    local deploy_base="$1"
    local previous_file="${deploy_base}/.previous_release"

    if [ ! -f "$previous_file" ]; then
        die "ロールバック先が見つかりません"
    fi

    local previous
    previous=$(cat "$previous_file")

    if [ ! -d "$previous" ]; then
        die "前のリリースディレクトリが存在しません: $previous"
    fi

    log "INFO" "ロールバック: $(basename "$previous")"
    run ln -sfn "$previous" "${deploy_base}/current"
    log "INFO" "ロールバック完了"
}

# ==================== メイン処理 ====================

usage() {
    cat <<EOF
使い方: $(basename "$0") [オプション]

オプション:
  -h HOST     デプロイ先ホスト
  -u USER     デプロイユーザー
  -p PATH     デプロイパス
  -r          ロールバック
  -n          ドライラン（実行せずに表示のみ）
  -v          詳細出力
  --help      このヘルプを表示

環境変数:
  DEPLOY_HOST   デプロイ先ホスト
  DEPLOY_USER   デプロイユーザー
  DEPLOY_PATH   デプロイパス
EOF
}

main() {
    # オプション解析
    while getopts ":h:u:p:rnv-:" opt; do
        case "$opt" in
            h) DEPLOY_HOST="$OPTARG" ;;
            u) DEPLOY_USER="$OPTARG" ;;
            p) DEPLOY_PATH="$OPTARG" ;;
            r) ROLLBACK=true ;;
            n) DRY_RUN=true ;;
            v) VERBOSE=true ;;
            -)
                case "$OPTARG" in
                    help) usage; exit 0 ;;
                    *) die "不明なオプション: --$OPTARG" ;;
                esac
                ;;
            :) die "オプション -$OPTARG には値が必要です" ;;
            *) die "不明なオプション: -$OPTARG" ;;
        esac
    done

    # 設定読み込み
    load_config

    # ロック取得 & クリーンアップ登録
    acquire_lock
    trap 'release_lock' EXIT

    local deploy_base="$DEPLOY_PATH"
    local releases_dir="${deploy_base}/releases"

    if $ROLLBACK; then
        rollback "$deploy_base"
        return 0
    fi

    # デプロイ実行
    local release_name
    release_name=$(date "+%Y%m%d_%H%M%S")
    local release_dir="${releases_dir}/${release_name}"

    log "INFO" "===== デプロイ開始 ====="
    log "INFO" "リリース: $release_name"

    prepare_release "$release_dir"
    switch_release "$release_dir"
    cleanup_releases "$releases_dir" "$KEEP_RELEASES"

    log "INFO" "===== デプロイ完了 ====="
}

main "$@"
```

## 解答例

<details>
<summary>設問1〜5の解答</summary>

### 設問1
スクリプト自身が存在するディレクトリの**絶対パス**を取得しています。

1. `${BASH_SOURCE[0]}` → 実行中のスクリプトのパス
2. `dirname` → ディレクトリ部分を抽出
3. `cd ... && pwd` → そのディレクトリに移動して絶対パスを取得

`$(pwd)` だけだと**スクリプトを実行した場所**（カレントディレクトリ）が返ります。`/home/user` から `/opt/deploy/deploy.sh` を実行すると、`$(pwd)` は `/home/user` になり、スクリプトと同じディレクトリにある `deploy.conf` を見つけられません。

### 設問2
`${変数名:-デフォルト値}` は「変数が未定義または空文字の場合にデフォルト値を使う」パラメータ展開です。

`DEPLOY_USER=admin` を設定して実行すると、`${DEPLOY_USER:-deploy}` は `admin` を返します。つまり環境変数で設定を上書きできます。これにより、設定ファイルを変更せずに一時的に設定を変えることが可能です。

### 設問3
`-t 2` はファイルディスクリプタ2（標準エラー出力）が**端末（ターミナル）に接続されているか**をチェックしています。

- 端末の場合 → カラーコード（`\033[32m` 等）を付けて見やすく出力
- ファイルやパイプの場合 → カラーコードなしのプレーンテキストで出力

`./deploy.sh 2>deploy.log` のようにリダイレクトされた場合、ログファイルにカラーコードが混入するのを防いでいます。

### 設問4
`kill -0 "$pid"` はプロセスを**終了させません**。シグナル0は「シグナルを送れるか（＝プロセスが存在するか）」をチェックするだけです。

- プロセスが存在する → 終了コード0（成功）→ 別のデプロイが実行中
- プロセスが存在しない → 終了コード非0（失敗）→ 古いロックファイルなので削除して良い

これにより、前回のデプロイがクラッシュして残ったロックファイルを安全に処理できます。

### 設問5
`run` 関数の役割は **コマンド実行の抽象化** です。

1. **ドライラン対応**: `DRY_RUN=true` のとき実際のコマンドを実行せず、何が実行されるかを表示するだけ
2. **詳細ログ**: `VERBOSE=true` のとき実行するコマンドをログに出力
3. **一元管理**: 将来的にリモート実行（SSH経由）への切り替えも `run` 関数の変更だけで対応可能

すべての副作用のあるコマンドを `run` 経由にすることで、テストやデバッグが容易になります。

### 設問6
```
/var/www/app/                    ← DEPLOY_PATH（deploy_base）
├── releases/                    ← リリースのバージョン管理
│   ├── 20240315_100000/         ← 過去のリリース
│   ├── 20240315_120000/         ← 過去のリリース
│   └── 20240315_140000/         ← 最新リリース
├── shared/                      ← リリース間で共有するファイル
│   └── config/
│       └── database.yml         ← 共有設定ファイル
├── current -> releases/20240315_140000/  ← シンボリックリンク
└── .previous_release            ← 前のリリースのパスを記録
```

- `releases/` : 各デプロイのスナップショットをタイムスタンプ付きで保持
- `current` : 現在使用中のリリースへのシンボリックリンク。Webサーバーはここを参照
- `shared/` : DB設定やアップロードファイルなど、リリース間で共有するもの

この構造はCapistranoスタイルと呼ばれ、業界標準のデプロイパターンです。

### 設問7
`tr -d '[:space:]'` ですべての空白を除去しているため、パスの中のスペースも消えてしまいます。

例: `deploy_path = /var/www/my app` → `value` は `/var/www/myapp` になる

修正案：先頭・末尾の空白だけ除去する。
```bash
value=$(echo "$value" | sed 's/^[[:space:]]*//; s/[[:space:]]*$//')
```

### 設問8
`trap 'release_lock' EXIT` は **スクリプトが終了するとき** に実行されます。

- 正常終了（`exit 0`）→ 実行される
- `die` でのエラー終了（`exit 1`）→ **実行される**
- `set -e` によるエラー終了 → **実行される**
- `kill` でプロセスが終了された場合 → **EXIT trapは実行される**
- `kill -9`（SIGKILL）の場合 → 実行**されない**

`EXIT` は最も強力なtrapで、ほぼすべての終了パターンで発火します。そのためロックの解放に適しています。

### 設問9
`run` 関数を修正してリモート実行に対応させるのが最も効率的です。

```bash
run() {
    if $DRY_RUN; then
        log "DRY" "$*"
        return 0
    fi
    verbose "実行: $*"

    if [ -n "$DEPLOY_HOST" ]; then
        # リモート実行
        ssh "${DEPLOY_USER}@${DEPLOY_HOST}" "$@"
    else
        # ローカル実行
        "$@"
    fi
}
```

また、`prepare_release` の `rsync` 部分はリモートパスに変更が必要です：
```bash
run rsync -a ... "$source_dir/" "${DEPLOY_USER}@${DEPLOY_HOST}:${release_dir}/"
```

### 設問10
```bash
# フック実行関数
run_hooks() {
    local hook_name="$1"    # "pre-deploy", "post-deploy" 等
    local hook_dir="${SCRIPT_DIR}/hooks/${hook_name}.d"

    if [ ! -d "$hook_dir" ]; then
        verbose "フックディレクトリなし: $hook_dir"
        return 0
    fi

    log "INFO" "フック実行: $hook_name"
    for hook_script in "$hook_dir"/*; do
        [ -x "$hook_script" ] || continue
        verbose "フック: $(basename "$hook_script")"
        run "$hook_script" "$release_dir" || {
            log "WARN" "フック失敗: $(basename "$hook_script")"
            return 1
        }
    done
}
```

呼び出し箇所：
```bash
# main 関数内
run_hooks "pre-deploy"          # prepare_release の前
prepare_release "$release_dir"
run_hooks "post-build"          # prepare_release の後、switch の前
switch_release "$release_dir"
run_hooks "post-deploy"         # switch_release の後
```

```
hooks/
├── pre-deploy.d/
│   └── 01_check_dependencies.sh
├── post-build.d/
│   └── 01_run_tests.sh
└── post-deploy.d/
    ├── 01_restart_app.sh
    └── 02_notify_slack.sh
```
