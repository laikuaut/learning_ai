## 読解対象コード

```bash
#!/bin/bash
# =============================================================
# server_monitor.sh - サーバー監視スクリプト
# 用途: CPU、メモリ、ディスク、プロセス、ネットワークの状態を監視し、
#       異常を検知した場合にアラートを発行する
# =============================================================

set -euo pipefail

# ==================== 設定セクション ====================

readonly CONFIG_FILE="${HOME}/.server_monitor.conf"
readonly LOG_DIR="/var/log/server_monitor"
readonly LOG_FILE="${LOG_DIR}/monitor.log"
readonly ALERT_FILE="${LOG_DIR}/alerts.log"
readonly PID_FILE="/tmp/server_monitor.pid"

# デフォルト閾値
CPU_THRESHOLD=80
MEM_THRESHOLD=85
DISK_THRESHOLD=90
LOAD_THRESHOLD_MULTIPLIER=2

# 監視対象プロセス（スペース区切り）
WATCH_PROCESSES="nginx mysql sshd"

# チェック対象ポート（スペース区切り）
WATCH_PORTS="80 443 3306 22"

# 設定ファイルの読み込み
load_config() {
    if [ -f "$CONFIG_FILE" ]; then
        # shellcheck source=/dev/null
        source "$CONFIG_FILE"
        log "INFO" "設定ファイルを読み込みました: $CONFIG_FILE"
    else
        log "WARN" "設定ファイルが見つかりません。デフォルト値を使用します"
    fi
}

# ==================== ユーティリティ関数 ====================

log() {
    local level="$1"
    local message="$2"
    local timestamp
    timestamp=$(date '+%Y-%m-%d %H:%M:%S')
    echo "[$timestamp] [$level] $message" | tee -a "$LOG_FILE"
}

alert() {
    local component="$1"
    local message="$2"
    local timestamp
    timestamp=$(date '+%Y-%m-%d %H:%M:%S')
    local alert_msg="[$timestamp] [ALERT] [$component] $message"
    echo "$alert_msg" | tee -a "$ALERT_FILE"
    log "ALERT" "[$component] $message"
}

# ==================== 初期化 ====================

initialize() {
    # ログディレクトリの作成
    mkdir -p "$LOG_DIR"

    # 多重起動防止
    if [ -f "$PID_FILE" ]; then
        local old_pid
        old_pid=$(cat "$PID_FILE")
        if kill -0 "$old_pid" 2>/dev/null; then
            echo "エラー: 既に実行中です (PID: $old_pid)" >&2
            exit 1
        else
            rm -f "$PID_FILE"
        fi
    fi
    echo $$ > "$PID_FILE"

    # 終了時のクリーンアップ
    trap cleanup EXIT INT TERM
}

cleanup() {
    rm -f "$PID_FILE"
    log "INFO" "監視スクリプト終了 (PID: $$)"
}

# ==================== 監視関数 ====================

check_cpu() {
    local cpu_usage
    cpu_usage=$(top -bn1 | grep "Cpu(s)" | awk '{print int($2 + $4)}')

    if [ "$cpu_usage" -ge "$CPU_THRESHOLD" ]; then
        alert "CPU" "使用率が ${cpu_usage}% です（閾値: ${CPU_THRESHOLD}%）"
        return 1
    fi
    log "INFO" "CPU使用率: ${cpu_usage}% [OK]"
    return 0
}

check_memory() {
    local mem_info
    mem_info=$(free | awk '/Mem:/ {printf "%.1f %d %d", $3/$2*100, $3/1024, $2/1024}')
    local mem_percent mem_used_mb mem_total_mb
    read -r mem_percent mem_used_mb mem_total_mb <<< "$mem_info"
    local mem_int=${mem_percent%.*}

    if [ "$mem_int" -ge "$MEM_THRESHOLD" ]; then
        alert "MEMORY" "使用率 ${mem_percent}%（${mem_used_mb}MB/${mem_total_mb}MB）閾値: ${MEM_THRESHOLD}%"
        return 1
    fi
    log "INFO" "メモリ使用率: ${mem_percent}%（${mem_used_mb}MB/${mem_total_mb}MB） [OK]"
    return 0
}

check_disk() {
    local has_alert=false
    while IFS= read -r line; do
        local usage mount
        usage=$(echo "$line" | awk '{print $5}' | tr -d '%')
        mount=$(echo "$line" | awk '{print $6}')

        if [ -n "$usage" ] && [ "$usage" -ge "$DISK_THRESHOLD" ] 2>/dev/null; then
            alert "DISK" "${mount} 使用率 ${usage}%（閾値: ${DISK_THRESHOLD}%）"
            has_alert=true
        fi
    done < <(df -h | tail -n +2)

    if [ "$has_alert" = true ]; then
        return 1
    fi
    log "INFO" "ディスク使用率: 全パーティション正常 [OK]"
    return 0
}

check_load() {
    local load_1m cpu_count threshold
    load_1m=$(cat /proc/loadavg | awk '{print $1}')
    cpu_count=$(nproc)
    threshold=$(echo "$cpu_count * $LOAD_THRESHOLD_MULTIPLIER" | bc)

    local load_int=${load_1m%.*}
    local threshold_int=${threshold%.*}

    if [ "${load_int:-0}" -ge "${threshold_int:-1}" ]; then
        alert "LOAD" "ロードアベレージ ${load_1m}（閾値: ${threshold}, CPU数: ${cpu_count}）"
        return 1
    fi
    log "INFO" "ロードアベレージ: ${load_1m}（CPU数: ${cpu_count}, 閾値: ${threshold}） [OK]"
    return 0
}

check_processes() {
    local has_alert=false
    for proc in $WATCH_PROCESSES; do
        if ! pgrep -x "$proc" > /dev/null 2>&1; then
            alert "PROCESS" "プロセス '${proc}' が実行されていません"
            has_alert=true
        else
            local pid_count
            pid_count=$(pgrep -cx "$proc" 2>/dev/null || echo 0)
            log "INFO" "プロセス ${proc}: 稼働中（${pid_count} プロセス） [OK]"
        fi
    done

    if [ "$has_alert" = true ]; then
        return 1
    fi
    return 0
}

check_ports() {
    local has_alert=false
    for port in $WATCH_PORTS; do
        if ! ss -tlnp | grep -q ":${port} "; then
            alert "PORT" "ポート ${port} がリッスンしていません"
            has_alert=true
        else
            log "INFO" "ポート ${port}: リッスン中 [OK]"
        fi
    done

    if [ "$has_alert" = true ]; then
        return 1
    fi
    return 0
}

check_zombie_processes() {
    local zombie_count
    zombie_count=$(ps aux | awk '$8 ~ /^Z/ {count++} END {print count+0}')

    if [ "$zombie_count" -gt 0 ]; then
        alert "ZOMBIE" "ゾンビプロセスが ${zombie_count} 個検出されました"
        ps aux | awk '$8 ~ /^Z/ {printf "  PID=%s CMD=%s\n", $2, $11}' >> "$ALERT_FILE"
        return 1
    fi
    log "INFO" "ゾンビプロセス: なし [OK]"
    return 0
}

# ==================== メイン処理 ====================

generate_report() {
    local total_checks=$1
    local failed_checks=$2
    local status

    if [ "$failed_checks" -eq 0 ]; then
        status="ALL OK"
    else
        status="ALERT (${failed_checks}/${total_checks} checks failed)"
    fi

    cat << EOF

======================================
  サーバー監視レポート
  日時: $(date '+%Y-%m-%d %H:%M:%S')
  ホスト: $(hostname)
  ステータス: $status
======================================
EOF
}

main() {
    initialize
    load_config

    log "INFO" "監視スクリプト開始 (PID: $$)"

    local total_checks=0
    local failed_checks=0

    local checks=(
        check_cpu
        check_memory
        check_disk
        check_load
        check_processes
        check_ports
        check_zombie_processes
    )

    for check_func in "${checks[@]}"; do
        ((total_checks++))
        if ! "$check_func"; then
            ((failed_checks++))
        fi
    done

    generate_report "$total_checks" "$failed_checks"

    if [ "$failed_checks" -gt 0 ]; then
        log "WARN" "監視完了: ${failed_checks}件のアラートがあります"
        exit 2
    else
        log "INFO" "監視完了: 全チェック正常"
        exit 0
    fi
}

main "$@"
```

## 解答例

<details>
<summary>設問1の解答</summary>

`set -euo pipefail` は、シェルスクリプトの堅牢性を高めるための設定です。

- **`-e`（errexit）**: コマンドが非ゼロの終了ステータス（exit status）を返した場合、スクリプトを即座に終了する。ただし `if` 文の条件や `||` の左側では適用されない
- **`-u`（nounset）**: 未定義の変数を参照しようとした場合にエラーとする。タイポ（typo）で変数名を間違えた場合にすぐ気づける
- **`-o pipefail`**: パイプラインの中でどれか1つのコマンドが失敗した場合、パイプライン全体の終了ステータスを非ゼロにする。通常は最後のコマンドの終了ステータスだけが使われるため、途中の失敗が見逃される

この3つは「Bashスクリプトの安全な書き方」として実務で広く推奨されています。

</details>

<details>
<summary>設問2の解答</summary>

`readonly` で定義された変数は、後から値を変更できません（定数扱い）。

```bash
readonly LOG_DIR="/var/log/server_monitor"
LOG_DIR="/tmp"  # エラー: readonly variable
```

設定値に `readonly` を使う理由：
- **意図しない上書きの防止**: スクリプトのどこかで誤ってパスを変更してしまうリスクを排除する
- **コードの意図の明示**: 「この値は変わらない」ということを読み手に伝える
- **バグの早期発見**: 変更しようとした時点でエラーになるため、問題を早く検知できる

</details>

<details>
<summary>設問3の解答</summary>

多重起動防止の流れ：

1. PIDファイル（`/tmp/server_monitor.pid`）が存在するかチェック
2. 存在する場合、ファイルに記録されたPIDを読み取る
3. `kill -0 "$old_pid"` でそのPIDのプロセスが生きているか確認
   - `kill -0` はシグナルを送らず、プロセスの存在だけを確認する特殊なシグナル
   - プロセスが存在すれば終了ステータス0（成功）、存在しなければ非ゼロ
4. ��ロセスが生きていれば「既に実行中」としてエラー終了
5. プロセスが死んでいれば（前回の異常終了など）、古いPIDファイルを削除して続行
6. 自分のPID（`$$`）をPIDファイルに書き込む

これにより、cronで5分ごとに実行される場合に前回の処理がまだ終わっていないケースを安全に検出できます。

</details>

<details>
<summary>設問4の解答</summary>

`trap cleanup EXIT INT TERM` は、特定のシグナル（signal）を受け取った時に `cleanup` 関数を実行するという設定です。

- **EXIT**: スクリプトが終了する時（正常終了・エラー終了どちらでも）
- **INT**: Ctrl+C が押された時（割り込みシグナル）
- **TERM**: `kill` コマンドで終了シグナルが送られた時

`trap` がなかった場合の問題：
- Ctrl+C で中断した場合、PIDファイルが残ったままになる
- 次回実行時に「既に実行中です」と誤判定され、起動できなくなる
- PIDファイルの手動削除が必要になる

</details>

<details>
<summary>設問5の解答</summary>

`< <(コマンド)` は **プロセス置換**（process substitution）と呼ばれる構文です。

`<(df -h | tail -n +2)` はコマンドの出力を一時的なファイルディスクリプタとして扱い、`while read` にリダイレクトで渡しています。

通常のパイプ `df -h | while read ...` を使わない理由：
- パイプの右側はサブシェル（subshell）で実行される
- サブシェル内で変更した変数（`has_alert=true`）は、パイプの外に反映されない
- プロセス置換を使えば `while` ループが現在のシェルで実行されるため、変数の変更が正しく反映される

```bash
# パイプ版（バグ：has_alert が外に反映されない）
df -h | while read line; do
    has_alert=true  # サブシェル内の変数
done
echo "$has_alert"   # false のまま

# プロセス置換版（正常：has_alert が反映される）
while read line; do
    has_alert=true  # 現在のシェルの変数
done < <(df -h)
echo "$has_alert"   # true
```

</details>

<details>
<summary>設問6の解答</summary>

`<<<` は **ヒアストリング**（here string）と呼ばれる構文です。文字列を標準入力としてコマンドに渡します。

```bash
read -r a b c <<< "1 2 3"
# a=1, b=2, c=3
```

`awk` の出力をスペース区切りにして `read` で受ける理由：
- `awk` は1つの `printf` で3つの値を計算してスペース区切りで出力
- `read` の単語分割機能で、3つの変数に一度に代入できる
- 各値を別々に `awk` で取得するよりも、`free` コマンドを1回だけ実行すればよいため効率的
- パイプを使うとサブシェル問題（設問5と同じ）が発生するが、ヒアストリングなら問題ない

</details>

<details>
<summary>設問7の解答</summary>

`pgrep -x` の `-x` は「完全一致」（exact match）を意味します。

`-x` がない場合、部分一致でプロセスを検索するため誤検知が起きます。

具体例：
- `WATCH_PROCESSES` に `ssh` を設定した場合
  - `-x` あり: `ssh` という名前のプロセスだけにマッチ
  - `-x` なし: `sshd`, `ssh-agent`, `autossh` など、`ssh` を含むすべてのプロセスにマッチ
- `mysql` を監視したい場合
  - `-x` なし: `mysqld`, `mysql_safe`, `mysqldump` にもマッチしてしまい、`mysql` プロセスが停止していても「稼働中」と誤判定される可能性がある

</details>

<details>
<summary>設問8の解答</summary>

配列でチェック関数を管理するメリット：

1. **拡張性**: 新しいチェック項目は配列に1行追加するだけ
2. **一貫性**: すべてのチェックが同じインターフェース（成功なら0、失敗なら非0を返す）で統一されている
3. **カウントの自動化**: ループで total_checks と failed_checks を自動カウントできる
4. **順序の変更が容易**: 配列の順番を変えるだけでチェック順を変更できる
5. **条件付き無効化が容易**: 配列から要素を削除すればチェックをスキップできる

新しいチェック項目（例: スワップ使用率チェック）を追加する場合：
1. `check_swap()` 関数を新たに定義する（成功なら `return 0`、失敗なら `return 1`）
2. `checks` 配列に `check_swap` を追加する

メイン処理のループは変更不要です。

</details>

<summary>設問9の解答</summary>

改善案（3つ以上）：

1. **メール/Slack通知機能の追加**
   - 現状: アラートはログファイルに記録されるだけで、人間が気づかない
   - 改善: `alert()` 関数にメール送信やSlack Webhookでの通知を追加
   - 理由: リアルタイムで異常に対応できるようになる

2. **設定ファイルの検証**
   - 現状: `source "$CONFIG_FILE"` で設定を読み込むが、値の妥当性チェックがない
   - 改善: 閾値が数値であること、0〜100の範囲内であることを検証する
   - 理由: 不正な設定値でスクリプトが予期しない動作をするのを防ぐ

3. **チェック結果のJSON出力**
   - 現状: テキスト形式のログのみ
   - 改善: JSON形式で結果を出力するオプションを追加
   - 理由: 監視ツール（Zabbix, Prometheus等）と連携しやすくなる

4. **タイムアウトの設定**
   - 現状: 各チェック関数にタイムアウトがない
   - 改善: `timeout` コマンドで各チェックに制限時間を設ける
   - 理由: ネットワーク関連のチェックが応答しない場合にスクリプト全体がハングすることを防ぐ

5. **前回の状態との比較**
   - 現状: 毎回の実行が独立しており、状態変化を検知できない
   - 改善: 前回の結果をファイルに保存し、状態が変わった時だけアラートを出す
   - 理由: 同じアラートが毎回出続けるのを防ぎ、ノイズを減らす

**メール通知の追加方法:**

`alert()` 関数を以下のように拡張します。

```bash
ALERT_EMAIL="admin@example.com"

alert() {
    local component="$1"
    local message="$2"
    local timestamp
    timestamp=$(date '+%Y-%m-%d %H:%M:%S')
    local alert_msg="[$timestamp] [ALERT] [$component] $message"
    echo "$alert_msg" | tee -a "$ALERT_FILE"
    log "ALERT" "[$component] $message"

    # メール通知
    if command -v mail > /dev/null 2>&1; then
        echo "$alert_msg" | mail -s "[ALERT] $(hostname): $component 異常検知" "$ALERT_EMAIL"
    fi
}
```

あるいは、`main()` 関数の末尾で、アラートがあった場合にまとめて通知する方法もあります。

```bash
if [ "$failed_checks" -gt 0 ]; then
    {
        generate_report "$total_checks" "$failed_checks"
        echo ""
        echo "--- アラート詳細 ---"
        tail -20 "$ALERT_FILE"
    } | mail -s "[ALERT] $(hostname): ${failed_checks}件の異常" "$ALERT_EMAIL"
fi
```

まとめて通知する方法の方がメールの数が抑えられるため、実運用ではこちらが好まれることが多いです。
