**シナリオ:**
中規模のECサイト「ShopNow」を運営するA社。従業員100名、年商20億円。以下のシステム構成で運用しています。

- Webサーバ（2台、ロードバランサ配下）
- アプリケーションサーバ（2台）
- データベースサーバ（1台、日次バックアップ）

- 全従業員にノートPC支給（リモートワーク対応）

**直近1年で発生したインシデント:**
- フィッシングメールにより3名がパスワードを漏洩（被害は軽微）

**課題4:** 選んだ3つのリスクについて、リスク対応戦略（回避・低減・転嫁・受容）と具体的な対策を提案してください。

<summary>パート1のガイド：リスク対応戦略の考え方</summary>

| 戦略 | 英語 | 説明 | 例 |
|------|------|------|-----|
| 回避（Avoidance） | Risk Avoidance | リスクの原因となる活動を中止する | 個人情報の収集をやめる |
| 低減（Mitigation） | Risk Mitigation | 発生可能性や影響度を下げる対策をとる | ファイアウォール導入 |
| 転嫁（Transfer） | Risk Transfer | リスクを第三者に移転する | サイバー保険加入 |
| 受容（Acceptance） | Risk Acceptance | リスクをそのまま受け入れる | コスト対効果が合わない場合 |

**判断基準の目安:**
```
リスクスコア 20-25 → 回避 or 低減（即時対応必須）
リスクスコア 12-19 → 低減（計画的に対応）
リスクスコア  6-11 → 低減 or 転嫁（コスト対効果を検討）
リスクスコア  1-5  → 受容 or 低減（モニタリング）
```

<summary>パート2：リスクアセスメントツール（初心者向け）</summary>

```python
# セキュリティリスクアセスメントツール
# 学べる内容：リスク分析手法、リスクマトリクス、リスク対応戦略
# 実行方法：python risk_assessment.py

print("===== セキュリティリスクアセスメントツール =====\n")

# --- 資産の登録 ---
print("--- 資産の登録 ---")
asset_name = input("資産名: ")
asset_value = int(input("資産価値（1-5）: "))

# --- 脅威の登録 ---
print("\n--- 脅威と脆弱性の特定 ---")
threats = []
while True:
    threat_name = input(f"\n脅威{len(threats)+1}の名前（終了はEnter）: ").strip()
    if not threat_name:
        break
    likelihood = int(input(f"  発生可能性（1-5）: "))
    vulnerability = input(f"  関連する脆弱性: ")
    impact = int(input(f"  影響度（1-5）: "))
    score = likelihood * impact
    threats.append({
        "name": threat_name,
        "likelihood": likelihood,
        "vulnerability": vulnerability,
        "impact": impact,
        "score": score,
    })

if not threats:
    print("脅威が登録されていません。")
else:
    # --- リスクレベルの判定 ---
    def risk_level(score):
        if score >= 20:
            return "極高"
        elif score >= 12:
            return "高"
        elif score >= 6:
            return "中"
        elif score >= 3:
            return "低"
        else:
            return "極低"

    def recommended_action(score):
        if score >= 20:
            return "リスク回避 or 即時低減"
        elif score >= 12:
            return "リスク低減（計画的に対応）"
        elif score >= 6:
            return "リスク低減 or 転嫁"
        else:
            return "リスク受容（モニタリング）"

    # スコア降順でソート
    threats.sort(key=lambda t: t["score"], reverse=True)

    # --- リスクマトリクス ---
    print(f"\n========== リスクアセスメント結果 ==========")
    print(f"\n対象資産: {asset_name}（資産価値: {asset_value}）\n")

    print("--- リスクマトリクス ---")
    print(f"{'':>12} 影響度→ ", end="")
    for i in range(1, 6):
        print(f" {i:>3} ", end="")
    print()
    print(f"発生可能性↓")

    # 脅威の位置を記録
    threat_positions = {}
    for t in threats:
        key = (t["likelihood"], t["impact"])
        if key not in threat_positions:
            threat_positions[key] = []
        threat_positions[key].append(t["name"][:4])

    for likelihood in range(5, 0, -1):
        print(f"    {likelihood:>1}        ", end="")
        for impact in range(1, 6):
            score = likelihood * impact
            if (likelihood, impact) in threat_positions:
                print(f"★{score:>2}★", end="")
            else:
                print(f"[{score:>2}]", end=" ")
        print()

    # --- リスク一覧 ---
    print(f"\n--- リスク一覧（スコア降順） ---")
    print(f"{'#':>2} | {'脅威':<28} | {'可能性':>6} | {'影響度':>6} | {'スコア':>6} | レベル")
    print("-" * 75)
    for i, t in enumerate(threats, 1):
        level = risk_level(t["score"])
        print(f"{i:2d} | {t['name']:<28} | {t['likelihood']:>6} | {t['impact']:>6} | {t['score']:>6} | {level}")

    # --- リスク対応策 ---
    print(f"\n--- リスク対応策の推奨 ---")
    for t in threats:
        level = risk_level(t["score"])
        action = recommended_action(t["score"])
        print(f"\n[{level}] {t['name']} (スコア: {t['score']})")
        print(f"  脆弱性: {t['vulnerability']}")
        print(f"  推奨対応: {action}")
```

<summary>パート2：リスクアセスメントツール（改良版 ─ 複数資産対応＆レポート出力）</summary>

```python
# セキュリティリスクアセスメントツール（改良版）
# 学べる内容：体系的なリスクアセスメント、優先度判定、対応計画策定
# 実行方法：python risk_assessment_advanced.py

def risk_level(score):
    """リスクスコアからレベルを返す"""
    levels = [(20, "極高", "赤"), (12, "高", "橙"), (6, "中", "黄"), (3, "低", "緑")]
    for threshold, level, color in levels:
        if score >= threshold:
            return level, color
    return "極低", "青"

def risk_strategy(score):
    """リスクスコアから推奨戦略を返す"""
    if score >= 20:
        return ["回避", "即時低減"]
    elif score >= 12:
        return ["低減（計画的）"]
    elif score >= 6:
        return ["低減", "転嫁"]
    else:
        return ["受容", "モニタリング"]

# --- プリセットデータ ---
print("===== セキュリティリスクアセスメント（改良版） =====\n")

use_preset = input("ECサイトのプリセットデータを使用しますか？(y/n): ").strip().lower()

if use_preset == "y":
    assets = [
        {"name": "顧客データベース", "value": 5},
        {"name": "Webサーバ群", "value": 4},
        {"name": "社内ファイルサーバ", "value": 4},
        {"name": "従業員ノートPC", "value": 3},
    ]
    all_threats = [
        {"asset": "顧客データベース", "name": "SQLインジェクション", "likelihood": 3, "impact": 5, "vuln": "入力バリデーション不備"},
        {"asset": "顧客データベース", "name": "ランサムウェア", "likelihood": 3, "impact": 5, "vuln": "マルウェア対策不足"},
        {"asset": "顧客データベース", "name": "内部犯行", "likelihood": 2, "impact": 5, "vuln": "アクセス制御不備"},
        {"asset": "Webサーバ群", "name": "DDoS攻撃", "likelihood": 3, "impact": 4, "vuln": "DDoS対策未導入"},
        {"asset": "Webサーバ群", "name": "ハードウェア障害", "likelihood": 3, "impact": 3, "vuln": "冗長化不十分"},
        {"asset": "社内ファイルサーバ", "name": "不正アクセス（退職者）", "likelihood": 4, "impact": 4, "vuln": "ID管理プロセス不備"},
        {"asset": "従業員ノートPC", "name": "フィッシング", "likelihood": 4, "impact": 3, "vuln": "セキュリティ教育不足"},
        {"asset": "従業員ノートPC", "name": "紛失・盗難", "likelihood": 2, "impact": 4, "vuln": "暗号化未適用"},
    ]
else:
    assets = []
    n_assets = int(input("資産数: "))
    for i in range(n_assets):
        name = input(f"資産{i+1}の名前: ")
        value = int(input(f"  資産価値（1-5）: "))
        assets.append({"name": name, "value": value})

    all_threats = []
    for asset in assets:
        print(f"\n--- {asset['name']}の脅威 ---")
        while True:
            t_name = input(f"  脅威名（終了はEnter）: ").strip()
            if not t_name:
                break
            lh = int(input(f"    発生可能性（1-5）: "))
            imp = int(input(f"    影響度（1-5）: "))
            vuln = input(f"    脆弱性: ")
            all_threats.append({"asset": asset["name"], "name": t_name,
                                "likelihood": lh, "impact": imp, "vuln": vuln})

# スコア計算
for t in all_threats:
    t["score"] = t["likelihood"] * t["impact"]
    t["level"], t["color"] = risk_level(t["score"])

# スコア降順ソート
all_threats.sort(key=lambda x: x["score"], reverse=True)

# --- レポート出力 ---
print("\n" + "=" * 70)
print("  リスクアセスメントレポート")
print("=" * 70)

# 資産一覧
print("\n【1. 情報資産一覧】")
for a in assets:
    stars = "★" * a["value"] + "☆" * (5 - a["value"])
    print(f"  {a['name']:<20} 価値: {stars}")

# リスク一覧
print(f"\n【2. リスク一覧（全{len(all_threats)}件）】")
print(f"{'優先':>4} | {'資産':<16} | {'脅威':<22} | {'可能性':>4} | {'影響':>4} | {'スコア':>4} | レベル")
print("-" * 80)
for i, t in enumerate(all_threats, 1):
    print(f"{i:4d} | {t['asset']:<16} | {t['name']:<22} | {t['likelihood']:>4} | "
          f"{t['impact']:>4} | {t['score']:>4} | {t['level']}")

# リスク分布
print(f"\n【3. リスク分布】")
level_counts = {}
for t in all_threats:
    level_counts[t["level"]] = level_counts.get(t["level"], 0) + 1
for level in ["極高", "高", "中", "低", "極低"]:
    count = level_counts.get(level, 0)
    bar = "█" * count
    print(f"  {level:<4}: {bar} ({count}件)")

# 対応計画
print(f"\n【4. リスク対応計画】")
for t in all_threats[:5]:  # 上位5件
    strategies = risk_strategy(t["score"])
    print(f"\n  [{t['level']}] {t['name']}（{t['asset']}）スコア: {t['score']}")
    print(f"    脆弱性: {t['vuln']}")
    print(f"    推奨戦略: {', '.join(strategies)}")

# 総括
high_risk_count = sum(1 for t in all_threats if t["score"] >= 12)
print(f"\n{'='*70}")
print(f"【総括】")
print(f"  評価対象: {len(assets)}資産、{len(all_threats)}脅威")
print(f"  高リスク以上: {high_risk_count}件 → 早急な対応が必要")
print(f"  最優先対応: {all_threats[0]['name']}（スコア: {all_threats[0]['score']}）")
```
