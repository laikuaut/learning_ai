import math

def calc_average(scores):
    """平均値を計算する"""
    return sum(scores) / len(scores)

def calc_median(scores):
    """中央値を計算する"""
    sorted_scores = sorted(scores)
    n = len(sorted_scores)
    if n % 2 == 1:
        return sorted_scores[n // 2]
    else:
        return (sorted_scores[n // 2 - 1] + sorted_scores[n // 2]) / 2

def calc_stddev(scores):
    """標準偏差を計算する"""
    avg = calc_average(scores)
    variance = sum((x - avg) ** 2 for x in scores) / len(scores)
    return math.sqrt(variance)

def insertion_sort(data, key_func=None):
    """挿入ソート（キー関数指定可能）"""
    arr = data.copy()
    for i in range(1, len(arr)):
        current = arr[i]
        key = key_func(current) if key_func else current
        j = i - 1
        while j >= 0:
            compare = key_func(arr[j]) if key_func else arr[j]
            if compare <= key:
                break
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = current
    return arr

def binary_search_range(sorted_data, low_score, high_score, key_func):
    """ソート済みデータから点数範囲に該当する要素を二分探索で見つける"""
    # 下限の位置を二分探索
    lo, hi = 0, len(sorted_data)
    while lo < hi:
        mid = (lo + hi) // 2
        if key_func(sorted_data[mid]) < low_score:
            lo = mid + 1
        else:
            hi = mid

    start = lo

    # 上限の位置を二分探索
    lo, hi = start, len(sorted_data)
    while lo < hi:
        mid = (lo + hi) // 2
        if key_func(sorted_data[mid]) <= high_score:
            lo = mid + 1
        else:
            hi = mid

    return sorted_data[start:lo]

def insertion_sort_records(records, key_index, reverse=False):
    """レコードのリストを挿入ソートで並べ替える"""
    arr = records.copy()
    comparisons = 0
    for i in range(1, len(arr)):
        current = arr[i]
        key = current[key_index]
        j = i - 1
        while j >= 0:
            comparisons += 1
            cmp = arr[j][key_index]
            if (not reverse and cmp <= key) or (reverse and cmp >= key):
                break
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = current
    return arr, comparisons


def binary_search_lower(sorted_data, target, key_index):
    """二分探索で target 以上の最初の位置を返す"""
    lo, hi = 0, len(sorted_data)
    while lo < hi:
        mid = (lo + hi) // 2
        if sorted_data[mid][key_index] < target:
            lo = mid + 1
        else:
            hi = mid
    return lo


def binary_search_upper(sorted_data, target, key_index):
    """二分探索で target 以下の最後の位置+1を返す"""
    lo, hi = 0, len(sorted_data)
    while lo < hi:
        mid = (lo + hi) // 2
        if sorted_data[mid][key_index] <= target:
            lo = mid + 1
        else:
            hi = mid
    return lo


class Statistics:
    """統計計算クラス（標準ライブラリの statistics を使わず自作）"""

    @staticmethod
    def average(values):
        return sum(values) / len(values) if values else 0

    @staticmethod
    def median(values):
        s = sorted(values)
        n = len(s)
        if n == 0:
            return 0
        if n % 2 == 1:
            return s[n // 2]
        return (s[n // 2 - 1] + s[n // 2]) / 2

    @staticmethod
    def stddev(values):
        if len(values) < 2:
            return 0
        avg = Statistics.average(values)
        variance = sum((x - avg) ** 2 for x in values) / len(values)
        return math.sqrt(variance)

    @staticmethod
    def distribution(values, bins=None):
        if bins is None:
            bins = [(90, 100), (80, 89), (70, 79), (60, 69), (50, 59), (0, 49)]
        result = []
        for low, high in bins:
            count = sum(1 for v in values if low <= v <= high)
            result.append((low, high, count))
        return result

def menu_search(conn):

def menu_export(conn):
    """ハッシュ付きエクスポート"""
    c = conn.cursor()
    c.execute("""SELECT s.student_id, s.name, sub.subject_name, sc.score
                 FROM scores sc JOIN students s ON sc.student_id=s.student_id
                 JOIN subjects sub ON sc.subject_id=sub.subject_id""")
    lines = ["student_id,name,subject,score"]
    for r in c.fetchall():
        lines.append(f"{r[0]},{r[1]},{r[2]},{r[3]}")

    csv = "\n".join(lines)
    h = hashlib.sha256(csv.encode()).hexdigest()

    print(f"\n  データ: {len(lines)-1}件")
    print(f"  SHA-256: {h}")
    print(f"\n  ハッシュ値をデータと一緒に保存すれば、")
    print(f"  後から改ざんされていないか検証できます。")

    # 改ざんシミュレーション
    demo = input("\n  改ざん検出デモを実行しますか？ (y/n): ")
    if demo.lower() == "y":
        tampered = csv.replace("100", "999", 1)  # 1箇所改ざん
        h2 = hashlib.sha256(tampered.encode()).hexdigest()
        print(f"\n  元ハッシュ:   {h}")
        print(f"  改ざん後:     {h2}")
        print(f"  一致: {'✓' if h == h2 else '✗ 不一致！改ざんを検出しました'}")
