# 第7章：ルーティング

## この章のゴール

- SPA（Single Page Application）の概念を理解する
- `ngRoute` モジュールでルーティングを設定できる
- ルートパラメータ（`$routeParams`）を使ってページ間でデータを受け渡せる
- `resolve` でデータの事前読み込みができる
- `$location` サービスでURLを操作できる
- 完全なマルチページSPAを構築できる

---

## 7.1 SPAとルーティング

SPA（Single Page Application）は、**ページ遷移なしにコンテンツを切り替える**Webアプリケーションの構成パターンです。

```
【従来のWebアプリ】
ページA ──クリック──→ サーバーにリクエスト ──→ ページB（全体を再読み込み）

【SPA】
┌────────────────────────────┐
│ 共通部分（ヘッダー、ナビ）    │  ← そのまま
│ ┌──────────────────────┐   │
│ │  ng-view             │   │  ← ここだけ切り替わる
│ │  (コンテンツ領域)      │   │
│ └──────────────────────┘   │
│ 共通部分（フッター）         │  ← そのまま
└────────────────────────────┘
```

### ngRoute のセットアップ

AngularJS のルーティングには `angular-route.js` を追加で読み込みます。

```html
<!-- angular.min.js の後に読み込む -->
<script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.3/angular.min.js"></script>
<script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.3/angular-route.min.js"></script>
```

```javascript
// モジュールの依存関係に 'ngRoute' を追加
var app = angular.module('myApp', ['ngRoute']);
```

## 7.2 基本的なルーティング設定

### $routeProvider

```javascript
app.config(['$routeProvider', function($routeProvider) {
  $routeProvider
    .when('/', {
      template: '<h2>ホームページ</h2><p>ようこそ！</p>',
      controller: 'HomeCtrl'
    })
    .when('/about', {
      template: '<h2>このサイトについて</h2>',
      controller: 'AboutCtrl'
    })
    .otherwise({
      redirectTo: '/'  // マッチしないURLはホームにリダイレクト
    });
}]);
```

### 完全な例

```html
<!DOCTYPE html>
<html ng-app="myApp">
<head>
  <meta charset="UTF-8">
  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.3/angular.min.js"></script>
  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.3/angular-route.min.js"></script>
  <style>
    body { font-family: sans-serif; margin: 0; }
    nav { background: #35495e; padding: 0; }
    nav a { color: white; text-decoration: none; padding: 12px 20px; display: inline-block; }
    nav a:hover, nav a.active { background: #42b883; }
    .content { padding: 20px; max-width: 800px; margin: 0 auto; }
  </style>
</head>
<body>
  <nav>
    <a href="#!/">ホーム</a>
    <a href="#!/about">概要</a>
    <a href="#!/contact">お問い合わせ</a>
  </nav>
  
  <!-- ここにルートに応じたコンテンツが表示される -->
  <div class="content">
    <ng-view></ng-view>
  </div>

  <script>
    var app = angular.module('myApp', ['ngRoute']);
    
    app.config(['$routeProvider', function($routeProvider) {
      $routeProvider
        .when('/', {
          template: '<h2>ホーム</h2><p>メインページです。</p>'
        })
        .when('/about', {
          template: '<h2>概要</h2><p>AngularJS ルーティングのデモサイトです。</p>'
        })
        .when('/contact', {
          template: '<h2>お問い合わせ</h2><p>email@example.com</p>'
        })
        .otherwise({
          template: '<h2>404</h2><p>ページが見つかりません。</p>'
        });
    }]);
  </script>
</body>
</html>
```

> **注意：** AngularJS 1.6以降、URLのハッシュプレフィックスはデフォルトで `!` が付きます（`#!/path`）。`$locationProvider.hashPrefix('')` で `#/path` に変更できます。

## 7.3 ルートパラメータ

URLにパラメータを含め、`$routeParams` で取得できます。

```javascript
// ルート定義（:id がパラメータ）
.when('/users/:id', {
  templateUrl: '...',
  controller: 'UserDetailCtrl'
})

// コントローラで取得
.controller('UserDetailCtrl', ['$scope', '$routeParams', 
  function($scope, $routeParams) {
    var userId = $routeParams.id;  // URLのパラメータを取得
  }
]);
```

### ユーザー一覧・詳細の例

```html
<!DOCTYPE html>
<html ng-app="myApp">
<head>
  <meta charset="UTF-8">
  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.3/angular.min.js"></script>
  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.3/angular-route.min.js"></script>
  <style>
    body { font-family: sans-serif; margin: 0; }
    nav { background: #35495e; padding: 12px 20px; }
    nav a { color: white; text-decoration: none; margin-right: 16px; }
    .content { padding: 20px; max-width: 800px; margin: 0 auto; }
    .user-card { border: 1px solid #ddd; border-radius: 8px; padding: 12px; margin: 8px 0; cursor: pointer; }
    .user-card:hover { background: #f5f5f5; }
    .back-link { color: #42b883; cursor: pointer; }
  </style>
</head>
<body>
  <nav>
    <a href="#!/">ホーム</a>
    <a href="#!/users">ユーザー一覧</a>
  </nav>
  <div class="content">
    <ng-view></ng-view>
  </div>

  <!-- テンプレート: ホーム -->
  <script type="text/ng-template" id="home.html">
    <h2>ホーム</h2>
    <p>ユーザー管理デモアプリです。</p>
  </script>

  <!-- テンプレート: ユーザー一覧 -->
  <script type="text/ng-template" id="users.html">
    <h2>ユーザー一覧</h2>
    <div class="user-card" ng-repeat="user in users" ng-click="goToUser(user.id)">
      <strong>{{ user.name }}</strong> — {{ user.email }}
    </div>
  </script>

  <!-- テンプレート: ユーザー詳細 -->
  <script type="text/ng-template" id="user-detail.html">
    <p><a class="back-link" href="#!/users">← 一覧に戻る</a></p>
    <h2>{{ user.name }}</h2>
    <table>
      <tr><td><strong>ID:</strong></td><td>{{ user.id }}</td></tr>
      <tr><td><strong>メール:</strong></td><td>{{ user.email }}</td></tr>
      <tr><td><strong>電話:</strong></td><td>{{ user.phone }}</td></tr>
      <tr><td><strong>会社:</strong></td><td>{{ user.company }}</td></tr>
    </table>
  </script>

  <script>
    var app = angular.module('myApp', ['ngRoute']);
    
    // ユーザーデータサービス
    app.factory('UserService', function() {
      var users = [
        { id: 1, name: '田中太郎', email: 'tanaka@example.com', phone: '090-1234-5678', company: 'ABC株式会社' },
        { id: 2, name: '鈴木花子', email: 'suzuki@example.com', phone: '080-2345-6789', company: 'XYZ株式会社' },
        { id: 3, name: '佐藤一郎', email: 'sato@example.com', phone: '070-3456-7890', company: 'DEF株式会社' }
      ];
      return {
        getAll: function() { return users; },
        getById: function(id) {
          return users.find(function(u) { return u.id === parseInt(id); });
        }
      };
    });
    
    // ルーティング設定
    app.config(['$routeProvider', function($routeProvider) {
      $routeProvider
        .when('/', {
          templateUrl: 'home.html'
        })
        .when('/users', {
          templateUrl: 'users.html',
          controller: 'UserListCtrl'
        })
        .when('/users/:id', {
          templateUrl: 'user-detail.html',
          controller: 'UserDetailCtrl'
        })
        .otherwise({ redirectTo: '/' });
    }]);
    
    app.controller('UserListCtrl', ['$scope', '$location', 'UserService',
      function($scope, $location, UserService) {
        $scope.users = UserService.getAll();
        $scope.goToUser = function(id) {
          $location.path('/users/' + id);
        };
      }
    ]);
    
    app.controller('UserDetailCtrl', ['$scope', '$routeParams', 'UserService',
      function($scope, $routeParams, UserService) {
        $scope.user = UserService.getById($routeParams.id);
      }
    ]);
  </script>
</body>
</html>
```

## 7.4 resolve — データの事前読み込み

`resolve` を使うと、**ルートが表示される前にデータを読み込む**ことができます。

```javascript
.when('/users', {
  templateUrl: 'users.html',
  controller: 'UserListCtrl',
  resolve: {
    // userData が解決されるまでページは表示されない
    userData: ['$http', function($http) {
      return $http.get('https://jsonplaceholder.typicode.com/users')
        .then(function(response) {
          return response.data;
        });
    }]
  }
})

// resolve で取得したデータはコントローラに注入される
.controller('UserListCtrl', ['$scope', 'userData', function($scope, userData) {
  $scope.users = userData;  // すでにデータが入っている
}]);
```

### resolve のメリット

| メリット | 説明 |
|----------|------|
| 画面のちらつき防止 | データ取得完了後にページを表示 |
| ローディング表示 | `$routeChangeStart` イベントでローディングを表示可能 |
| エラーハンドリング | データ取得失敗時にページ遷移を中止できる |

## 7.5 $location サービス

`$location` はURLの操作を行うサービスです。

```javascript
// 現在のパスを取得
var path = $location.path();     // "/users"

// パスを変更（ページ遷移）
$location.path('/users/1');

// クエリパラメータ
$location.search('q', 'angular'); // ?q=angular
var q = $location.search().q;     // "angular"

// ハッシュ
$location.hash('section1');       // #section1
```

## 7.6 ナビゲーションのアクティブ状態

```html
<!DOCTYPE html>
<html ng-app="myApp">
<head>
  <meta charset="UTF-8">
  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.3/angular.min.js"></script>
  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.3/angular-route.min.js"></script>
  <style>
    nav a { padding: 10px 16px; text-decoration: none; color: #333; display: inline-block; }
    nav a.active { background: #42b883; color: white; border-radius: 4px; }
  </style>
</head>
<body ng-controller="NavCtrl">
  <nav>
    <a href="#!/" ng-class="{ active: isActive('/') }">ホーム</a>
    <a href="#!/about" ng-class="{ active: isActive('/about') }">概要</a>
    <a href="#!/contact" ng-class="{ active: isActive('/contact') }">連絡先</a>
  </nav>
  <ng-view></ng-view>

  <script>
    var app = angular.module('myApp', ['ngRoute']);
    
    app.config(['$routeProvider', function($routeProvider) {
      $routeProvider
        .when('/', { template: '<h2>ホーム</h2>' })
        .when('/about', { template: '<h2>概要</h2>' })
        .when('/contact', { template: '<h2>連絡先</h2>' })
        .otherwise({ redirectTo: '/' });
    }]);
    
    app.controller('NavCtrl', ['$scope', '$location', function($scope, $location) {
      $scope.isActive = function(path) {
        return $location.path() === path;
      };
    }]);
  </script>
</body>
</html>
```

## 7.7 ルーティングイベント

| イベント | タイミング |
|----------|-----------|
| `$routeChangeStart` | ルート変更開始時 |
| `$routeChangeSuccess` | ルート変更成功時 |
| `$routeChangeError` | ルート変更失敗時（resolve のエラーなど） |

```javascript
app.run(['$rootScope', function($rootScope) {
  $rootScope.$on('$routeChangeStart', function(event, next, current) {
    $rootScope.loading = true;  // ローディング表示
  });
  
  $rootScope.$on('$routeChangeSuccess', function() {
    $rootScope.loading = false;
  });
  
  $rootScope.$on('$routeChangeError', function() {
    $rootScope.loading = false;
    $rootScope.error = 'ページの読み込みに失敗しました';
  });
}]);
```

> **よくある間違い：** `resolve` で Promise が reject された場合、ページは表示されず `$routeChangeError` が発火します。エラーハンドリングを忘れないようにしましょう。

## 7.8 ポイントまとめ

| トピック | ポイント |
|----------|----------|
| ngRoute | `angular-route.min.js` を追加し、モジュール依存に `'ngRoute'` を指定 |
| $routeProvider | `.when()` でルートを定義、`.otherwise()` で404処理 |
| ng-view | ルートに応じたコンテンツが表示される領域 |
| $routeParams | URLパラメータの取得（`:id` → `$routeParams.id`） |
| resolve | ページ表示前にデータを事前読み込み |
| $location | URLの取得・変更を行うサービス |
| ハッシュプレフィックス | 1.6以降はデフォルトで `#!/` |
