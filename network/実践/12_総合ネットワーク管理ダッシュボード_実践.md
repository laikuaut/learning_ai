### 必須機能（8つ）

1. **ホスト情報表示**: 自機のIPアドレス、ホスト名、クラス判定、プライベートIP判定、2進数表示

```python
# ==================== IPユーティリティ ====================

def ip_to_int(ip_str):
    """IPアドレスを32ビット整数に変換する"""
    result = 0
    for o in ip_str.split("."):
        result = result * 256 + int(o)
    return result


def int_to_ip(ip_int):
    """32ビット整数をIPアドレスに変換する"""
    return ".".join(str((ip_int >> (8 * i)) & 0xFF) for i in range(3, -1, -1))


def ip_to_binary(ip_str):
    """IPアドレスをドット区切り2進数に変換する"""
    return ".".join(format(int(o), "08b") for o in ip_str.split("."))


def validate_ip(ip_str):
    """IPアドレスの形式を検証する"""
    parts = ip_str.split(".")
    if len(parts) != 4:
        return False
    return all(p.isdigit() and 0 <= int(p) <= 255 for p in parts)


def classify_ip(ip_str):
    """IPアドレスのクラスを判定する"""
    first = int(ip_str.split(".")[0])
    if first <= 127:
        return "A"
    elif first <= 191:
        return "B"
    elif first <= 223:
        return "C"
    return "D/E"


def is_private_ip(ip_str):
    """プライベートIPかどうか判定する"""
    octets = [int(o) for o in ip_str.split(".")]
    return (octets[0] == 10 or
            (octets[0] == 172 and 16 <= octets[1] <= 31) or
            (octets[0] == 192 and octets[1] == 168))

def calculate_subnet():
    """サブネット計算を行う"""
    ip_str = input("  IPアドレス: ").strip()
    if not validate_ip(ip_str):
        print("  エラー: 無効なIPアドレスです。")
        return

    prefix = int(input("  プレフィックス長 (/0-32): ").strip())
    if not (0 <= prefix <= 32):
        print("  エラー: プレフィックス長は0〜32です。")
        return

    ip_int = ip_to_int(ip_str)
    mask_int = (0xFFFFFFFF << (32 - prefix)) & 0xFFFFFFFF
    network_int = ip_int & mask_int
    broadcast_int = ip_int | (~mask_int & 0xFFFFFFFF)

    print(f"\n===== サブネット計算結果 =====")
    print(f"  IPアドレス       : {ip_str}")
    print(f"  サブネットマスク : {int_to_ip(mask_int)} (/{prefix})")
    print(f"  ネットワーク     : {int_to_ip(network_int)}")
    print(f"  ブロードキャスト : {int_to_ip(broadcast_int)}")
    print(f"  ホスト範囲       : {int_to_ip(network_int + 1)} 〜 {int_to_ip(broadcast_int - 1)}")
    print(f"  利用可能ホスト数 : {broadcast_int - network_int - 1}")


def port_scan():
    """ポートスキャンを行う"""
    host = input("  対象ホスト: ").strip() or "localhost"
    print(f"\n  スキャン中: {host}")

    common_ports = [21, 22, 23, 25, 53, 80, 110, 143, 443, 993, 995,
                    3306, 3389, 5432, 8080, 8443]

    start_time = time.time()
    open_ports = []

    for port in common_ports:
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(0.5)
        result = sock.connect_ex((host, port))
        sock.close()

        if result == 0:
            try:
                service = socket.getservbyport(port, "tcp")
            except OSError:
                service = "不明"
            open_ports.append({"port": port, "service": service})
            print(f"  ポート {port:<6} ({service:<12}) OPEN")

    elapsed = time.time() - start_time
    print(f"\n  完了（{elapsed:.1f}秒）  開いているポート: {len(open_ports)}個")
    return open_ports


def dns_resolve():
    """DNS解決を行う"""
    print("\n  1. 正引き  2. 逆引き  3. 一括正引き")
    mode = input("  モード: ").strip()

    if mode == "1":

            print(f"  IPアドレス : {', '.join(addresses)}")

    elif mode == "2":
        ip = input("  IPアドレス: ").strip()

            print(f"\n  ホスト名: {hostname}")

def run_service_monitor():
    """サービス監視を実行する"""
    if not monitor_targets:
        print("  監視対象が登録されていません。まず追加してください。")
        add = input("  今すぐ追加しますか？ (y/n): ").strip()
        if add.lower() == "y":
            add_monitor_target()
        else:
            return

    print(f"\n===== サービス監視 =====")
    print(f"  監視対象: {len(monitor_targets)}件\n")

    timestamp = datetime.now().isoformat()

    for target in monitor_targets:
        name = target["name"]
        if name not in monitor_history:
            monitor_history[name] = []

        if target["type"] == "tcp":
            host = target["host"]
            port = target["port"]
            sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            sock.settimeout(2)
            start = time.time()
            result = sock.connect_ex((host, port))
            ms = (time.time() - start) * 1000
            sock.close()

            ok = result == 0
            status = "OK" if ok else "FAIL"
            detail = f"{ms:.0f}ms" if ok else "接続失敗"
            print(f"  [{status:<4}] {name:<25} ({host}:{port})  {detail}")

            monitor_history[name].append({"timestamp": timestamp, "ok": ok, "ms": ms})

        elif target["type"] == "http":
            url = target["url"]
            try:
                req = Request(url, headers={"User-Agent": "Monitor/1.0"})
                start = time.time()
                response = urlopen(req, timeout=5)
                ms = (time.time() - start) * 1000
                response.read()
                response.close()
                print(f"  [OK  ] {name:<25} ({url})  {ms:.0f}ms  {response.status}")
                monitor_history[name].append({"timestamp": timestamp, "ok": True, "ms": ms})
            except Exception as e:
                print(f"  [FAIL] {name:<25} ({url})  {str(e)[:30]}")
                monitor_history[name].append({"timestamp": timestamp, "ok": False, "ms": 0})

    # 稼働率サマリ
    print(f"\n  稼働率サマリ:")
    for name, history in monitor_history.items():
        if not history:
            continue
        total = len(history)
        ok_count = sum(1 for h in history if h["ok"])
        rate = ok_count / total * 100
        bar_len = int(rate / 5)
        bar = "█" * bar_len + "░" * (20 - bar_len)
        print(f"    {name:<20}: {bar} {rate:.1f}% ({ok_count}/{total})")


def show_network_stats():
    """ネットワーク統計を表示する"""
    print("\n===== ネットワーク統計 =====")

    # 監視履歴の統計

                continue
            total = len(history)
            ok_count = sum(1 for h in history if h["ok"])
            latencies = [h["ms"] for h in history if h["ok"] and h["ms"] > 0]

            print(f"\n  {name}:")
            print(f"    チェック回数  : {total}")
            print(f"    成功/失敗     : {ok_count}/{total - ok_count}")

            if latencies:
                avg = sum(latencies) / len(latencies)
                print(f"    平均遅延      : {avg:.1f}ms")
                print(f"    最小/最大     : {min(latencies):.1f}ms / {max(latencies):.1f}ms")
    else:
        print("  監視履歴がありません。サービス監視を実行してください。")

SECURITY_RULES = {

           "message": "RDPが開いています → VPN経由のみに制限を推奨"},

def security_check():
    """セキュリティチェックを行う"""
    host = input("  対象ホスト: ").strip() or "localhost"
    print(f"\n===== セキュリティチェック: {host} =====\n")

    check_ports = list(SECURITY_RULES.keys()) + [110, 143, 993, 995, 8080, 8443]
    check_ports = sorted(set(check_ports))

    score = 100
    findings = []

    for port in check_ports:
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(0.5)
        result = sock.connect_ex((host, port))
        sock.close()

        if result == 0:
            rule = SECURITY_RULES.get(port)
            if rule:
                severity = rule["severity"]
                message = rule["message"]
                if severity == "WARN":
                    score -= 10
                elif severity == "INFO":
                    score -= 3
                findings.append((severity, port, message))
                print(f"  [{severity:<4}] ポート {port:<5} ({rule['service']:<12}) {message}")
            else:
                try:
                    service = socket.getservbyport(port, "tcp")
                except OSError:
                    service = "不明"
                print(f"  [INFO] ポート {port:<5} ({service:<12}) が開いています")
                findings.append(("INFO", port, f"{service}が開いています"))

    score = max(0, score)
    print(f"\n  セキュリティスコア: {score}/100")

    if score >= 80:
        print(f"  評価: 良好 ─ 大きなリスクは検出されませんでした。")
    elif score >= 50:
        print(f"  評価: 注意 ─ いくつかの改善ポイントがあります。")
    else:
        print(f"  評価: 要改善 ─ セキュリティリスクが高い状態です。")

    return {"host": host, "score": score, "findings": findings}

def generate_report():
    """統合レポートを生成する"""
    filename = f"network_report_{datetime.now():%Y%m%d_%H%M%S}.txt"

    hostname = socket.gethostname()
    ip = socket.gethostbyname(hostname)

    with open(filename, "w", encoding="utf-8") as f:
        f.write("=" * 60 + "\n")
        f.write("  ネットワーク管理レポート\n")
        f.write(f"  生成日時: {datetime.now():%Y-%m-%d %H:%M:%S}\n")
        f.write("=" * 60 + "\n\n")

        # ホスト情報
        f.write("【ホスト情報】\n")
        f.write(f"  ホスト名: {hostname}\n")
        f.write(f"  IP: {ip}\n")
        f.write(f"  クラス: {classify_ip(ip)}\n")
        f.write(f"  プライベート: {'はい' if is_private_ip(ip) else 'いいえ'}\n\n")

        # 監視結果
        if monitor_history:
            f.write("【サービス監視結果】\n")
            for name, history in monitor_history.items():
                if not history:
                    continue
                total = len(history)
                ok_count = sum(1 for h in history if h["ok"])
                rate = ok_count / total * 100
                f.write(f"  {name}: 稼働率 {rate:.1f}% ({ok_count}/{total})\n")
            f.write("\n")

        # 監視対象
        if monitor_targets:
            f.write("【監視対象一覧】\n")
            for t in monitor_targets:
                if t["type"] == "tcp":
                    f.write(f"  TCP: {t['name']} ({t['host']}:{t['port']})\n")
                else:
                    f.write(f"  HTTP: {t['name']} ({t['url']})\n")
            f.write("\n")

        f.write("=" * 60 + "\n")

    print(f"\n  レポートを保存しました: {filename}")
```

```python
# ネットワーク管理ダッシュボード（改良版）
# 学べる内容：ネットワーク全般、クラス設計、JSON永続化、統合ツール開発
# 実行方法：python network_dashboard_v2.py

import socket
import time
import json
import math
from datetime import datetime
from urllib.request import urlopen, Request
from urllib.error import URLError, HTTPError


class IPUtils:
    """IPアドレス関連のユーティリティクラス"""

    @staticmethod
    def to_int(ip_str):
        result = 0
        for o in ip_str.split("."):
            result = result * 256 + int(o)
        return result

    @staticmethod
    def from_int(ip_int):
        return ".".join(str((ip_int >> (8 * i)) & 0xFF) for i in range(3, -1, -1))

    @staticmethod
    def to_binary(ip_str):
        return ".".join(format(int(o), "08b") for o in ip_str.split("."))

    @staticmethod
    def validate(ip_str):
        parts = ip_str.split(".")
        return (len(parts) == 4 and
                all(p.isdigit() and 0 <= int(p) <= 255 for p in parts))

    @staticmethod
    def classify(ip_str):
        first = int(ip_str.split(".")[0])
        for upper, cls in [(127, "A"), (191, "B"), (223, "C"), (239, "D")]:
            if first <= upper:
                return cls
        return "E"

    @staticmethod
    def is_private(ip_str):
        o = [int(x) for x in ip_str.split(".")]
        return (o[0] == 10 or
                (o[0] == 172 and 16 <= o[1] <= 31) or
                (o[0] == 192 and o[1] == 168))


class SubnetCalc:
    """サブネット計算クラス"""

    def __init__(self, ip_str, prefix):
        self.ip_int = IPUtils.to_int(ip_str)
        self.prefix = prefix
        self.mask = (0xFFFFFFFF << (32 - prefix)) & 0xFFFFFFFF
        self.network = self.ip_int & self.mask
        self.broadcast = self.ip_int | (~self.mask & 0xFFFFFFFF)

    @property
    def host_count(self):
        return self.broadcast - self.network - 1

    def info(self):
        return {
            "mask": IPUtils.from_int(self.mask),
            "network": IPUtils.from_int(self.network),
            "broadcast": IPUtils.from_int(self.broadcast),
            "first_host": IPUtils.from_int(self.network + 1),
            "last_host": IPUtils.from_int(self.broadcast - 1),
            "host_count": self.host_count,
        }


class PortScanner:
    """ポートスキャンクラス"""

    COMMON_PORTS = [21, 22, 23, 25, 53, 80, 110, 143, 443,
                    993, 995, 3306, 3389, 5432, 8080, 8443]

    def __init__(self, host, timeout=0.5):
        self.host = host
        self.timeout = timeout

    def scan(self, ports=None):
        if ports is None:
            ports = self.COMMON_PORTS
        results = []
        for port in ports:
            sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            sock.settimeout(self.timeout)
            code = sock.connect_ex((self.host, port))
            sock.close()
            if code == 0:
                try:
                    service = socket.getservbyport(port, "tcp")
                except OSError:
                    service = "不明"
                results.append({"port": port, "service": service, "open": True})
        return results


class ServiceMonitor:
    """サービス監視クラス"""

    def __init__(self):
        self.targets = []
        self.history = {}

    def add_tcp(self, name, host, port):
        self.targets.append({"name": name, "host": host, "port": port, "type": "tcp"})
        self.history.setdefault(name, [])

    def add_http(self, name, url):
        self.targets.append({"name": name, "url": url, "type": "http"})
        self.history.setdefault(name, [])

    def check_all(self):
        results = []
        ts = datetime.now().isoformat()
        for t in self.targets:
            if t["type"] == "tcp":
                r = self._check_tcp(t["host"], t["port"])
            else:
                r = self._check_http(t["url"])
            r["name"] = t["name"]
            r["timestamp"] = ts
            self.history[t["name"]].append(r)
            results.append(r)
        return results

    def _check_tcp(self, host, port):
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(2)
        start = time.time()
        code = sock.connect_ex((host, port))
        ms = (time.time() - start) * 1000
        sock.close()
        return {"ok": code == 0, "ms": round(ms, 1), "detail": "" if code == 0 else "接続失敗"}

    def _check_http(self, url):
        try:
            req = Request(url, headers={"User-Agent": "Monitor/1.0"})
            start = time.time()
            resp = urlopen(req, timeout=5)
            ms = (time.time() - start) * 1000
            status = resp.status
            resp.read()
            resp.close()
            return {"ok": True, "ms": round(ms, 1), "detail": f"{status}"}
        except Exception as e:
            return {"ok": False, "ms": 0, "detail": str(e)[:40]}

    def get_availability(self, name):
        h = self.history.get(name, [])
        if not h:
            return 0.0
        return sum(1 for r in h if r["ok"]) / len(h) * 100

    def to_dict(self):
        return {"targets": self.targets, "history": self.history}

    def from_dict(self, data):
        self.targets = data.get("targets", [])
        self.history = data.get("history", {})


class SecurityChecker:
    """セキュリティチェッククラス"""

    RULES = {
        21: ("WARN", "FTP → SFTP/SCPに移行推奨"),
        23: ("WARN", "Telnet → SSHに移行推奨"),
        25: ("INFO", "SMTP → 外部アクセスを制限推奨"),
        80: ("INFO", "HTTP → HTTPSに移行推奨"),
        3306: ("WARN", "MySQL → FW制限推奨"),
        5432: ("WARN", "PostgreSQL → FW制限推奨"),
        3389: ("WARN", "RDP → VPN経由のみに制限推奨"),
        22: ("OK", "SSH（推奨プロトコル）"),
        443: ("OK", "HTTPS（推奨プロトコル）"),
    }

    def check(self, host):
        scanner = PortScanner(host)
        open_ports = scanner.scan(sorted(self.RULES.keys()))
        score = 100
        findings = []

        for r in open_ports:
            port = r["port"]
            if port in self.RULES:
                severity, msg = self.RULES[port]
                if severity == "WARN":
                    score -= 10
                elif severity == "INFO":
                    score -= 3
                findings.append({"severity": severity, "port": port, "message": msg})

        return {"host": host, "score": max(0, score), "findings": findings}


class NetworkDashboard:
    """ネットワーク管理ダッシュボードのメインクラス"""

    CONFIG_FILE = "dashboard_config.json"

    def __init__(self):
        self.monitor = ServiceMonitor()
        self.security = SecurityChecker()

    def run(self):
        print("===== ネットワーク管理ダッシュボード =====")

        while True:
            print("\n[メニュー]")
            print("  1. ホスト情報    5. HTTP監視     9. セキュリティ")
            print("  2. サブネット    6. 監視追加    10. レポート")
            print("  3. ポートスキャン 7. サービス監視 11. 保存/読込")
            print("  4. DNS解決       8. 統計        12. 終了")

            choice = input("\n操作: ").strip()
            actions = {
                "1": self._host_info, "2": self._subnet, "3": self._port_scan,
                "4": self._dns, "5": self._http, "6": self._add_target,
                "7": self._service_monitor, "8": self._stats, "9": self._security,
                "10": self._report, "11": self._config, "12": None,
            }

            if choice == "12":
                print("終了します。")
                break
            elif choice in actions:
                actions[choice]()
            else:
                print("  1〜12で選択してください。")

    def _host_info(self):
        hostname = socket.gethostname()
        ip = socket.gethostbyname(hostname)
        print(f"\n  ホスト名: {hostname}  IP: {ip}  クラス: {IPUtils.classify(ip)}")
        print(f"  プライベート: {'はい' if IPUtils.is_private(ip) else 'いいえ'}")
        print(f"  2進数: {IPUtils.to_binary(ip)}")

    def _subnet(self):
        ip = input("  IP: ").strip()
        prefix = int(input("  プレフィックス: ").strip())
        info = SubnetCalc(ip, prefix).info()
        print(f"\n  ネットワーク: {info['network']}/{prefix}")
        print(f"  ブロードキャスト: {info['broadcast']}")
        print(f"  ホスト範囲: {info['first_host']} 〜 {info['last_host']} ({info['host_count']}台)")

    def _port_scan(self):
        host = input("  ホスト: ").strip() or "localhost"
        start = time.time()
        results = PortScanner(host).scan()
        elapsed = time.time() - start
        for r in results:
            print(f"  ポート {r['port']:<6} ({r['service']:<12}) OPEN")
        print(f"  完了 ({elapsed:.1f}秒) 開いている: {len(results)}個")

    def _dns(self):
        domain = input("  ドメイン名: ").strip()
        try:
            _, _, addrs = socket.gethostbyname_ex(domain)
            print(f"  {domain} → {', '.join(addrs)}")
        except socket.gaierror:
            print("  解決できませんでした。")

    def _http(self):
        url = input("  URL: ").strip()
        try:
            start = time.time()
            resp = urlopen(Request(url, headers={"User-Agent": "Dashboard/1.0"}), timeout=10)
            ms = (time.time() - start) * 1000
            size = len(resp.read())
            print(f"  {resp.status} {resp.reason}  {ms:.0f}ms  {size:,}B")
        except Exception as e:
            print(f"  エラー: {e}")

    def _add_target(self):
        mode = input("  1.TCP 2.HTTP: ").strip()
        name = input("  名前: ").strip()
        if mode == "1":
            host = input("  ホスト: ").strip()
            port = int(input("  ポート: ").strip())
            self.monitor.add_tcp(name, host, port)
        else:
            url = input("  URL: ").strip()
            self.monitor.add_http(name, url)
        print(f"  '{name}' を追加しました。")

    def _service_monitor(self):
        if not self.monitor.targets:
            print("  監視対象がありません。先に追加してください。")
            return
        results = self.monitor.check_all()
        for r in results:
            s = "OK" if r["ok"] else "NG"
            print(f"  [{s:<2}] {r['name']:<20} {r['ms']:>7.1f}ms  {r['detail']}")
        print("\n  稼働率:")
        for t in self.monitor.targets:
            rate = self.monitor.get_availability(t["name"])
            bar = "█" * int(rate / 5) + "░" * (20 - int(rate / 5))
            print(f"    {t['name']:<20} {bar} {rate:.1f}%")

    def _stats(self):
        for name, hist in self.monitor.history.items():
            if not hist:
                continue
            ok = sum(1 for h in hist if h["ok"])
            lats = [h["ms"] for h in hist if h["ok"] and h["ms"] > 0]
            print(f"\n  {name}: 稼働率 {ok/len(hist)*100:.1f}% ({len(hist)}回)")
            if lats:
                print(f"    遅延 平均: {sum(lats)/len(lats):.1f}ms 最小: {min(lats):.1f}ms 最大: {max(lats):.1f}ms")

    def _security(self):
        host = input("  ホスト: ").strip() or "localhost"
        result = self.security.check(host)
        for f in result["findings"]:
            print(f"  [{f['severity']:<4}] ポート {f['port']}: {f['message']}")
        print(f"\n  スコア: {result['score']}/100")

    def _report(self):
        fn = f"report_{datetime.now():%Y%m%d_%H%M%S}.txt"
        hostname = socket.gethostname()
        ip = socket.gethostbyname(hostname)
        with open(fn, "w", encoding="utf-8") as f:
            f.write(f"ネットワークレポート ({datetime.now():%Y-%m-%d %H:%M})\n{'='*50}\n")
            f.write(f"ホスト: {hostname} ({ip})\n\n")
            for name, hist in self.monitor.history.items():
                if hist:
                    ok = sum(1 for h in hist if h["ok"])
                    f.write(f"{name}: {ok/len(hist)*100:.1f}%\n")
        print(f"  保存: {fn}")

    def _config(self):
        mode = input("  1.保存 2.読込: ").strip()
        if mode == "1":
            with open(self.CONFIG_FILE, "w", encoding="utf-8") as f:
                json.dump(self.monitor.to_dict(), f, ensure_ascii=False, indent=2)
            print("  保存しました。")
        elif mode == "2":
            try:
                with open(self.CONFIG_FILE, "r", encoding="utf-8") as f:
                    self.monitor.from_dict(json.load(f))
                print(f"  読み込みました（{len(self.monitor.targets)}件）。")
            except (FileNotFoundError, json.JSONDecodeError) as e:
                print(f"  読み込みエラー: {e}")


if __name__ == "__main__":
    NetworkDashboard().run()
```
