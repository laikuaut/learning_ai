# 実践課題03：買い物リスト管理アプリ ★2

> **難易度**: ★★☆☆☆（基礎）
> **前提知識**: 第1章〜第5章（変数・演算子・条件分岐・関数・配列）
> **課題の種類**: ミニプロジェクト
> **学習目標**: 配列の追加・削除・検索操作と関数の定義を組み合わせて、データを管理するアプリを作る

---

## 完成イメージ

```
===== 買い物リスト =====

[入力欄: たまご] [追加ボタン]

買い物リスト（3件）:
  ☐ 1. 牛乳
  ☑ 2. パン       ← クリックで購入済みに
  ☐ 3. たまご

購入済み: 1 / 3 件
合計金額: ---円（金額未設定）

[全てクリア]
```

---

## 課題の要件

1. HTMLファイル1つで完結すること
2. テキスト入力欄と「追加」ボタンで商品をリストに追加できる
3. リストの各項目をクリックすると「購入済み」と「未購入」を切り替えられる
4. 各項目に「削除」ボタンがあり、リストから取り除ける
5. 配列で商品リストを管理する
6. 関数を適切に分割する（追加・削除・表示更新・切り替えなど）
7. 空欄のまま追加しようとしたらエラーメッセージを出す

<summary>解答例（初心者向け）</summary>

```html
<!DOCTYPE html>
<html lang="ja">
<head>
  <meta charset="UTF-8">
  <title>買い物リスト</title>
  <style>
    body { font-family: sans-serif; max-width: 500px; margin: 40px auto; padding: 0 20px; }
    h1 { text-align: center; }
    .input-area { display: flex; gap: 8px; margin-bottom: 20px; }
    .input-area input { flex: 1; font-size: 16px; padding: 8px; }
    .input-area button { font-size: 16px; padding: 8px 16px; cursor: pointer; }
    ul { list-style: none; padding: 0; }
    li { display: flex; align-items: center; padding: 8px; border-bottom: 1px solid #eee; }
    li.done { text-decoration: line-through; color: #999; }
    .item-name { flex: 1; cursor: pointer; font-size: 16px; }
    .delete-btn { background: #e74c3c; color: white; border: none;
                  padding: 4px 10px; cursor: pointer; border-radius: 4px; font-size: 14px; }
    .summary { margin-top: 16px; color: #666; text-align: center; }
    .clear-btn { display: block; margin: 20px auto; padding: 8px 20px;
                 font-size: 14px; cursor: pointer; }
    .error { color: red; font-size: 14px; min-height: 20px; }
  </style>
</head>
<body>
  <h1>買い物リスト</h1>

  <div class="input-area">
    <input type="text" id="itemInput" placeholder="商品名を入力...">
    <button onclick="addItem()">追加</button>
  </div>
  <p id="error" class="error"></p>

  <ul id="list"></ul>
  <p id="summary" class="summary"></p>
  <button class="clear-btn" onclick="clearAll()">全てクリア</button>

  <script>
    // 買い物リストのデータ（配列）
    let items = [];

    // 商品を追加する関数
    function addItem() {
      const input = document.getElementById("itemInput");
      const name = input.value.trim();

      // バリデーション：空欄チェック
      if (name === "") {
        document.getElementById("error").textContent = "商品名を入力してください。";
        return;
      }

      // 配列に追加
      items.push({ name: name, done: false });
      input.value = "";
      document.getElementById("error").textContent = "";
      renderList();
      input.focus();
    }

    // 購入済み/未購入を切り替える関数
    function toggleItem(index) {
      if (items[index].done === true) {
        items[index].done = false;
      } else {
        items[index].done = true;
      }
      renderList();
    }

    // 商品を削除する関数
    function deleteItem(index) {
      items.splice(index, 1);  // index番目を1個削除
      renderList();
    }

    // リストを画面に描画する関数
    function renderList() {
      const listEl = document.getElementById("list");
      listEl.innerHTML = "";

      for (let i = 0; i < items.length; i++) {
        const item = items[i];

        // li要素を作成
        const li = document.createElement("li");
        if (item.done) {
          li.className = "done";
        }

        // チェックマーク
        const check = document.createElement("span");
        check.textContent = item.done ? "☑ " : "☐ ";

        // 商品名（クリックで切り替え）
        const nameSpan = document.createElement("span");
        nameSpan.className = "item-name";
        nameSpan.textContent = (i + 1) + ". " + item.name;
        nameSpan.onclick = function() { toggleItem(i); };

        // 削除ボタン
        const delBtn = document.createElement("button");
        delBtn.className = "delete-btn";
        delBtn.textContent = "削除";
        delBtn.onclick = function() { deleteItem(i); };

        li.appendChild(check);
        li.appendChild(nameSpan);
        li.appendChild(delBtn);
        listEl.appendChild(li);
      }

      // サマリーを更新
      updateSummary();
    }

    // 集計を更新する関数
    function updateSummary() {
      let doneCount = 0;
      for (let i = 0; i < items.length; i++) {
        if (items[i].done) {
          doneCount = doneCount + 1;
        }
      }

      const summaryEl = document.getElementById("summary");
      if (items.length === 0) {
        summaryEl.textContent = "リストは空です。";
      } else {
        summaryEl.textContent = "購入済み: " + doneCount + " / " + items.length + " 件";
      }
    }

    // 全てクリアする関数
    function clearAll() {
      if (items.length === 0) {
        return;
      }
      items = [];
      renderList();
    }

    // 初期表示
    renderList();
  </script>
</body>
</html>
```

<details>
<summary>解答例（改良版 ─ 第6章・第7章の知識を先取り）</summary>

オブジェクトのメソッドや配列の高階関数（`filter`, `map`）、テンプレートリテラルを活用した改良版です。

```html
<!DOCTYPE html>
<html lang="ja">
<head>
  <meta charset="UTF-8">
  <title>買い物リスト（改良版）</title>
  <style>
    body { font-family: sans-serif; max-width: 500px; margin: 40px auto; padding: 0 20px; }
    h1 { text-align: center; }
    .input-area { display: flex; gap: 8px; margin-bottom: 20px; }
    .input-area input { flex: 1; font-size: 16px; padding: 8px; }
    .input-area button { font-size: 16px; padding: 8px 16px; cursor: pointer; }
    ul { list-style: none; padding: 0; }
    li { display: flex; align-items: center; padding: 8px; border-bottom: 1px solid #eee; }
    li.done span.name { text-decoration: line-through; color: #999; }
    .name { flex: 1; cursor: pointer; font-size: 16px; }
    .del { background: #e74c3c; color: white; border: none;
           padding: 4px 10px; cursor: pointer; border-radius: 4px; }
    .summary { margin-top: 16px; color: #666; text-align: center; }
    .filter-area { text-align: center; margin-bottom: 10px; }
    .filter-area button { margin: 0 4px; padding: 4px 12px; cursor: pointer; }
    .filter-area button.active { background: #333; color: white; }
    .error { color: red; font-size: 14px; min-height: 20px; }
  </style>
</head>
<body>
  <h1>買い物リスト</h1>

  <div class="input-area">
    <input type="text" id="itemInput" placeholder="商品名を入力...">
    <button onclick="app.add()">追加</button>
  </div>
  <p id="error" class="error"></p>

  <div class="filter-area">
    <button onclick="app.setFilter('all')" class="active" id="filterAll">全て</button>
    <button onclick="app.setFilter('active')" id="filterActive">未購入</button>
    <button onclick="app.setFilter('done')" id="filterDone">購入済み</button>
  </div>

  <ul id="list"></ul>
  <p id="summary" class="summary"></p>

  <script>
    // アプリの状態と操作をオブジェクトにまとめる
    const app = {
      items: [],
      filter: "all",  // "all" | "active" | "done"

      add() {
        const input = document.getElementById("itemInput");
        const name = input.value.trim();

        if (name === "") {
          document.getElementById("error").textContent = "商品名を入力してください。";
          return;
        }

        this.items.push({ name: name, done: false });
        input.value = "";
        document.getElementById("error").textContent = "";
        this.render();
        input.focus();
      },

      toggle(index) {
        this.items[index].done = !this.items[index].done;
        this.render();
      },

      remove(index) {
        this.items.splice(index, 1);
        this.render();
      },

      setFilter(filter) {
        this.filter = filter;
        // ボタンのアクティブ状態を更新
        document.getElementById("filterAll").className = filter === "all" ? "active" : "";
        document.getElementById("filterActive").className = filter === "active" ? "active" : "";
        document.getElementById("filterDone").className = filter === "done" ? "active" : "";
        this.render();
      },

      getFilteredItems() {
        if (this.filter === "active") {
          return this.items.filter(function(item) { return !item.done; });
        }
        if (this.filter === "done") {
          return this.items.filter(function(item) { return item.done; });
        }
        return this.items;
      },

      render() {
        const listEl = document.getElementById("list");
        listEl.innerHTML = "";
        const filtered = this.getFilteredItems();

        for (let i = 0; i < this.items.length; i++) {
          const item = this.items[i];

          // フィルタに該当しない項目はスキップ
          if (this.filter === "active" && item.done) continue;
          if (this.filter === "done" && !item.done) continue;

          const li = document.createElement("li");
          if (item.done) li.className = "done";

          const check = document.createElement("span");
          check.textContent = item.done ? "☑ " : "☐ ";

          const nameSpan = document.createElement("span");
          nameSpan.className = "name";
          nameSpan.textContent = `${i + 1}. ${item.name}`;
          // クロージャで正しいインデックスを保持
          nameSpan.onclick = (function(idx) {
            return function() { app.toggle(idx); };
          })(i);

          const delBtn = document.createElement("button");
          delBtn.className = "del";
          delBtn.textContent = "削除";
          delBtn.onclick = (function(idx) {
            return function() { app.remove(idx); };
          })(i);

          li.appendChild(check);
          li.appendChild(nameSpan);
          li.appendChild(delBtn);
          listEl.appendChild(li);
        }

        // サマリー
        const doneCount = this.items.filter(function(item) { return item.done; }).length;
        const total = this.items.length;
        const summaryEl = document.getElementById("summary");
        summaryEl.textContent = total === 0
          ? "リストは空です。"
          : `購入済み: ${doneCount} / ${total} 件`;
      }
    };

    // Enterキーで追加
    document.getElementById("itemInput").addEventListener("keydown", function(e) {
      if (e.key === "Enter") {
        app.add();
      }
    });

    app.render();
  </script>
</body>
</html>
```

**初心者向けとの違い:**
- アプリの状態と操作を1つのオブジェクトに集約（関心の分離）
- `filter()` メソッドで条件にあうアイテムを抽出
- テンプレートリテラル（`` ` `` バッククォート）で文字列結合を簡潔に
- フィルタ機能（全て / 未購入 / 購入済み）を追加
- `addEventListener` でEnterキー入力にも対応
