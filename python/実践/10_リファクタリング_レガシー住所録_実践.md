## 課題の説明

以下は「住所録アプリケーション」の全コードです。**正しく動作します**が、以下の問題を抱えています。

- すべての処理が1つの長い関数に詰まっている
- 同じような処理がコピー&ペーストで繰り返されている
- エラー処理がない
- データ構造の管理が場当たり的
- マジックナンバー（意味不明な数値）がある

**このコードを段階的にリファクタリング（構造改善）してください。**

2. **ステップ2**: クラスを導入してデータと操作をまとめる

```python
# 住所録アプリケーション（リファクタリング：関数分離版）

GROUP_MAP = {"1": "家族", "2": "友人", "3": "仕事"}


def format_contact(index, contact):
    """連絡先を整形して文字列で返す"""
    return (f" {index}. 名前:{contact['name']} 電話:{contact['phone']} "
            f"住所:{contact['address']} グループ:{contact['group']}")


def input_group():
    """グループを入力して名前を返す"""
    num = input("グループ(1:家族 2:友人 3:仕事): ")
    return GROUP_MAP.get(num, "その他")


def add_contact(contacts):
    """連絡先を追加する"""
    name = input("名前: ")
    phone = input("電話番号: ")
    address = input("住所: ")
    group = input_group()
    contacts.append({"name": name, "phone": phone, "address": address, "group": group})
    print(f"{name}を追加しました。")


def show_all(contacts):
    """全件表示する"""
    if not contacts:
        print("データがありません。")
        return
    print("\n--- 住所録一覧 ---")
    for i, contact in enumerate(contacts, 1):
        print(format_contact(i, contact))


def search_contacts(contacts):
    """キーワード検索する"""
    keyword = input("検索キーワード: ")
    results = [
        (i, c) for i, c in enumerate(contacts, 1)
        if any(keyword in c[field] for field in ("name", "phone", "address", "group"))
    ]
    if results:
        for i, contact in results:
            print(format_contact(i, contact))
    else:
        print("見つかりませんでした。")


def edit_contact(contacts):
    """連絡先を編集する"""
    try:
        num = int(input("編集する番号: "))
    except ValueError:
        print("数字を入力してください。")
        return
    if not (1 <= num <= len(contacts)):
        print("無効な番号です。")
        return

    contact = contacts[num - 1]
    print(f"現在: {format_contact(num, contact)}")

    new_name = input("新しい名前(空欄でスキップ): ")
    new_phone = input("新しい電話番号(空欄でスキップ): ")
    new_address = input("新しい住所(空欄でスキップ): ")

    if new_name:
        contact["name"] = new_name
    if new_phone:
        contact["phone"] = new_phone
    if new_address:
        contact["address"] = new_address
    print("更新しました。")


def delete_contact(contacts):
    """連絡先を削除する"""
    try:
        num = int(input("削除する番号: "))
    except ValueError:
        print("数字を入力してください。")
        return
    if not (1 <= num <= len(contacts)):
        print("無効な番号です。")
        return
    removed = contacts.pop(num - 1)
    print(f"{removed['name']}を削除しました。")


def show_stats(contacts):
    """統計を表示する"""
    if not contacts:
        print("データがありません。")
        return
    print(f"\n合計: {len(contacts)}件")
    group_counts = {}
    for contact in contacts:
        group = contact["group"]
        group_counts[group] = group_counts.get(group, 0) + 1
    for group, count in group_counts.items():
        print(f"  {group}: {count}件")


# --- メイン処理 ---
contacts = []
print("===== 住所録 =====")

while True:
    print("\n1.追加 2.一覧 3.検索 4.編集 5.削除 6.統計 7.終了")
    choice = input("番号: ")

    if choice == "1":
        add_contact(contacts)
    elif choice == "2":
        show_all(contacts)
    elif choice == "3":
        search_contacts(contacts)
    elif choice == "4":
        edit_contact(contacts)
    elif choice == "5":
        delete_contact(contacts)
    elif choice == "6":
        show_stats(contacts)
    elif choice == "7":
        print("終了します。")
        break
    else:
        print("1〜7の番号を入力してください。")
```

```python
# 住所録アプリケーション（リファクタリング：クラス版）


class Contact:
    """1件の連絡先を表すクラス"""

    GROUP_MAP = {"1": "家族", "2": "友人", "3": "仕事"}

    def __init__(self, name, phone, address, group):
        self.name = name
        self.phone = phone
        self.address = address
        self.group = group

    def matches(self, keyword):
        """キーワードがいずれかのフィールドに含まれるか"""
        return any(
            keyword in value
            for value in (self.name, self.phone, self.address, self.group)
        )

    def format(self, index):
        """整形した文字列を返す"""
        return (f" {index}. 名前:{self.name} 電話:{self.phone} "
                f"住所:{self.address} グループ:{self.group}")

    @classmethod
    def from_input(cls):
        """ユーザー入力から連絡先を生成する"""
        name = input("名前: ")
        phone = input("電話番号: ")
        address = input("住所: ")
        group_num = input("グループ(1:家族 2:友人 3:仕事): ")
        group = cls.GROUP_MAP.get(group_num, "その他")
        return cls(name, phone, address, group)


class AddressBook:
    """住所録全体を管理するクラス"""

    def __init__(self):
        self.contacts = []

    def add(self):
        """連絡先を追加する"""
        contact = Contact.from_input()
        self.contacts.append(contact)
        print(f"{contact.name}を追加しました。")

    def show_all(self):
        """全件表示する"""
        if not self.contacts:
            print("データがありません。")
            return
        print("\n--- 住所録一覧 ---")
        for i, contact in enumerate(self.contacts, 1):
            print(contact.format(i))

    def search(self):
        """キーワード検索する"""
        keyword = input("検索キーワード: ")
        results = [
            (i, c) for i, c in enumerate(self.contacts, 1)
            if c.matches(keyword)
        ]
        if results:
            for i, contact in results:
                print(contact.format(i))
        else:
            print("見つかりませんでした。")

    def _get_valid_index(self, prompt):
        """有効なインデックスを取得する（ヘルパー）"""
        try:
            num = int(input(prompt))
        except ValueError:
            print("数字を入力してください。")
            return None
        if not (1 <= num <= len(self.contacts)):
            print("無効な番号です。")
            return None
        return num - 1

    def edit(self):
        """連絡先を編集する"""
        index = self._get_valid_index("編集する番号: ")
        if index is None:
            return

        contact = self.contacts[index]
        print(f"現在: {contact.format(index + 1)}")

        new_name = input("新しい名前(空欄でスキップ): ")
        new_phone = input("新しい電話番号(空欄でスキップ): ")
        new_address = input("新しい住所(空欄でスキップ): ")

        if new_name:
            contact.name = new_name
        if new_phone:
            contact.phone = new_phone
        if new_address:
            contact.address = new_address
        print("更新しました。")

    def delete(self):
        """連絡先を削除する"""
        index = self._get_valid_index("削除する番号: ")
        if index is None:
            return
        removed = self.contacts.pop(index)
        print(f"{removed.name}を削除しました。")

    def stats(self):
        """統計を表示する"""
        if not self.contacts:
            print("データがありません。")
            return
        print(f"\n合計: {len(self.contacts)}件")
        group_counts = {}
        for contact in self.contacts:
            group_counts[contact.group] = group_counts.get(contact.group, 0) + 1
        for group, count in group_counts.items():
            print(f"  {group}: {count}件")

    def run(self):
        """メインループを実行する"""
        print("===== 住所録 =====")

        # メニュー項目と対応するメソッド
        menu_actions = {
            "1": ("追加", self.add),
            "2": ("一覧", self.show_all),
            "3": ("検索", self.search),
            "4": ("編集", self.edit),
            "5": ("削除", self.delete),
            "6": ("統計", self.stats),
        }

        while True:
            menu = "  ".join(f"{k}.{v[0]}" for k, v in menu_actions.items())
            print(f"\n{menu}  7.終了")
            choice = input("番号: ")

            if choice == "7":
                print("終了します。")
                break
            elif choice in menu_actions:
                menu_actions[choice][1]()
            else:
                print("1〜7の番号を入力してください。")


# --- 実行 ---
if __name__ == "__main__":
    book = AddressBook()
    book.run()
```

**レガシー版からの改善点まとめ:**

| 改善項目 | レガシー版 | リファクタリング後 |
|---|---|---|
| 変数名 | `c`, `n`, `p`, `g` | `choice`, `name`, `phone`, `group` |
| データ構造 | `data[i][0]` | `contact.name` |
| 表示処理 | コピペ3箇所 | `contact.format()` 1箇所 |
| ループ | `while i < len(data)` | `for i, c in enumerate(...)` |
| グループ変換 | if-elif ハードコード | `GROUP_MAP` 辞書 |
| エラー処理 | なし | `try-except` + バリデーション |
| 統計カウント | 個別変数4つ | `dict.get()` で汎用集計 |
| 全体構造 | ベタ書き1ブロック | `Contact` + `AddressBook` クラス |
