## 課題の要件

### 必須機能

1. **認証**: ログイン/ログアウト（Cookie ベース、ミドルウェアでルート保護）
2. **タスクCRUD**: Server Actions で作成・更新・削除・完了切替
3. **タスク一覧**: サーバーコンポーネントでデータ取得、フィルタリング・ソート
4. **フィルタリング**: ステータス（未着手/進行中/完了）、優先度（高/中/低）、キーワード検索
5. **ダッシュボード**: 統計情報（全タスク数、ステータス別集計、優先度別集計）
6. **メタデータ**: 各ページに適切な `<title>` と `<meta>` を設定
7. **エラーハンドリング**: `loading.tsx`、`error.tsx`、`not-found.tsx` を配置
8. **レスポンシブ**: サイドバーがモバイルで折りたたまれる

### 設計要件

- **サーバーコンポーネント**でデータ取得と静的な表示を行う
- **クライアントコンポーネント**はインタラクティブな部分のみ（フィルタ、モーダル、サイドバーの開閉）
- **Server Actions** でデータ変更（`revalidatePath` で再検証）
- **ミドルウェア**で認証チェック
- **ネストされたレイアウト**でサイドバー + ヘッダーを共有
- コンポーネントは最低10個に分割する

```tsx
  return [...tasks].sort(
    (a, b) => new Date(b.updatedAt).getTime() - new Date(a.updatedAt).getTime()
  )

export function getFilteredTasks(filter: TaskFilter) {
  let result = getAllTasks()

  if (filter.status && filter.status !== 'all') {
    result = result.filter(function (task) {
      return task.status === filter.status
    })
  }

  if (filter.priority && filter.priority !== 'all') {
    result = result.filter(function (task) {
      return task.priority === filter.priority
    })
  }

  if (filter.query) {
    const q = filter.query.toLowerCase()
    result = result.filter(function (task) {
      return (
        task.title.toLowerCase().includes(q) ||
        task.description.toLowerCase().includes(q)
      )
    })
  }

  const page = filter.page || 1
  const limit = filter.limit || 5
  const start = (page - 1) * limit
  const paginatedTasks = result.slice(start, start + limit)

  return {
    tasks: paginatedTasks,
    total: result.length,
    page,
    totalPages: Math.ceil(result.length / limit),
  }

    ...data,
    id: Date.now().toString(),
    createdAt: now,
    updatedAt: now,

  id: string,
  data: Partial<Omit<Task, 'id' | 'createdAt'>>
): Task | null {

    return task.id === id

  if (index === -1) return null
  tasks[index] = {
    ...tasks[index],
    ...data,
    updatedAt: new Date().toISOString().split('T')[0],
  }
  return tasks[index]
}

    return task.id !== id
```

### app/(authenticated)/tasks/actions.ts

```tsx
'use server'

import { revalidatePath } from 'next/cache'
import { redirect } from 'next/navigation'
import { createTask, updateTaskById, deleteTaskById } from '@/lib/tasks'

export async function createTaskAction(
  prevState: { error?: string } | null,
  formData: FormData
) {
  const title = formData.get('title') as string
  const description = formData.get('description') as string
  const priority = formData.get('priority') as 'high' | 'medium' | 'low'
  const assigneeId = formData.get('assigneeId') as string
  const dueDate = formData.get('dueDate') as string

  // バリデーション
  if (!title || title.trim().length === 0) {
    return { error: 'タイトルは必須です' }
  }
  if (title.length > 200) {
    return { error: 'タイトルは200文字以内にしてください' }
  }
  if (!dueDate) {
    return { error: '期限を指定してください' }
  }

  createTask({
    title: title.trim(),
    description: description?.trim() || '',
    status: 'todo',
    priority: priority || 'medium',
    assigneeId: assigneeId || '1',
    dueDate,
  })

  revalidatePath('/tasks')
  redirect('/tasks')
}

export async function updateTaskAction(formData: FormData) {
  const id = formData.get('id') as string
  const title = formData.get('title') as string
  const description = formData.get('description') as string
  const status = formData.get('status') as 'todo' | 'in-progress' | 'done'
  const priority = formData.get('priority') as 'high' | 'medium' | 'low'
  const dueDate = formData.get('dueDate') as string

  if (!title || title.trim().length === 0) {
    return { error: 'タイトルは必須です' }
  }

  updateTaskById(id, {
    title: title.trim(),
    description: description?.trim() || '',
    status,
    priority,
    dueDate,
  })

  revalidatePath('/tasks')
  redirect('/tasks')
}

export async function deleteTaskAction(formData: FormData) {
  const id = formData.get('id') as string
  deleteTaskById(id)
  revalidatePath('/tasks')
}

export async function toggleTaskStatusAction(formData: FormData) {
  const id = formData.get('id') as string
  const currentStatus = formData.get('currentStatus') as string

  const nextStatus =
    currentStatus === 'todo'
      ? 'in-progress'
      : currentStatus === 'in-progress'
        ? 'done'
        : 'todo'

  updateTaskById(id, { status: nextStatus as 'todo' | 'in-progress' | 'done' })
  revalidatePath('/tasks')
}
```

```tsx
'use client'

import { useRouter, useSearchParams } from 'next/navigation'
import { useState, useCallback } from 'react'

export default function TaskFilter() {
  const router = useRouter()
  const searchParams = useSearchParams()
  const [query, setQuery] = useState(searchParams.get('q') || '')

  const updateFilter = useCallback(
    function (key: string, value: string) {
      const params = new URLSearchParams(searchParams.toString())
      if (value && value !== 'all') {
        params.set(key, value)
      } else {
        params.delete(key)
      }
      params.delete('page') // フィルタ変更時はページをリセット
      router.push(`/tasks?${params.toString()}`)
    },
    [router, searchParams]
  )

  function handleSearch() {
    updateFilter('q', query)
  }

  function handleKeyDown(e: React.KeyboardEvent) {
    if (e.key === 'Enter') {
      handleSearch()
    }
  }

  return (
    <div
      style={{
        display: 'flex',
        gap: '12px',
        marginBottom: '24px',
        flexWrap: 'wrap',
        alignItems: 'center',
      }}
    >
      <select
        defaultValue={searchParams.get('status') || 'all'}
        onChange={function (e) { updateFilter('status', e.target.value) }}
```

### 改善4: ミドルウェアの強化

```tsx
// middleware.ts
import { NextResponse } from 'next/server'
import type { NextRequest } from 'next/server'

const protectedPaths = ['/dashboard', '/tasks', '/members']

export function middleware(request: NextRequest) {
  const { pathname } = request.nextUrl
  const token = request.cookies.get('session-token')?.value

  // 認証チェック
  const isProtected = protectedPaths.some((path) => pathname.startsWith(path))
  if (isProtected && !token) {
    const url = new URL('/login', request.url)
    url.searchParams.set('redirect', pathname)
    return NextResponse.redirect(url)
  }

  // ログイン済みユーザーの/loginアクセスをリダイレクト
  if (pathname === '/login' && token) {
    return NextResponse.redirect(new URL('/dashboard', request.url))
  }

  // レスポンスヘッダーにパス情報を追加（デバッグ用）
  const response = NextResponse.next()
  response.headers.set('x-pathname', pathname)
  return response
}

export const config = {
  matcher: ['/((?!api|_next/static|_next/image|favicon.ico).*)'],
}
```

- フィルタ条件をURL（`searchParams`）で管理し、状態の永続化とURL共有を実現

- `Suspense` でダッシュボードのセクションを個別にストリーミング
- ミドルウェアで保護パスをリスト化し、拡張性を向上
