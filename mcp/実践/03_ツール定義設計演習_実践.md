## 完成イメージ

「メモ管理MCPサーバー」のツール群を設計し、以下のような動作するPythonコードを完成させます。

```
サーバーが提供するツール一覧：
┌────────────────────────────────────────────────────┐
│ 1. add_memo(title, content, tags)                  │
│    → メモを追加する                                 │
│                                                    │
│ 2. search_memos(keyword, tag)                      │
│    → キーワードやタグでメモを検索する                │
│                                                    │
│ 3. delete_memo(memo_id)                            │
│    → 指定IDのメモを削除する                          │
│                                                    │
│ 4. list_tags()                                     │
│    → 使用中のタグ一覧を取得する                      │
└────────────────────────────────────────────────────┘
```

---

## 課題の要件

1. FastMCPを使って「メモ管理MCPサーバー」を実装する
2. 以下の4つのツールを定義する
   - `add_memo`: メモの追加（タイトル・本文・タグ）
   - `search_memos`: キーワードまたはタグによる検索
   - `delete_memo`: IDを指定してメモを削除
   - `list_tags`: 登録済みタグの一覧取得
3. 各ツールに適切な型アノテーション（type annotation）と説明文（docstring）を付ける
4. 必須引数とオプション引数を適切に設計する
5. 存在しないIDの削除など、エラーケースを処理する

MCPツールでエラーを返す方法は主に2つあります。

**方法1：文字列でエラーメッセージを返す（シンプル）**
```python
def delete_memo(memo_id: int) -> str:
    # 存在チェック
    memo = find_memo(memo_id)
    if memo is None:
        return f"エラー: ID {memo_id} のメモは存在しません"
    # 削除処理
    ...
```

**方法2：例外を発生させる（推奨）**
```python
from mcp.server.fastmcp import FastMCP

def delete_memo(memo_id: int) -> str:
    memo = find_memo(memo_id)
    if memo is None:
        raise ValueError(f"ID {memo_id} のメモは存在しません")
    # 削除処理
    ...
```

例外を使う方が、MCPフレームワークが `isError: true` を自動でセットしてくれるため推奨です。

## 解答例

<details>
<summary>解答例（初心者向け）</summary>

```python
"""
メモ管理MCPサーバー（初心者向け）
学べること：FastMCPでのツール定義、型アノテーション、エラーハンドリング
実行方法：uv run memo_server.py
"""
from mcp.server.fastmcp import FastMCP

mcp = FastMCP("memo-server")

# ── データストア（インメモリ） ──
memos: list[dict] = []
next_id: int = 1


@mcp.tool()
def add_memo(title: str, content: str, tags: list[str] | None = None) -> str:
    """メモを新規追加します。

    Args:
        title: メモのタイトル（必須、例: "買い物リスト"）
        content: メモの本文（必須）
        tags: 分類用のタグのリスト（オプション、例: ["買い物", "日用品"]）
    """
    global next_id

    memo = {
        "id": next_id,
        "title": title,
        "content": content,
        "tags": tags or [],
    }
    memos.append(memo)
    next_id += 1

    tag_text = ", ".join(memo["tags"]) if memo["tags"] else "なし"
    return f"メモを追加しました（ID: {memo['id']}, タグ: {tag_text}）"


@mcp.tool()
def search_memos(keyword: str | None = None, tag: str | None = None) -> str:
    """キーワードやタグでメモを検索します。

    少なくとも keyword か tag のどちらかを指定してください。
    両方指定した場合はAND条件で検索します。

    Args:
        keyword: 検索キーワード（タイトルと本文を検索対象とする）
        tag: タグ名（完全一致で検索）
    """
    if keyword is None and tag is None:
        return "エラー: keyword または tag のどちらかを指定してください"

    results = memos

    # キーワードフィルタ
    if keyword is not None:
        keyword_lower = keyword.lower()
        results = [
            m for m in results
            if keyword_lower in m["title"].lower()
            or keyword_lower in m["content"].lower()
        ]

    # タグフィルタ
    if tag is not None:
        results = [m for m in results if tag in m["tags"]]

    if not results:
        return "該当するメモはありません"

    # 結果のフォーマット
    lines = [f"検索結果: {len(results)}件"]
    for m in results:
        tag_text = ", ".join(m["tags"]) if m["tags"] else "なし"
        lines.append(f"  ID:{m['id']} [{tag_text}] {m['title']}")
        lines.append(f"    {m['content'][:50]}...")
    return "\n".join(lines)


@mcp.tool()
def delete_memo(memo_id: int) -> str:
    """指定されたIDのメモを削除します。

    Args:
        memo_id: 削除するメモのID（add_memoで返されたID）
    """
    for i, m in enumerate(memos):
        if m["id"] == memo_id:
            deleted = memos.pop(i)
            return f"メモ「{deleted['title']}」（ID: {memo_id}）を削除しました"

    return f"エラー: ID {memo_id} のメモは存在しません"


@mcp.tool()
def list_tags() -> str:
    """登録済みのタグ一覧を取得します。

    メモに使用されているすべてのタグをアルファベット順で返します。
    """
    all_tags: set[str] = set()
    for m in memos:
        all_tags.update(m["tags"])

    if not all_tags:
        return "タグはまだ登録されていません"

    sorted_tags = sorted(all_tags)
    return f"タグ一覧（{len(sorted_tags)}件）: " + ", ".join(sorted_tags)


if __name__ == "__main__":
    mcp.run()
```

<details>
<summary>解答例（改良版）</summary>

改良版では、以下の点を強化しています。
- `Annotated` を使った詳細なスキーマ定義
- `pydantic.Field` による入力バリデーション
- 例外による適切なエラーハンドリング
- タイムスタンプの付与

```python
"""
メモ管理MCPサーバー（改良版）
学べること：Annotated/Fieldによる型制約、例外ベースのエラー処理、構造化出力
実行方法：uv run memo_server.py
"""
from datetime import datetime, timezone
from typing import Annotated

from mcp.server.fastmcp import FastMCP
from pydantic import Field

mcp = FastMCP("memo-server")

# ── データストア（インメモリ） ──
memos: list[dict] = []
next_id: int = 1


@mcp.tool()
def add_memo(
    title: Annotated[str, Field(description="メモのタイトル", min_length=1, max_length=100)],
    content: Annotated[str, Field(description="メモの本文", min_length=1)],
    tags: Annotated[
        list[str] | None,
        Field(description="分類用タグのリスト（例: ['仕事', '重要']）", default=None),
    ] = None,
) -> str:
    """メモを新規追加します。タイトルと本文は必須です。"""
    global next_id

    now = datetime.now(timezone.utc).isoformat()
    memo = {
        "id": next_id,
        "title": title.strip(),
        "content": content.strip(),
        "tags": [t.strip() for t in (tags or []) if t.strip()],
        "created_at": now,
        "updated_at": now,
    }
    memos.append(memo)
    next_id += 1

    tag_text = ", ".join(memo["tags"]) if memo["tags"] else "なし"
    return (
        f"メモを追加しました\n"
        f"  ID: {memo['id']}\n"
        f"  タイトル: {memo['title']}\n"
        f"  タグ: {tag_text}\n"
        f"  作成日時: {now}"
    )


@mcp.tool()
def search_memos(
    keyword: Annotated[
        str | None,
        Field(description="検索キーワード（タイトル・本文を対象）", default=None),
    ] = None,
    tag: Annotated[
        str | None,
        Field(description="タグ名で絞り込み（完全一致）", default=None),
    ] = None,
) -> str:
    """キーワードやタグでメモを検索します。

    少なくとも keyword か tag のどちらかを指定してください。
    両方指定するとAND条件になります。
    """
    if keyword is None and tag is None:
        raise ValueError("keyword または tag のどちらかを指定してください")

    results = list(memos)  # コピーを作成

    if keyword is not None:
        kw = keyword.lower()
        results = [
            m for m in results
            if kw in m["title"].lower() or kw in m["content"].lower()
        ]

    if tag is not None:
        results = [m for m in results if tag in m["tags"]]

    if not results:
        return "該当するメモはありません"

    lines = [f"検索結果: {len(results)}件\n"]
    for m in results:
        tag_text = ", ".join(m["tags"]) if m["tags"] else "なし"
        preview = m["content"][:80].replace("\n", " ")
        lines.append(
            f"  [{m['id']}] {m['title']}\n"
            f"       タグ: {tag_text}\n"
            f"       {preview}...\n"
        )
    return "\n".join(lines)


@mcp.tool()
def delete_memo(
    memo_id: Annotated[int, Field(description="削除するメモのID（add_memoの戻り値に含まれるID）", gt=0)],
) -> str:
    """指定IDのメモを削除します。"""
    for i, m in enumerate(memos):
        if m["id"] == memo_id:
            deleted = memos.pop(i)
            return f"メモ「{deleted['title']}」（ID: {memo_id}）を削除しました"

    raise ValueError(f"ID {memo_id} のメモは存在しません")


@mcp.tool()
def list_tags() -> str:
    """使用中のタグ一覧を件数付きで返します。"""
    tag_counts: dict[str, int] = {}
    for m in memos:
        for t in m["tags"]:
            tag_counts[t] = tag_counts.get(t, 0) + 1

    if not tag_counts:
        return "タグはまだ登録されていません"

    lines = [f"タグ一覧（{len(tag_counts)}種類）:"]
    for tag, count in sorted(tag_counts.items()):
        lines.append(f"  ・{tag}（{count}件）")
    return "\n".join(lines)


if __name__ == "__main__":
    mcp.run()
```

### 初心者向けとの違い

| ポイント | 初心者向け | 改良版 |
|---------|-----------|--------|
| 型定義 | 基本的な型アノテーション | `Annotated` + `Field` で制約（min_length等）を追加 |
| エラー処理 | 文字列でエラーを返す | `raise ValueError` で例外を発生（`isError: true` が自動セット） |
| 入力処理 | そのまま保存 | `strip()` で前後の空白を除去 |
| タグ一覧 | タグ名のみ | タグ名 + 使用件数 |
| メタデータ | なし | タイムスタンプ（created_at / updated_at）を付与 |

**実務のポイント**: `Annotated` と `Field` を使うと、JSON Schema として正確な制約が公開され、LLMがより適切にツールを使えるようになります。これはMCPツール設計で最も効果的なテクニックの一つです。

</details>

## よくある間違い

| 間違い | 正しい理解 |
|--------|-----------|
| docstringを書かない | docstringはLLMがツールの用途を判断する**最重要情報**です。必ず記述してください |
| 全引数を必須にする | 検索のフィルタ条件のように、オプションにすべき引数はデフォルト値を設定します |
| 戻り値を辞書にする | MCPツールの戻り値は文字列（テキストコンテンツ）が基本です。構造化データは文字列にフォーマットして返します |
| エラー時に空文字を返す | エラーメッセージを明示するか、例外を発生させましょう。LLMが問題を把握できなくなります |
| ツール名が曖昧（例: `process`） | ツール名は動詞+名詞で具体的に（例: `add_memo`, `search_memos`）。LLMが使い分けを判断しやすくなります |
