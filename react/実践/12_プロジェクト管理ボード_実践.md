## 課題の要件

### 必須機能

1. **3カラムのカンバンボード**: TODO / 進行中 / 完了
2. **タスク追加**: タイトル・説明・優先度・担当者を設定
3. **ステータス移動**: ボタンクリックでタスクを次/前のカラムに移動
4. **タスク削除**: 個別削除（確認ダイアログ付き）
5. **タスク編集**: タイトルと説明をインライン編集
6. **統計表示**: 各カラムのタスク数、全体の進捗率
7. **フィルタ**: 担当者・優先度でフィルタリング
8. **ダーク/ライトモード**: テーマ切替

### 設計要件

- **useReducer** でタスクの全操作を管理する
- **Context** を2つ以上使う（テーマ、フィルタ設定）
- **React.memo** と **React.useCallback** で最適化する（最低2箇所）
- コンポーネントは最低8つに分割する

## ステップガイド

<details>
<summary>ステップ1：Reducer を設計する</summary>

```jsx
const STATUSES = ["todo", "inProgress", "done"];

function boardReducer(state, action) {
  switch (action.type) {
    case "ADD_TASK":
      return { ...state, tasks: [...state.tasks, { ...action.payload, id: Date.now(), status: "todo" }] };

    case "MOVE_TASK": {
      var task = state.tasks.find(t => t.id === action.payload.id);
      var currentIndex = STATUSES.indexOf(task.status);
      var newIndex = currentIndex + action.payload.direction; // +1: 右へ, -1: 左へ
      if (newIndex < 0 || newIndex >= STATUSES.length) return state;
      return {
        ...state,
        tasks: state.tasks.map(t =>
          t.id === action.payload.id ? { ...t, status: STATUSES[newIndex] } : t
        ),
      };
    }

    case "DELETE_TASK":
      return { ...state, tasks: state.tasks.filter(t => t.id !== action.payload) };

    case "UPDATE_TASK":
      return { ...state, tasks: state.tasks.map(t =>
        t.id === action.payload.id ? { ...t, ...action.payload.changes } : t
      )};

    default: return state;
  }
}
```

<summary>ステップ3：useCallbackでdispatchラッパーを最適化する</summary>

```jsx
// Appコンポーネント内
const handleMoveTask = React.useCallback(function(id, direction) {
  dispatch({ type: "MOVE_TASK", payload: { id, direction } });
}, []);

const handleDeleteTask = React.useCallback(function(id) {
  dispatch({ type: "DELETE_TASK", payload: id });
}, []);
```

`React.memo` でラップした `TaskCard` に渡すことで、他タスクの変更時に再レンダリングを防ぎます。

```html
<!DOCTYPE html>
<html lang="ja">
<head>
  <meta charset="UTF-8">

  <style>

    .task-card:hover { opacity: 0.95; }
    .modal-overlay { position: fixed; top: 0; left: 0; right: 0; bottom: 0; background: rgba(0,0,0,0.5); display: flex; align-items: center; justify-content: center; z-index: 100; }
    .modal { border-radius: 12px; padding: 24px; width: 400px; max-width: 90vw; }
  </style>
</head>
<body>
  <div id="root"></div>
  <script type="text/babel">
    // ==================== 定数 ====================
    const STATUSES = ["todo", "inProgress", "done"];
    const STATUS_LABELS = { todo: "TODO", inProgress: "進行中", done: "完了" };
    const STATUS_COLORS = { todo: "#e3f2fd", inProgress: "#fff3e0", done: "#e8f5e9" };
    const PRIORITY_CONFIG = { high: { label: "高", icon: "🔴" }, medium: { label: "中", icon: "🟡" }, low: { label: "低", icon: "🟢" } };
    const MEMBERS = ["太郎", "花子", "次郎", "未割当"];

    const THEMES = {
      light: { bg: "#f5f5f5", card: "#fff", text: "#333", sub: "#888", border: "#e0e0e0", headerBg: "#fff" },
      dark: { bg: "#1a1a2e", card: "#16213e", text: "#e0e0e0", sub: "#8899aa", border: "#2a3a5a", headerBg: "#0f3460" },
    };

    // ==================== Context ====================
    const ThemeContext = React.createContext();
    const FilterContext = React.createContext();
    function useTheme() { return React.useContext(ThemeContext); }
    function useFilter() { return React.useContext(FilterContext); }

    // ==================== Reducer ====================
    const initialState = {
      tasks: [
        { id: 1, title: "API設計", description: "RESTful APIのエンドポイント設計", priority: "high", assignee: "太郎", status: "todo" },
        { id: 2, title: "UI実装", description: "ダッシュボード画面のコンポーネント実装", priority: "medium", assignee: "花子", status: "inProgress" },
        { id: 3, title: "DB設計", description: "テーブル定義とER図の作成", priority: "high", assignee: "次郎", status: "done" },
        { id: 4, title: "テスト計画", description: "単体テストと結合テストの計画", priority: "low", assignee: "未割当", status: "todo" },
        { id: 5, title: "認証機能", description: "JWT認証の実装", priority: "high", assignee: "太郎", status: "todo" },
      ],
    };

    function boardReducer(state, action) {
      switch (action.type) {
        case "ADD_TASK":
          return { ...state, tasks: [...state.tasks, { ...action.payload, id: Date.now() }] };
        case "MOVE_TASK": {
          return { ...state, tasks: state.tasks.map(function(t) {
            if (t.id !== action.payload.id) return t;
            var idx = STATUSES.indexOf(t.status) + action.payload.direction;
            if (idx < 0 || idx >= STATUSES.length) return t;
            return { ...t, status: STATUSES[idx] };
          })};
        }
        case "DELETE_TASK":
          return { ...state, tasks: state.tasks.filter(function(t) { return t.id !== action.payload; }) };
        case "UPDATE_TASK":
          return { ...state, tasks: state.tasks.map(function(t) {
            return t.id === action.payload.id ? { ...t, ...action.payload.changes } : t;
          })};
        default: return state;
      }
    }

    // ==================== コンポーネント ====================
    function Header({ onAddClick }) {
      var themeCtx = useTheme();
      var colors = THEMES[themeCtx.current];
      return (
        <header style={{ display: "flex", justifyContent: "space-between", alignItems: "center", padding: "12px 20px", background: colors.headerBg, borderBottom: "1px solid " + colors.border }}>
          <h1 style={{ fontSize: 18, color: colors.text }}>📋 プロジェクトボード</h1>
          <div style={{ display: "flex", gap: 8 }}>
            <button onClick={themeCtx.toggle} style={{ background: "none", border: "none", fontSize: 18, cursor: "pointer" }}>
              {themeCtx.current === "light" ? "🌙" : "☀️"}
            </button>
            <button onClick={onAddClick} style={{ padding: "6px 16px", background: THEMES[themeCtx.current].text === "#333" ? "#1976d2" : "#2d88ff", color: "white", border: "none", borderRadius: 6, cursor: "pointer", fontSize: 13, fontWeight: "bold" }}>+ タスク追加</button>
          </div>
        </header>
      );
    }

    function FilterBar() {
      var themeCtx = useTheme();
      var filterCtx = useFilter();
      var colors = THEMES[themeCtx.current];

      function chip(label, isActive, onClick) {
        return <button key={label} onClick={onClick} style={{ padding: "4px 12px", border: "1px solid " + (isActive ? colors.text : colors.border), borderRadius: 16, background: isActive ? colors.text + "22" : "transparent", color: colors.text, cursor: "pointer", fontSize: 12 }}>{label}</button>;
      }

      return (
        <div style={{ display: "flex", gap: 8, padding: "8px 20px", flexWrap: "wrap", alignItems: "center" }}>
          <span style={{ color: colors.sub, fontSize: 13 }}>担当:</span>
          {chip("全員", !filterCtx.assignee, function() { filterCtx.setAssignee(null); })}
          {MEMBERS.map(function(m) { return chip(m, filterCtx.assignee === m, function() { filterCtx.setAssignee(m); }); })}
          <span style={{ color: colors.sub, fontSize: 13, marginLeft: 8 }}>優先度:</span>
          {chip("すべて", !filterCtx.priority, function() { filterCtx.setPriority(null); })}
          {Object.keys(PRIORITY_CONFIG).map(function(p) {
            return chip(PRIORITY_CONFIG[p].icon + " " + PRIORITY_CONFIG[p].label, filterCtx.priority === p, function() { filterCtx.setPriority(p); });
          })}
        </div>
      );
    }

    var TaskCard = React.memo(function TaskCard({ task, onMove, onDelete, onUpdate }) {
      var themeCtx = useTheme();
      var colors = THEMES[themeCtx.current];
      const [editing, setEditing] = React.useState(false);
      const [editTitle, setEditTitle] = React.useState(task.title);

      var statusIdx = STATUSES.indexOf(task.status);
      var canMoveRight = statusIdx < STATUSES.length - 1;
      var canMoveLeft = statusIdx > 0;
      var pri = PRIORITY_CONFIG[task.priority];

      function handleSave() {
        if (editTitle.trim()) {
          onUpdate(task.id, { title: editTitle.trim() });
          setEditing(false);
        }
      }

      return (
        <div className="task-card" style={{ background: colors.card, border: "1px solid " + colors.border }}>
          {editing ? (
            <div>
              <input value={editTitle} onChange={function(e) { setEditTitle(e.target.value); }}
                onKeyDown={function(e) { if (e.key === "Enter") handleSave(); if (e.key === "Escape") setEditing(false); }}
                style={{ width: "100%", padding: 4, marginBottom: 6, border: "1px solid " + colors.border, borderRadius: 4, background: colors.bg, color: colors.text }} autoFocus />
              <button onClick={handleSave} style={{ fontSize: 11, padding: "2px 8px", background: "#4caf50", color: "white", border: "none", borderRadius: 4, cursor: "pointer", marginRight: 4 }}>保存</button>
              <button onClick={function() { setEditing(false); setEditTitle(task.title); }} style={{ fontSize: 11, padding: "2px 8px", background: "#999", color: "white", border: "none", borderRadius: 4, cursor: "pointer" }}>取消</button>
            </div>
          ) : (
            <div onDoubleClick={function() { setEditing(true); }} style={{ fontWeight: "bold", fontSize: 14, color: colors.text, marginBottom: 4 }}>{task.title}</div>
          )}
          {task.description && <div style={{ fontSize: 12, color: colors.sub, marginBottom: 8 }}>{task.description}</div>}
          <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center", fontSize: 12 }}>
            <span>{pri.icon} {task.assignee}</span>
            <div style={{ display: "flex", gap: 4 }}>
              {canMoveLeft && <button onClick={function() { onMove(task.id, -1); }} title="左へ移動" style={{ background: "none", border: "1px solid " + colors.border, borderRadius: 4, cursor: "pointer", padding: "2px 6px", color: colors.text }}>←</button>}
              {canMoveRight && <button onClick={function() { onMove(task.id, 1); }} title="右へ移動" style={{ background: "none", border: "1px solid " + colors.border, borderRadius: 4, cursor: "pointer", padding: "2px 6px", color: colors.text }}>→</button>}
              <button onClick={function() { if (confirm("「" + task.title + "」を削除しますか？")) onDelete(task.id); }} title="削除" style={{ background: "none", border: "1px solid #f44336", borderRadius: 4, cursor: "pointer", padding: "2px 6px", color: "#f44336" }}>✗</button>
            </div>
          </div>
        </div>
      );
    });

    function Column({ status, tasks, onMove, onDelete, onUpdate }) {
      var themeCtx = useTheme();
      var colors = THEMES[themeCtx.current];
      var bgColor = themeCtx.current === "light" ? STATUS_COLORS[status] : colors.bg;

      return (
        <div className="column" style={{ background: bgColor }}>
          <div className="column-title" style={{ color: colors.text }}>
            <span>{STATUS_LABELS[status]}</span>
            <span style={{ background: colors.card, borderRadius: 10, padding: "1px 8px", fontSize: 12 }}>{tasks.length}</span>
          </div>
          {tasks.map(function(task) {
            return <TaskCard key={task.id} task={task} onMove={onMove} onDelete={onDelete} onUpdate={onUpdate} />;
          })}
          {tasks.length === 0 && <div style={{ textAlign: "center", color: colors.sub, padding: 20, fontSize: 13 }}>タスクなし</div>}
        </div>
      );
    }

    function StatsBar({ tasks }) {
      var themeCtx = useTheme();
      var colors = THEMES[themeCtx.current];
      var total = tasks.length;
      var done = tasks.filter(function(t) { return t.status === "done"; }).length;
      var percent = total > 0 ? Math.round(done / total * 100) : 0;

      return (
        <div style={{ padding: "12px 20px", background: colors.card, borderTop: "1px solid " + colors.border, display: "flex", justifyContent: "space-between", alignItems: "center", fontSize: 13, color: colors.sub }}>
          <span>全{total}件 — TODO: {tasks.filter(function(t) { return t.status === "todo"; }).length} / 進行中: {tasks.filter(function(t) { return t.status === "inProgress"; }).length} / 完了: {done}</span>
          <span>進捗: {percent}%</span>
        </div>
      );
    }

    function AddTaskModal({ onClose, onAdd }) {
      var themeCtx = useTheme();
      var colors = THEMES[themeCtx.current];
      const [title, setTitle] = React.useState("");
      const [description, setDescription] = React.useState("");
      const [priority, setPriority] = React.useState("medium");
      const [assignee, setAssignee] = React.useState("未割当");

      function handleSubmit(e) {
        e.preventDefault();
        if (!title.trim()) return;
        onAdd({ title: title.trim(), description: description.trim(), priority: priority, assignee: assignee, status: "todo" });
        onClose();
      }

      var inputStyle = { width: "100%", padding: 8, border: "1px solid " + colors.border, borderRadius: 6, marginBottom: 12, background: colors.bg, color: colors.text, fontSize: 14 };

      return (
        <div className="modal-overlay" onClick={onClose}>
          <div className="modal" style={{ background: colors.card }} onClick={function(e) { e.stopPropagation(); }}>
            <h2 style={{ color: colors.text, marginBottom: 16 }}>タスク追加</h2>
            <form onSubmit={handleSubmit}>
              <input placeholder="タイトル" value={title} onChange={function(e) { setTitle(e.target.value); }} style={inputStyle} autoFocus />
              <textarea placeholder="説明（任意）" value={description} onChange={function(e) { setDescription(e.target.value); }} style={{ ...inputStyle, minHeight: 60, resize: "vertical" }} />
              <div style={{ display: "flex", gap: 12, marginBottom: 12 }}>
                <select value={priority} onChange={function(e) { setPriority(e.target.value); }} style={{ ...inputStyle, flex: 1, marginBottom: 0 }}>
                  <option value="high">🔴 高</option>
                  <option value="medium">🟡 中</option>
                  <option value="low">🟢 低</option>
                </select>
                <select value={assignee} onChange={function(e) { setAssignee(e.target.value); }} style={{ ...inputStyle, flex: 1, marginBottom: 0 }}>
                  {MEMBERS.map(function(m) { return <option key={m} value={m}>{m}</option>; })}
                </select>
              </div>
              <div style={{ display: "flex", gap: 8, justifyContent: "flex-end" }}>
                <button type="button" onClick={onClose} style={{ padding: "8px 20px", background: "#999", color: "white", border: "none", borderRadius: 6, cursor: "pointer" }}>キャンセル</button>
                <button type="submit" disabled={!title.trim()} style={{ padding: "8px 20px", background: title.trim() ? "#1976d2" : "#ccc", color: "white", border: "none", borderRadius: 6, cursor: title.trim() ? "pointer" : "default" }}>追加</button>
              </div>
            </form>
          </div>
        </div>
      );
    }

    // ==================== App ====================
    function App() {
      const [state, dispatch] = React.useReducer(boardReducer, initialState);
      const [theme, setTheme] = React.useState("light");
      const [showModal, setShowModal] = React.useState(false);
      const [filterAssignee, setFilterAssignee] = React.useState(null);
      const [filterPriority, setFilterPriority] = React.useState(null);

      var themeValue = { current: theme, toggle: function() { setTheme(function(t) { return t === "light" ? "dark" : "light"; }); } };
      var filterValue = { assignee: filterAssignee, setAssignee: setFilterAssignee, priority: filterPriority, setPriority: setFilterPriority };

      // フィルタ適用
      var filteredTasks = state.tasks.filter(function(t) {
        if (filterAssignee && t.assignee !== filterAssignee) return false;
        if (filterPriority && t.priority !== filterPriority) return false;
        return true;
      });

      // useCallbackで安定した参照を提供
      var handleMove = React.useCallback(function(id, direction) {
        dispatch({ type: "MOVE_TASK", payload: { id: id, direction: direction } });
      }, []);
      var handleDelete = React.useCallback(function(id) {
        dispatch({ type: "DELETE_TASK", payload: id });
      }, []);
      var handleUpdate = React.useCallback(function(id, changes) {
        dispatch({ type: "UPDATE_TASK", payload: { id: id, changes: changes } });
      }, []);
      var handleAdd = React.useCallback(function(taskData) {
        dispatch({ type: "ADD_TASK", payload: taskData });
      }, []);

      var colors = THEMES[theme];

      return (
        <ThemeContext.Provider value={themeValue}>
          <FilterContext.Provider value={filterValue}>
            <div style={{ minHeight: "100vh", background: colors.bg, display: "flex", flexDirection: "column" }}>
              <Header onAddClick={function() { setShowModal(true); }} />
              <FilterBar />
              <div className="board" style={{ flex: 1 }}>
                {STATUSES.map(function(status) {
                  var columnTasks = filteredTasks.filter(function(t) { return t.status === status; });
                  return <Column key={status} status={status} tasks={columnTasks} onMove={handleMove} onDelete={handleDelete} onUpdate={handleUpdate} />;
                })}
              </div>
              <StatsBar tasks={state.tasks} />
              {showModal && <AddTaskModal onClose={function() { setShowModal(false); }} onAdd={handleAdd} />}
            </div>
          </FilterContext.Provider>
        </ThemeContext.Provider>
      );
    }

    ReactDOM.createRoot(document.getElementById("root")).render(<App />);
  </script>
</body>
</html>
```
