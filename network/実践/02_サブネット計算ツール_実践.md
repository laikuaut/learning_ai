## ステップガイド

<details>
<summary>ステップ1：IPアドレスを32ビット整数に変換する</summary>

IPアドレスの各オクテットを8ビットずつシフトして足し合わせると、32ビットの整数値になります。

```python
def ip_to_int(ip_str):
    """IPアドレス文字列を32ビット整数に変換する"""
    octets = ip_str.split(".")
    result = 0
    for octet in octets:
        result = result * 256 + int(octet)
    return result
```

</details>

<details>
<summary>ステップ2：サブネットマスクを作る</summary>

CIDR表記の `/24` は「上位24ビットが1」という意味です。ビットシフトで作れます。

```python
def cidr_to_mask(prefix_len):
    """CIDR表記からサブネットマスクの整数値を返す"""
    mask = (0xFFFFFFFF << (32 - prefix_len)) & 0xFFFFFFFF
    return mask
```

</details>

<details>
<summary>ステップ3：ネットワークアドレスとブロードキャストを計算する</summary>

AND演算でネットワークアドレス、OR演算でブロードキャストアドレスが求まります。

```python
network = ip_int & mask_int                    # AND演算
broadcast = ip_int | (~mask_int & 0xFFFFFFFF)  # OR演算（マスクの反転）
```

</details>

## 解答例

<details>
<summary>解答例（初心者向け）</summary>

```python
# サブネット計算ツール
# 学べる内容：IPアドレスのサブネット計算、ビット演算
# 実行方法：python subnet_calc.py

# --- IPアドレスを32ビット整数に変換 ---
def ip_to_int(ip_str):
    """IPアドレス文字列を32ビット整数に変換する"""
    octets = ip_str.split(".")
    result = 0
    for octet in octets:
        result = result * 256 + int(octet)
    return result


# --- 32ビット整数をIPアドレス文字列に変換 ---
def int_to_ip(ip_int):
    """32ビット整数をIPアドレス文字列に変換する"""
    octets = []
    for i in range(4):
        octets.append(str(ip_int & 0xFF))
        ip_int = ip_int >> 8
    octets.reverse()
    return ".".join(octets)


# --- 32ビット整数を2進数ドット表記に変換 ---
def int_to_binary(ip_int):
    """32ビット整数を8ビットごとにドット区切りの2進数に変換する"""
    binary = format(ip_int, "032b")
    parts = [binary[i:i+8] for i in range(0, 32, 8)]
    return ".".join(parts)


# --- CIDR表記からサブネットマスクを生成 ---
def cidr_to_mask(prefix_len):
    """CIDR表記のプレフィックス長からサブネットマスクを返す"""
    mask = (0xFFFFFFFF << (32 - prefix_len)) & 0xFFFFFFFF
    return mask


# --- メイン処理 ---
print("===== サブネット計算ツール =====")
print()

ip_str = input("IPアドレスを入力してください: ")
prefix_len = int(input("サブネットマスク（CIDR表記）を入力してください: "))

# 各値の計算
ip_int = ip_to_int(ip_str)
mask_int = cidr_to_mask(prefix_len)
network_int = ip_int & mask_int
broadcast_int = ip_int | (~mask_int & 0xFFFFFFFF)
host_min_int = network_int + 1
host_max_int = broadcast_int - 1
host_count = broadcast_int - network_int - 1

# 文字列に変換
mask_str = int_to_ip(mask_int)
network_str = int_to_ip(network_int)
broadcast_str = int_to_ip(broadcast_int)
host_min_str = int_to_ip(host_min_int)
host_max_str = int_to_ip(host_max_int)

# 結果表示
print()
print("【計算結果】")
print(f"  IPアドレス           : {ip_str}")
print(f"  サブネットマスク     : {mask_str} (/{prefix_len})")
print(f"  ネットワークアドレス : {network_str}")
print(f"  ブロードキャスト     : {broadcast_str}")
print(f"  ホスト範囲           : {host_min_str} 〜 {host_max_str}")
print(f"  利用可能ホスト数     : {host_count}")
print()

# 2進数表示
print("【2進数表示】")
print(f"  IPアドレス       : {int_to_binary(ip_int)}")
print(f"  サブネットマスク : {int_to_binary(mask_int)}")
print(f"  ネットワーク部   : {int_to_binary(network_int)}")
```

<details>
<summary>解答例（改良版 ─ 入力検証と繰り返し対応）</summary>

入力値のバリデーションと繰り返し計算に対応したバージョンです。

```python
# サブネット計算ツール（改良版）
# 学べる内容：サブネット計算、入力検証、ビット演算
# 実行方法：python subnet_calc_v2.py

def ip_to_int(ip_str):
    """IPアドレス文字列を32ビット整数に変換する"""
    octets = ip_str.split(".")
    result = 0
    for octet in octets:
        result = result * 256 + int(octet)
    return result


def int_to_ip(ip_int):
    """32ビット整数をIPアドレス文字列に変換する"""
    return ".".join(str((ip_int >> (8 * i)) & 0xFF) for i in range(3, -1, -1))


def int_to_binary(ip_int):
    """32ビット整数をドット区切り2進数に変換する"""
    binary = format(ip_int, "032b")
    return ".".join(binary[i:i+8] for i in range(0, 32, 8))


def cidr_to_mask(prefix_len):
    """CIDR表記からサブネットマスクの整数値を返す"""
    return (0xFFFFFFFF << (32 - prefix_len)) & 0xFFFFFFFF


def validate_ip(ip_str):
    """IPアドレスの形式を検証する"""
    parts = ip_str.split(".")
    if len(parts) != 4:
        return False
    for part in parts:
        if not part.isdigit():
            return False
        num = int(part)
        if num < 0 or num > 255:
            return False
    return True


def validate_prefix(prefix_str):
    """プレフィックス長を検証する"""
    if not prefix_str.isdigit():
        return False
    prefix = int(prefix_str)
    return 0 <= prefix <= 32


def calculate_subnet(ip_str, prefix_len):
    """サブネット情報を計算して辞書で返す"""
    ip_int = ip_to_int(ip_str)
    mask_int = cidr_to_mask(prefix_len)
    network_int = ip_int & mask_int
    broadcast_int = ip_int | (~mask_int & 0xFFFFFFFF)

    return {
        "ip": ip_str,
        "mask": int_to_ip(mask_int),
        "prefix": prefix_len,
        "network": int_to_ip(network_int),
        "broadcast": int_to_ip(broadcast_int),
        "host_min": int_to_ip(network_int + 1),
        "host_max": int_to_ip(broadcast_int - 1),
        "host_count": broadcast_int - network_int - 1,
        "ip_binary": int_to_binary(ip_int),
        "mask_binary": int_to_binary(mask_int),
        "network_binary": int_to_binary(network_int),
    }


def display_result(info):
    """計算結果を表示する"""
    print()
    print("【計算結果】")
    print(f"  IPアドレス           : {info['ip']}")
    print(f"  サブネットマスク     : {info['mask']} (/{info['prefix']})")
    print(f"  ネットワークアドレス : {info['network']}")
    print(f"  ブロードキャスト     : {info['broadcast']}")
    print(f"  ホスト範囲           : {info['host_min']} 〜 {info['host_max']}")
    print(f"  利用可能ホスト数     : {info['host_count']}")
    print()
    print("【2進数表示】")
    print(f"  IPアドレス       : {info['ip_binary']}")
    print(f"  サブネットマスク : {info['mask_binary']}")
    print(f"  ネットワーク部   : {info['network_binary']}")


def main():
    print("===== サブネット計算ツール =====")

    while True:
        print()
        ip_str = input("IPアドレスを入力してください（qで終了）: ").strip()
        if ip_str.lower() == "q":
            print("終了します。")
            break

        if not validate_ip(ip_str):
            print("エラー: 無効なIPアドレス形式です。")
            continue

        prefix_str = input("サブネットマスク（CIDR表記）を入力してください: ").strip()
        if not validate_prefix(prefix_str):
            print("エラー: プレフィックス長は0〜32の整数で入力してください。")
            continue

        result = calculate_subnet(ip_str, int(prefix_str))
        display_result(result)


if __name__ == "__main__":
    main()
```

**初心者向けとの違い:**
- 入力値のバリデーション（不正な入力でクラッシュしない）
- 繰り返し計算に対応（`q` で終了）
- 計算ロジックを `calculate_subnet()` 関数に集約し、結果を辞書で返す
- 表示を `display_result()` に分離し、責務を分割
