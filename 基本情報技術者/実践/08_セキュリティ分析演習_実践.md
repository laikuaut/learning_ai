## 解答例

<details>
<summary>セキュリティ知識問題の解答</summary>

**Q1.**

| 方式 | 分類 |
|------|------|
| AES | **共通鍵暗号** |
| RSA | **公開鍵暗号** |
| SHA-256 | **ハッシュ関数** |
| DES | **共通鍵暗号** |
| 楕円曲線暗号 | **公開鍵暗号** |
| MD5 | **ハッシュ関数**（※現在は安全でない） |

**Q2.** デジタル署名の仕組み:
1. 送信者がメッセージの**ハッシュ値**を計算する
2. ハッシュ値を**送信者の秘密鍵**で暗号化する → これが**デジタル署名**
3. メッセージと署名を受信者に送る
4. 受信者は**送信者の公開鍵**で署名を復号し、ハッシュ値を取得する
5. 受信者がメッセージから独自にハッシュ値を計算する
6. 両方のハッシュ値が一致すれば、メッセージが改ざんされていないことと、送信者が本人であることが確認できる

**Q3.**

| 攻撃手法の説明 | 攻撃名 | 対策 |
|---------------|--------|------|
| SQL文を入力 | **SQLインジェクション** | プレースホルダ（バインド変数）の使用 |
| 偽サイトに誘導 | **フィッシング** | URLの確認、多要素認証 |
| 大量リクエスト | **DoS/DDoS攻撃** | ファイアウォール、CDN、レート制限 |
| スクリプト埋め込み | **XSS（クロスサイトスクリプティング）** | 出力のエスケープ処理 |
| 意図しないリクエスト | **CSRF（クロスサイトリクエストフォージェリ）** | トークンによる検証 |

**Q4.** 組み合わせ数: 62^8 = **218,340,105,584,896**（約218兆通り）
- 秒間10億回: 218,340,105,584,896 / 1,000,000,000 = 218,340秒 ≒ **約2.5日**
- ※8文字では不十分。12文字以上を推奨。

**Q5.**

| 要素 | 脅威の例 | 対策の例 |
|------|---------|---------|
| 機密性 | **不正アクセスによる情報漏洩** | アクセス制御、暗号化、認証強化 |
| 完全性 | **Webページの改ざん、データ破壊** | デジタル署名、改ざん検知、アクセスログ |
| 可用性 | **DDoS攻撃によるサービス停止** | 冗長化、負荷分散、バックアップ |

```python
# パスワード強度チェッカー＆ハッシュ計算
# 学べる内容：セキュリティ基礎、文字列処理、hashlib
# 実行方法：python 08_security_tool.py

import hashlib

def check_password(password):
    """パスワードの強度をチェックする"""
    length = len(password)
    has_upper = False
    has_lower = False
    has_digit = False
    has_symbol = False

    for char in password:
        if char.isupper():
            has_upper = True
        elif char.islower():
            has_lower = True
        elif char.isdigit():
            has_digit = True
        else:
            has_symbol = True

    # 結果表示
    print(f"\n--- 分析結果 ---")
    print(f"文字数:     {length}文字")
    print(f"大文字:     {'✓ 含む' if has_upper else '✗ なし'}")
    print(f"小文字:     {'✓ 含む' if has_lower else '✗ なし'}")
    print(f"数字:       {'✓ 含む' if has_digit else '✗ なし'}")
    print(f"記号:       {'✓ 含む' if has_symbol else '✗ なし'}")

    # 文字種の数を計算
    charset_size = 0
    if has_upper:
        charset_size += 26
    if has_lower:
        charset_size += 26
    if has_digit:
        charset_size += 10
    if has_symbol:
        charset_size += 33

    # 組み合わせ数
    combinations = charset_size ** length

    # 解読推定時間（秒間10億回試行）
    seconds = combinations / 1_000_000_000

    if seconds < 60:
        time_str = f"約 {seconds:.1f}秒"
    elif seconds < 3600:
        time_str = f"約 {seconds / 60:.1f}分"
    elif seconds < 86400:
        time_str = f"約 {seconds / 3600:.1f}時間"
    elif seconds < 86400 * 365:
        time_str = f"約 {seconds / 86400:.1f}日"
    else:
        time_str = f"約 {seconds / (86400 * 365):.1f}年"

    print(f"\n総当り攻撃の推定時間: {time_str}（秒間10億回試行）")
    print(f"組み合わせ数: {combinations:,}通り")

    # 強度判定（スコア制）
    score = 0
    if length >= 8:
        score += 1
    if length >= 12:
        score += 1
    if has_upper and has_lower:
        score += 1
    if has_digit:
        score += 1
    if has_symbol:
        score += 1

    stars = "★" * score + "☆" * (5 - score)
    labels = ["非常に弱い", "弱い", "普通", "強い", "非常に強い"]
    label = labels[min(score, 4)]
    print(f"\n強度: {stars} ({label})")


def hash_calculator():
    """ハッシュ値を計算する"""
    text = input("\nハッシュ化する文字列: ")

    algorithms = ["md5", "sha1", "sha256", "sha512"]
    print(f"\n--- ハッシュ値 ---")
    for algo in algorithms:
        h = hashlib.new(algo, text.encode()).hexdigest()
        print(f"  {algo.upper():<8}: {h}")


# --- メイン ---
print("===== セキュリティツール =====")
print("  1. パスワード強度チェック")
print("  2. ハッシュ値計算")

choice = input("\n選択 (1-2): ")

if choice == "1":
    password = input("\nパスワードを入力: ")
    check_password(password)
elif choice == "2":
    hash_calculator()
else:
    print("無効な選択です。")
```

<summary>解答例（改良版 ─ 辞書攻撃検出・ハッシュ照合）</summary>

```python
# セキュリティツール（改良版）
# 辞書攻撃の弱点検出、ハッシュ値の照合機能付き

import hashlib


# よくある弱いパスワードの一部（辞書）
COMMON_WORDS = [
    "password", "pass", "admin", "login", "welcome",
    "qwerty", "abc", "test", "user", "root",
    "1234", "0000", "master", "dragon", "monkey",
    "iloveyou", "sunshine", "princess", "football",
]


def analyze_password(password):
    """パスワードの詳細分析"""
    length = len(password)
    categories = {
        "upper": any(c.isupper() for c in password),
        "lower": any(c.islower() for c in password),
        "digit": any(c.isdigit() for c in password),
        "symbol": any(not c.isalnum() for c in password),
    }

    # 文字種のサイズ
    sizes = {"upper": 26, "lower": 26, "digit": 10, "symbol": 33}
    charset = sum(sizes[k] for k, v in categories.items() if v)

    # 連続同文字チェック
    consecutive = False
    for i in range(len(password) - 2):
        if password[i] == password[i+1] == password[i+2]:
            consecutive = True
            break

    # 辞書攻撃の弱点チェック
    dict_matches = []
    pw_lower = password.lower()
    for word in COMMON_WORDS:
        if word in pw_lower:
            dict_matches.append(word)

    # 組み合わせ計算
    combinations = charset ** length if charset > 0 else 0
    seconds = combinations / 1_000_000_000 if combinations > 0 else 0

    # 時間のフォーマット
    time_units = [
        (86400 * 365 * 1000, "千年"), (86400 * 365, "年"),
        (86400, "日"), (3600, "時間"), (60, "分"), (1, "秒"),
    ]
    time_str = "一瞬"
    for divisor, unit in time_units:
        if seconds >= divisor:
            time_str = f"約 {seconds / divisor:.1f}{unit}"
            break

    # スコア計算
    score = 0
    if length >= 8: score += 1
    if length >= 12: score += 1
    if categories["upper"] and categories["lower"]: score += 1
    if categories["digit"]: score += 1
    if categories["symbol"]: score += 1
    if consecutive: score -= 1
    if dict_matches: score -= 1
    score = max(0, min(5, score))

    # 表示
    print(f"\n--- 分析結果 ---")
    print(f"文字数:     {length}文字")
    for name, label in [("upper","大文字"), ("lower","小文字"), ("digit","数字"), ("symbol","記号")]:
        status = "✓ 含む" if categories[name] else "✗ なし"
        print(f"{label}:       {status}")
    print(f"連続同文字: {'✗ あり（弱点）' if consecutive else '✓ なし'}")

    if dict_matches:
        print(f"\n⚠ 辞書攻撃の弱点: {', '.join(dict_matches)} が含まれています")

    stars = "★" * score + "☆" * (5 - score)
    labels = {0: "非常に弱い", 1: "弱い", 2: "やや弱い", 3: "普通", 4: "強い", 5: "非常に強い"}
    print(f"\n強度: {stars} ({labels[score]})")
    print(f"総当り推定時間: {time_str}（秒間10億回試行）")
    print(f"組み合わせ数: {combinations:,}通り")

    # 改善アドバイス
    print("\n--- 改善アドバイス ---")
    if length < 12:
        print("  ・12文字以上にすると大幅に安全性が向上します")
    if not categories["symbol"]:
        print("  ・記号（!@#$%など）を追加しましょう")
    if dict_matches:
        print("  ・辞書にある単語をそのまま使わないようにしましょう")


def hash_tool():
    """ハッシュ計算と照合"""
    print("\n--- ハッシュツール ---")
    print("  1. ハッシュ値を計算")
    print("  2. ハッシュ値を照合（改ざんチェック）")

    choice = input("選択 (1-2): ").strip()

    if choice == "1":
        text = input("文字列: ")
        for algo in ["md5", "sha1", "sha256"]:
            h = hashlib.new(algo, text.encode()).hexdigest()
            print(f"  {algo.upper():<8}: {h}")

    elif choice == "2":
        text = input("元の文字列: ")
        expected = input("期待されるSHA-256ハッシュ値: ").strip()
        actual = hashlib.sha256(text.encode()).hexdigest()

        if actual == expected.lower():
            print("  ✓ 一致！ データは改ざんされていません。")
        else:
            print("  ✗ 不一致！ データが改ざんされている可能性があります。")
            print(f"  計算値: {actual}")


def main():
    print("===== セキュリティ分析ツール =====")
    print("  1. パスワード強度チェック")
    print("  2. ハッシュツール")

    choice = input("\n選択 (1-2): ").strip()
    if choice == "1":
        pw = input("\nパスワードを入力: ")
        analyze_password(pw)
    elif choice == "2":
        hash_tool()


if __name__ == "__main__":
    main()
```
