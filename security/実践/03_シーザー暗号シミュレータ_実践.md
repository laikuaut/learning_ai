## 課題の要件

1. アルファベット（A-Z, a-z）をシフト数だけずらして暗号化する
2. アルファベット以外の文字（数字、記号、スペース）はそのまま残す
3. 暗号化と復号の両方に対応する
4. ブルートフォース（brute force）解読機能を実装する（シフト1〜25を全て試す）
5. 大文字・小文字の区別を維持する

## ステップガイド

<details>
<summary>ステップ1：1文字をシフトする方法を考える</summary>

ASCII コードを使ってシフトします。`ord()` で文字を数値に、`chr()` で数値を文字に変換します。

```python
# 'A' は 65、'Z' は 90
# 'a' は 97、'z' は 122

ch = 'H'
shift = 3
# H(72) + 3 = K(75)
new_code = ord(ch) - ord('A')    # 0〜25 の範囲に変換
shifted = (new_code + shift) % 26  # 26で割った余りで循環させる
result = chr(shifted + ord('A'))   # 文字に戻す
```

</details>

<details>
<summary>ステップ2：文字列全体に対してシフト処理を適用する</summary>

```python
def caesar_encrypt(text, shift):
    result = ""
    for ch in text:
        if ch.isupper():
            new_code = (ord(ch) - ord('A') + shift) % 26
            result += chr(new_code + ord('A'))
        elif ch.islower():
            # 小文字も同様に処理
            pass
        else:
            result += ch  # アルファベット以外はそのまま
    return result
```

</details>

<details>
<summary>ステップ3：復号はシフト方向を逆にする</summary>

復号は「26 - シフト数」でシフトすることと同じです。

```python
def caesar_decrypt(text, shift):
    return caesar_encrypt(text, 26 - shift)
```

```python
# シーザー暗号シミュレータ
# 学べる内容：古典暗号、文字コード操作、ブルートフォース攻撃の概念
# 実行方法：python caesar_cipher.py

def caesar_encrypt(text, shift):
    """シーザー暗号で暗号化します。"""
    result = ""
    for ch in text:
        if ch.isupper():
            new_code = (ord(ch) - ord('A') + shift) % 26
            result += chr(new_code + ord('A'))
        elif ch.islower():
            new_code = (ord(ch) - ord('a') + shift) % 26
            result += chr(new_code + ord('a'))
        else:
            result += ch
    return result


def caesar_decrypt(text, shift):
    """シーザー暗号を復号します。"""
    return caesar_encrypt(text, 26 - shift)


def brute_force(ciphertext):
    """シフト1〜25を全て試して表示します。"""
    print("\n--- 総当たり結果 ---")
    for shift in range(1, 26):
        decrypted = caesar_decrypt(ciphertext, shift)
        print(f"シフト {shift:2d}: {decrypted}")


# --- メイン処理 ---
print("===== シーザー暗号シミュレータ =====")

while True:

    print("4. 終了")

    choice = input("\n選択 (1-4): ")

    if choice == "1":

    elif choice == "2":

        if 1 <= shift <= 25:
            decrypted = caesar_decrypt(ciphertext, shift)
            print(f"\n平文: {decrypted}")
        else:
            print("シフト数は1〜25で指定してください。")
```

<summary>解答例（改良版 ─ 頻度分析による自動解読）</summary>

```python
# シーザー暗号シミュレータ（改良版）
# 学べる内容：頻度分析、暗号解読の基礎、統計的手法
# 実行方法：python caesar_cipher_v2.py

# 英語テキストにおける文字の出現頻度（上位）
# 'E' が最も多く、次に 'T', 'A', 'O', 'I', 'N', 'S' と続きます
ENGLISH_FREQ_ORDER = "ETAOINSHRDLCUMWFGYPBVKJXQZ"


def caesar_encrypt(text, shift):
    """シーザー暗号で暗号化します。"""
    result = []
    for ch in text:
        if ch.isalpha():
            base = ord('A') if ch.isupper() else ord('a')
            result.append(chr((ord(ch) - base + shift) % 26 + base))
        else:
            result.append(ch)
    return "".join(result)


def caesar_decrypt(text, shift):
    """シーザー暗号を復号します。"""
    return caesar_encrypt(text, 26 - shift)


def brute_force(ciphertext):
    """シフト1〜25を全て試して表示します。"""
    print("\n--- 総当たり結果 ---")
    for shift in range(1, 26):
        decrypted = caesar_decrypt(ciphertext, shift)
        print(f"シフト {shift:2d}: {decrypted}")


def frequency_analysis(ciphertext):
    """頻度分析（frequency analysis）で最も可能性の高いシフト数を推定します。"""
    # 暗号文中の各アルファベットの出現回数を数えます
    freq = {}
    for ch in ciphertext.upper():
        if ch.isalpha():
            freq[ch] = freq.get(ch, 0) + 1

    if not freq:
        print("アルファベットが含まれていません。")
        return

    # 最も多く出現する文字を取得
    most_common = max(freq, key=freq.get)
    # 英語では 'E' が最頻出なので、最頻出文字が 'E' のシフトだと仮定
    guessed_shift = (ord(most_common) - ord('E')) % 26

    print(f"\n--- 頻度分析結果 ---")
    print(f"最頻出文字: {most_common}（{freq[most_common]}回）")
    print(f"推定シフト数: {guessed_shift}")
    print(f"復号結果: {caesar_decrypt(ciphertext, guessed_shift)}")
    print()
    print("※ テキストが短い場合や英語以外の場合、推定が外れることがあります。")
    print("  その場合は総当たり（ブルートフォース）を使ってください。")


def main():
    print("===== シーザー暗号シミュレータ =====")

    while True:
        print("\n1. 暗号化")
        print("2. 復号")
        print("3. ブルートフォース（総当たり）で解読")
        print("4. 頻度分析で自動解読")
        print("5. 終了")

        choice = input("\n選択 (1-5): ")

        if choice == "1":
            plaintext = input("平文を入力してください: ")
            try:
                shift = int(input("シフト数 (1-25): "))
                if 1 <= shift <= 25:
                    print(f"\n暗号文: {caesar_encrypt(plaintext, shift)}")
                else:
                    print("シフト数は1〜25で指定してください。")
            except ValueError:
                print("数値を入力してください。")

        elif choice == "2":
            ciphertext = input("暗号文を入力してください: ")
            try:
                shift = int(input("シフト数 (1-25): "))
                if 1 <= shift <= 25:
                    print(f"\n平文: {caesar_decrypt(ciphertext, shift)}")
                else:
                    print("シフト数は1〜25で指定してください。")
            except ValueError:
                print("数値を入力してください。")

        elif choice == "3":
            ciphertext = input("暗号文を入力してください: ")
            brute_force(ciphertext)

        elif choice == "4":
            ciphertext = input("暗号文を入力してください: ")
            frequency_analysis(ciphertext)

        elif choice == "5":
            print("終了します。")
            break

        else:
            print("1〜5の番号を入力してください。")


if __name__ == "__main__":
    main()
```

**初心者向けとの違い:**
- リスト + `join()` で文字列結合 → 文字列の `+=` より効率的
- 頻度分析（frequency analysis）機能を追加 → 暗号解読の古典的手法を体験できる
- `try-except ValueError` で不正入力を処理 → 堅牢なプログラム
- 暗号学の歴史的背景（英語の文字頻度）を学べる
