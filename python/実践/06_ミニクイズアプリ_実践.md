## 課題の要件

1. 最低5問のクイズデータを辞書のリストで用意する
2. 各問題に4つの選択肢を持たせる
3. 問題を順番に出題し、ユーザーの回答を受け付ける
4. 正解・不正解をその場で表示する
5. 全問終了後、正解数・正答率・評価を表示する
6. 間違えた問題の一覧と正解を復習用に表示する

<summary>解答例（初心者向け）</summary>

```python
# ミニクイズアプリ
# 学べる内容：辞書のリスト、ループ、条件分岐、スコア集計

# --- クイズデータ ---
quizzes = [
    {
        "question": "Pythonでコメントを書くときに使う記号は？",
        "choices": ["//", "#", "--", "/*"],
        "answer": 2,
    },
    {
        "question": "リストの末尾に要素を追加するメソッドは？",
        "choices": ["add()", "insert()", "append()", "push()"],
        "answer": 3,
    },
    {
        "question": "Pythonで「もし〜なら」を表すキーワードは？",
        "choices": ["when", "case", "if", "switch"],
        "answer": 3,
    },
    {
        "question": "文字列を数値に変換する関数は？",
        "choices": ["str()", "int()", "num()", "convert()"],
        "answer": 2,
    },
    {
        "question": "Pythonのインデントに使うのは？",
        "choices": ["中括弧 {}", "セミコロン ;", "スペースまたはタブ", "括弧 ()"],
        "answer": 3,
    },
]

# --- クイズ開始 ---
print("===== Python ミニクイズ =====")
print(f"全{len(quizzes)}問に挑戦しましょう！")

correct_count = 0
wrong_list = []

for i in range(len(quizzes)):
    quiz = quizzes[i]
    print(f"\n--- 第{i + 1}問 ---")
    print(quiz["question"])

    for j in range(len(quiz["choices"])):
        print(f"  {j + 1}. {quiz['choices'][j]}")

    user_answer = int(input(f"\nあなたの回答 (1-{len(quiz['choices'])}): "))

    if user_answer == quiz["answer"]:
        print("✓ 正解！")
        correct_count = correct_count + 1
    else:
        correct_choice = quiz["choices"][quiz["answer"] - 1]
        print(f"✗ 不正解... 正解は {quiz['answer']}. {correct_choice} でした。")
        wrong_list.append({
            "number": i + 1,
            "question": quiz["question"],
            "correct": correct_choice,
        })

# --- 結果発表 ---
total = len(quizzes)
percentage = int(correct_count / total * 100)

print("\n===== 結果発表 =====")
print(f"{total}問中 {correct_count}問正解！")
print(f"スコア: {percentage}%")

if percentage == 100:
    print("評価: パーフェクト！")
elif percentage >= 80:
    print("評価: すばらしい！")
elif percentage >= 60:
    print("評価: がんばりました！")
else:
    print("評価: もう一度挑戦してみましょう！")

# 間違えた問題の復習
if len(wrong_list) > 0:
    print("\n--- 間違えた問題 ---")
    for item in wrong_list:
        print(f"第{item['number']}問: {item['question']}")
        print(f"  → 正解: {item['correct']}")
```

```python
# ミニクイズアプリ（改良版）
# ランダム出題、リトライ機能、カテゴリ表示付き

import random


def create_quizzes():
    """クイズデータを返す"""
    return [
        {
            "question": "Pythonでコメントを書くときに使う記号は？",
            "choices": ["//", "#", "--", "/*"],
            "answer": 2,
        },
        {
            "question": "リストの末尾に要素を追加するメソッドは？",
            "choices": ["add()", "insert()", "append()", "push()"],
            "answer": 3,
        },
        {
            "question": "Pythonで「もし〜なら」を表すキーワードは？",
            "choices": ["when", "case", "if", "switch"],
            "answer": 3,
        },
        {
            "question": "文字列を数値に変換する関数は？",
            "choices": ["str()", "int()", "num()", "convert()"],
            "answer": 2,
        },
        {
            "question": "Pythonのインデントに使うのは？",
            "choices": ["中括弧 {}", "セミコロン ;", "スペースまたはタブ", "括弧 ()"],
            "answer": 3,
        },
    ]


def ask_question(quiz, number):
    """1問出題して正誤を返す"""
    print(f"\n--- 第{number}問 ---")
    print(quiz["question"])
    for i, choice in enumerate(quiz["choices"], 1):
        print(f"  {i}. {choice}")

    while True:
        try:
            user_answer = int(input(f"\nあなたの回答 (1-{len(quiz['choices'])}): "))
            if 1 <= user_answer <= len(quiz["choices"]):
                break
            print("範囲内の数字を入力してください。")
        except ValueError:
            print("数字を入力してください。")

    correct = user_answer == quiz["answer"]
    if correct:
        print("✓ 正解！")
    else:
        correct_choice = quiz["choices"][quiz["answer"] - 1]
        print(f"✗ 不正解... 正解は {quiz['answer']}. {correct_choice} でした。")

    return correct


def show_results(total, correct_count, wrong_list):
    """結果を表示する"""
    percentage = int(correct_count / total * 100) if total > 0 else 0

    print("\n===== 結果発表 =====")
    print(f"{total}問中 {correct_count}問正解！")
    print(f"スコア: {percentage}%")

    labels = [(100, "パーフェクト！"), (80, "すばらしい！"),
              (60, "がんばりました！"), (0, "もう一度挑戦してみましょう！")]
    label = next(l for threshold, l in labels if percentage >= threshold)
    print(f"評価: {label}")

    if wrong_list:
        print("\n--- 間違えた問題 ---")
        for num, quiz in wrong_list:
            correct_choice = quiz["choices"][quiz["answer"] - 1]
            print(f"第{num}問: {quiz['question']}")
            print(f"  → 正解: {correct_choice}")


# --- メイン処理 ---
while True:
    quizzes = create_quizzes()
    random.shuffle(quizzes)  # ランダム出題

    print("===== Python ミニクイズ =====")
    print(f"全{len(quizzes)}問に挑戦しましょう！")

    correct_count = 0
    wrong_list = []

    for i, quiz in enumerate(quizzes, 1):
        if ask_question(quiz, i):
            correct_count += 1
        else:
            wrong_list.append((i, quiz))

    show_results(len(quizzes), correct_count, wrong_list)

    retry = input("\nもう一度挑戦しますか？ (y/n): ")
    if retry.lower() != "y":
        print("お疲れさまでした！")
        break
```

**初心者向けとの違い:**
- `random.shuffle()` でランダム出題
- 入力バリデーション（数字以外を弾く）
- リトライ機能付き
- 関数分離で再利用性が高い
