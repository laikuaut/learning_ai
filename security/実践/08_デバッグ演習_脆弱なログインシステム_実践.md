## 課題の説明

以下の「ログインシステム」には **8個のセキュリティ上の問題** が埋め込まれています。
プログラムとしては動作しますが、実際に運用すると深刻な脆弱性になります。

すべての問題を見つけて修正してください。

### 進め方

1. まず下のコードをファイルに保存する
2. コードを読み、セキュリティ上の問題を探す
3. 1つずつ問題を修正する
4. 修正後のコードが正しく動作することを確認する

## 脆弱なコード

以下のコードをそのまま `vulnerable_login.py` として保存してください。

```python
# ログインシステム（脆弱版）
# このコードには8つのセキュリティ上の問題があります。すべて修正してください。

import hashlib

# ユーザーデータベース（問題1: パスワードの保存方法）
users_db = {
    "admin": {"password": "admin123", "role": "admin"},          # 問題1
    "user1": {"password": "password", "role": "user"},
    "user2": {"password": "letmein", "role": "user"},
}

login_attempts = {}  # IPアドレスごとのログイン試行回数


def register(username, password):
    """新規ユーザー登録"""
    if username in users_db:
        print("そのユーザー名は既に使われています。")
        return False

    # 問題2: パスワードの強度チェックがない
    users_db[username] = {"password": password, "role": "user"}  # 問題3: 平文保存
    print(f"ユーザー '{username}' を登録しました。")
    return True


def login(username, password):
    """ログイン認証"""
    # 問題4: ログイン試行回数の制限がない
    if username not in users_db:
        print(f"ユーザー '{username}' は存在しません。")   # 問題5: 情報漏洩
        return False

    stored_password = users_db[username]["password"]
    if password == stored_password:                          # 問題6: 平文比較
        print(f"ようこそ、{username}さん！")
        print(f"あなたの権限: {users_db[username]['role']}")
        return True
    else:
        print("パスワードが間違っています。")               # 問題5: 情報漏洩
        return False


def change_password(username, old_password, new_password):
    """パスワード変更"""
    if username not in users_db:
        print("ユーザーが見つかりません。")
        return False

    if users_db[username]["password"] != old_password:
        print("現在のパスワードが正しくありません。")
        return False

    users_db[username]["password"] = new_password  # 問題7: 新パスワードの強度チェックなし
    print("パスワードを変更しました。")

    # 問題8: パスワード変更のログが残らない
    return True


def show_all_users():
    """全ユーザー情報を表示（管理者向け）"""
    print("\n--- ユーザー一覧 ---")
    for username, data in users_db.items():
        print(f"  {username}: パスワード={data['password']}, 権限={data['role']}")


# --- メイン処理 ---
print("===== ログインシステム =====")

while True:
    print("\n1.ログイン  2.ユーザー登録  3.パスワード変更  4.ユーザー一覧  5.終了")
    choice = input("操作: ")

    if choice == "1":
        username = input("ユーザー名: ")
        password = input("パスワード: ")
        login(username, password)

    elif choice == "2":
        username = input("新しいユーザー名: ")
        password = input("新しいパスワード: ")
        register(username, password)

    elif choice == "3":
        username = input("ユーザー名: ")
        old_pw = input("現在のパスワード: ")
        new_pw = input("新しいパスワード: ")
        change_password(username, old_pw, new_pw)

    elif choice == "4":
        show_all_users()

    elif choice == "5":
        print("終了します。")
        break
```

## セキュリティ問題の種類（ヒントレベル別）

<details>
<summary>ヒント（軽め）：問題の種類だけ知りたい</summary>

8つの問題は以下のカテゴリに分類されます。

| カテゴリ | 件数 |
|---|:---:|
| パスワード保存の問題 | 2 |
| パスワードポリシーの欠如 | 2 |
| 情報漏洩（エラーメッセージ） | 1 |
| 認証制御の不備 | 1 |
| 監査ログの欠如 | 1 |
| アクセス制御の不備 | 1 |

</details>

<details>
<summary>ヒント（詳細）：各問題の場所と簡単な説明</summary>

| # | 場所 | ヒント |
|---|---|---|
| 1 | `users_db` の初期データ | パスワードがそのまま読める状態で保存されている |
| 2 | `register()` 関数 | どんな弱いパスワードでも受け入れてしまう |
| 3 | `register()` 関数 | 新規登録時もパスワードを平文で保存している |
| 4 | `login()` 関数 | 何回失敗してもログインを試行できる |
| 5 | `login()` のエラーメッセージ | ユーザー名の存在有無が推測できてしまう |
| 6 | `login()` のパスワード比較 | タイミング攻撃（timing attack）に弱い |
| 7 | `change_password()` | 新しいパスワードの強度チェックがない |
| 8 | `change_password()` | 変更履歴が記録されない |

</details>

<summary>修正済みの完全なコード</summary>

```python
# ログインシステム（修正済み版）
# 学べる内容：安全な認証実装、パスワード管理、監査ログ
# 実行方法：python secure_login.py

import hashlib
import hmac
import os
from datetime import datetime

# --- 定数 ---
MAX_LOGIN_ATTEMPTS = 5
MIN_PASSWORD_LENGTH = 8
SALT_LENGTH = 16

# --- データ構造 ---
# ユーザーDB: {username: {"hash": str, "salt": bytes, "role": str}}
users_db = {}
login_attempts = {}  # {username: 失敗回数}
audit_log = []       # 監査ログ


def log_event(event_type, username, detail=""):
    """監査ログに記録します。（修正8）"""
    entry = {
        "timestamp": datetime.now().strftime("%Y-%m-%d %H:%M:%S"),
        "event": event_type,
        "user": username,
        "detail": detail,
    }
    audit_log.append(entry)
    # 実務ではファイルやログ管理システムに出力します


def hash_password(password, salt):
    """パスワードをソルト付きSHA-256でハッシュ化します。"""
    data = salt + password.encode("utf-8")
    return hashlib.sha256(data).hexdigest()


def check_password_strength(password):
    """パスワードの強度をチェックします。（修正2, 7）"""
    errors = []
    if len(password) < MIN_PASSWORD_LENGTH:
        errors.append(f"{MIN_PASSWORD_LENGTH}文字以上にしてください")
    if not any(c.isupper() for c in password):
        errors.append("大文字を含めてください")
    if not any(c.islower() for c in password):
        errors.append("小文字を含めてください")
    if not any(c.isdigit() for c in password):
        errors.append("数字を含めてください")
    if not any(c in "!@#$%^&*()-_=+[]{}|;:,.<>?" for c in password):
        errors.append("特殊文字を含めてください")
    return len(errors) == 0, errors


def register(username, password):
    """新規ユーザー登録（修正1, 2, 3）"""
    if username in users_db:
        print("そのユーザー名は既に使われています。")
        return False

    # パスワード強度チェック（修正2）
    is_strong, errors = check_password_strength(password)
    if not is_strong:
        print("パスワードが弱すぎます:")
        for err in errors:
            print(f"  - {err}")
        return False

    # ソルト付きハッシュで保存（修正1, 3）
    salt = os.urandom(SALT_LENGTH)
    pw_hash = hash_password(password, salt)
    users_db[username] = {"hash": pw_hash, "salt": salt, "role": "user"}

    log_event("REGISTER", username, "新規ユーザー登録")
    print(f"ユーザー '{username}' を登録しました。")
    return True


def login(username, password):
    """ログイン認証（修正4, 5, 6）"""
    # ログイン試行回数チェック（修正4）
    attempts = login_attempts.get(username, 0)
    if attempts >= MAX_LOGIN_ATTEMPTS:
        print("アカウントがロックされています。管理者に連絡してください。")
        log_event("LOGIN_LOCKED", username, "ロック中のログイン試行")
        return False

    # 統一されたエラーメッセージ（修正5）
    error_message = "ユーザー名またはパスワードが正しくありません。"

    if username not in users_db:
        # ユーザーが存在しない場合もダミーのハッシュ計算を行う（タイミング攻撃対策）
        dummy_salt = os.urandom(SALT_LENGTH)
        hash_password(password, dummy_salt)
        print(error_message)
        log_event("LOGIN_FAIL", username, "存在しないユーザー")
        login_attempts[username] = attempts + 1
        return False

    user = users_db[username]
    input_hash = hash_password(password, user["salt"])

    # 定時間比較（修正6: タイミング攻撃対策）
    if hmac.compare_digest(input_hash, user["hash"]):
        login_attempts[username] = 0  # 成功したらリセット
        log_event("LOGIN_SUCCESS", username)
        print(f"ようこそ、{username}さん！")
        print(f"あなたの権限: {user['role']}")
        return True
    else:
        login_attempts[username] = attempts + 1
        remaining = MAX_LOGIN_ATTEMPTS - attempts - 1
        print(error_message)
        if remaining > 0:
            print(f"残り試行回数: {remaining}回")
        log_event("LOGIN_FAIL", username, "パスワード不一致")
        return False


def change_password(username, old_password, new_password):
    """パスワード変更（修正7, 8）"""
    if username not in users_db:
        print("ユーザーが見つかりません。")
        return False

    user = users_db[username]
    old_hash = hash_password(old_password, user["salt"])

    if not hmac.compare_digest(old_hash, user["hash"]):
        print("現在のパスワードが正しくありません。")
        log_event("PASSWORD_CHANGE_FAIL", username, "旧パスワード不一致")
        return False

    # 新パスワードの強度チェック（修正7）
    is_strong, errors = check_password_strength(new_password)
    if not is_strong:
        print("新しいパスワードが弱すぎます:")
        for err in errors:
            print(f"  - {err}")
        return False

    # 新しいソルトで保存
    new_salt = os.urandom(SALT_LENGTH)
    users_db[username]["hash"] = hash_password(new_password, new_salt)
    users_db[username]["salt"] = new_salt

    log_event("PASSWORD_CHANGE", username, "パスワード変更成功")  # 修正8
    print("パスワードを変更しました。")
    return True


def show_all_users():
    """ユーザー一覧（パスワードハッシュは表示しない）"""
    print("\n--- ユーザー一覧 ---")
    for username, data in users_db.items():
        print(f"  {username}: 権限={data['role']}, ハッシュ={data['hash'][:16]}...")
    # パスワードそのものは表示しない（元のコードでは平文パスワードが見えていた）


def show_audit_log():
    """監査ログを表示します。"""
    if not audit_log:
        print("監査ログは空です。")
        return

    print("\n--- 監査ログ ---")
    for entry in audit_log:
        print(f"  [{entry['timestamp']}] {entry['event']:<20} "
              f"ユーザー: {entry['user']:<10} {entry['detail']}")


# --- 初期ユーザーの安全な登録 ---
def init_users():
    """初期ユーザーを安全にセットアップします。"""
    initial_users = [
        ("admin", "Adm!n2024Secure", "admin"),
        ("user1", "Us3r1@Pass2024", "user"),
    ]
    for uname, pw, role in initial_users:
        salt = os.urandom(SALT_LENGTH)
        pw_hash = hash_password(pw, salt)
        users_db[uname] = {"hash": pw_hash, "salt": salt, "role": role}


init_users()

# --- メイン処理 ---
print("===== ログインシステム（セキュア版） =====")
print("初期ユーザー: admin / Adm!n2024Secure")

while True:
    print("\n1.ログイン  2.登録  3.パスワード変更  4.ユーザー一覧  5.監査ログ  6.終了")
    choice = input("操作: ")

    if choice == "1":
        username = input("ユーザー名: ")
        password = input("パスワード: ")
        login(username, password)

    elif choice == "2":
        username = input("新しいユーザー名: ")
        password = input("新しいパスワード: ")
        register(username, password)

    elif choice == "3":
        username = input("ユーザー名: ")
        old_pw = input("現在のパスワード: ")
        new_pw = input("新しいパスワード: ")
        change_password(username, old_pw, new_pw)

    elif choice == "4":
        show_all_users()

    elif choice == "5":
        show_audit_log()

    elif choice == "6":
        print("終了します。")
        break
```
