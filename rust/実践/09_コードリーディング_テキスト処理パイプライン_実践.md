## 読解対象コード

```rust
// アクセスログ解析パイプライン
// 各ステージが独立した関数になっており、パイプライン的にデータを変換していきます。

use std::collections::HashMap;

// ==================== データ定義 ====================

#[derive(Debug, Clone)]
enum HttpMethod {
    Get,
    Post,
    Put,
    Delete,
    Unknown(String),
}

impl HttpMethod {
    fn from_str(s: &str) -> HttpMethod {
        match s {
            "GET" => HttpMethod::Get,
            "POST" => HttpMethod::Post,
            "PUT" => HttpMethod::Put,
            "DELETE" => HttpMethod::Delete,
            other => HttpMethod::Unknown(other.to_string()),
        }
    }

    fn label(&self) -> &str {
        match self {
            HttpMethod::Get => "GET",
            HttpMethod::Post => "POST",
            HttpMethod::Put => "PUT",
            HttpMethod::Delete => "DELETE",
            HttpMethod::Unknown(s) => s,
        }
    }
}

#[derive(Debug, Clone)]
struct LogEntry {
    timestamp: String,
    method: HttpMethod,
    path: String,
    status: u16,
    size: u64,
}

#[derive(Debug)]
struct PathStats {
    count: u32,
    total_size: u64,
    error_count: u32,
}

impl PathStats {
    fn new() -> PathStats {
        PathStats {
            count: 0,
            total_size: 0,
            error_count: 0,
        }
    }

    fn record(&mut self, size: u64, is_error: bool) {
        self.count += 1;
        self.total_size += size;
        if is_error {
            self.error_count += 1;
        }
    }

    fn avg_size(&self) -> f64 {
        if self.count == 0 {
            0.0
        } else {
            self.total_size as f64 / self.count as f64
        }
    }
}

// ==================== ステージ1: データ生成 ====================

fn generate_sample_log() -> Vec<String> {
    vec![
        "2024-03-01T09:15:23 GET /index.html 200 1234".to_string(),
        "2024-03-01T09:15:45 GET /about.html 200 2345".to_string(),
        "2024-03-01T09:16:01 POST /api/login 200 567".to_string(),
        "2024-03-01T09:16:15 GET /index.html 200 1234".to_string(),
        "2024-03-01T09:17:00 GET /products.html 404 0".to_string(),
        "2024-03-01T10:00:12 GET /index.html 200 1234".to_string(),
        "2024-03-01T10:05:30 POST /api/login 401 234".to_string(),
        "2024-03-01T10:05:45 POST /api/login 401 234".to_string(),
        "2024-03-01T10:06:00 POST /api/login 200 567".to_string(),
        "2024-03-01T11:00:00 GET /about.html 200 2345".to_string(),
        "2024-03-01T11:30:00 GET /dashboard.html 500 0".to_string(),
        "2024-03-01T11:30:05 GET /dashboard.html 500 0".to_string(),
        "2024-03-01T14:00:00 GET /index.html 200 1234".to_string(),
        "2024-03-01T14:15:00 DELETE /api/users/5 403 0".to_string(),
        "2024-03-01T15:00:00 GET /products.html 200 3456".to_string(),
    ]
}

// ==================== ステージ2: パース ====================

fn parse_log_line(line: &str) -> Option<LogEntry> {
    let parts: Vec<&str> = line.split_whitespace().collect();
    if parts.len() != 5 {
        return None;
    }
    Some(LogEntry {
        timestamp: parts[0].to_string(),
        method: HttpMethod::from_str(parts[1]),
        path: parts[2].to_string(),
        status: parts[3].parse().ok()?,
        size: parts[4].parse().ok()?,
    })
}

fn parse_all(lines: &[String]) -> Vec<LogEntry> {
    lines.iter().filter_map(|line| parse_log_line(line)).collect()
}

// ==================== ステージ3: フィルタ・変換 ====================

fn filter_by_status_range(entries: &[LogEntry], min: u16, max: u16) -> Vec<&LogEntry> {
    entries
        .iter()
        .filter(|e| e.status >= min && e.status <= max)
        .collect()
}

fn filter_by_method(entries: &[LogEntry], method: &str) -> Vec<&LogEntry> {
    entries
        .iter()
        .filter(|e| e.method.label() == method)
        .collect()
}

// ==================== ステージ4: 集計 ====================

fn count_by_status(entries: &[LogEntry]) -> HashMap<u16, u32> {
    let mut counts = HashMap::new();
    for entry in entries {
        *counts.entry(entry.status).or_insert(0) += 1;
    }
    counts
}

fn count_by_method(entries: &[LogEntry]) -> HashMap<String, u32> {
    let mut counts = HashMap::new();
    for entry in entries {
        *counts.entry(entry.method.label().to_string()).or_insert(0) += 1;
    }
    counts
}

fn aggregate_by_path(entries: &[LogEntry]) -> HashMap<String, PathStats> {
    let mut stats: HashMap<String, PathStats> = HashMap::new();
    for entry in entries {
        let path_stats = stats.entry(entry.path.clone()).or_insert_with(PathStats::new);
        let is_error = entry.status >= 400;
        path_stats.record(entry.size, is_error);
    }
    stats
}

// ==================== ステージ5: レポート生成 ====================

fn generate_report(entries: &[LogEntry]) -> String {
    let mut report = String::new();

    report.push_str("====== アクセスログ解析レポート ======\n\n");
    report.push_str(&format!("総リクエスト数: {}\n\n", entries.len()));

    // ステータスコード別集計
    report.push_str("--- ステータスコード別 ---\n");
    let status_counts = count_by_status(entries);
    let mut status_sorted: Vec<(&u16, &u32)> = status_counts.iter().collect();
    status_sorted.sort_by_key(|(k, _)| *k);
    for (status, count) in &status_sorted {
        report.push_str(&format!("  {}: {}件\n", status, count));
    }

    // メソッド別集計
    report.push_str("\n--- HTTPメソッド別 ---\n");
    let method_counts = count_by_method(entries);
    let mut method_sorted: Vec<(&String, &u32)> = method_counts.iter().collect();
    method_sorted.sort_by(|a, b| b.1.cmp(a.1));
    for (method, count) in &method_sorted {
        report.push_str(&format!("  {}: {}件\n", method, count));
    }

    // パス別詳細
    report.push_str("\n--- パス別詳細 ---\n");
    let path_stats = aggregate_by_path(entries);
    let mut path_sorted: Vec<(&String, &PathStats)> = path_stats.iter().collect();
    path_sorted.sort_by(|a, b| b.1.count.cmp(&a.1.count));
    for (path, stats) in &path_sorted {
        report.push_str(&format!(
            "  {:<20} アクセス: {:>2}  平均サイズ: {:>6.0}B  エラー: {}\n",
            path,
            stats.count,
            stats.avg_size(),
            stats.error_count
        ));
    }

    // エラー一覧
    let errors = filter_by_status_range(entries, 400, 599);
    if !errors.is_empty() {
        report.push_str("\n--- エラーリクエスト一覧 ---\n");
        for entry in &errors {
            report.push_str(&format!(
                "  {} {} {} → {}\n",
                entry.timestamp,
                entry.method.label(),
                entry.path,
                entry.status
            ));
        }
    }

    report
}

fn main() {
    let raw_lines = generate_sample_log();
    let entries = parse_all(&raw_lines);
    let report = generate_report(&entries);
    println!("{}", report);
}
```

## 設問

以下の10問に答えてください。

### Q1：データの流れ

`main()` 関数の中でデータはどのような型に変換されながら処理されていますか？
`generate_sample_log()` から `println!` までの型の変遷を書いてください。

### Q2：`Option` の活用

`parse_log_line()` が `Option<LogEntry>` を返す理由は何ですか？
また、`parse_all()` の `filter_map` は `None` をどのように扱いますか？

### Q3：所有権と借用

`filter_by_status_range()` は `Vec<&LogEntry>` を返し、`parse_all()` は `Vec<LogEntry>` を返します。なぜ返す型が異なるのですか？設計上の理由を説明してください。

### Q4：`entry` API

`count_by_status()` の `*counts.entry(entry.status).or_insert(0) += 1;` は何をしていますか？
先頭の `*` がないとどうなりますか？

### Q5：`HttpMethod` の設計

`HttpMethod` 列挙型に `Unknown(String)` バリアントがある理由は何ですか？
もし `Unknown` がなかったらどのような問題が生じますか？

### Q6：`PathStats::record()` のシグネチャ

`record()` メソッドが `&mut self` を取る理由を説明してください。
`&self` に変更するとどうなりますか？

### Q7：`Clone` トレイトの役割

### Q8：ソートの設計

`generate_report()` の中で、ステータスコード・メソッド・パスがそれぞれ異なるソート基準でソートされています。各ソート基準とその理由を説明してください。

### Q9：エラーの閾値

## 解答例

<details>
<summary>全10問の解答</summary>

### A1：データの流れ

```
Vec<String>          ← generate_sample_log()
    ↓ parse_all()
Vec<LogEntry>        ← 構造化データに変換
    ↓ generate_report()
String               ← テキストレポートに変換
    ↓ println!()
標準出力
```

`Vec<String>` → `Vec<LogEntry>` → `String` の3段階です。
生のテキストを構造化データにパースし、集計した結果をテキストに戻すパイプラインになっています。

### A2：`Option` の活用

`parse_log_line()` が `Option<LogEntry>` を返す理由は、ログの形式が不正な場合（フィールドが5つでない、数値に変換できない等）に安全にスキップするためです。

`filter_map` は各要素にクロージャを適用し、`Some(value)` の場合のみ `value` を結果に含め、`None` は自動的に除外します。これにより不正な行が無視され、正常なログだけが `Vec<LogEntry>` に集まります。

### A3：所有権と借用

- `parse_all()` は `Vec<LogEntry>` を返す：元の `Vec<String>` から新しい `LogEntry` を**生成**するため、所有権を持つ必要がある
- `filter_by_status_range()` は `Vec<&LogEntry>` を返す：既存の `Vec<LogEntry>` から条件に合うものを**参照**するだけで、新しいデータを作る必要がない

参照を返すことで不要なクローンを避け、メモリ効率が良くなります。元データを加工せず「フィルタしたビュー」を提供する設計です。

### A4：`entry` API

```rust
*counts.entry(entry.status).or_insert(0) += 1;
```

1. `counts.entry(entry.status)` → HashMapのエントリを取得
2. `.or_insert(0)` → キーが存在しなければ `0` を挿入し、値への可変参照（`&mut u32`）を返す
3. `*` → 参照を外して実際の値にアクセス
4. `+= 1` → 値をインクリメント

`*` がないと、`&mut u32` 型に対して `+= 1` を適用することになり、コンパイルエラーになります。

### A5：`HttpMethod` の設計

`Unknown(String)` があることで、想定外のHTTPメソッド（例：`PATCH`, `OPTIONS`）が来ても `from_str()` がパニックせずに処理を続けられます。

もし `Unknown` がなければ、未知のメソッドに対して `panic!` するか、`Option` を返す必要があり、呼び出し元のコードが複雑になります。「閉じた列挙型」より「拡張可能な列挙型」の方がログ解析のような用途に適しています。

### A6：`PathStats::record()` のシグネチャ

`record()` は `count`、`total_size`、`error_count` フィールドを更新するため、`&mut self`（可変借用）が必要です。

`&self`（不変借用）に変更すると、フィールドへの代入（`self.count += 1` 等）がコンパイルエラーになります。Rustの所有権システムでは、データを変更するには明示的に可変参照を取得しなければなりません。

### A7：`Clone` トレイトの役割

`entry.path.clone()` が `aggregate_by_path()` 内で呼ばれています。

```rust
let path_stats = stats.entry(entry.path.clone()).or_insert_with(PathStats::new);
```

`HashMap::entry()` はキーの所有権を要求するため、`entry.path`（`&String`）をそのまま渡すことはできません。`clone()` で新しい `String` を作成してHashMapに渡しています。

### A8：ソートの設計

| 対象 | ソート基準 | 理由 |
|------|-----------|------|
| ステータスコード | 数値の昇順 | HTTPステータスコードは番号順に見る方が直感的（200系→300系→400系→500系） |
| HTTPメソッド | 件数の降順 | どのメソッドが最も多く使われているかを一目で把握するため |
| パス | アクセス数の降順 | 人気のあるページ（アクセスが多いパス）を上に表示するため |

### A9：エラーの閾値

HTTPステータスコード400番台はクライアントエラー（400 Bad Request, 401 Unauthorized, 403 Forbidden, 404 Not Found等）、500番台はサーバーエラー（500 Internal Server Error等）を表します。

`status >= 400` はこれらのエラーレスポンスをすべて検出する条件です。200番台（成功）や300番台（リダイレクト）は正常応答として除外されます。

### A10：改善提案

1. **時間帯別のアクセス集計**: タイムスタンプを解析して時間帯別のアクセス数を表示する
2. **エラー率の計算**: パスごとのエラー率（`error_count / count * 100`）を表示し、問題のあるページを特定しやすくする
3. **`Display` トレイトの実装**: `HttpMethod` や `LogEntry` に `std::fmt::Display` トレイトを実装し、`format!` マクロでの表示を統一する
4. **`Result` 型によるエラーハンドリング**: `parse_log_line()` が `Option` ではなく `Result<LogEntry, ParseError>` を返すようにし、エラー理由を記録する
5. **ジェネリックなフィルタ関数**: `filter_by_status_range` と `filter_by_method` を統合し、クロージャを受け取る汎用フィルタ関数にする

</details>
