## 1. 複雑なフォームの操作

### 入力→バリデーション→送信の一連の流れ

```typescript
import { test, expect } from '@playwright/test';

test('ユーザー登録フォーム', async ({ page }) => {
  await page.goto('/register');

  // テキスト入力
  await page.getByLabel('ユーザー名').fill('taro_tanaka');
  await page.getByLabel('メールアドレス').fill('taro@example.com');
  await page.getByLabel('パスワード').fill('SecureP@ss123');
  await page.getByLabel('パスワード（確認）').fill('SecureP@ss123');

  // セレクトボックス
  await page.getByLabel('��道府県').selectOption({ label: '東京都' });

  // ラジオボタン
  await page.getByRole('radio', { name: '男性' }).check();

  // チェックボックス
  await page.getByRole('checkbox', { name: '利用規約に同意する' }).check();

  // 送信
  await page.getByRole('button', { name: '登録' }).click();

  // 成功メッセージの確認
  await expect(page.getByText('登録が完了しました')).toBeVisible();
});
```

## 2. ダイアログの処理

ブラウザのダイアログ（`alert`、`confirm`、`prompt`）は、**事前にイベントリスナーを設定**して処理します。

### alert ダイアログ

```typescript
test('alertダイアログの処理', async ({ page }) => {
  await page.goto('/dialog-demo');

  // ダイアログのハンドラーを設定（先に設定する）
  page.on('dialog', async (dialog) => {
    // ダイアログのメッセージを確認
    expect(dialog.message()).toBe('処理が完了しました');
    // OKをクリック
    await dialog.accept();
  });

  // ダイアログを発生させる操作
  await page.getByRole('button', { name: '実行' }).click();
});
```

### confirm ダイアログ

```typescript
test('confirmダイアログでOKを押す', async ({ page }) => {
  await page.goto('/delete-item');

  // OKを押す場合
  page.on('dialog', async (dialog) => {
    expect(dialog.message()).toBe('本当に削除しますか？');
    await dialog.accept();  // OK
  });

  await page.getByRole('button', { name: '削除' }).click();
  await expect(page.getByText('削除しました')).toBeVisible();
});

test('confirmダイアログでキャンセルを押す', async ({ page }) => {
  await page.goto('/delete-item');

  page.on('dialog', async (dialog) => {
    await dialog.dismiss();  // キャンセル
  });

  await page.getByRole('button', { name: '削除' }).click();
  // 削除されていないことを確認
  await expect(page.getByText('削除しました')).not.toBeVisible();
});
```

### prompt ダイアログ

```typescript
test('promptダイアログにテキストを入力', async ({ page }) => {
  page.on('dialog', async (dialog) => {
    expect(dialog.message()).toBe('名前を入力してください');
    await dialog.accept('太郎');  // テキストを入力してOK
  });

  await page.getByRole('button', { name: '名前を設定' }).click();
  await expect(page.getByText('こんにちは、太郎さん')).toBeVisible();
});
```

### 1回だけのダイアログ処理

```typescript
// once() で1回だけ処理（繰り返し登録を避ける）
page.once('dialog', async (dialog) => {
  await dialog.accept();
});
```

## 3. iframe の操作

`<iframe>` 内の要素を操作するには、まず **frame** オブジェクトを取得します。

```typescript
test('iframe内のフォームを操作', async ({ page }) => {
  await page.goto('/iframe-demo');

  // iframeを取得する方法
  // 方法1: frameLocator で取得
  const iframe = page.frameLocator('#my-iframe');

  // iframe内の要素を操作
  await iframe.getByLabel('名前').fill('太郎');
  await iframe.getByRole('button', { name: '送信' }).click();
  await expect(iframe.getByText('送信完了')).toBeVisible();
});
```

### ネストしたiframe

```typescript
// iframe内のiframe
const outerFrame = page.frameLocator('#outer');
const innerFrame = outerFrame.frameLocator('#inner');
await innerFrame.getByRole('button').click();
```

## 6. ファイルダウンロード

```typescript
test('ファイルをダウンロードする', async ({ page }) => {
  await page.goto('/download-page');

  // ダウンロードイベントを待ちながらクリック
  const [download] = await Promise.all([
    page.waitForEvent('download'),
    page.getByRole('link', { name: 'レポートをダウンロード' }).click(),
  ]);

  // ダウンロードされたファイルの情報
  const filename = download.suggestedFilename();
  expect(filename).toBe('report.pdf');

  // ファイルを保存
  await download.saveAs(`./downloads/${filename}`);
});
```

## 8. 実践：ログインフォームの完全なテスト

```typescript
import { test, expect } from '@playwright/test';

test.describe('ログインフォーム', () => {
  test.beforeEach(async ({ page }) => {
    await page.goto('/login');
  });

  test('正常なログイン', async ({ page }) => {
    await page.getByLabel('メールアドレス').fill('user@example.com');
    await page.getByLabel('パスワード').fill('password123');
    await page.getByRole('button', { name: 'ログイン' }).click();

    await expect(page).toHaveURL('/dashboard');
    await expect(page.getByText('よう���そ')).toBeVisible();
  });

  test('パスワードが間違っている', async ({ page }) => {
    await page.getByLabel('メールアドレス').fill('user@example.com');
    await page.getByLabel('パスワード').fill('wrongpass');
    await page.getByRole('button', { name: 'ログイン' }).click();

    await expect(page.getByText('メールアドレスまたはパスワードが違います')).toBeVisible();
    await expect(page).toHaveURL('/login');
  });

  test('メールアドレスが未入力', async ({ page }) => {
    await page.getByLabel('パスワード').fill('password123');
    await page.getByRole('button', { name: 'ログイン' }).click();

    await expect(page.getByText('メールアドレスを入力してください')).toBeVisible();
  });

  test('パスワードの表示切替', async ({ page }) => {
    const passwordInput = page.getByLabel('パスワード');
    await passwordInput.fill('secret');

    // 初期状態はpassword（非表示）
    await expect(passwordInput).toHaveAttribute('type', 'password');

    // 表示ボタンをクリック
    await page.getByRole('button', { name: 'パスワードを表示' }).click();

    // テキストタイプに変わる
    await expect(passwordInput).toHaveAttribute('type', 'text');
  });

  test('Enterキーでログインできる', async ({ page }) => {
    await page.getByLabel('メールアドレス').fill('user@example.com');
    await page.getByLabel('パスワード').fill('password123');
    await page.getByLabel('パスワード').press('Enter');

    await expect(page).toHaveURL('/dashboard');
  });
});
```

## まとめ

| 項目 | ポイント |
|------|---------|
| フォーム操作 | `fill()`, `check()`, `selectOption()` の組み合わせ |
| ダイアログ | `page.on('dialog')` で事前にハンドラーを設定 |
| alert | `dialog.accept()` でOK |
| confirm | `accept()` でOK、`dismiss()` でキャンセル |
| prompt | `accept('テキスト')` で入力してOK |
| iframe | `page.frameLocator()` でフレームを取得 |
| ドラッグ＆ドロップ | `.dragTo()` またはマウス操作 |
| ホバー | `.hover()` でマウスオーバー |
| ダウンロード | `page.waitForEvent('download')` で検出 |
