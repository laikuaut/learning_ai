# 実践課題07：メモ帳アプリ ★3

> **難易度**: ★★★☆☆（中級）
> **前提知識**: 第1章〜第11章（変数・演算子・条件分岐・関数・配列・オブジェクト・文字列操作・DOM操作・イベント処理・非同期処理・エラー処理）
> **課題の種類**: ミニプロジェクト
> **学習目標**: localStorage による永続化、DOM操作によるCRUD（作成・読取・更新・削除）、エラー処理を組み合わせた実用的アプリを構築する

---

## 完成イメージ

```
===== メモ帳 =====

[新規メモ]  検索: [________]

─────────────────────────────────
メモ一覧（3件）

  買い物リスト              2024-04-01 14:30
  プロジェクトのアイデア    2024-04-02 09:15
  読書メモ                  2024-04-03 18:45

─────────────────────────────────

タイトル: [買い物リスト        ]
───────────────────────
牛乳、パン、たまご
バター（特売日に買う）
洗剤

最終更新: 2024-04-01 14:30
[保存] [削除]
```

---

## 課題の要件

1. HTMLファイル1つで完結すること
2. メモの作成・表示・編集・削除（CRUD）ができる
3. メモはタイトルと本文を持つ
4. `localStorage` でメモデータを永続化する（ブラウザを閉じても残る）
5. メモ一覧から選択して編集できる
6. キーワードでメモを検索（フィルタ）できる
7. 最終更新日時を記録して表示する
8. 不正な操作（空のタイトルで保存など）に対するエラー処理を含む

<summary>ステップ1：データ構造とlocalStorageの基本</summary>

メモデータをオブジェクトの配列として管理し、`localStorage` で保存/読み込みします。

```javascript
// メモのデータ構造
// { id: 1, title: "買い物リスト", body: "牛乳、パン", updatedAt: "2024-04-01T14:30:00" }

// 保存
function saveToStorage(memos) {
  const json = JSON.stringify(memos);
  localStorage.setItem("memos", json);
}

// 読み込み
function loadFromStorage() {
  const json = localStorage.getItem("memos");
  if (json === null) {
    return [];
  }
  return JSON.parse(json);
}
```

<details>
<summary>ステップ3：CRUD操作の実装</summary>

作成・読取・更新・削除の各操作を関数で実装します。

```javascript
// 新規作成
function createMemo() {
  const newMemo = {
    id: Date.now(),  // ユニークID（タイムスタンプ）
    title: "新しいメモ",
    body: "",
    updatedAt: new Date().toISOString()
  };
  memos.push(newMemo);
  saveToStorage(memos);
}

// 更新
function updateMemo(id, title, body) {
  for (let i = 0; i < memos.length; i++) {
    if (memos[i].id === id) {
      memos[i].title = title;
      memos[i].body = body;
      memos[i].updatedAt = new Date().toISOString();
      break;
    }
  }
  saveToStorage(memos);
}
```

  <script>
    // メモ管理クラス
    class MemoStore {
      constructor(storageKey) {
        this.storageKey = storageKey;
        this.memos = this._load();
      }

      _load() {
        try {
          const json = localStorage.getItem(this.storageKey);
          return json ? JSON.parse(json) : [];
        } catch {
          return [];
        }
      }

      _save() {
        localStorage.setItem(this.storageKey, JSON.stringify(this.memos));
      }

      getAll() {
        // 更新日順（新しいものが先）
        return [...this.memos].sort((a, b) => b.updatedAt.localeCompare(a.updatedAt));
      }

      find(id) {
        return this.memos.find(m => m.id === id) || null;
      }

      search(keyword) {
        const kw = keyword.toLowerCase();
        return this.getAll().filter(m =>
          m.title.toLowerCase().includes(kw) || m.body.toLowerCase().includes(kw)
        );
      }

      create() {
        const memo = {
          id: Date.now(),
          title: "新しいメモ",
          body: "",
          updatedAt: new Date().toISOString()
        };
        this.memos.push(memo);
        this._save();
        return memo;
      }

      update(id, title, body) {
        const memo = this.find(id);
        if (!memo) throw new Error("メモが見つかりません");
        if (!title.trim()) throw new Error("タイトルを入力してください");
        memo.title = title.trim();
        memo.body = body;
        memo.updatedAt = new Date().toISOString();
        this._save();
        return memo;
      }

      remove(id) {
        this.memos = this.memos.filter(m => m.id !== id);
        this._save();
      }
    }

    // アプリケーション
    const store = new MemoStore("memo-app-data");
    let selectedId = null;

    // DOM要素の参照
    const elements = {
      list: document.getElementById("memoList"),
      search: document.getElementById("searchInput"),
      title: document.getElementById("memoTitle"),
      body: document.getElementById("memoBody"),
      status: document.getElementById("status"),
      emptyState: document.getElementById("emptyState"),
      editorContent: document.getElementById("editorContent"),
    };

    function formatDate(iso) {
      const d = new Date(iso);
      return `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, "0")}-${String(d.getDate()).padStart(2, "0")} ${String(d.getHours()).padStart(2, "0")}:${String(d.getMinutes()).padStart(2, "0")}`;
    }

    function renderList() {
      const keyword = elements.search.value;
      const memos = keyword ? store.search(keyword) : store.getAll();

      elements.list.innerHTML = memos.map(m => `
        <div class="memo-item ${m.id === selectedId ? "selected" : ""}" data-id="${m.id}">
          <div class="title">${escapeHtml(m.title || "(無題)")}</div>
          <div class="date">${formatDate(m.updatedAt)}</div>
          <div class="preview">${escapeHtml(m.body.slice(0, 40))}</div>
        </div>
      `).join("");

      // 各メモアイテムにクリックイベントを登録
      elements.list.querySelectorAll(".memo-item").forEach(el => {
        el.addEventListener("click", () => {
          selectMemo(Number(el.dataset.id));
        });
      });
    }

    function escapeHtml(str) {
      const div = document.createElement("div");
      div.textContent = str;
      return div.innerHTML;
    }

    function selectMemo(id) {
      selectedId = id;
      const memo = store.find(id);
      if (!memo) return;

      elements.emptyState.style.display = "none";
      elements.editorContent.style.display = "flex";
      elements.title.value = memo.title;
      elements.body.value = memo.body;
      elements.status.textContent = `最終更新: ${formatDate(memo.updatedAt)}`;
      renderList();
      elements.title.focus();
    }

    function saveCurrent() {
      if (selectedId === null) return;
      try {
        const memo = store.update(selectedId, elements.title.value, elements.body.value);
        elements.status.textContent = `保存しました (${formatDate(memo.updatedAt)})`;
        renderList();

        elements.status.textContent = e.message;
        elements.status.style.color = "#c62828";
        setTimeout(() => { elements.status.style.color = "#888"; }, 2000);
      }

      const memo = store.create();
      selectMemo(memo.id);

      if ((e.ctrlKey || e.metaKey) && e.key === "s") {
        e.preventDefault();
        saveCurrent();

**初心者向けとの違い:**
- `class` 構文でデータ管理を `MemoStore` クラスにカプセル化
- `find()`, `filter()`, `includes()` などの配列メソッドを活用
- スプレッド構文 `[...array]` で元の配列を壊さずソート
- テンプレートリテラルで `innerHTML` を一括生成（`map` + `join`）
- `dataset` 属性でDOM要素にデータIDを付与
