# 実践課題05：タイマーアプリ ★2

> **難易度**: ★★☆☆☆（基礎）
> **前提知識**: 第2章（リアクティビティ）、第3章（テンプレート構文）、第5章（Composition API ─ ライフサイクルフック）
> **課題の種類**: ミニプロジェクト
> **学習目標**: `onMounted` / `onUnmounted` ライフサイクルフック、`setInterval` / `clearInterval` の管理、`watch` によるリアクティブデータの監視を実践する

---

## 完成イメージ

ストップウォッチとカウントダウンタイマーの2機能を持つアプリです。

```
┌──────────────────────────────────────┐
│     ⏱️ タイマーアプリ                 │
├──────────────────────────────────────┤
│  [ストップウォッチ]  [カウントダウン]  │
│                                      │
│  ── ストップウォッチ ──               │
│                                      │
│          00:05.32                     │
│                                      │
│    [スタート]  [リセット]             │
│                                      │
│  ── ラップタイム ──                   │
│  #1  00:01.24                        │
│  #2  00:03.56                        │
│  #3  00:05.32                        │
└──────────────────────────────────────┘
```

---

## 課題の要件

1. **ストップウォッチモード**: スタート/ストップ/リセット/ラップ記録
2. **カウントダウンモード**: 分・秒を設定してカウントダウン、0になったら通知
3. タブでモードを切り替えられる
4. 時間表示は `MM:SS.mm`（分:秒.ミリ秒）形式
5. `onUnmounted` でタイマーのクリーンアップを行う
6. `watch` でカウントダウンが0になったことを検知する

---

## ステップガイド

<details>
<summary>ステップ1：ストップウォッチのロジックを作る</summary>

```js
const elapsed = ref(0)      // 経過ミリ秒
const isRunning = ref(false)
let intervalId = null

function start() {
  if (isRunning.value) return
  isRunning.value = true
  const startTime = Date.now() - elapsed.value
  intervalId = setInterval(() => {
    elapsed.value = Date.now() - startTime
  }, 10)
}

function stop() {
  isRunning.value = false
  clearInterval(intervalId)
}
```

</details>

<details>
<summary>ステップ2：時間のフォーマット関数を作る</summary>

```js
function formatTime(ms) {
  const minutes = Math.floor(ms / 60000)
  const seconds = Math.floor((ms % 60000) / 1000)
  const centiseconds = Math.floor((ms % 1000) / 10)
  return (
    String(minutes).padStart(2, '0') + ':' +
    String(seconds).padStart(2, '0') + '.' +
    String(centiseconds).padStart(2, '0')
  )
}
```

</details>

<details>
<summary>ステップ3：onUnmounted でクリーンアップする</summary>

```js
onUnmounted(() => {
  if (intervalId) clearInterval(intervalId)
})
```

コンポーネントが破棄されたとき、タイマーを止めないとメモリリーク（memory leak）になります。

</details>

    const { createApp, ref, watch, onUnmounted } = Vue

    createApp({
      setup() {
        var mode = ref("stopwatch")

        // ==================== ストップウォッチ ====================
        var swElapsed = ref(0)
        var swRunning = ref(false)
        var laps = ref([])
        var swIntervalId = null

        function swStart() {
          if (swRunning.value) return
          swRunning.value = true
          var startTime = Date.now() - swElapsed.value
          swIntervalId = setInterval(function() {
            swElapsed.value = Date.now() - startTime
          }, 10)
        }

        function swStop() {
          swRunning.value = false
          clearInterval(swIntervalId)
        }

        function swLap() {
          if (swRunning.value) {
            laps.value.push(swElapsed.value)
          }
        }

        function swReset() {
          swStop()
          swElapsed.value = 0
          laps.value = []
        }

        // ==================== カウントダウン ====================
        var cdMinutes = ref(1)
        var cdSeconds = ref(0)
        var cdRemaining = ref(0)
        var cdRunning = ref(false)
        var cdFinished = ref(false)
        var cdIntervalId = null

        function cdStart() {
          if (cdRunning.value) return
          if (cdRemaining.value <= 0) {
            // 新規スタート
            cdRemaining.value = (cdMinutes.value * 60 + cdSeconds.value) * 1000
          }
          if (cdRemaining.value <= 0) return

          cdRunning.value = true
          cdFinished.value = false
          var targetTime = Date.now() + cdRemaining.value
          cdIntervalId = setInterval(function() {
            var remaining = targetTime - Date.now()
            if (remaining <= 0) {
              cdRemaining.value = 0
              cdRunning.value = false
              cdFinished.value = true
              clearInterval(cdIntervalId)
            } else {
              cdRemaining.value = remaining
            }
          }, 10)
        }

        function cdStop() {
          cdRunning.value = false
          clearInterval(cdIntervalId)
        }

        function cdReset() {
          cdStop()
          cdRemaining.value = 0
          cdFinished.value = false
        }

        // watch でカウントダウン終了を検知
        watch(cdFinished, function(finished) {
          if (finished) {
            // ブラウザ通知（Notification API が利用可能な場合）
            if (typeof Notification !== 'undefined' && Notification.permission === 'granted') {
              new Notification('タイムアップ！', { body: 'カウントダウンが終了しました' })
            }
          }
        })

        // ==================== 共通 ====================
        function formatTime(ms) {
          if (ms < 0) ms = 0
          var minutes = Math.floor(ms / 60000)
          var seconds = Math.floor((ms % 60000) / 1000)
          var centiseconds = Math.floor((ms % 1000) / 10)
          return (
            String(minutes).padStart(2, '0') + ':' +
            String(seconds).padStart(2, '0') + '.' +
            String(centiseconds).padStart(2, '0')
          )
        }

        // クリーンアップ
        onUnmounted(function() {
          if (swIntervalId) clearInterval(swIntervalId)
          if (cdIntervalId) clearInterval(cdIntervalId)
        })

        return {
          mode, formatTime,
          swElapsed, swRunning, laps, swStart, swStop, swLap, swReset,
          cdMinutes, cdSeconds, cdRemaining, cdRunning, cdFinished,
          cdStart, cdStop, cdReset
        }
      }
    }).mount('#app')
  </script>

    const { createApp, ref, reactive, computed, watch, onUnmounted } = Vue

    // ---- composable: useStopwatch ----
    function useStopwatch(formatFn) {
      const elapsed = ref(0)
      const isRunning = ref(false)
      const laps = ref([])
      let intervalId = null

      const displayTime = computed(() => formatFn(elapsed.value))

      function start() {
        if (isRunning.value) return
        isRunning.value = true
        const base = Date.now() - elapsed.value
        intervalId = setInterval(() => { elapsed.value = Date.now() - base }, 10)
      }
      function stop() { isRunning.value = false; clearInterval(intervalId) }
      function reset() { stop(); elapsed.value = 0; laps.value = [] }
      function lap() { if (isRunning.value) laps.value.push(elapsed.value) }
      function cleanup() { clearInterval(intervalId) }

      return reactive({ elapsed, isRunning, laps, displayTime, start, stop, reset, lap, cleanup })
    }

    // ---- composable: useCountdown ----
    function useCountdown(formatFn) {
      const inputMin = ref(1)
      const inputSec = ref(0)
      const remaining = ref(0)
      const isRunning = ref(false)
      const isFinished = ref(false)
      let intervalId = null

      function start() {
        if (isRunning.value) return
        if (remaining.value <= 0) {
          remaining.value = (inputMin.value * 60 + inputSec.value) * 1000
        }
        if (remaining.value <= 0) return
        isRunning.value = true
        isFinished.value = false
        const target = Date.now() + remaining.value
        intervalId = setInterval(() => {
          const r = target - Date.now()
          if (r <= 0) { remaining.value = 0; isRunning.value = false; isFinished.value = true; clearInterval(intervalId) }
          else { remaining.value = r }
        }, 10)
      }
      function stop() { isRunning.value = false; clearInterval(intervalId) }
      function reset() { stop(); remaining.value = 0; isFinished.value = false }
      function cleanup() { clearInterval(intervalId) }

      return reactive({ inputMin, inputSec, remaining, isRunning, isFinished, start, stop, reset, cleanup })
    }

    createApp({
      setup() {
        const mode = ref("stopwatch")

        function formatMs(ms) {
          if (ms < 0) ms = 0
          const m = Math.floor(ms / 60000)
          const s = Math.floor((ms % 60000) / 1000)
          const cs = Math.floor((ms % 1000) / 10)
          return String(m).padStart(2, '0') + ':' + String(s).padStart(2, '0') + '.' + String(cs).padStart(2, '0')
        }

        const sw = useStopwatch(formatMs)
        const cd = useCountdown(formatMs)

          { label: "1分", min: 1, sec: 0 },
          { label: "3分", min: 3, sec: 0 },
          { label: "5分", min: 5, sec: 0 },
          { label: "10分", min: 10, sec: 0 },
        ]

        onUnmounted(() => { sw.cleanup(); cd.cleanup() })

        return { mode, sw, cd, presets, formatMs }
      }
    }).mount('#app')

**初心者向けとの違い:**
- `useStopwatch` / `useCountdown` という **composable**（コンポーザブル）関数にロジックを抽出
- `reactive` でまとめたオブジェクトを返すことで、テンプレートで `sw.start()` のように呼べる
- `computed` でフォーマット済み時間を提供
- カウントダウンのプリセットボタン（30秒、1分、3分...）
- ダークテーマで見やすいUI
- 一時停止からの再開にも対応
