## 課題の要件

1. サンプルのアクセスログデータをプログラム内で生成する（外部ファイル不要）
2. IPアドレス別、ステータスコード別、時間帯別の集計機能を実装する
3. ブルートフォース攻撃（brute force attack）の検知機能を実装する（短時間に多数の認証失敗）
4. 攻撃パターン（SQLインジェクション、XSS等）をURLから検知する
5. 分析結果をテキストファイルに出力するレポート機能を含める

```python
# アクセスログ分析ツール
# 学べる内容：ログ分析、不正アクセス検知、ブルートフォース検知
# 実行方法：python log_analyzer.py

import random
from datetime import datetime, timedelta


def generate_sample_logs():
    """サンプルのアクセスログを生成します。
    正常なアクセスに加えて、攻撃パターンを混ぜています。
    """
    logs = []
    normal_ips = ["192.168.1.100", "192.168.1.101", "10.0.0.50"]
    attacker_ip = "192.168.1.105"
    scanner_ip = "172.16.0.200"

    normal_paths = ["/index.html", "/about", "/api/users", "/products", "/contact"]
    attack_paths_sqli = [
        "/search?q=' OR 1=1--",
        "/login?user=admin'--",
        "/api/users?id=1 UNION SELECT * FROM passwords",
    ]
    attack_paths_xss = [
        "/search?q=<script>alert('xss')</script>",
        "/comment?text=<img onerror=alert(1)>",
    ]

    base_time = datetime(2024, 1, 15, 10, 0, 0)
    random.seed(42)  # 再現性のためシードを固定

    # 正常アクセス（30件）
    for i in range(30):
        logs.append({
            "timestamp": base_time + timedelta(minutes=random.randint(0, 480)),
            "ip": random.choice(normal_ips),
            "method": "GET",
            "path": random.choice(normal_paths),
            "status": random.choice([200, 200, 200, 200, 301, 304]),
        })

    # ブルートフォース攻撃（15件 - 短時間にログイン試行）
    bf_base = base_time + timedelta(hours=4, minutes=23)
    for i in range(15):
        logs.append({
            "timestamp": bf_base + timedelta(seconds=random.randint(0, 280)),
            "ip": attacker_ip,
            "method": "POST",
            "path": "/login",
            "status": random.choice([401, 401, 401, 403]),
        })

    # SQLインジェクション攻撃（3件）
    for i in range(3):
        logs.append({
            "timestamp": base_time + timedelta(hours=2, minutes=random.randint(0, 30)),
            "ip": scanner_ip,
            "method": "GET",
            "path": random.choice(attack_paths_sqli),
            "status": 500,
        })

    # XSS攻撃（2件）
    for i in range(2):
        logs.append({
            "timestamp": base_time + timedelta(hours=3, minutes=random.randint(0, 30)),
            "ip": scanner_ip,
            "method": "GET",
            "path": random.choice(attack_paths_xss),
            "status": 200,
        })

    # タイムスタンプ順にソート
    logs.sort(key=lambda x: x["timestamp"])
    return logs


def show_logs(logs):
    """ログ一覧を表示します。"""
    print(f"\n--- アクセスログ一覧（{len(logs)}件） ---")
    for log in logs:
        ts = log["timestamp"].strftime("%Y-%m-%d %H:%M:%S")
        print(f"  {ts} | {log['ip']:<16} | {log['method']:<4} | {log['status']} | {log['path'][:50]}")


def count_by_ip(logs):
    """IPアドレス別のアクセス数を集計します。"""
    counts = {}
    for log in logs:
        ip = log["ip"]
        if ip in counts:
            counts[ip] += 1
        else:
            counts[ip] = 1

    print("\n--- IPアドレス別アクセス数 ---")
    for ip, count in sorted(counts.items(), key=lambda x: x[1], reverse=True):
        bar = "#" * count
        print(f"  {ip:<16} : {count:3}件 {bar}")


def count_by_status(logs):
    """ステータスコード別の集計を表示します。"""
    counts = {}
    for log in logs:
        status = log["status"]
        if status in counts:
            counts[status] += 1
        else:
            counts[status] = 1

    print("\n--- ステータスコード別集計 ---")
    status_descriptions = {
        200: "OK", 301: "Moved", 304: "Not Modified",
        401: "Unauthorized", 403: "Forbidden",
        404: "Not Found", 500: "Internal Server Error",
    }
    for status in sorted(counts.keys()):
        desc = status_descriptions.get(status, "Unknown")
        count = counts[status]
        print(f"  {status} ({desc:<20}): {count}件")


def detect_brute_force(logs, threshold=10):
    """ブルートフォース攻撃を検知します。"""
    print(f"\n--- ブルートフォース攻撃検知 ---")
    print(f"閾値: 同一IPから {threshold} 回以上の認証失敗（401/403）")

    # 401/403のログをIPごとにグループ化
    failures_by_ip = {}
    for log in logs:
        if log["status"] in [401, 403]:
            ip = log["ip"]
            if ip not in failures_by_ip:
                failures_by_ip[ip] = []
            failures_by_ip[ip].append(log)

    found = False
    for ip, failures in failures_by_ip.items():
        if len(failures) >= threshold:
            found = True
            first_ts = failures[0]["timestamp"].strftime("%Y-%m-%d %H:%M:%S")
            last_ts = failures[-1]["timestamp"].strftime("%Y-%m-%d %H:%M:%S")

            # アクセス先の集計
            path_counts = {}
            for f in failures:
                path = f["path"]
                path_counts[path] = path_counts.get(path, 0) + 1

            print(f"\n  警告: {ip} から {len(failures)} 回の認証失敗を検知")
            print(f"    最初の失敗: {first_ts}")
            print(f"    最後の失敗: {last_ts}")
            for path, count in path_counts.items():
                print(f"    対象パス:   {path} ({count}回)")
            print("    → ブルートフォース攻撃の可能性があります")

    if not found:
        print("\n  ブルートフォース攻撃の兆候は検知されませんでした。")


def detect_attack_patterns(logs):
    """SQLインジェクション/XSSなどの攻撃パターンを検知します。"""
    print("\n--- 攻撃パターン検知 ---")

    sqli_keywords = ["UNION", "SELECT", "DROP", "DELETE", "OR 1=1", "'--", "' OR"]
    xss_keywords = ["<script", "javascript:", "onerror=", "onload=", "alert("]

    sqli_found = []
    xss_found = []

    for log in logs:
        path_upper = log["path"].upper()
        path_lower = log["path"].lower()

        for keyword in sqli_keywords:
            if keyword.upper() in path_upper:
                sqli_found.append(log)
                break

        for keyword in xss_keywords:
            if keyword.lower() in path_lower:
                xss_found.append(log)
                break

    if sqli_found:
        print(f"\n  SQLインジェクション疑い: {len(sqli_found)}件")
        for log in sqli_found:
            ts = log["timestamp"].strftime("%H:%M:%S")
            print(f"    [{ts}] {log['ip']} → {log['path'][:60]}")
    else:
        print("\n  SQLインジェクション: 検知なし")

    if xss_found:
        print(f"\n  XSS疑い: {len(xss_found)}件")
        for log in xss_found:
            ts = log["timestamp"].strftime("%H:%M:%S")
            print(f"    [{ts}] {log['ip']} → {log['path'][:60]}")
    else:
        print("\n  XSS: 検知なし")


def analyze_by_hour(logs):
    """時間帯別のアクセス分布を表示します。"""
    print("\n--- 時間帯別アクセス分布 ---")
    hour_counts = {}
    for log in logs:
        hour = log["timestamp"].hour
        hour_counts[hour] = hour_counts.get(hour, 0) + 1

    for hour in range(24):
        count = hour_counts.get(hour, 0)
        bar = "#" * count
        print(f"  {hour:02d}時: {count:3}件 {bar}")


def export_report(logs):
    """分析レポートをファイルに出力します。"""
    filename = "security_report.txt"
    with open(filename, "w", encoding="utf-8") as f:
        f.write("===== セキュリティ分析レポート =====\n")
        f.write(f"分析日時: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}\n")
        f.write(f"ログ件数: {len(logs)}件\n\n")

        # IP別集計
        ip_counts = {}
        for log in logs:
            ip_counts[log["ip"]] = ip_counts.get(log["ip"], 0) + 1
        f.write("--- IPアドレス別アクセス数 ---\n")
        for ip, count in sorted(ip_counts.items(), key=lambda x: x[1], reverse=True):
            f.write(f"  {ip:<16}: {count}件\n")

        # 認証失敗
        fail_count = sum(1 for log in logs if log["status"] in [401, 403])
        f.write(f"\n認証失敗数（401/403）: {fail_count}件\n")

        f.write(f"\nレポート出力完了\n")

    print(f"\nレポートを {filename} に出力しました。")


# --- メイン処理 ---
print("===== アクセスログ分析ツール =====")
logs = generate_sample_logs()
print(f"サンプルログを生成しました（{len(logs)}件）")

while True:
    print("\n1. ログ一覧表示")
    print("2. IPアドレス別アクセス集計")
    print("3. ステータスコード別集計")
    print("4. 不正アクセス検知（ブルートフォース）")
    print("5. 攻撃パターン検知（SQLi/XSS）")
    print("6. 時間帯別アクセス分布")
    print("7. レポート出力")
    print("8. 終了")

    choice = input("\n選択 (1-8): ")

    if choice == "1":
        show_logs(logs)
    elif choice == "2":
        count_by_ip(logs)
    elif choice == "3":
        count_by_status(logs)
    elif choice == "4":
        detect_brute_force(logs)
    elif choice == "5":

    elif choice == "6":
        analyze_by_hour(logs)
    elif choice == "7":
        export_report(logs)
    elif choice == "8":
        print("終了します。")
        break
    else:
        print("1〜8の番号を入力してください。")
```

```python
# アクセスログ分析ツール（改良版）
# 学べる内容：クラス設計、統計分析、セキュリティ監視
# 実行方法：python log_analyzer_v2.py

import random
from datetime import datetime, timedelta
from collections import Counter


class AccessLog:
    """1件のアクセスログを表すクラスです。"""

    def __init__(self, timestamp, ip, method, path, status):
        self.timestamp = timestamp
        self.ip = ip
        self.method = method
        self.path = path
        self.status = status

    def __str__(self):
        ts = self.timestamp.strftime("%Y-%m-%d %H:%M:%S")
        return f"{ts} | {self.ip:<16} | {self.method:<4} | {self.status} | {self.path[:50]}"

    @property
    def is_error(self):
        return self.status >= 400

    @property
    def is_auth_failure(self):
        return self.status in (401, 403)


class LogAnalyzer:
    """アクセスログを分析するクラスです。"""

    SQLI_PATTERNS = ["UNION", "SELECT", "DROP", "DELETE", "OR 1=1", "'--", "' OR", "INSERT"]
    XSS_PATTERNS = ["<script", "javascript:", "onerror=", "onload=", "alert(", "<iframe"]

    def __init__(self, logs):
        self.logs = logs

    def count_by(self, key_func):
        """任意のキー関数で集計します。"""
        return Counter(key_func(log) for log in self.logs)

    def filter_logs(self, predicate):
        """条件に一致するログを抽出します。"""
        return [log for log in self.logs if predicate(log)]

    def detect_brute_force(self, threshold=10, window_minutes=5):
        """ブルートフォース攻撃を検知します。"""
        auth_failures = self.filter_logs(lambda log: log.is_auth_failure)
        ip_failures = {}
        for log in auth_failures:
            ip_failures.setdefault(log.ip, []).append(log)

        alerts = []
        for ip, failures in ip_failures.items():
            failures.sort(key=lambda x: x.timestamp)
            # スライディングウィンドウで判定
            window = timedelta(minutes=window_minutes)
            for i, start_log in enumerate(failures):
                count = sum(
                    1 for f in failures[i:]
                    if f.timestamp - start_log.timestamp <= window
                )
                if count >= threshold:
                    alerts.append({
                        "ip": ip,
                        "count": count,
                        "first": failures[i].timestamp,
                        "last": failures[-1].timestamp,
                        "paths": Counter(f.path for f in failures),
                    })
                    break  # 同一IPは1回だけ報告
        return alerts

    def detect_sqli(self):
        """SQLインジェクションの兆候を検知します。"""
        results = []
        for log in self.logs:
            path_upper = log.path.upper()
            matched = [p for p in self.SQLI_PATTERNS if p.upper() in path_upper]
            if matched:
                results.append({"log": log, "patterns": matched})
        return results

    def detect_xss(self):
        """XSSの兆候を検知します。"""
        results = []
        for log in self.logs:
            path_lower = log.path.lower()
            matched = [p for p in self.XSS_PATTERNS if p.lower() in path_lower]
            if matched:
                results.append({"log": log, "patterns": matched})
        return results

    def generate_report(self):
        """分析レポートの文字列を生成します。"""
        lines = []
        lines.append("===== セキュリティ分析レポート =====")
        lines.append(f"分析日時: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}")
        lines.append(f"ログ件数: {len(self.logs)}件")
        lines.append("")

        # IP別集計
        ip_counts = self.count_by(lambda log: log.ip)
        lines.append("--- IPアドレス別アクセス数 ---")
        for ip, count in ip_counts.most_common():
            lines.append(f"  {ip:<16}: {count}件")

        # 攻撃検知サマリ
        bf_alerts = self.detect_brute_force()
        sqli = self.detect_sqli()
        xss = self.detect_xss()
        lines.append("")
        lines.append("--- 攻撃検知サマリ ---")
        lines.append(f"  ブルートフォース: {len(bf_alerts)}件")
        lines.append(f"  SQLインジェクション疑い: {len(sqli)}件")
        lines.append(f"  XSS疑い: {len(xss)}件")

        error_rate = sum(1 for log in self.logs if log.is_error) / len(self.logs) * 100
        lines.append(f"\n  エラー率: {error_rate:.1f}%")

        return "\n".join(lines)


def generate_sample_logs():
    """サンプルログを生成します。"""
    logs = []
    normal_ips = ["192.168.1.100", "192.168.1.101", "10.0.0.50"]
    attacker_ip = "192.168.1.105"
    scanner_ip = "172.16.0.200"
    normal_paths = ["/index.html", "/about", "/api/users", "/products", "/contact"]

    base_time = datetime(2024, 1, 15, 10, 0, 0)
    random.seed(42)

    for _ in range(30):
        logs.append(AccessLog(
            base_time + timedelta(minutes=random.randint(0, 480)),
            random.choice(normal_ips), "GET",
            random.choice(normal_paths),
            random.choice([200, 200, 200, 200, 301, 304]),
        ))

    bf_base = base_time + timedelta(hours=4, minutes=23)
    for _ in range(15):
        logs.append(AccessLog(
            bf_base + timedelta(seconds=random.randint(0, 280)),
            attacker_ip, "POST", "/login",
            random.choice([401, 401, 401, 403]),
        ))

    for _ in range(3):
        logs.append(AccessLog(
            base_time + timedelta(hours=2, minutes=random.randint(0, 30)),
            scanner_ip, "GET",
            random.choice(["/search?q=' OR 1=1--", "/api/users?id=1 UNION SELECT *"]),
            500,
        ))

    for _ in range(2):
        logs.append(AccessLog(
            base_time + timedelta(hours=3, minutes=random.randint(0, 30)),
            scanner_ip, "GET",
            "/search?q=<script>alert('xss')</script>",
            200,
        ))

    logs.sort(key=lambda x: x.timestamp)
    return logs


def main():
    print("===== アクセスログ分析ツール =====")
    logs = generate_sample_logs()
    analyzer = LogAnalyzer(logs)
    print(f"サンプルログを生成しました（{len(logs)}件）")

    while True:
        print("\n1. ログ一覧  2. IP別集計  3. ステータス別集計")
        print("4. ブルートフォース検知  5. 攻撃パターン検知")
        print("6. 時間帯別分布  7. レポート出力  8. 終了")

        choice = input("\n選択 (1-8): ")

        if choice == "1":
            for log in logs:
                print(f"  {log}")
        elif choice == "2":
            for ip, count in analyzer.count_by(lambda l: l.ip).most_common():
                print(f"  {ip:<16}: {count:3}件 {'#' * count}")
        elif choice == "3":
            for status, count in sorted(analyzer.count_by(lambda l: l.status).items()):
                print(f"  {status}: {count}件")
        elif choice == "4":
            alerts = analyzer.detect_brute_force()
            if alerts:
                for a in alerts:
                    print(f"  警告: {a['ip']} から {a['count']} 回の認証失敗")
            else:
                print("  検知なし")
        elif choice == "5":
            sqli = analyzer.detect_sqli()
            xss = analyzer.detect_xss()
            print(f"  SQLi: {len(sqli)}件、XSS: {len(xss)}件")
            for r in sqli + xss:
                print(f"    {r['log']}")
        elif choice == "6":
            for hour, count in sorted(analyzer.count_by(lambda l: l.timestamp.hour).items()):
                print(f"  {hour:02d}時: {count:3}件 {'#' * count}")
        elif choice == "7":
            report = analyzer.generate_report()
            print(report)
            with open("security_report.txt", "w", encoding="utf-8") as f:
                f.write(report)
            print("\nsecurity_report.txt に保存しました。")
        elif choice == "8":
            print("終了します。")
            break


if __name__ == "__main__":
    main()
```
