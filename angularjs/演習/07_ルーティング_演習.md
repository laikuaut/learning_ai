## 演習7-1：基本的なルーティング（基本）

### 問題

ngRoute を使って以下の3ページを持つ SPA を作成してください。

1. ホーム（`/`）— 「ようこそ」メッセージを表示
2. 概要（`/about`）— サイトの説明を表示
3. お問い合わせ（`/contact`）— メールアドレスを表示
4. ナビゲーションバーに各ページへのリンクを配置
5. 存在しないURLは `/` にリダイレクト

<details>
<summary>解答例</summary>

```html
<!DOCTYPE html>
<html ng-app="myApp">
<head>
  <meta charset="UTF-8">
  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.3/angular.min.js"></script>
  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.3/angular-route.min.js"></script>
  <style>
    body { font-family: sans-serif; margin: 0; }
    nav { background: #35495e; padding: 0; }
    nav a { color: white; text-decoration: none; padding: 14px 20px; display: inline-block; }
    nav a:hover { background: #42b883; }
    .content { padding: 24px; max-width: 700px; margin: 0 auto; }
  </style>
</head>
<body>
  <!-- ナビゲーション -->
  <nav>
    <a href="#!/">ホーム</a>
    <a href="#!/about">概要</a>
    <a href="#!/contact">お問い合わせ</a>
  </nav>
  
  <!-- ルートに応じたコンテンツ -->
  <div class="content">
    <ng-view></ng-view>
  </div>

  <!-- インラインテンプレート -->
  <script type="text/ng-template" id="home.html">
    <h2>ようこそ</h2>
    <p>AngularJS ルーティングのデモサイトです。上のナビゲーションからページを移動できます。</p>
  </script>

  <script type="text/ng-template" id="about.html">
    <h2>概要</h2>
    <p>このサイトは ngRoute を使ったSPAのサンプルです。</p>
    <p>ページ遷移時にサーバーへのリクエストは発生しません。</p>
  </script>

  <script type="text/ng-template" id="contact.html">
    <h2>お問い合わせ</h2>
    <p>メール: contact@example.com</p>
    <p>電話: 03-1234-5678</p>
  </script>

  <script>
    var app = angular.module('myApp', ['ngRoute']);
    
    // ルーティング設定
    app.config(['$routeProvider', function($routeProvider) {
      $routeProvider
        .when('/', { templateUrl: 'home.html' })
        .when('/about', { templateUrl: 'about.html' })
        .when('/contact', { templateUrl: 'contact.html' })
        .otherwise({ redirectTo: '/' }); // 不明なURLはホームへ
    }]);
  </script>
</body>
</html>
```

</details>

## 演習7-2：ルートパラメータを使う（基本）

### 問題

フルーツの一覧ページと詳細ページを作成してください。

1. `/fruits` — フルーツ一覧を表示（クリックで詳細へ）
2. `/fruits/:id` — 選択したフルーツの詳細情報を表示
3. `$routeParams` でIDを取得し、対応するデータを表示
4. 詳細ページに「一覧に戻る」リンクを配置

<details>
<summary>解答例</summary>

```html
<!DOCTYPE html>
<html ng-app="myApp">
<head>
  <meta charset="UTF-8">
  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.3/angular.min.js"></script>
  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.3/angular-route.min.js"></script>
  <style>
    body { font-family: sans-serif; margin: 0; }
    nav { background: #35495e; padding: 14px 20px; }
    nav a { color: white; text-decoration: none; }
    .content { padding: 24px; max-width: 600px; margin: 0 auto; }
    .fruit-item { padding: 12px; border: 1px solid #ddd; border-radius: 8px; margin: 8px 0; cursor: pointer; }
    .fruit-item:hover { background: #f5f5f5; }
    .back { color: #42b883; }
    .detail { border: 2px solid #42b883; border-radius: 12px; padding: 20px; }
  </style>
</head>
<body>
  <nav><a href="#!/fruits">フルーツ図鑑</a></nav>
  <div class="content"><ng-view></ng-view></div>

  <script type="text/ng-template" id="list.html">
    <h2>フルーツ一覧</h2>
    <div class="fruit-item" ng-repeat="f in fruits" ng-click="goTo(f.id)">
      {{ f.name }} — {{ f.color }}
    </div>
  </script>

  <script type="text/ng-template" id="detail.html">
    <a class="back" href="#!/fruits">← 一覧に戻る</a>
    <div class="detail" ng-if="fruit">
      <h2>{{ fruit.name }}</h2>
      <p><strong>色:</strong> {{ fruit.color }}</p>
      <p><strong>味:</strong> {{ fruit.taste }}</p>
      <p><strong>旬:</strong> {{ fruit.season }}</p>
    </div>
    <p ng-if="!fruit" style="color: #e74c3c;">フルーツが見つかりません</p>
  </script>

  <script>
    var app = angular.module('myApp', ['ngRoute']);
    
    // データサービス
    app.factory('FruitService', function() {
      var fruits = [
        { id: 1, name: 'りんご', color: '赤', taste: '甘酸っぱい', season: '秋' },
        { id: 2, name: 'バナナ', color: '黄', taste: '甘い', season: '通年' },
        { id: 3, name: 'みかん', color: 'オレンジ', taste: '甘い', season: '冬' },
        { id: 4, name: 'ぶどう', color: '紫', taste: '甘い', season: '秋' },
        { id: 5, name: 'いちご', color: '赤', taste: '甘酸っぱい', season: '春' }
      ];
      return {
        getAll: function() { return fruits; },
        getById: function(id) { return fruits.find(function(f) { return f.id === parseInt(id); }); }
      };
    });
    
    app.config(['$routeProvider', function($routeProvider) {
      $routeProvider
        .when('/fruits', { templateUrl: 'list.html', controller: 'ListCtrl' })
        .when('/fruits/:id', { templateUrl: 'detail.html', controller: 'DetailCtrl' })
        .otherwise({ redirectTo: '/fruits' });
    }]);
    
    app.controller('ListCtrl', ['$scope', '$location', 'FruitService',
      function($scope, $location, FruitService) {
        $scope.fruits = FruitService.getAll();
        $scope.goTo = function(id) { $location.path('/fruits/' + id); };
      }
    ]);
    
    // $routeParams でURLパラメータを取得
    app.controller('DetailCtrl', ['$scope', '$routeParams', 'FruitService',
      function($scope, $routeParams, FruitService) {
        $scope.fruit = FruitService.getById($routeParams.id);
      }
    ]);
  </script>
</body>
</html>
```

</details>

## 演習7-3：アクティブ状態のナビゲーション（応用）

### 問題

現在のページに応じてナビゲーションリンクをハイライトしてください。

1. `$location.path()` で現在のパスを取得
2. `ng-class` で現在のルートと一致するリンクに `active` クラスを付与
3. 4ページ以上のナビゲーションを作成

<details>
<summary>解答例</summary>

```html
<!DOCTYPE html>
<html ng-app="myApp">
<head>
  <meta charset="UTF-8">
  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.3/angular.min.js"></script>
  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.3/angular-route.min.js"></script>
  <style>
    body { font-family: sans-serif; margin: 0; }
    nav { background: #35495e; display: flex; }
    nav a { color: rgba(255,255,255,0.7); text-decoration: none; padding: 14px 20px; transition: all 0.2s; }
    nav a:hover { color: white; background: rgba(255,255,255,0.1); }
    nav a.active { color: white; background: #42b883; }
    .content { padding: 24px; max-width: 700px; margin: 0 auto; }
  </style>
</head>
<body ng-controller="NavCtrl">
  <nav>
    <a href="#!/" ng-class="{ active: isActive('/') }">ホーム</a>
    <a href="#!/products" ng-class="{ active: isActive('/products') }">商品</a>
    <a href="#!/about" ng-class="{ active: isActive('/about') }">概要</a>
    <a href="#!/contact" ng-class="{ active: isActive('/contact') }">連絡先</a>
  </nav>
  <div class="content"><ng-view></ng-view></div>

  <script>
    var app = angular.module('myApp', ['ngRoute']);
    
    app.config(['$routeProvider', function($routeProvider) {
      $routeProvider
        .when('/', { template: '<h2>ホーム</h2><p>メインページです。</p>' })
        .when('/products', { template: '<h2>商品</h2><p>商品一覧です。</p>' })
        .when('/about', { template: '<h2>概要</h2><p>サイト概要です。</p>' })
        .when('/contact', { template: '<h2>連絡先</h2><p>お問い合わせ先です。</p>' })
        .otherwise({ redirectTo: '/' });
    }]);
    
    // $location を使って現在のパスを判定
    app.controller('NavCtrl', ['$scope', '$location', function($scope, $location) {
      $scope.isActive = function(path) {
        return $location.path() === path;
      };
    }]);
  </script>
</body>
</html>
```

</details>

## 演習7-4：resolve でデータを事前読み込み（応用）

### 問題

resolve を使って、ページ表示前に API データを読み込んでください。

1. JSONPlaceholder API (`/posts?userId=1`) からデータを取得
2. resolve でデータを取得してからページを表示
3. データ取得中はローディング表示を出す（`$routeChangeStart` イベント使用）

<details>
<summary>解答例</summary>

```html
<!DOCTYPE html>
<html ng-app="myApp">
<head>
  <meta charset="UTF-8">
  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.3/angular.min.js"></script>
  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.3/angular-route.min.js"></script>
  <style>
    body { font-family: sans-serif; margin: 0; }
    nav { background: #35495e; padding: 0; }
    nav a { color: white; text-decoration: none; padding: 14px 20px; display: inline-block; }
    nav a:hover { background: #42b883; }
    .content { padding: 24px; max-width: 700px; margin: 0 auto; }
    .loading { text-align: center; padding: 40px; color: #999; font-size: 18px; }
    .post { border-left: 3px solid #42b883; padding: 8px 16px; margin: 10px 0; }
  </style>
</head>
<body>
  <nav>
    <a href="#!/">ホーム</a>
    <a href="#!/posts">投稿一覧（resolve）</a>
  </nav>
  
  <!-- ローディング表示 -->
  <div class="loading" ng-show="loading">読み込み中...</div>
  
  <div class="content" ng-hide="loading">
    <ng-view></ng-view>
  </div>

  <script type="text/ng-template" id="home.html">
    <h2>ホーム</h2>
    <p>「投稿一覧」をクリックすると、resolve でデータを取得してから表示します。</p>
  </script>

  <script type="text/ng-template" id="posts.html">
    <h2>投稿一覧（{{ posts.length }}件）</h2>
    <div class="post" ng-repeat="post in posts">
      <h3>{{ post.title }}</h3>
      <p>{{ post.body | limitTo:80 }}...</p>
    </div>
  </script>

  <script>
    var app = angular.module('myApp', ['ngRoute']);
    
    app.config(['$routeProvider', function($routeProvider) {
      $routeProvider
        .when('/', { templateUrl: 'home.html' })
        .when('/posts', {
          templateUrl: 'posts.html',
          controller: 'PostsCtrl',
          // resolve: ページ表示前にデータを取得
          resolve: {
            postData: ['$http', function($http) {
              return $http.get('https://jsonplaceholder.typicode.com/posts?userId=1')
                .then(function(res) { return res.data; });
            }]
          }
        })
        .otherwise({ redirectTo: '/' });
    }]);
    
    // resolve で取得したデータはコントローラに注入される
    app.controller('PostsCtrl', ['$scope', 'postData', function($scope, postData) {
      $scope.posts = postData;
    }]);
    
    // ルーティングイベントでローディング表示
    app.run(['$rootScope', function($rootScope) {
      $rootScope.loading = false;
      $rootScope.$on('$routeChangeStart', function() {
        $rootScope.loading = true;
      });
      $rootScope.$on('$routeChangeSuccess', function() {
        $rootScope.loading = false;
      });
      $rootScope.$on('$routeChangeError', function() {
        $rootScope.loading = false;
      });
    }]);
  </script>
</body>
</html>
```

</details>

## 演習7-5：ブログ SPA の構築（チャレンジ）

### 問題

以下の機能を持つブログ SPA を作成してください。

1. 記事一覧ページ（`/`）— タイトルと概要を表示
2. 記事詳細ページ（`/post/:id`）— 記事全文を表示
3. 新規作成ページ（`/new`）— フォームで記事を追加
4. 存在しない記事IDは404ページを表示
5. ナビゲーションのアクティブ状態
6. 記事データは `PostService` で管理

<details>
<summary>解答例</summary>

```html
<!DOCTYPE html>
<html ng-app="blogApp">
<head>
  <meta charset="UTF-8">
  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.3/angular.min.js"></script>
  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.3/angular-route.min.js"></script>
  <style>
    body { font-family: sans-serif; margin: 0; background: #f5f5f5; }
    nav { background: #35495e; display: flex; }
    nav a { color: rgba(255,255,255,0.7); text-decoration: none; padding: 14px 20px; }
    nav a.active { color: white; background: #42b883; }
    .content { max-width: 700px; margin: 20px auto; padding: 0 20px; }
    .card { background: white; border-radius: 8px; padding: 20px; margin: 12px 0; box-shadow: 0 1px 4px rgba(0,0,0,0.1); cursor: pointer; }
    .card:hover { box-shadow: 0 2px 8px rgba(0,0,0,0.15); }
    .card h3 { margin: 0 0 8px; color: #35495e; }
    .card p { color: #666; margin: 0; }
    .card .date { font-size: 12px; color: #999; }
    .back { color: #42b883; text-decoration: none; }
    .form-group { margin: 12px 0; }
    .form-group label { display: block; font-weight: bold; margin-bottom: 4px; }
    .form-group input, .form-group textarea { width: 100%; padding: 10px; border: 1px solid #ddd; border-radius: 4px; box-sizing: border-box; font-family: inherit; }
    .form-group textarea { height: 150px; }
    .btn { padding: 10px 20px; background: #42b883; color: white; border: none; border-radius: 4px; cursor: pointer; }
    .btn:disabled { background: #ccc; }
  </style>
</head>
<body ng-controller="AppCtrl">
  <nav>
    <a href="#!/" ng-class="{ active: isActive('/') }">記事一覧</a>
    <a href="#!/new" ng-class="{ active: isActive('/new') }">新規作成</a>
  </nav>
  <div class="content"><ng-view></ng-view></div>

  <script type="text/ng-template" id="list.html">
    <h2>ブログ記事（{{ posts.length }}件）</h2>
    <div class="card" ng-repeat="post in posts | orderBy:'-id'" ng-click="goTo(post.id)">
      <span class="date">{{ post.date }}</span>
      <h3>{{ post.title }}</h3>
      <p>{{ post.body | limitTo:60 }}...</p>
    </div>
  </script>

  <script type="text/ng-template" id="detail.html">
    <a class="back" href="#!/">← 記事一覧</a>
    <div ng-if="post" style="background:white; padding:24px; border-radius:8px; margin-top:12px;">
      <span style="color:#999; font-size:13px;">{{ post.date }}</span>
      <h2>{{ post.title }}</h2>
      <p style="white-space:pre-wrap;">{{ post.body }}</p>
    </div>
    <div ng-if="!post" style="text-align:center; padding:40px;">
      <h2>404 — 記事が見つかりません</h2>
      <a class="back" href="#!/">ホームに戻る</a>
    </div>
  </script>

  <script type="text/ng-template" id="new.html">
    <h2>新規記事作成</h2>
    <form name="postForm" ng-submit="create()" novalidate>
      <div class="form-group">
        <label>タイトル</label>
        <input type="text" ng-model="newPost.title" required>
      </div>
      <div class="form-group">
        <label>本文</label>
        <textarea ng-model="newPost.body" required ng-minlength="10"></textarea>
      </div>
      <button class="btn" type="submit" ng-disabled="postForm.$invalid">投稿する</button>
    </form>
  </script>

  <script>
    var app = angular.module('blogApp', ['ngRoute']);
    
    app.factory('PostService', function() {
      var nextId = 4;
      var posts = [
        { id: 1, title: 'AngularJSを始めよう', body: 'AngularJSは強力なMVCフレームワークです。双方向データバインディングにより、効率的にWebアプリケーションを構築できます。', date: '2024-03-01' },
        { id: 2, title: 'ルーティングの基本', body: 'ngRouteを使うと、SPAのページ遷移を簡単に実装できます。$routeProviderで各ルートを定義し、ng-viewでコンテンツを切り替えます。', date: '2024-03-10' },
        { id: 3, title: 'サービスの活用', body: 'サービスはコントローラ間でデータを共有する仕組みです。factoryパターンが最も一般的に使われます。', date: '2024-03-15' }
      ];
      return {
        getAll: function() { return posts; },
        getById: function(id) { return posts.find(function(p) { return p.id === parseInt(id); }); },
        create: function(post) {
          post.id = nextId++;
          post.date = new Date().toISOString().slice(0, 10);
          posts.push(post);
          return post;
        }
      };
    });
    
    app.config(['$routeProvider', function($routeProvider) {
      $routeProvider
        .when('/', { templateUrl: 'list.html', controller: 'ListCtrl' })
        .when('/new', { templateUrl: 'new.html', controller: 'NewCtrl' })
        .when('/post/:id', { templateUrl: 'detail.html', controller: 'DetailCtrl' })
        .otherwise({ redirectTo: '/' });
    }]);
    
    app.controller('AppCtrl', ['$scope', '$location', function($scope, $location) {
      $scope.isActive = function(path) { return $location.path() === path; };
    }]);
    
    app.controller('ListCtrl', ['$scope', '$location', 'PostService',
      function($scope, $location, PostService) {
        $scope.posts = PostService.getAll();
        $scope.goTo = function(id) { $location.path('/post/' + id); };
      }
    ]);
    
    app.controller('DetailCtrl', ['$scope', '$routeParams', 'PostService',
      function($scope, $routeParams, PostService) {
        $scope.post = PostService.getById($routeParams.id);
      }
    ]);
    
    app.controller('NewCtrl', ['$scope', '$location', 'PostService',
      function($scope, $location, PostService) {
        $scope.newPost = {};
        $scope.create = function() {
          PostService.create($scope.newPost);
          $location.path('/');
        };
      }
    ]);
  </script>
</body>
</html>
```

</details>
