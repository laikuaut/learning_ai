# 第11章: エラー処理

## この章のゴール

- JavaScriptのエラーの種類と原因を理解する
- try...catch...finally を使った適切なエラー処理ができる
- カスタムエラーを作成・活用できる
- 非同期処理のエラーハンドリングパターンを身につける
- 防御的プログラミングの考え方を習得する

---

## 11.1 エラーの種類

```
JavaScriptの主なエラー型:

  Error               すべてのエラーの基底クラス
    ├── SyntaxError      構文エラー（コードの書き方が間違い）
    ├── ReferenceError   未定義の変数を参照
    ├── TypeError        型が不正（nullのプロパティにアクセス等）
    ├── RangeError       値が範囲外
    └── URIError         URI関連のエラー
```

```javascript
// SyntaxError — 構文エラー（実行前に検出される）
// const x = ; // SyntaxError: Unexpected token ';'

// ReferenceError — 未定義の変数
// console.log(undeclaredVar); // ReferenceError: undeclaredVar is not defined

// TypeError — 型の不一致
// null.toString(); // TypeError: Cannot read properties of null
// (123).toUpperCase(); // TypeError: (123).toUpperCase is not a function

// RangeError — 範囲外
// new Array(-1); // RangeError: Invalid array length
// (1.5).toFixed(200); // RangeError: toFixed() digits argument must be between 0 and 100

// Errorオブジェクトのプロパティ
try {
  undeclaredVar;
} catch (e) {
  console.log(e.name);    // "ReferenceError"
  console.log(e.message); // "undeclaredVar is not defined"
  console.log(e.stack);   // スタックトレース（デバッグに重要）
}
```

## 11.2 try...catch...finally

```javascript
try {
  // エラーが発生する可能性のあるコード
  const data = JSON.parse('{ invalid json }');
} catch (error) {
  // エラーが発生した場合に実行される
  console.error("JSONのパースに失敗:", error.message);
} finally {
  // 成功でも失敗でも必ず実行される（省略可能）
  console.log("処理完了");
}
```

```
try...catch の流れ:

  try ブロック
    │ エラーなし → catch をスキップ → finally → 次の処理
    │ エラー発生 → catch ブロック → finally → 次の処理
```

```javascript
// 実用例: ファイル読み込みのような処理
function parseConfig(jsonString) {
  try {
    const config = JSON.parse(jsonString);

    if (!config.apiUrl) {
      throw new Error("apiUrl は必須です");
    }

    return config;
  } catch (error) {
    if (error instanceof SyntaxError) {
      console.error("設定ファイルのJSON形式が不正です");
    } else {
      console.error("設定エラー:", error.message);
    }
    // デフォルト設定を返す
    return { apiUrl: "https://default.api.com", timeout: 5000 };
  }
}

console.log(parseConfig('{"apiUrl": "https://api.example.com"}'));
console.log(parseConfig("invalid")); // デフォルト設定が返る
```

## 11.3 throw — エラーを意図的に発生させる

```javascript
// throw文でエラーを投げる
function divide(a, b) {
  if (typeof a !== "number" || typeof b !== "number") {
    throw new TypeError("引数は数値でなければなりません");
  }
  if (b === 0) {
    throw new RangeError("ゼロで割ることはできません");
  }
  return a / b;
}

try {
  console.log(divide(10, 2));  // 5
  console.log(divide(10, 0));  // RangeError が投げられる
} catch (error) {
  console.error(`${error.name}: ${error.message}`);
  // "RangeError: ゼロで割ることはできません"
}

// エラーの型による分岐
function processInput(input) {
  try {
    const data = JSON.parse(input);
    if (!data.name) throw new Error("名前は必須です");
    return data;
  } catch (error) {
    if (error instanceof SyntaxError) {
      console.error("不正なJSON形式です");
    } else if (error instanceof Error) {
      console.error("バリデーションエラー:", error.message);
    }
    return null;
  }
}
```

## 11.4 カスタムエラー

```javascript
// カスタムエラークラスの作成
class ValidationError extends Error {
  constructor(field, message) {
    super(message);
    this.name = "ValidationError";
    this.field = field;
  }
}

class NotFoundError extends Error {
  constructor(resource, id) {
    super(`${resource}（ID: ${id}）が見つかりません`);
    this.name = "NotFoundError";
    this.resource = resource;
    this.id = id;
  }
}

class AuthenticationError extends Error {
  constructor(message = "認証に失敗しました") {
    super(message);
    this.name = "AuthenticationError";
  }
}

// 使用例
function findUser(id) {
  const users = [
    { id: 1, name: "太郎" },
    { id: 2, name: "花子" }
  ];

  const user = users.find(u => u.id === id);
  if (!user) {
    throw new NotFoundError("ユーザー", id);
  }
  return user;
}

function validateAge(age) {
  if (typeof age !== "number") {
    throw new ValidationError("age", "年齢は数値で入力してください");
  }
  if (age < 0 || age > 150) {
    throw new ValidationError("age", "年齢は0〜150の範囲で入力してください");
  }
  return true;
}

// エラーの型で処理を分岐
try {
  const user = findUser(99);
} catch (error) {
  if (error instanceof NotFoundError) {
    console.log(`404: ${error.message}`);
  } else if (error instanceof ValidationError) {
    console.log(`入力エラー（${error.field}）: ${error.message}`);
  } else {
    console.log("予期しないエラー:", error.message);
    throw error; // 未知のエラーは再スロー
  }
}
```

## 11.5 非同期処理のエラーハンドリング

### Promiseのエラー処理

```javascript
// .catch() でエラーを捕捉
fetch("https://api.example.com/data")
  .then(response => {
    if (!response.ok) {
      throw new Error(`HTTP Error: ${response.status}`);
    }
    return response.json();
  })
  .then(data => console.log(data))
  .catch(error => console.error("エラー:", error.message));
```

### async/await のエラー処理

```javascript
// try...catch で捕捉
async function fetchUserData(userId) {
  try {
    const response = await fetch(`https://api.example.com/users/${userId}`);

    if (!response.ok) {
      if (response.status === 404) {
        throw new NotFoundError("ユーザー", userId);
      }
      throw new Error(`HTTP Error: ${response.status}`);
    }

    return await response.json();
  } catch (error) {
    if (error instanceof NotFoundError) {
      console.warn(error.message);
      return null;
    }
    if (error.name === "TypeError") {
      console.error("ネットワークエラー: APIに接続できません");
      return null;
    }
    throw error; // 未知のエラーは再スロー
  }
}
```

### Promise.allSettled でエラーを個別処理

```javascript
async function fetchMultipleAPIs() {
  const urls = [
    "https://api.example.com/users",
    "https://invalid-url.example.com/data",
    "https://api.example.com/posts"
  ];

  const results = await Promise.allSettled(
    urls.map(url => fetch(url).then(r => r.json()))
  );

  const successes = [];
  const failures = [];

  results.forEach((result, i) => {
    if (result.status === "fulfilled") {
      successes.push({ url: urls[i], data: result.value });
    } else {
      failures.push({ url: urls[i], error: result.reason.message });
    }
  });

  console.log(`成功: ${successes.length}件, 失敗: ${failures.length}件`);
  return { successes, failures };
}
```

## 11.6 防御的プログラミング

エラーが起きる前にチェックする考え方です。

```javascript
// オプショナルチェーン（?. ）— nullやundefinedでもエラーにならない
const user = { name: "太郎", address: null };

// NG: TypeError が発生する
// console.log(user.address.city);

// OK: undefinedが返る
console.log(user.address?.city); // undefined
console.log(user.profile?.avatar?.url); // undefined

// Null合体演算子（?? ）— nullまたはundefinedの場合のデフォルト値
const port = undefined;
console.log(port ?? 3000);  // 3000
console.log(port || 3000);  // 3000

// ★ ?? と || の違い
console.log(0 ?? 3000);     // 0（0はnullでもundefinedでもない）
console.log(0 || 3000);     // 3000（0はfalsyなので）
console.log("" ?? "デフォルト"); // ""
console.log("" || "デフォルト"); // "デフォルト"
```

### 入力値のバリデーション

```javascript
function createUser(params) {
  // 早期リターン（ガード節）パターン
  if (!params) {
    throw new ValidationError("params", "パラメータが必要です");
  }

  const { name, email, age } = params;

  if (!name || typeof name !== "string") {
    throw new ValidationError("name", "名前は文字列で入力してください");
  }

  if (name.trim().length < 2) {
    throw new ValidationError("name", "名前は2文字以上で入力してください");
  }

  if (!email || !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) {
    throw new ValidationError("email", "正しいメールアドレスを入力してください");
  }

  if (age !== undefined && (typeof age !== "number" || age < 0)) {
    throw new ValidationError("age", "年齢は0以上の数値で入力してください");
  }

  return {
    name: name.trim(),
    email: email.toLowerCase().trim(),
    age: age ?? null,
    createdAt: new Date().toISOString()
  };
}
```

## 11.7 エラーログとデバッグ

```javascript
// consoleのレベル使い分け
console.log("一般的な情報");     // 通常のログ
console.info("参考情報");        // 情報
console.warn("警告");            // 警告（黄色）
console.error("エラー");         // エラー（赤色）
console.debug("デバッグ情報");   // デバッグ用

// console.trace — スタックトレースの出力
function funcA() { funcB(); }
function funcB() { funcC(); }
function funcC() { console.trace("ここまでの呼び出し"); }
funcA();
// Console:
//   ここまでの呼び出し
//     funcC
//     funcB
//     funcA

// console.time — 処理時間の計測
console.time("処理時間");
for (let i = 0; i < 1000000; i++) {} // 重い処理
console.timeEnd("処理時間"); // "処理時間: 5.123ms"

// console.assert — 条件がfalseの場合のみ出力
const age = 15;
console.assert(age >= 18, "未成年です:", age);
// Assertion failed: 未成年です: 15

// console.table — オブジェクトや配列を表形式で表示
const users = [
  { name: "太郎", age: 25 },
  { name: "花子", age: 22 }
];
console.table(users);
```

## 11.8 現場で使われるパターン

### エラーハンドリングの統一

```javascript
// Result型パターン（エラーを値として扱う）
function safeParseJSON(jsonString) {
  try {
    return { success: true, data: JSON.parse(jsonString) };
  } catch (error) {
    return { success: false, error: error.message };
  }
}

const result1 = safeParseJSON('{"name": "太郎"}');
if (result1.success) {
  console.log(result1.data); // { name: "太郎" }
}

const result2 = safeParseJSON("invalid");
if (!result2.success) {
  console.error(result2.error);
}
```

### グローバルエラーハンドラー

```javascript
// ブラウザ: 未処理のエラーをキャッチ
window.addEventListener("error", (event) => {
  console.error("グローバルエラー:", event.message);
  // エラーログサーバーに送信など
});

// 未処理のPromise rejectionをキャッチ
window.addEventListener("unhandledrejection", (event) => {
  console.error("未処理のPromise rejection:", event.reason);
  event.preventDefault(); // デフォルトのコンソールエラーを抑制
});

// Node.js
process.on("uncaughtException", (error) => {
  console.error("未処理の例外:", error);
  process.exit(1);
});

process.on("unhandledRejection", (reason) => {
  console.error("未処理のPromise rejection:", reason);
});
```

### エラーの再スロー

```javascript
async function processOrder(orderId) {
  try {
    const order = await fetchOrder(orderId);
    return await processPayment(order);
  } catch (error) {
    // ログに記録
    console.error(`注文処理エラー [${orderId}]:`, error.message);

    // 特定のエラーは処理して、それ以外は再スロー
    if (error instanceof NotFoundError) {
      return { status: "not_found", message: error.message };
    }

    // 未知のエラーは上位に伝搬させる
    throw error;
  }
}
```

## よくある間違い

### 1. catchで何もしない

```javascript
// NG: エラーを握りつぶす（サイレントキャッチ）
try {
  riskyOperation();
} catch (e) {
  // 何もしない ← バグの原因が見つからなくなる！
}

// OK: 少なくともログを出す
try {
  riskyOperation();
} catch (error) {
  console.error("操作に失敗:", error);
}
```

### 2. 必要以上に広い範囲をtryで囲む

```javascript
// NG: 全てをtryで囲む
try {
  const name = getName();
  const age = getAge();
  const result = calculate(name, age);
  display(result);
} catch (e) {
  console.error("エラー"); // どこでエラーが起きたかわからない
}

// OK: エラーが起きうる箇所だけ囲む
const name = getName();
const age = getAge();

try {
  const result = calculate(name, age);
} catch (e) {
  console.error("計算エラー:", e.message);
}

display(result);
```

### 3. async関数のエラーを捕捉し忘れる

```javascript
// NG: エラーが未処理になる
async function riskyAsync() {
  throw new Error("非同期エラー");
}
riskyAsync(); // UnhandledPromiseRejection!

// OK
riskyAsync().catch(error => console.error(error));

// または
try {
  await riskyAsync();
} catch (error) {
  console.error(error);
}
```

## ポイントまとめ

- エラーには `SyntaxError`, `TypeError`, `ReferenceError` など種類があり、それぞれ原因が異なります
- `try...catch...finally` でエラーを安全に処理します
- `throw new Error("メッセージ")` で意図的にエラーを発生させられます
- カスタムエラークラスを作ると、エラーの型による分岐処理が可能になります
- `?.`（オプショナルチェーン）と `??`（Null合体演算子）で防御的に書けます
- エラーは**握りつぶさない**（最低限ログを出す）ことが重要です
- 非同期処理のエラーは `try...catch` または `.catch()` で必ず捕捉しましょう
- 未知のエラーは再スロー（`throw error`）で上位に伝搬させるのが基本です
