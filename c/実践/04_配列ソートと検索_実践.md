> **難易度**: ★★☆☆☆（基礎）

> **学習目標**: 配列操作の基本（代入・参照・ソート・検索）を実践し、バブルソートと線形探索のアルゴリズムを理解する

```

## 課題の要件

1. ユーザーから5個の整数を入力で受け取り、配列に格納する
2. 入力されたデータをそのまま表示する
3. バブルソート（bubble sort）で昇順に並べ替えて表示する
4. 最大値・最小値・平均値を計算して表示する
5. ユーザーが指定した値を線形探索（linear search）で検索する

```

<details>
<summary>ステップ2：バブルソートを実装する</summary>

隣り合う要素を比較して交換を繰り返すアルゴリズムです。

```
バブルソートのイメージ（昇順）:
  [38, 15, 72, 4, 51]
   ↓ 38>15 → 交換
  [15, 38, 72, 4, 51]
   ↓ 72>4 → 交換
  [15, 38, 4, 72, 51]
   ... これを繰り返す
```

```c
int temp;
for (i = 0; i < SIZE - 1; i++) {
    for (int j = 0; j < SIZE - 1 - i; j++) {
        if (data[j] > data[j + 1]) {
            temp = data[j];
            data[j] = data[j + 1];
            data[j + 1] = temp;
        }
    }
}
```

## 解答例

<details>
<summary>解答例（初心者向け）</summary>

```c
/*
 * 配列ソートと検索
 * 学べる内容：配列、バブルソート、線形探索、平均値計算
 * コンパイル：gcc -o sort_search sort_search.c
 * 実行：./sort_search
 */
#include <stdio.h>

#define SIZE 5

int main(void) {
    int data[SIZE];
    int i, j, temp;
    int sum = 0;
    int target;
    int found = 0;

    /* データ入力 */
    printf("===== 配列ソート＆検索ツール =====\n\n");
    printf("%d個の整数を入力してください:\n", SIZE);
    for (i = 0; i < SIZE; i++) {
        printf("%d番目: ", i + 1);
        scanf("%d", &data[i]);
    }

    /* 入力データ表示 */
    printf("\n--- 入力データ ---\n");
    for (i = 0; i < SIZE; i++) {
        printf("%d ", data[i]);
    }
    printf("\n");

    /* バブルソート（昇順） */
    for (i = 0; i < SIZE - 1; i++) {
        for (j = 0; j < SIZE - 1 - i; j++) {
            if (data[j] > data[j + 1]) {
                temp = data[j];
                data[j] = data[j + 1];
                data[j + 1] = temp;
            }
        }
    }

    /* ソート結果表示 */
    printf("\n--- ソート結果（昇順）---\n");
    for (i = 0; i < SIZE; i++) {
        printf("%d ", data[i]);
    }
    printf("\n");

    /* 統計情報 */
    for (i = 0; i < SIZE; i++) {
        sum += data[i];
    }
    printf("\n最大値: %d\n", data[SIZE - 1]);
    printf("最小値: %d\n", data[0]);
    printf("平均値: %.1f\n", (double)sum / SIZE);

    /* 線形探索 */
    printf("\n検索する値を入力してください: ");
    scanf("%d", &target);

    for (i = 0; i < SIZE; i++) {
        if (data[i] == target) {
            printf("%d は %d番目に見つかりました。\n", target, i + 1);
            found = 1;
            break;
        }
    }
    if (!found) {
        printf("%d は見つかりませんでした。\n", target);
    }

    return 0;
}
```

<details>
<summary>解答例（改良版 ─ 関数分割＆二分探索）</summary>

関数に処理を分割し、ソート済み配列に対してより効率的な二分探索（binary search）を使うバージョンです。
第5章（関数）を学んだ後に挑戦してみてください。

```c
/*
 * 配列ソートと検索（改良版）
 * 学べる内容：関数分割、二分探索、配列のポインタ渡し
 * コンパイル：gcc -o sort_search2 sort_search2.c
 * 実行：./sort_search2
 */
#include <stdio.h>

#define SIZE 5

/* 配列を表示する */
void print_array(int arr[], int size) {
    int i;
    for (i = 0; i < size; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
}

/* バブルソート（昇順） */
void bubble_sort(int arr[], int size) {
    int i, j, temp;
    for (i = 0; i < size - 1; i++) {
        for (j = 0; j < size - 1 - i; j++) {
            if (arr[j] > arr[j + 1]) {
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

/* 二分探索（ソート済み配列が前提） */
int binary_search(int arr[], int size, int target) {
    int low = 0;
    int high = size - 1;

    while (low <= high) {
        int mid = (low + high) / 2;
        if (arr[mid] == target) {
            return mid;  /* 見つかったインデックスを返す */
        } else if (arr[mid] < target) {
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }
    return -1;  /* 見つからなかった */
}

int main(void) {
    int data[SIZE];
    int i, sum = 0;
    int target, result;

    printf("===== 配列ソート＆検索ツール（改良版）=====\n\n");
    printf("%d個の整数を入力してください:\n", SIZE);
    for (i = 0; i < SIZE; i++) {
        printf("%d番目: ", i + 1);
        scanf("%d", &data[i]);
    }

    printf("\n--- 入力データ ---\n");
    print_array(data, SIZE);

    bubble_sort(data, SIZE);

    printf("\n--- ソート結果（昇順）---\n");
    print_array(data, SIZE);

    for (i = 0; i < SIZE; i++) {
        sum += data[i];
    }
    printf("\n最大値: %d\n", data[SIZE - 1]);
    printf("最小値: %d\n", data[0]);
    printf("平均値: %.1f\n", (double)sum / SIZE);

    printf("\n検索する値を入力してください: ");
    scanf("%d", &target);

    result = binary_search(data, SIZE, target);
    if (result >= 0) {
        printf("%d は %d番目に見つかりました。\n", target, result + 1);
    } else {
        printf("%d は見つかりませんでした。\n", target);
    }

    return 0;
}
```

**初心者向けとの違い:**
- 処理を関数に分割 → 各関数が独立したテスト可能な単位になっている
- 二分探索を導入 → データ量が多いときに線形探索よりも高速（O(log n) vs O(n)）
- 配列を関数に渡す書き方 (`int arr[]`) を習得できる

</details>
