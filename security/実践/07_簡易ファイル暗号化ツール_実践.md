## 前提知識：XOR演算とは

XOR（排他的論理和）は、2つのビットが異なるとき1、同じとき0を返す演算です。

```
XOR の真理値表:
  0 XOR 0 = 0
  0 XOR 1 = 1
  1 XOR 0 = 1
  1 XOR 1 = 0

暗号化への応用:
  平文   XOR 鍵 = 暗号文
  暗号文 XOR 鍵 = 平文     ← 同じ鍵で元に戻る！
```

## ステップガイド

<details>
<summary>ステップ1：1バイトのXOR演算を理解する</summary>

Python では `^` 演算子で XOR を計算できます。

```python
plaintext_byte = ord('A')      # 65 = 01000001
key_byte = ord('K')            # 75 = 01001011
encrypted = plaintext_byte ^ key_byte  # 10 = 00001010
decrypted = encrypted ^ key_byte       # 65 = 01000001 → 'A' に戻る

print(f"平文: {chr(plaintext_byte)} ({plaintext_byte})")
print(f"鍵:   {chr(key_byte)} ({key_byte})")
print(f"暗号: {encrypted}")
print(f"復号: {chr(decrypted)} ({decrypted})")
```

</details>

<details>
<summary>ステップ2：文字列全体をXOR暗号化する</summary>

鍵が短い場合は、繰り返し使います。

```python
def xor_encrypt(data_bytes, key_bytes):
    result = bytearray()
    for i in range(len(data_bytes)):
        key_byte = key_bytes[i % len(key_bytes)]  # 鍵を繰り返す
        result.append(data_bytes[i] ^ key_byte)
    return bytes(result)
```

</details>

<details>
<summary>ステップ3：ファイルの読み書き（バイナリモード）</summary>

暗号化はバイト単位で行うため、ファイルはバイナリモード（`"rb"` / `"wb"`）で開きます。

```python
# 読み込み
with open("input.txt", "rb") as f:
    data = f.read()

# 暗号化
encrypted = xor_encrypt(data, key_bytes)

# 書き込み
with open("input.txt.enc", "wb") as f:
    f.write(encrypted)
```

```python
# 簡易ファイル暗号化ツール
# 学べる内容：XOR暗号、バイナリファイル操作、共通鍵暗号の基本
# 実行方法：python file_encryptor.py

import os


def xor_crypt(data_bytes, key_bytes):
    """XOR演算でデータを暗号化/復号します。
    XOR暗号は、同じ鍵で暗号化と復号の両方を行えます。
    """
    result = bytearray()
    key_len = len(key_bytes)
    for i in range(len(data_bytes)):
        # 鍵を繰り返し使用
        key_byte = key_bytes[i % key_len]
        result.append(data_bytes[i] ^ key_byte)
    return bytes(result)


def encrypt_text():
    """テキストの暗号化・復号デモを行います。"""
    plaintext = input("暗号化する文字列: ")
    key = input("暗号鍵を入力: ")

    if not key:
        print("エラー: 鍵が空です。")
        return

    plain_bytes = plaintext.encode("utf-8")
    key_bytes = key.encode("utf-8")

    # 暗号化
    encrypted = xor_crypt(plain_bytes, key_bytes)
    hex_str = encrypted.hex()
    print(f"\n暗号化結果（16進数表記）: {hex_str}")

    # 復号（同じ鍵でもう一度XOR）
    decrypted = xor_crypt(encrypted, key_bytes)
    decrypted_text = decrypted.decode("utf-8")
    print(f"復号結果: {decrypted_text}")

    if plaintext == decrypted_text:
        print("→ 同じ鍵で暗号化と復号ができることを確認しました。")
    else:
        print("→ エラー: 復号結果が元の文字列と一致しません。")


def encrypt_file():
    """ファイルを暗号化します。"""
    filepath = input("暗号化するファイルのパス: ")
    key = input("暗号鍵を入力: ")

    if not key:
        print("エラー: 鍵が空です。")
        return

    if not os.path.isfile(filepath):
        print(f"エラー: ファイル '{filepath}' が見つかりません。")
        return

    with open(filepath, "rb") as f:
        data = f.read()

    key_bytes = key.encode("utf-8")
    encrypted = xor_crypt(data, key_bytes)

    output_path = filepath + ".enc"
    with open(output_path, "wb") as f:
        f.write(encrypted)

    print(f"\n暗号化完了: {output_path}")
    print(f"元のサイズ: {len(data)} bytes")
    print(f"暗号化後:   {len(encrypted)} bytes")


def decrypt_file():
    """暗号化されたファイルを復号します。"""
    filepath = input("復号するファイルのパス（.encファイル）: ")
    key = input("暗号鍵を入力: ")

    if not key:
        print("エラー: 鍵が空です。")
        return

    if not os.path.isfile(filepath):
        print(f"エラー: ファイル '{filepath}' が見つかりません。")
        return

    with open(filepath, "rb") as f:
        data = f.read()

    key_bytes = key.encode("utf-8")
    decrypted = xor_crypt(data, key_bytes)

    # .enc を取り除いた名前で出力
    if filepath.endswith(".enc"):
        output_path = filepath[:-4] + ".dec"
    else:
        output_path = filepath + ".dec"

    with open(output_path, "wb") as f:
        f.write(decrypted)

    print(f"\n復号完了: {output_path}")
    print(f"サイズ: {len(decrypted)} bytes")


def comparison_demo():
    """暗号化方式の比較デモを行います。"""
    print("\n--- 暗号化方式の比較デモ ---")

    plaintext = "AAAAAAAAAA"  # 同じ文字の繰り返し
    key_short = "K"           # 1文字の鍵
    key_long = "MyStr0ngK3y!" # 長い鍵

    plain_bytes = plaintext.encode("utf-8")

    # 短い鍵での暗号化
    enc_short = xor_crypt(plain_bytes, key_short.encode("utf-8"))
    print(f"\n平文:         {plaintext}")
    print(f"鍵（1文字）:  '{key_short}'")
    print(f"暗号文(hex):  {enc_short.hex()}")
    print("→ 同じバイトの繰り返しになり、パターンが見える（弱い）")

    # 長い鍵での暗号化
    enc_long = xor_crypt(plain_bytes, key_long.encode("utf-8"))
    print(f"\n鍵（12文字）: '{key_long}'")
    print(f"暗号文(hex):  {enc_long.hex()}")
    print("→ パターンが見えにくくなる（鍵が長いほど安全）")

    print("\n--- 重要な注意事項 ---")
    print("XOR暗号は教育目的の簡易暗号です。実務では使用しないでください。")
    print("実務ではAES（Advanced Encryption Standard）などの")
    print("標準的な暗号アルゴリズムを使用してください。")


# --- メイン処理 ---
print("===== 簡易ファイル暗号化ツール =====")

while True:
    print("\n1. テキストの暗号化・復号（XOR方式）")
    print("2. ファイルの暗号化")
    print("3. ファイルの復号")
    print("4. 暗号化方式の比較デモ")
    print("5. 終了")

    choice = input("\n選択 (1-5): ")

    if choice == "1":
        encrypt_text()
    elif choice == "2":
        encrypt_file()
    elif choice == "3":
        decrypt_file()
    elif choice == "4":
        comparison_demo()
    elif choice == "5":
        print("終了します。")
        break
    else:
        print("1〜5の番号を入力してください。")
```

```python
# 簡易ファイル暗号化ツール（改良版）
# 学べる内容：鍵導出関数（KDF）、暗号化ヘッダ、安全な鍵管理
# 実行方法：python file_encryptor_v2.py

import hashlib
import os
import struct

# マジックバイト（暗号化ファイルの識別用）
MAGIC_BYTES = b"ENC1"
SALT_LENGTH = 16
KDF_ITERATIONS = 100_000


def derive_key(password, salt, key_length=32):
    """パスワードから暗号鍵を導出します（PBKDF2を使用）。
    パスワードをそのまま鍵として使うのではなく、
    鍵導出関数（KDF: Key Derivation Function）を経由することで
    安全性を高めます。
    """
    return hashlib.pbkdf2_hmac(
        "sha256",
        password.encode("utf-8"),
        salt,
        KDF_ITERATIONS,
        dklen=key_length,
    )


def xor_crypt(data_bytes, key_bytes):
    """XOR演算でデータを暗号化/復号します。"""
    result = bytearray(len(data_bytes))
    key_len = len(key_bytes)
    for i in range(len(data_bytes)):
        result[i] = data_bytes[i] ^ key_bytes[i % key_len]
    return bytes(result)


def encrypt_text():
    """テキストの暗号化・復号デモを行います。"""
    plaintext = input("暗号化する文字列: ")
    password = input("パスワードを入力: ")

    if not password:
        print("エラー: パスワードが空です。")
        return

    salt = os.urandom(SALT_LENGTH)
    key = derive_key(password, salt)

    plain_bytes = plaintext.encode("utf-8")
    encrypted = xor_crypt(plain_bytes, key)

    print(f"\nソルト（hex）:   {salt.hex()}")
    print(f"導出鍵（hex）:   {key.hex()[:32]}...")
    print(f"暗号文（hex）:   {encrypted.hex()}")

    # 復号
    key_again = derive_key(password, salt)
    decrypted = xor_crypt(encrypted, key_again)
    print(f"復号結果:        {decrypted.decode('utf-8')}")


def encrypt_file():
    """ファイルを暗号化します。ヘッダにソルトを含めます。"""
    filepath = input("暗号化するファイルのパス: ")
    password = input("パスワードを入力: ")

    if not password:
        print("エラー: パスワードが空です。")
        return
    if not os.path.isfile(filepath):
        print(f"エラー: ファイル '{filepath}' が見つかりません。")
        return

    with open(filepath, "rb") as f:
        data = f.read()

    salt = os.urandom(SALT_LENGTH)
    key = derive_key(password, salt)
    encrypted = xor_crypt(data, key)

    # ヘッダ構造: MAGIC(4) + SALT_LEN(4) + SALT(16) + DATA_LEN(4) + DATA
    output_path = filepath + ".enc"
    with open(output_path, "wb") as f:
        f.write(MAGIC_BYTES)
        f.write(struct.pack(">I", SALT_LENGTH))
        f.write(salt)
        f.write(struct.pack(">I", len(data)))
        f.write(encrypted)

    print(f"\n暗号化完了: {output_path}")
    print(f"元のサイズ:   {len(data)} bytes")
    print(f"暗号化後:     {os.path.getsize(output_path)} bytes（ヘッダ含む）")


def decrypt_file():
    """暗号化されたファイルを復号します。"""
    filepath = input("復号するファイルのパス（.encファイル）: ")
    password = input("パスワードを入力: ")

    if not password:
        print("エラー: パスワードが空です。")
        return
    if not os.path.isfile(filepath):
        print(f"エラー: ファイル '{filepath}' が見つかりません。")
        return

    with open(filepath, "rb") as f:
        # ヘッダ読み取り
        magic = f.read(4)
        if magic != MAGIC_BYTES:
            print("エラー: このファイルは本ツールで暗号化されたものではありません。")
            return

        salt_len = struct.unpack(">I", f.read(4))[0]
        salt = f.read(salt_len)
        data_len = struct.unpack(">I", f.read(4))[0]
        encrypted = f.read()

    key = derive_key(password, salt)
    decrypted = xor_crypt(encrypted, key)

    if filepath.endswith(".enc"):
        output_path = filepath[:-4] + ".dec"
    else:
        output_path = filepath + ".dec"

    with open(output_path, "wb") as f:
        f.write(decrypted[:data_len])

    print(f"\n復号完了: {output_path}")
    print(f"サイズ: {data_len} bytes")


def comparison_demo():
    """鍵導出の有無による安全性の違いを示します。"""
    print("\n--- 鍵導出関数（KDF）の効果デモ ---")

    plaintext = "AAAAAAAAAA"
    password = "password"
    plain_bytes = plaintext.encode("utf-8")

    # 方式1: パスワードをそのまま鍵として使用
    key_raw = password.encode("utf-8")
    enc_raw = xor_crypt(plain_bytes, key_raw)
    print(f"\n平文:             {plaintext}")
    print(f"パスワード:       '{password}'")
    print(f"\n【方式1】パスワードをそのまま鍵に使用:")
    print(f"  鍵(hex):        {key_raw.hex()}")
    print(f"  暗号文(hex):    {enc_raw.hex()}")
    print("  問題: 鍵が短く、パターンが見える")

    # 方式2: KDFで導出した鍵を使用
    salt = os.urandom(16)
    key_derived = derive_key(password, salt, key_length=len(plain_bytes))
    enc_derived = xor_crypt(plain_bytes, key_derived)
    print(f"\n【方式2】PBKDF2で鍵を導出:")
    print(f"  ソルト(hex):    {salt.hex()}")
    print(f"  導出鍵(hex):    {key_derived.hex()}")
    print(f"  暗号文(hex):    {enc_derived.hex()}")
    print("  改善: 鍵が十分長く、ランダムに見える")

    print("\n--- 注意事項 ---")
    print("XOR暗号は教育目的です。実務では AES-256-GCM 等を使用してください。")
    print("Python では `cryptography` ライブラリが推奨されます。")


def main():
    print("===== 簡易ファイル暗号化ツール =====")

    while True:
        print("\n1. テキストの暗号化・復号")
        print("2. ファイルの暗号化")
        print("3. ファイルの復号")
        print("4. 暗号化方式の比較デモ")
        print("5. 終了")

        choice = input("\n選択 (1-5): ")

        if choice == "1":
            encrypt_text()
        elif choice == "2":
            encrypt_file()
        elif choice == "3":
            decrypt_file()
        elif choice == "4":
            comparison_demo()
        elif choice == "5":
            print("終了します。")
            break
        else:
            print("1〜5の番号を入力してください。")


if __name__ == "__main__":
    main()
```

- PBKDF2による鍵導出関数（KDF）を使用 → パスワードから安全な暗号鍵を生成
- 暗号化ファイルにヘッダ（マジックバイト、ソルト、データ長）を付加 → ファイル形式の識別と復号に必要な情報を埋め込み
- `struct` モジュールでバイナリデータのパック/アンパック → ファイルフォーマット設計の基本
- ソルトをファイルに保存 → 毎回異なる暗号文が生成される
