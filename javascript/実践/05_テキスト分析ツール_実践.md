## 課題の要件

1. HTMLファイル1つで完結すること
2. テキストエリアにユーザーが文章を入力できる
3. 「分析する」ボタンで以下を計算して表示する：
   - 文字数（空白込み/空白なし）
   - 行数
   - 文の数（句点「。」やピリオド「.」で区切り）
4. 文字種別（ひらがな・カタカナ・漢字・英字・数字・その他）の割合を計算する
5. 各分析処理をそれぞれ関数に分割する

## ステップガイド

<details>
<summary>ステップ1：基本統計の計算</summary>

文字列のプロパティとメソッドを使って基本的な統計を出します。

```javascript
function analyzeBasic(text) {
  const charCount = text.length;
  const charCountNoSpace = text.replace(/ /g, "").replace(/　/g, "").length;
  const lines = text.split("\n");
  const lineCount = lines.length;
  // 文の数: 句点やピリオドで区切る
  const sentences = text.split(/[。.!?！？]/).filter(function(s) {
    return s.trim() !== "";
  });
  return { charCount, charCountNoSpace, lineCount, sentenceCount: sentences.length };
}
```

</details>

<details>
<summary>ステップ2：文字種別の判定</summary>

各文字がどの種別に属するかを正規表現（RegExp）で判定します。

```javascript
function classifyChar(ch) {
  if (/[ぁ-ん]/.test(ch)) return "hiragana";
  if (/[ァ-ヶー]/.test(ch)) return "katakana";
  if (/[一-龥]/.test(ch)) return "kanji";
  if (/[a-zA-Zａ-ｚＡ-Ｚ]/.test(ch)) return "alpha";
  if (/[0-9０-９]/.test(ch)) return "digit";
  return "other";
}
```

    // メインの分析関数
    function analyze() {
      const text = document.getElementById("inputText").value;

      if (text.trim() === "") {
        document.getElementById("results").textContent = "テキストを入力してください。";
        return;
      }

      // 各分析を実行
      const basic = analyzeBasic(text);
      const charTypes = analyzeCharTypes(text);

      // 結果を組み立てて表示
      let output = "";

      // 基本統計
      output += "=== 基本統計 ===\n";
      output += "文字数:    " + basic.charCountNoSpace + "文字";
      output += "（空白含む: " + basic.charCount + "文字）\n";
      output += "行数:      " + basic.lineCount + "行\n";
      output += "文の数:    " + basic.sentenceCount + "文\n";
      output += "\n";

      // 文字種別
      output += "=== 文字種別 ===\n";
      const labels = {
        hiragana: "ひらがな",
        katakana: "カタカナ",
        kanji:    "漢字",
        alpha:    "英字",
        digit:    "数字",
        other:    "記号他"
      };
      const totalChars = basic.charCountNoSpace;
      const types = ["hiragana", "katakana", "kanji", "alpha", "digit", "other"];

      for (let i = 0; i < types.length; i++) {
        const type = types[i];
        const count = charTypes[type];
        let percent = 0;
        if (totalChars > 0) {
          percent = (count / totalChars * 100).toFixed(1);
        }
        output += labels[type] + ": " + padLeft(count + "文字", 8) + " (" + percent + "%)\n";
      }

      document.getElementById("results").textContent = output;
    }

    // 基本統計を計算する関数
    function analyzeBasic(text) {
      // 文字数
      const charCount = text.length;
      // 空白を除いた文字数
      const noSpace = text.replace(/ /g, "").replace(/　/g, "").replace(/\n/g, "").replace(/\t/g, "");
      const charCountNoSpace = noSpace.length;
      // 行数
      const lines = text.split("\n");
      const lineCount = lines.length;
      // 文の数（句点・ピリオド・感嘆符・疑問符で区切る）
      const sentences = text.split(/[。.!?！？]/);
      let sentenceCount = 0;
      for (let i = 0; i < sentences.length; i++) {
        if (sentences[i].trim() !== "") {
          sentenceCount = sentenceCount + 1;
        }
      }

      return {
        charCount: charCount,
        charCountNoSpace: charCountNoSpace,
        lineCount: lineCount,
        sentenceCount: sentenceCount
      };
    }

    // 文字種別を集計する関数
    function analyzeCharTypes(text) {
      const counts = {
        hiragana: 0,
        katakana: 0,
        kanji: 0,
        alpha: 0,
        digit: 0,
        other: 0
      };

      for (let i = 0; i < text.length; i++) {
        const ch = text[i];

        // 空白・改行はスキップ
        if (ch === " " || ch === "　" || ch === "\n" || ch === "\t") {
          continue;
        }

        // 文字種別を判定
        if (/[ぁ-ん]/.test(ch)) {
          counts.hiragana = counts.hiragana + 1;
        } else if (/[ァ-ヶー]/.test(ch)) {
          counts.katakana = counts.katakana + 1;
        } else if (/[一-龥々]/.test(ch)) {
          counts.kanji = counts.kanji + 1;
        } else if (/[a-zA-Zａ-ｚＡ-Ｚ]/.test(ch)) {
          counts.alpha = counts.alpha + 1;
        } else if (/[0-9０-９]/.test(ch)) {
          counts.digit = counts.digit + 1;
        } else {
          counts.other = counts.other + 1;
        }
      }

      return counts;
    }

    // 文字列を右寄せする補助関数
    function padLeft(str, length) {
      while (str.length < length) {
        str = " " + str;
      }
      return str;
    }
  </script>

<summary>解答例（改良版 ─ 第12章の知識を先取り）</summary>

ES6+の機能（`Map`, アロー関数, テンプレートリテラル, `padStart`）と単語頻度分析を追加した改良版です。

```html
<!DOCTYPE html>
<html lang="ja">
<head>
  <meta charset="UTF-8">
  <title>テキスト分析ツール（改良版）</title>
  <style>
    body { font-family: sans-serif; max-width: 700px; margin: 30px auto; padding: 0 20px; }
    h1 { text-align: center; }
    textarea { width: 100%; height: 150px; font-size: 16px; padding: 10px; box-sizing: border-box; }
    button { font-size: 16px; padding: 10px 24px; margin: 10px 0; cursor: pointer;
             background: #1565c0; color: white; border: none; border-radius: 4px; }
    .section { background: #f5f5f5; padding: 16px; margin: 10px 0; border-radius: 8px; }
    .section h3 { margin-top: 0; }
    pre { margin: 0; font-family: monospace; font-size: 14px; line-height: 1.8; white-space: pre-wrap; }
  </style>
</head>
<body>
  <h1>テキスト分析ツール</h1>

  <textarea id="inputText">JavaScriptは楽しい言語です。JavaScriptを使えばウェブサイトに動きを付けることができます。プログラミングの世界へようこそ！初心者でも楽しく学べます。</textarea>
  <br>
  <button onclick="analyze()">分析する</button>
  <div id="results"></div>

  <script>
    // 文字種別の判定マップ
    const charClassifiers = [
      { name: "ひらがな", test: ch => /[ぁ-ん]/.test(ch) },
      { name: "カタカナ", test: ch => /[ァ-ヶー]/.test(ch) },
      { name: "漢字",     test: ch => /[一-龥々]/.test(ch) },
      { name: "英字",     test: ch => /[a-zA-Zａ-ｚＡ-Ｚ]/.test(ch) },
      { name: "数字",     test: ch => /[0-9０-９]/.test(ch) },
    ];

    function analyze() {
      const text = document.getElementById("inputText").value;
      if (text.trim() === "") {
        document.getElementById("results").innerHTML =
          '<div class="section">テキストを入力してください。</div>';
        return;
      }

      const results = document.getElementById("results");
      results.innerHTML = "";

      // 基本統計
      results.innerHTML += buildBasicSection(text);
      // 文字種別
      results.innerHTML += buildCharTypeSection(text);
      // 単語頻度
      results.innerHTML += buildWordFreqSection(text);
    }

    function buildBasicSection(text) {
      const charCount = text.length;
      const noSpace = text.replace(/[\s　]/g, "");
      const lines = text.split("\n").length;
      const sentences = text.split(/[。.!?！？]/).filter(s => s.trim() !== "").length;

      return `<div class="section">
        <h3>基本統計</h3>
        <pre>文字数: ${noSpace.length}文字（空白含む: ${charCount}文字）
行数:   ${lines}行
文数:   ${sentences}文</pre>
      </div>`;
    }

    function buildCharTypeSection(text) {
      const noSpace = text.replace(/[\s　]/g, "");
      const total = noSpace.length;

      // 各種別のカウント
      const counts = new Map();
      charClassifiers.forEach(c => counts.set(c.name, 0));
      counts.set("記号他", 0);

      for (const ch of noSpace) {
        let matched = false;
        for (const classifier of charClassifiers) {
          if (classifier.test(ch)) {
            counts.set(classifier.name, counts.get(classifier.name) + 1);
            matched = true;
            break;
          }
        }
        if (!matched) {
          counts.set("記号他", counts.get("記号他") + 1);
        }
      }

      let lines = "";
      counts.forEach((count, name) => {
        const pct = total > 0 ? (count / total * 100).toFixed(1) : "0.0";
        const bar = "█".repeat(Math.round(count / total * 30));
        lines += `${name.padEnd(6, "　")}: ${String(count).padStart(4)}文字 ${bar} (${pct}%)\n`;
      });

      return `<div class="section">
        <h3>文字種別</h3>
        <pre>${lines}</pre>
      </div>`;
    }

    function buildWordFreqSection(text) {
      // 英単語を抽出して頻度を数える
      const words = text.toLowerCase().match(/[a-zA-Z]+/g);
      if (!words || words.length === 0) {
        return `<div class="section"><h3>英単語頻度</h3><pre>英単語が見つかりませんでした。</pre></div>`;
      }

      // 頻度を集計
      const freq = {};
      words.forEach(word => {
        freq[word] = (freq[word] || 0) + 1;
      });

      // 頻度順にソート（上位10件）
      const sorted = Object.entries(freq)
        .sort((a, b) => b[1] - a[1])
        .slice(0, 10);

      let lines = "";
      sorted.forEach(([word, count]) => {
        const bar = "█".repeat(count * 3);
        lines += `${word.padEnd(15)}: ${count}回 ${bar}\n`;
      });

      return `<div class="section">
        <h3>英単語頻度 TOP${sorted.length}</h3>
        <pre>${lines}</pre>
      </div>`;
    }
  </script>
</body>
</html>
```

**初心者向けとの違い:**
- `Map` オブジェクトでカウントを管理（挿入順序が保証される）
- `for...of` ループで文字列を1文字ずつ処理
- アロー関数で簡潔なコールバック
- `padStart` / `padEnd` で文字列の桁揃え（自作の `padLeft` が不要に）
- `Object.entries` + `sort` で頻度順のランキングを実現
- 棒グラフによるビジュアル表示を追加
- 英単語の頻度分析機能を追加
