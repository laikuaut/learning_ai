# 実践課題10：リファクタリング ─ 天気予報アプリ ★4

> **難易度**: ★★★★☆（上級）
> **前提知識**: 第2章（コンポーネントとprops）、第3章（state）、第5章（useEffect）、第6章（カスタムフック）、第7章（useContext）
> **課題の種類**: リファクタリング
> **学習目標**: 巨大な単一コンポーネントを分割し、カスタムフックとContextで整理する

---

## 課題の説明

以下の「天気予報アプリ」は**正しく動作しますが**、すべてのロジックが1つのコンポーネントに詰まっています。段階的にリファクタリングしてください。

### 問題点

1. `App` コンポーネントが200行以上で肥大化している
2. 状態管理ロジックとUIが混在している
3. 同じスタイル定義が繰り返されている
4. データ取得ロジックが再利用できない
5. 単位設定（℃/℉）がpropsバケツリレーになる構造

### ゴール

1. **ステップ1**: UIコンポーネントを分割する
2. **ステップ2**: データ取得をカスタムフックに抽出する
3. **ステップ3**: 単位設定をContextで管理する

---

## リファクタリング対象のコード

```html
<!DOCTYPE html>
<html lang="ja">
<head>
  <meta charset="UTF-8">
  <title>天気予報アプリ（リファクタリング前）</title>
  <script src="https://unpkg.com/react@18/umd/react.development.js"></script>
  <script src="https://unpkg.com/react-dom@18/umd/react-dom.development.js"></script>
  <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
  <style>
    body { font-family: sans-serif; background: #e3f2fd; padding: 20px; }
    .app { max-width: 500px; margin: 0 auto; }
    .card { background: white; border-radius: 12px; padding: 20px; margin-bottom: 16px; box-shadow: 0 2px 8px rgba(0,0,0,0.1); }
  </style>
</head>
<body>
  <div id="root"></div>
  <script type="text/babel">
    // すべてが1つのコンポーネントに入っている
    function App() {
      const [city, setCity] = React.useState("東京");
      const [weather, setWeather] = React.useState(null);
      const [forecast, setForecast] = React.useState([]);
      const [loading, setLoading] = React.useState(false);
      const [error, setError] = React.useState(null);
      const [unit, setUnit] = React.useState("celsius");
      const [favorites, setFavorites] = React.useState(["東京", "大阪", "札幌"]);
      const [inputValue, setInputValue] = React.useState("");

      // ダミーの天気データ生成（API呼び出しの代わり）
      function fetchWeatherData(cityName) {
        return new Promise(function(resolve) {
          setTimeout(function() {
            var temps = { "東京": 22, "大阪": 24, "札幌": 15, "福岡": 25, "名古屋": 23 };
            var conditions = { "東京": "晴れ", "大阪": "曇り", "札幌": "雨", "福岡": "晴れ", "名古屋": "曇り" };
            var humidity = { "東京": 55, "大阪": 65, "札幌": 70, "福岡": 50, "名古屋": 60 };

            if (temps[cityName] !== undefined) {
              resolve({
                current: { temp: temps[cityName], condition: conditions[cityName], humidity: humidity[cityName], wind: Math.floor(Math.random() * 10) + 1 },
                forecast: [
                  { day: "明日", temp: temps[cityName] + Math.floor(Math.random() * 5) - 2, condition: "晴れ" },
                  { day: "明後日", temp: temps[cityName] + Math.floor(Math.random() * 5) - 2, condition: "曇り" },
                  { day: "3日後", temp: temps[cityName] + Math.floor(Math.random() * 5) - 2, condition: "雨" },
                ],
              });
            } else {
              resolve(null);
            }
          }, 800);
        });
      }

      // 天気データ取得
      React.useEffect(function() {
        setLoading(true);
        setError(null);
        fetchWeatherData(city).then(function(data) {
          if (data) {
            setWeather(data.current);
            setForecast(data.forecast);
          } else {
            setWeather(null);
            setForecast([]);
            setError(city + " の天気データが見つかりません");
          }
          setLoading(false);
        });
      }, [city]);

      // 温度変換
      function convertTemp(celsius) {
        if (unit === "fahrenheit") {
          return Math.round(celsius * 9 / 5 + 32);
        }
        return celsius;
      }

      function getTempUnit() {
        return unit === "celsius" ? "℃" : "℉";
      }

      // 天気アイコン
      function getWeatherIcon(condition) {
        if (condition === "晴れ") return "☀️";
        if (condition === "曇り") return "☁️";
        if (condition === "雨") return "🌧️";
        return "🌤️";
      }

      // お気に入り追加
      function handleAddFavorite() {
        if (inputValue && !favorites.includes(inputValue)) {
          setFavorites([...favorites, inputValue]);
          setInputValue("");
        }
      }

      // お気に入り削除
      function handleRemoveFavorite(name) {
        setFavorites(favorites.filter(function(f) { return f !== name; }));
      }

      return (
        <div className="app">
          <h1 style={{ textAlign: "center" }}>天気予報</h1>

          {/* 単位切替 */}
          <div style={{ textAlign: "right", marginBottom: 12 }}>
            <button
              onClick={function() { setUnit("celsius"); }}
              style={{ padding: "4px 12px", marginRight: 4, background: unit === "celsius" ? "#2196f3" : "#e0e0e0", color: unit === "celsius" ? "white" : "#333", border: "none", borderRadius: 4, cursor: "pointer" }}
            >℃</button>
            <button
              onClick={function() { setUnit("fahrenheit"); }}
              style={{ padding: "4px 12px", background: unit === "fahrenheit" ? "#2196f3" : "#e0e0e0", color: unit === "fahrenheit" ? "white" : "#333", border: "none", borderRadius: 4, cursor: "pointer" }}
            >℉</button>
          </div>

          {/* お気に入り都市 */}
          <div className="card">
            <h3>お気に入り都市</h3>
            <div style={{ display: "flex", flexWrap: "wrap", gap: 8, marginBottom: 12 }}>
              {favorites.map(function(fav) {
                return (
                  <span key={fav} style={{ display: "inline-flex", alignItems: "center", padding: "4px 12px", background: city === fav ? "#2196f3" : "#e8e8e8", color: city === fav ? "white" : "#333", borderRadius: 16, cursor: "pointer", fontSize: 14 }}>
                    <span onClick={function() { setCity(fav); }}>{fav}</span>
                    <span onClick={function() { handleRemoveFavorite(fav); }} style={{ marginLeft: 6, cursor: "pointer" }}>×</span>
                  </span>
                );
              })}
            </div>
            <div style={{ display: "flex", gap: 8 }}>
              <input value={inputValue} onChange={function(e) { setInputValue(e.target.value); }} placeholder="都市名を追加" style={{ flex: 1, padding: "6px 12px", border: "1px solid #ddd", borderRadius: 6 }} />
              <button onClick={handleAddFavorite} style={{ padding: "6px 16px", background: "#4caf50", color: "white", border: "none", borderRadius: 6, cursor: "pointer" }}>追加</button>
            </div>
          </div>

          {/* 現在の天気 */}
          <div className="card">
            <h2>{city}の天気</h2>
            {loading && <p>読み込み中...</p>}
            {error && <p style={{ color: "red" }}>{error}</p>}
            {weather && !loading && (
              <div>
                <div style={{ fontSize: 48, textAlign: "center" }}>{getWeatherIcon(weather.condition)}</div>
                <div style={{ fontSize: 36, textAlign: "center", fontWeight: "bold" }}>{convertTemp(weather.temp)}{getTempUnit()}</div>
                <div style={{ textAlign: "center", color: "#666" }}>{weather.condition}</div>
                <div style={{ display: "flex", justifyContent: "center", gap: 24, marginTop: 12, color: "#888" }}>
                  <span>湿度: {weather.humidity}%</span>
                  <span>風速: {weather.wind}m/s</span>
                </div>
              </div>
            )}
          </div>

          {/* 3日間予報 */}
          {forecast.length > 0 && !loading && (
            <div className="card">
              <h3>3日間予報</h3>
              <div style={{ display: "flex", justifyContent: "space-around" }}>
                {forecast.map(function(day, index) {
                  return (
                    <div key={index} style={{ textAlign: "center" }}>
                      <div style={{ fontWeight: "bold" }}>{day.day}</div>
                      <div style={{ fontSize: 24 }}>{getWeatherIcon(day.condition)}</div>
                      <div>{convertTemp(day.temp)}{getTempUnit()}</div>
                    </div>
                  );
                })}
              </div>
            </div>
          )}
        </div>
      );
    }

    ReactDOM.createRoot(document.getElementById("root")).render(<App />);
  </script>
</body>
</html>
```

<summary>解答例（リファクタリング後）</summary>

```html
<!DOCTYPE html>
<html lang="ja">
<head>
  <meta charset="UTF-8">
  <title>天気予報アプリ（リファクタリング後）</title>
  <script src="https://unpkg.com/react@18/umd/react.development.js"></script>
  <script src="https://unpkg.com/react-dom@18/umd/react-dom.development.js"></script>
  <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
  <style>
    body { font-family: sans-serif; background: #e3f2fd; padding: 20px; }
    .app { max-width: 500px; margin: 0 auto; }
    .card { background: white; border-radius: 12px; padding: 20px; margin-bottom: 16px; box-shadow: 0 2px 8px rgba(0,0,0,0.1); }
    .toggle-btn { padding: 4px 12px; border: none; border-radius: 4px; cursor: pointer; margin-left: 4px; }
    .chip { display: inline-flex; align-items: center; padding: 4px 12px; border-radius: 16px; cursor: pointer; font-size: 14px; margin: 2px; }
  </style>
</head>
<body>
  <div id="root"></div>
  <script type="text/babel">
    // ==================== ユーティリティ ====================
    function getWeatherIcon(condition) {
      var icons = { "晴れ": "☀️", "曇り": "☁️", "雨": "🌧️" };
      return icons[condition] || "🌤️";
    }

    // ==================== データ取得（ダミーAPI） ====================
    function fetchWeatherData(cityName) {
      return new Promise(function(resolve) {
        setTimeout(function() {
          var temps = { "東京": 22, "大阪": 24, "札幌": 15, "福岡": 25, "名古屋": 23 };
          var conditions = { "東京": "晴れ", "大阪": "曇り", "札幌": "雨", "福岡": "晴れ", "名古屋": "曇り" };
          var humidity = { "東京": 55, "大阪": 65, "札幌": 70, "福岡": 50, "名古屋": 60 };
          if (temps[cityName] !== undefined) {
            resolve({
              current: { temp: temps[cityName], condition: conditions[cityName], humidity: humidity[cityName], wind: Math.floor(Math.random() * 10) + 1 },
              forecast: [
                { day: "明日", temp: temps[cityName] + Math.floor(Math.random() * 5) - 2, condition: "晴れ" },
                { day: "明後日", temp: temps[cityName] + Math.floor(Math.random() * 5) - 2, condition: "曇り" },
                { day: "3日後", temp: temps[cityName] + Math.floor(Math.random() * 5) - 2, condition: "雨" },
              ],
            });
          } else {
            resolve(null);
          }
        }, 800);
      });
    }

    // ==================== カスタムフック ====================
    function useWeather(city) {
      const [weather, setWeather] = React.useState(null);
      const [forecast, setForecast] = React.useState([]);
      const [loading, setLoading] = React.useState(false);
      const [error, setError] = React.useState(null);

      React.useEffect(function() {
        setLoading(true);
        setError(null);
        fetchWeatherData(city).then(function(data) {
          if (data) {
            setWeather(data.current);
            setForecast(data.forecast);
          } else {
            setWeather(null);
            setForecast([]);
            setError(city + " の天気データが見つかりません");
          }
          setLoading(false);
        });
      }, [city]);

      return { weather, forecast, loading, error };
    }

    function useFavorites(initial) {
      const [favorites, setFavorites] = React.useState(initial);

      function add(name) {
        if (name && !favorites.includes(name)) {
          setFavorites([...favorites, name]);
        }
      }

      function remove(name) {
        setFavorites(favorites.filter(function(f) { return f !== name; }));
      }

      return { favorites, add, remove };
    }

    // ==================== Context ====================
    const UnitContext = React.createContext();

    function UnitProvider({ children }) {
      const [unit, setUnit] = React.useState("celsius");

      function convertTemp(celsius) {
        if (unit === "fahrenheit") return Math.round(celsius * 9 / 5 + 32);
        return celsius;
      }

      function getSymbol() {
        return unit === "celsius" ? "℃" : "℉";
      }

      return (
        <UnitContext.Provider value={{ unit, setUnit, convertTemp, getSymbol }}>
          {children}
        </UnitContext.Provider>
      );
    }

    function useUnit() {
      return React.useContext(UnitContext);
    }

    // ==================== UIコンポーネント ====================
    function UnitToggle() {
      var ctx = useUnit();
      function btn(value, label) {
        var isActive = ctx.unit === value;
        return (
          <button className="toggle-btn"
            style={{ background: isActive ? "#2196f3" : "#e0e0e0", color: isActive ? "white" : "#333" }}
            onClick={function() { ctx.setUnit(value); }}>{label}</button>
        );
      }
      return <div style={{ textAlign: "right", marginBottom: 12 }}>{btn("celsius", "℃")}{btn("fahrenheit", "℉")}</div>;
    }

    function FavoriteList({ favorites, currentCity, onSelect, onRemove, onAdd }) {
      const [inputValue, setInputValue] = React.useState("");
      function handleAdd() {
        onAdd(inputValue);
        setInputValue("");
      }

      return (
        <div className="card">
          <h3>お気に入り都市</h3>
          <div style={{ display: "flex", flexWrap: "wrap", gap: 8, marginBottom: 12 }}>
            {favorites.map(function(fav) {
              var isActive = currentCity === fav;
              return (
                <span key={fav} className="chip" style={{ background: isActive ? "#2196f3" : "#e8e8e8", color: isActive ? "white" : "#333" }}>
                  <span onClick={function() { onSelect(fav); }}>{fav}</span>
                  <span onClick={function() { onRemove(fav); }} style={{ marginLeft: 6 }}>×</span>
                </span>
              );
            })}
          </div>
          <div style={{ display: "flex", gap: 8 }}>
            <input value={inputValue} onChange={function(e) { setInputValue(e.target.value); }}
                   onKeyDown={function(e) { if (e.key === "Enter") handleAdd(); }}
                   placeholder="都市名を追加" style={{ flex: 1, padding: "6px 12px", border: "1px solid #ddd", borderRadius: 6 }} />
            <button onClick={handleAdd} style={{ padding: "6px 16px", background: "#4caf50", color: "white", border: "none", borderRadius: 6, cursor: "pointer" }}>追加</button>
          </div>
        </div>
      );
    }

    function CurrentWeather({ city, weather, loading, error }) {
      var ctx = useUnit();

      return (
        <div className="card">
          <h2>{city}の天気</h2>
          {loading && <p>読み込み中...</p>}
          {error && <p style={{ color: "red" }}>{error}</p>}
          {weather && !loading && (
            <div>
              <div style={{ fontSize: 48, textAlign: "center" }}>{getWeatherIcon(weather.condition)}</div>
              <div style={{ fontSize: 36, textAlign: "center", fontWeight: "bold" }}>{ctx.convertTemp(weather.temp)}{ctx.getSymbol()}</div>
              <div style={{ textAlign: "center", color: "#666" }}>{weather.condition}</div>
              <div style={{ display: "flex", justifyContent: "center", gap: 24, marginTop: 12, color: "#888" }}>
                <span>湿度: {weather.humidity}%</span>
                <span>風速: {weather.wind}m/s</span>
              </div>
            </div>
          )}
        </div>
      );
    }

    function ForecastPanel({ forecast, loading }) {
      var ctx = useUnit();
      if (forecast.length === 0 || loading) return null;

      return (
        <div className="card">
          <h3>3日間予報</h3>
          <div style={{ display: "flex", justifyContent: "space-around" }}>
            {forecast.map(function(day, index) {
              return (
                <div key={index} style={{ textAlign: "center" }}>
                  <div style={{ fontWeight: "bold" }}>{day.day}</div>
                  <div style={{ fontSize: 24 }}>{getWeatherIcon(day.condition)}</div>
                  <div>{ctx.convertTemp(day.temp)}{ctx.getSymbol()}</div>
                </div>
              );
            })}
          </div>
        </div>
      );
    }

    // ==================== メインコンポーネント ====================
    function App() {
      const [city, setCity] = React.useState("東京");
      var weatherData = useWeather(city);
      var favs = useFavorites(["東京", "大阪", "札幌"]);

      return (
        <div className="app">
          <h1 style={{ textAlign: "center" }}>天気予報</h1>
          <UnitToggle />
          <FavoriteList favorites={favs.favorites} currentCity={city}
            onSelect={setCity} onRemove={favs.remove} onAdd={favs.add} />
          <CurrentWeather city={city} weather={weatherData.weather}
            loading={weatherData.loading} error={weatherData.error} />
          <ForecastPanel forecast={weatherData.forecast} loading={weatherData.loading} />
        </div>
      );
    }

    ReactDOM.createRoot(document.getElementById("root")).render(
      <UnitProvider><App /></UnitProvider>
    );
  </script>
</body>
</html>
```

**リファクタリングで改善した点：**

| 改善項目 | Before | After |
|---|---|---|
| Appの行数 | 約200行 | 約20行 |
| データ取得 | App内にベタ書き | `useWeather` カスタムフック |
| お気に入り管理 | App内に散在 | `useFavorites` カスタムフック |
| 温度単位 | propsで渡す | `UnitContext` + `useUnit` |
| UI部品 | 全部App内 | 5つのコンポーネントに分割 |
| 再利用性 | なし | フックとコンポーネントが独立 |
