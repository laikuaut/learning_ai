## 課題の要件

1. 全項目を正しく入力して送信し、成功メッセージが表示されることを検証する
2. セレクトボックス（プラン選択）で各値を選べることを検証する
3. ラジオボタンの選択を切り替えられることを検証する
4. チェックボックス（利用規約）のチェック/解除を検証する
5. 名前・メール・パスワード・プラン・同意が空/不正な状態で送信したとき、それぞれのエラーメッセージが表示されることを検証する
6. パラメータ化テストで複数のバリデーションパターンをまとめてテストする

// tests/form-validation.spec.ts
// 学べる内容：フォーム操作、バリデーションテスト、パラメータ化テスト
// 実行方法：npx playwright test tests/form-validation.spec.ts --project=chromium

import { test, expect } from '@playwright/test';
import path from 'path';

const TEST_PAGE = path.resolve(__dirname, '../register-form.html');

test.describe('正常系', () => {

  test.beforeEach(async ({ page }) => {
    await page.goto(`file://${TEST_PAGE}`);
  });

  test('すべての項目を正しく入力して送信できる', async ({ page }) => {
    // フォーム入力
    await page.getByLabel('名前').fill('田中太郎');
    await page.getByLabel('メールアドレス').fill('taro@example.com');
    await page.getByLabel('パスワード（8文字以上）').fill('password123');
    await page.getByLabel('プラン').selectOption('basic');
    await page.getByLabel('利用規約に同意する').check();

    // 送信
    await page.getByRole('button', { name: '登録する' }).click();

    // 成功メッセージの検証
    await expect(page.locator('#success-msg')).toBeVisible();
    await expect(page.locator('#success-msg')).toHaveText('登録が完了しました！');

    // フォームが非表示になることを検証
    await expect(page.locator('#register-form')).toBeHidden();
  });

  test('セレクトボックスとチェックボックスが動作する', async ({ page }) => {
    // セレクトボックス
    const planSelect = page.getByLabel('プラン');
    await planSelect.selectOption('free');
    await expect(planSelect).toHaveValue('free');

    await planSelect.selectOption('premium');
    await expect(planSelect).toHaveValue('premium');

    // チェックボックス
    const agreeCheck = page.getByLabel('利用規約に同意する');
    await agreeCheck.check();
    await expect(agreeCheck).toBeChecked();

    await agreeCheck.uncheck();
    await expect(agreeCheck).not.toBeChecked();

    // ラジオボタン
    const importantRadio = page.getByLabel('重要のみ');
    await importantRadio.check();
    await expect(importantRadio).toBeChecked();
    await expect(page.getByLabel('すべて受け取る')).not.toBeChecked();
  });
});

test.describe('バリデーション', () => {

  test.beforeEach(async ({ page }) => {
    await page.goto(`file://${TEST_PAGE}`);
  });

  test('名前が空のときエラーが表示される', async ({ page }) => {
    // 名前以外は入力
    await page.getByLabel('メールアドレス').fill('taro@example.com');
    await page.getByLabel('パスワード（8文字以上）').fill('password123');
    await page.getByLabel('プラン').selectOption('free');
    await page.getByLabel('利用規約に同意する').check();

    await page.getByRole('button', { name: '登録する' }).click();

    // エラーメッセージの検証
    await expect(page.locator('#name-error')).toBeVisible();
    await expect(page.locator('#name-error')).toHaveText('名前を入力してください');

    // 成功メッセージは表示されない
    await expect(page.locator('#success-msg')).not.toBeVisible();
  });

  test('メールアドレスが不正なときエラーが表示される', async ({ page }) => {
    await page.getByLabel('名前').fill('田中太郎');
    await page.getByLabel('メールアドレス').fill('invalid-email');
    await page.getByLabel('パスワード（8文字以上）').fill('password123');
    await page.getByLabel('プラン').selectOption('free');
    await page.getByLabel('利用規約に同意する').check();

    await page.getByRole('button', { name: '登録する' }).click();

    await expect(page.locator('#email-error')).toBeVisible();
  });

  test('パスワードが短すぎるときエラーが表示される', async ({ page }) => {

    await expect(page.locator('#password-error')).toBeVisible();
  });

  test('利用規約に同意しないと送信できない', async ({ page }) => {

<summary>解答例（改良版 ─ パラメータ化テストとヘルパー関数）</summary>

```typescript
// tests/form-validation-advanced.spec.ts
// 改良版：パラメータ化テストとヘルパー関数で重複を排除
// 実行方法：npx playwright test tests/form-validation-advanced.spec.ts --project=chromium

import { test, expect, Page } from '@playwright/test';
import path from 'path';

const TEST_PAGE = path.resolve(__dirname, '../register-form.html');

// --- ヘルパー関数 ---

/** 全フォーム項目を正しい値で埋める */
async function fillAllFields(page: Page) {
  await page.getByLabel('名前').fill('田中太郎');
  await page.getByLabel('メールアドレス').fill('taro@example.com');
  await page.getByLabel('パスワード（8文字以上）').fill('password123');
  await page.getByLabel('プラン').selectOption('basic');
  await page.getByLabel('すべて受け取る').check();
  await page.getByLabel('利用規約に同意する').check();
}

/** フォームを送信する */
async function submitForm(page: Page) {
  await page.getByRole('button', { name: '登録する' }).click();
}

// --- テスト ---

test.beforeEach(async ({ page }) => {
  await page.goto(`file://${TEST_PAGE}`);
});

test('正常系：全項目入力で登録成功', async ({ page }) => {
  await fillAllFields(page);
  await submitForm(page);

  await expect(page.locator('#success-msg')).toBeVisible();
  await expect(page.locator('#register-form')).toBeHidden();
});

// パラメータ化テスト：バリデーションパターン
const validationCases = [
  {
    name: '名前が空',
    setup: async (page: Page) => {
      await fillAllFields(page);
      await page.getByLabel('名前').clear();
    },
    errorSelector: '#name-error',
    errorMessage: '名前を入力してください',
  },
  {
    name: 'メールアドレスが不正',
    setup: async (page: Page) => {
      await fillAllFields(page);
      await page.getByLabel('メールアドレス').clear();
      await page.getByLabel('メールアドレス').fill('not-an-email');
    },
    errorSelector: '#email-error',
    errorMessage: '有効なメールアドレスを入力してください',
  },
  {
    name: 'パスワードが7文字',
    setup: async (page: Page) => {
      await fillAllFields(page);
      await page.getByLabel('パスワード（8文字以上）').clear();
      await page.getByLabel('パスワード（8文字以上）').fill('1234567');
    },
    errorSelector: '#password-error',
    errorMessage: 'パスワードは8文字以上で入力してください',
  },
  {
    name: 'プランが未選択',
    setup: async (page: Page) => {
      await fillAllFields(page);
      await page.getByLabel('プラン').selectOption('');
    },
    errorSelector: '#plan-error',
    errorMessage: 'プランを選択してください',
  },
  {
    name: '利用規約に未同意',
    setup: async (page: Page) => {
      await fillAllFields(page);
      await page.getByLabel('利用規約に同意する').uncheck();
    },
    errorSelector: '#agree-error',
    errorMessage: '利用規約への同意が必要です',
  },
];

for (const tc of validationCases) {
  test(`バリデーション：${tc.name}`, async ({ page }) => {
    await tc.setup(page);
    await submitForm(page);

    // エラーメッセージの検証
    await expect(page.locator(tc.errorSelector)).toBeVisible();
    await expect(page.locator(tc.errorSelector)).toHaveText(tc.errorMessage);

    // 成功メッセージは表示されない
    await expect(page.locator('#success-msg')).not.toBeVisible();
  });
}

test('エラー修正後に再送信すると成功する', async ({ page }) => {
  // 空のまま送信 → エラー
  await submitForm(page);
  await expect(page.locator('#name-error')).toBeVisible();

  // 全項目を入力して再送信 → 成功
  await fillAllFields(page);
  await submitForm(page);

  await expect(page.locator('#success-msg')).toBeVisible();
  await expect(page.locator('#name-error')).not.toBeVisible();
});
```

**初心者向けとの違い:**
- `fillAllFields` ヘルパー関数でフォーム入力の重複を排除
- パラメータ化テストで5つのバリデーションパターンを `for` ループで回す
- 「全部入力してから1つだけクリア」というパターンで、他のフィールドの影響を排除
- エラー→修正→再送信の流れをテストするシナリオを追加
