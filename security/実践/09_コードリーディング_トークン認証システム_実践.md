## 対象コード

```python
# 簡易トークン認証システム
# 実行方法：python token_auth.py

import hashlib
import hmac
import json
import os
import time
import base64

SECRET_KEY = os.urandom(32)
TOKEN_EXPIRY = 3600  # トークンの有効期限（秒）
REFRESH_WINDOW = 300  # リフレッシュ可能な猶予期間（秒）

users = {}
revoked_tokens = set()


def b64_encode(data):
    """バイト列をURL安全なBase64に変換します。"""
    return base64.urlsafe_b64encode(data).rstrip(b"=").decode("ascii")


def b64_decode(s):
    """URL安全なBase64をバイト列に戻します。"""
    padding = 4 - len(s) % 4
    if padding != 4:
        s += "=" * padding
    return base64.urlsafe_b64decode(s.encode("ascii"))


def create_signature(header_b64, payload_b64):
    """ヘッダとペイロードからHMAC署名を生成します。"""
    message = f"{header_b64}.{payload_b64}".encode("ascii")
    return hmac.new(SECRET_KEY, message, hashlib.sha256).digest()


def generate_token(username, role):
    """認証トークンを生成します。"""
    header = {"alg": "HS256", "typ": "JWT"}
    payload = {
        "sub": username,
        "role": role,
        "iat": int(time.time()),
        "exp": int(time.time()) + TOKEN_EXPIRY,
        "jti": os.urandom(16).hex(),
    }

    header_b64 = b64_encode(json.dumps(header).encode("utf-8"))
    payload_b64 = b64_encode(json.dumps(payload).encode("utf-8"))
    signature_b64 = b64_encode(create_signature(header_b64, payload_b64))

    return f"{header_b64}.{payload_b64}.{signature_b64}"


def verify_token(token):
    """トークンの署名と有効期限を検証します。"""
    parts = token.split(".")
    if len(parts) != 3:
        return None, "トークンの形式が不正です"

    header_b64, payload_b64, signature_b64 = parts

    expected_sig = create_signature(header_b64, payload_b64)
    actual_sig = b64_decode(signature_b64)

    if not hmac.compare_digest(expected_sig, actual_sig):
        return None, "署名が不正です"

    try:
        payload = json.loads(b64_decode(payload_b64))
    except (json.JSONDecodeError, Exception):
        return None, "ペイロードのデコードに失敗しました"

    if payload.get("jti") in revoked_tokens:
        return None, "トークンは無効化されています"

    now = int(time.time())
    if now > payload.get("exp", 0):
        return None, "トークンの有効期限が切れています"

    return payload, "OK"


def refresh_token(old_token):
    """有効期限が近いトークンを新しいトークンに交換します。"""
    payload, msg = verify_token(old_token)
    if payload is None:
        exp_check = None
        try:
            parts = old_token.split(".")
            exp_check = json.loads(b64_decode(parts[1]))
        except Exception:
            return None, "トークンの解析に失敗しました"

        if exp_check and int(time.time()) - exp_check.get("exp", 0) <= REFRESH_WINDOW:
            header_b64, payload_b64, signature_b64 = old_token.split(".")
            expected_sig = create_signature(header_b64, payload_b64)
            actual_sig = b64_decode(signature_b64)
            if hmac.compare_digest(expected_sig, actual_sig):
                revoked_tokens.add(exp_check["jti"])
                return generate_token(exp_check["sub"], exp_check["role"]), "リフレッシュ成功"

        return None, f"リフレッシュ不可: {msg}"

    remaining = payload["exp"] - int(time.time())
    if remaining > TOKEN_EXPIRY // 2:
        return None, "有効期限に十分な余裕があります。リフレッシュは不要です。"

    revoked_tokens.add(payload["jti"])
    new_token = generate_token(payload["sub"], payload["role"])
    return new_token, "リフレッシュ成功"


def register_user(username, password, role="user"):
    """ユーザーを登録します。"""
    if username in users:
        return False, "既に登録済みです"
    salt = os.urandom(16)
    pw_hash = hashlib.pbkdf2_hmac("sha256", password.encode(), salt, 100000)
    users[username] = {"hash": pw_hash.hex(), "salt": salt.hex(), "role": role}
    return True, "登録完了"


def authenticate(username, password):
    """ユーザー認証を行い、成功すればトークンを返します。"""
    if username not in users:
        return None, "認証失敗"
    user = users[username]
    salt = bytes.fromhex(user["salt"])
    pw_hash = hashlib.pbkdf2_hmac("sha256", password.encode(), salt, 100000)
    if hmac.compare_digest(pw_hash.hex(), user["hash"]):
        token = generate_token(username, user["role"])
        return token, "認証成功"
    return None, "認証失敗"


def check_permission(token, required_role):
    """トークンから権限をチェックします。"""
    payload, msg = verify_token(token)
    if payload is None:
        return False, msg
    role_hierarchy = {"admin": 3, "editor": 2, "user": 1}
    user_level = role_hierarchy.get(payload["role"], 0)
    required_level = role_hierarchy.get(required_role, 0)
    if user_level >= required_level:
        return True, f"アクセス許可（{payload['sub']}: {payload['role']}）"
    return False, "権限が不足しています"


# --- メイン処理 ---
print("===== トークン認証システム =====")
register_user("admin", "Adm!n2024#", "admin")
register_user("editor", "Ed!tor2024#", "editor")
register_user("viewer", "V!ewer2024#", "user")

current_token = None

while True:
    print("\n1.ログイン  2.トークン検証  3.権限チェック  4.リフレッシュ")
    print("5.ログアウト  6.トークン内容表示  7.終了")
    choice = input("操作: ")

    if choice == "1":
        username = input("ユーザー名: ")
        password = input("パスワード: ")
        token, msg = authenticate(username, password)
        if token:
            current_token = token
            print(f"{msg}")
            print(f"トークン: {token[:50]}...")
        else:
            print(msg)

    elif choice == "2":
        if not current_token:
            print("ログインしてください。")
            continue
        payload, msg = verify_token(current_token)
        if payload:
            remaining = payload["exp"] - int(time.time())
            print(f"有効（残り{remaining}秒）")
        else:
            print(f"無効: {msg}")

    elif choice == "3":
        if not current_token:
            print("ログインしてください。")
            continue
        role = input("必要な権限 (admin/editor/user): ")
        ok, msg = check_permission(current_token, role)
        print(msg)

    elif choice == "4":
        if not current_token:
            print("ログインしてください。")
            continue
        new_token, msg = refresh_token(current_token)
        if new_token:
            current_token = new_token
            print(f"{msg}")
        else:
            print(msg)

    elif choice == "5":
        if current_token:
            payload, _ = verify_token(current_token)
            if payload:
                revoked_tokens.add(payload["jti"])
            current_token = None
            print("ログアウトしました。")
        else:
            print("ログインしていません。")

    elif choice == "6":
        if not current_token:
            print("ログインしてください。")
            continue
        parts = current_token.split(".")
        try:
            header = json.loads(b64_decode(parts[0]))
            payload = json.loads(b64_decode(parts[1]))
            print(f"ヘッダ:   {json.dumps(header, indent=2)}")
            print(f"ペイロード: {json.dumps(payload, indent=2)}")
        except Exception as e:
            print(f"デコードエラー: {e}")

    elif choice == "7":
        print("終了します。")
        break
```

### 設問1（基礎）: トークンの構造

## 解答例

<details>
<summary>設問1の解答</summary>

トークンは以下の3つの部分で構成されています。

1. **ヘッダ（Header）**: アルゴリズム情報（`"alg": "HS256"`）とトークンの種類（`"typ": "JWT"`）
2. **ペイロード（Payload）**: ユーザー情報（`sub`）、権限（`role`）、発行時刻（`iat`）、有効期限（`exp`）、一意識別子（`jti`）
3. **署名（Signature）**: ヘッダとペイロードをHMAC-SHA256で署名した値

**3つに分かれている理由：**
- ヘッダとペイロードはBase64でエンコードされているだけで、誰でも読める（暗号化されていない）
- 署名はサーバーの秘密鍵でしか生成できない
- この分離により、「内容は読めるが改ざんできない」という性質を持つ
- 受信側は署名を検証するだけで、データベースに問い合わせることなくトークンの正当性を確認できる（ステートレス認証）

</details>

<details>
<summary>設問2の解答</summary>

**署名の目的：** トークンの完全性（Integrity）を保証すること。つまり、トークンが発行後に改ざんされていないことを証明します。

**署名がなかった場合に可能な攻撃：**
- ペイロードの `role` を `"user"` から `"admin"` に書き換える → 権限昇格（Privilege Escalation）
- `exp`（有効期限）を未来の時刻に書き換える → トークンの無期限化
- `sub`（ユーザー名）を他人に書き換える → なりすまし（Impersonation）
- トークン全体を自分で自由に作成する → トークン偽造（Token Forgery）

署名があるため、これらの改ざんを行うと `verify_token()` で `hmac.compare_digest()` が失敗し、不正なトークンとして拒否されます。

</details>

<details>
<summary>設問3の解答</summary>

古いトークンの `jti`（JSON Token Identifier）を `revoked_tokens` に追加することで、**トークンの再利用攻撃（Token Replay Attack）** を防いでいます。

もしこの処理がなかった場合：
1. ユーザーがトークンAでリフレッシュしてトークンBを取得
2. 攻撃者がトークンAを盗み出す
3. トークンAはまだ有効なので、攻撃者もリフレッシュしてトークンCを取得
4. 結果、正規ユーザーとは別のトークンが不正に発行される

`jti` を無効化リストに追加することで、一度リフレッシュに使用されたトークンは二度と使えなくなります。

さらに、ログアウト時にも `jti` を `revoked_tokens` に追加しているため、ログアウト後のトークン悪用も防止しています。

</details>

<details>
<summary>設問4の解答</summary>

`hmac.compare_digest()` が使われている箇所：

1. `verify_token()` 内の署名検証: `hmac.compare_digest(expected_sig, actual_sig)`
2. `refresh_token()` 内の署名再検証: `hmac.compare_digest(expected_sig, actual_sig)`
3. `authenticate()` 内のパスワード検証: `hmac.compare_digest(pw_hash.hex(), user["hash"])`

**通常の `==` 比較との違い：**

通常の文字列/バイト列比較（`==`）は、先頭から1文字ずつ比較し、不一致を見つけた時点で即座に `False` を返します。この性質を利用すると：

- 1文字目で不一致 → すぐ返る（短い応答時間）
- 10文字目で不一致 → 少し時間がかかる（長い応答時間）
- 全文字一致 → 最も時間がかかる

攻撃者は応答時間の微小な差を統計的に分析することで、正しい値を1文字ずつ推測できます。これを**タイミング攻撃（Timing Attack）**と呼びます。

`hmac.compare_digest()` は、文字列の長さが同じであれば常に一定の時間で比較を完了するため、応答時間から情報が漏洩しません。

</details>

<details>
<summary>設問5の解答</summary>

**改善すべき点（実運用に向けて）：**

1. **`SECRET_KEY` の永続化**
   現在は `os.urandom(32)` で毎回新しい鍵が生成されるため、サーバー再起動時に全トークンが無効になります。鍵は環境変数や鍵管理サービス（AWS KMS等）から読み込むべきです。

2. **`revoked_tokens` のメモリ管理**
   `set()` にトークンIDを追加し続けるため、長期間運用するとメモリが枯渇します。期限切れのトークンIDを定期的に削除するか、Redis等の外部ストレージを使うべきです。

3. **ユーザーデータの永続化**
   `users` がメモリ上の辞書なので、再起動でデータが消えます。データベースに保存する必要があります。

4. **HTTPS の必須化**
   トークンが通信経路で盗聴される可能性があります。実運用ではHTTPS（TLS）を必須にすべきです。

5. **トークンのスコープ制限**
   現在のトークンはシステム全体にアクセスできます。APIごとにスコープ（`"scope": ["read:users", "write:posts"]`）を設定すべきです。

6. **リフレッシュトークンの分離**
   現在はアクセストークンとリフレッシュトークンが同じです。実務では、短寿命のアクセストークンと長寿命のリフレッシュトークンを分けるのが一般的です。

7. **ログイン試行回数の制限**
   `authenticate()` にブルートフォース対策がありません。

</details>
