## 課題の要件

1. **スワップ関数**: ポインタ渡し（参照渡し）で2つの整数の値を交換する関数を作る
2. **配列逆転関数**: ポインタを使って配列の要素を反転させる関数を作る
3. **動的配列**: `malloc` で実行時にサイズを決めた配列を確保し、データを入力・集計する
4. メニュー形式で各機能を選択できるようにする
5. 動的メモリは必ず `free` で解放する

## ステップガイド

<details>
<summary>ステップ1：スワップ関数を作る</summary>

C言語では関数の引数は値渡し（call by value）なので、呼び出し元の変数を変更するにはポインタを使います。

```c
void swap(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

/* 呼び出し側 */
int x = 10, y = 20;
swap(&x, &y);  /* & でアドレスを渡す */
```

</details>

<details>
<summary>ステップ2：配列の逆転をポインタで実装する</summary>

先頭ポインタと末尾ポインタを使って、外側から内側に向かって交換していきます。

```
配列: [1, 2, 3, 4, 5]
       ↑              ↑
      left           right
      
交換 → [5, 2, 3, 4, 1]
          ↑        ↑
         left    right
         
交換 → [5, 4, 3, 2, 1]  完了
```

```c
void reverse_array(int *arr, int size) {
    int *left = arr;
    int *right = arr + size - 1;
    while (left < right) {
        int temp = *left;
        *left = *right;
        *right = temp;
        left++;
        right--;
    }
}
```

</details>

<details>
<summary>ステップ3：malloc で動的配列を確保する</summary>

`malloc`（memory allocation）は指定バイト数のメモリを確保して先頭アドレスを返します。

```c
#include <stdlib.h>

int n;
printf("要素数を入力してください: ");
scanf("%d", &n);

int *arr = (int *)malloc(sizeof(int) * n);
if (arr == NULL) {
    printf("メモリ確保に失敗しました。\n");
    return 1;
}

/* 使い終わったら解放 */
free(arr);
```

</details>

## 解答例

<details>
<summary>解答例（初心者向け）</summary>

```c
/*
 * ポインタで学ぶメモリ操作
 * 学べる内容：ポインタ渡し、配列とポインタ、malloc/free
 * コンパイル：gcc -o pointer_practice pointer_practice.c
 * 実行：./pointer_practice
 */
#include <stdio.h>
#include <stdlib.h>

/* 2つの整数を交換する */
void swap(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

/* 配列を逆順にする */
void reverse_array(int *arr, int size) {
    int *left = arr;
    int *right = arr + size - 1;
    while (left < right) {
        int temp = *left;
        *left = *right;
        *right = temp;
        left++;
        right--;
    }
}

/* 配列を表示する */
void print_array(int *arr, int size) {
    int i;
    for (i = 0; i < size; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
}

/* --- スワップのデモ --- */
void demo_swap(void) {
    int a = 10, b = 20;
    printf("\n--- スワップ ---\n");
    printf("交換前: a = %d, b = %d\n", a, b);
    swap(&a, &b);
    printf("交換後: a = %d, b = %d\n", a, b);
}

/* --- 配列逆転のデモ --- */
void demo_reverse(void) {
    int arr[] = {1, 2, 3, 4, 5};
    int size = sizeof(arr) / sizeof(arr[0]);

    printf("\n--- 配列の逆転 ---\n");
    printf("元の配列: ");
    print_array(arr, size);

    reverse_array(arr, size);

    printf("逆転後  : ");
    print_array(arr, size);
}

/* --- 動的配列のデモ --- */
void demo_dynamic(void) {
    int n, i;
    int sum = 0;

    printf("\n--- 動的配列 ---\n");
    printf("要素数を入力してください: ");
    scanf("%d", &n);

    if (n <= 0) {
        printf("要素数は1以上にしてください。\n");
        return;
    }

    int *arr = (int *)malloc(sizeof(int) * n);
    if (arr == NULL) {
        printf("メモリ確保に失敗しました。\n");
        return;
    }

    printf("値を入力してください:\n");
    for (i = 0; i < n; i++) {
        printf("[%d]: ", i);
        scanf("%d", &arr[i]);
    }

    printf("\n入力データ: ");
    print_array(arr, n);

    for (i = 0; i < n; i++) {
        sum += arr[i];
    }
    printf("合計: %d\n", sum);
    printf("平均: %.1f\n", (double)sum / n);

    free(arr);
    printf("メモリを解放しました。\n");
}

/* --- メイン --- */
int main(void) {
    int choice;

    printf("===== ポインタ実践メニュー =====\n");

    while (1) {
        printf("\n1. スワップ（値の交換）\n");
        printf("2. 配列の逆転\n");
        printf("3. 動的配列の作成\n");
        printf("0. 終了\n\n");
        printf("選択: ");
        scanf("%d", &choice);

        switch (choice) {
            case 0:
                printf("終了します。\n");
                return 0;
            case 1:
                demo_swap();
                break;
            case 2:
                demo_reverse();
                break;
            case 3:
                demo_dynamic();
                break;
            default:
                printf("無効な選択です。\n");
                break;
        }
    }

    return 0;
}
```

<details>
<summary>解答例（改良版 ─ ジェネリックなスワップと realloc）</summary>

`void *` ポインタと `memcpy` を使った汎用スワップ、`realloc` による配列の動的拡張を含むバージョンです。

```c
/*
 * ポインタで学ぶメモリ操作（改良版）
 * 学べる内容：void ポインタ、memcpy、realloc による動的拡張
 * コンパイル：gcc -o pointer_practice2 pointer_practice2.c
 * 実行：./pointer_practice2
 */
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

/* 任意の型に対応する汎用スワップ */
void generic_swap(void *a, void *b, size_t size) {
    void *temp = malloc(size);
    if (temp == NULL) return;
    memcpy(temp, a, size);
    memcpy(a, b, size);
    memcpy(b, temp, size);
    free(temp);
}

/* realloc で配列を動的に拡張しながらデータを読み込む */
void demo_dynamic_grow(void) {
    int capacity = 2;
    int count = 0;
    int *arr = (int *)malloc(sizeof(int) * capacity);
    int i, sum = 0;
    char line[64];

    if (arr == NULL) {
        printf("メモリ確保に失敗しました。\n");
        return;
    }

    printf("\n--- 動的配列（自動拡張）---\n");
    printf("整数を入力してください（'done' で終了）:\n");

    while (1) {
        printf("[%d]: ", count);
        fgets(line, sizeof(line), stdin);
        if (strncmp(line, "done", 4) == 0) {
            break;
        }

        int val;
        if (sscanf(line, "%d", &val) != 1) {
            printf("数値を入力してください。\n");
            continue;
        }

        /* 容量が足りなければ倍に拡張 */
        if (count >= capacity) {
            capacity *= 2;
            int *new_arr = (int *)realloc(arr, sizeof(int) * capacity);
            if (new_arr == NULL) {
                printf("メモリ拡張に失敗しました。\n");
                free(arr);
                return;
            }
            arr = new_arr;
            printf("  (容量を %d に拡張しました)\n", capacity);
        }

        arr[count] = val;
        count++;
    }

    if (count == 0) {
        printf("データがありません。\n");
        free(arr);
        return;
    }

    printf("\n入力データ: ");
    for (i = 0; i < count; i++) {
        printf("%d ", arr[i]);
        sum += arr[i];
    }
    printf("\n要素数: %d\n", count);
    printf("合計: %d\n", sum);
    printf("平均: %.1f\n", (double)sum / count);

    free(arr);
    printf("メモリを解放しました。\n");
}

int main(void) {
    /* 汎用スワップのデモ */
    printf("--- 汎用スワップ ---\n");

    int a = 10, b = 20;
    printf("int: 交換前 a=%d, b=%d\n", a, b);
    generic_swap(&a, &b, sizeof(int));
    printf("int: 交換後 a=%d, b=%d\n", a, b);

    double x = 3.14, y = 2.72;
    printf("double: 交換前 x=%.2f, y=%.2f\n", x, y);
    generic_swap(&x, &y, sizeof(double));
    printf("double: 交換後 x=%.2f, y=%.2f\n", x, y);

    /* 動的配列拡張のデモ */
    demo_dynamic_grow();

    return 0;
}
```

**初心者向けとの違い:**
- `void *` と `memcpy` による汎用スワップ → `int` でも `double` でも同じ関数で交換できる
- `realloc` による動的拡張 → データ数を事前に決めなくてよい（実務で非常に重要なパターン）
- 容量を倍々に増やす戦略（amortized doubling）→ 動的配列の標準的な実装方法
