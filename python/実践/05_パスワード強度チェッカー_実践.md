## 課題の要件

1. パスワードを入力として受け取る
2. 以下の基準で個別に判定する
   - **長さ**: 8文字以上で合格、12文字以上で高評価
   - **大文字**: 1文字以上含む
   - **小文字**: 1文字以上含む
   - **数字**: 1文字以上含む
   - **特殊文字**: `!@#$%^&*()-_=+` のいずれかを含む
   - **連続同一文字**: 同じ文字が3回以上連続していないか
3. 各基準に配点し、合計スコア（0〜100）を算出する
4. スコアに応じて強度ラベルを付ける（弱い / 普通 / 強い / 非常に強い）
5. 不合格の項目についてアドバイスを表示する
6. **各判定は個別の関数として実装すること**

<summary>ステップ1：個別の判定関数を作る</summary>

各判定を個別の関数にします。戻り値は `True` / `False` にすると扱いやすいです。

```python
def has_uppercase(password):
    """大文字が含まれているか判定"""
    for char in password:
        if char.isupper():
            return True
    return False

def has_digit(password):
    """数字が含まれているか判定"""
    for char in password:
        if char.isdigit():
            return True
    return False
```

# パスワード強度チェッカー
# 学べる内容：関数設計、条件分岐、文字列メソッド、スコアリング

def check_length(password):
    """長さをチェックし、(得点, メッセージ)を返す"""
    length = len(password)
    if length >= 12:
        return 30, f"✓ OK ({length}文字)"
    elif length >= 8:
        return 20, f"⚠ 12文字以上でさらに安全 ({length}文字)"
    else:
        return 0, f"✗ 8文字以上にしてください ({length}文字)"


def has_uppercase(password):
    """大文字が含まれているか"""
    for char in password:
        if char.isupper():
            return True
    return False


def has_lowercase(password):
    """小文字が含まれているか"""
    for char in password:
        if char.islower():
            return True
    return False


def has_digit(password):
    """数字が含まれているか"""
    for char in password:
        if char.isdigit():
            return True
    return False


def has_special(password):
    """特殊文字が含まれているか"""
    special_chars = "!@#$%^&*()-_=+"
    for char in password:
        if char in special_chars:
            return True
    return False


def check_consecutive(password):
    """同じ文字が3回以上連続していないかチェック"""
    for i in range(len(password) - 2):
        if password[i] == password[i + 1] == password[i + 2]:
            return False, f"⚠ 「{password[i] * 3}」が連続しています"
    return True, "✓ OK"


def get_strength_label(score):
    """スコアから強度ラベルを返す"""
    if score >= 90:
        return "非常に強い"
    elif score >= 70:
        return "強い"
    elif score >= 40:
        return "普通"
    else:
        return "弱い"


# --- メイン処理 ---
print("===== パスワード強度チェッカー =====")
password = input("パスワードを入力してください: ")

score = 0
advices = []

# 長さチェック
length_score, length_msg = check_length(password)
score = score + length_score
print(f"\n--- 判定結果 ---")
print(f"長さ ({len(password)}文字)       : {length_msg}")
if len(password) < 8:
    advices.append("8文字以上にするとさらに安全です")

# 大文字
if has_uppercase(password):
    score = score + 15
    print(f"大文字              : ✓ OK")
else:
    print(f"大文字              : ✗ 含まれていません")
    advices.append("大文字を含めてください")

# 小文字
if has_lowercase(password):
    score = score + 15
    print(f"小文字              : ✓ OK")
else:
    print(f"小文字              : ✗ 含まれていません")
    advices.append("小文字を含めてください")

# 数字
if has_digit(password):
    score = score + 15
    print(f"数字                : ✓ OK")
else:
    print(f"数字                : ✗ 含まれていません")
    advices.append("数字を含めてください")

# 特殊文字
if has_special(password):
    score = score + 15
    print(f"特殊文字            : ✓ OK")
else:
    print(f"特殊文字            : ✗ 含まれていません")
    advices.append("特殊文字（!@#$%など）を含めてください")

# 連続文字
consec_ok, consec_msg = check_consecutive(password)
if consec_ok:
    score = score + 10
print(f"連続同一文字        : {consec_msg}")

# 総合結果
label = get_strength_label(score)
print(f"\n総合スコア: {score}/100")
print(f"強度: 【{label}】")

if advices:
    print("\n--- アドバイス ---")
    for advice in advices:
        print(f"・{advice}")

```python
# パスワード強度チェッカー（改良版）
# 判定ルールをデータとして定義し、ループで処理するバージョン

def has_upper(pw):
    return any(c.isupper() for c in pw)

def has_lower(pw):
    return any(c.islower() for c in pw)

def has_digit(pw):
    return any(c.isdigit() for c in pw)

def has_special(pw):
    return any(c in "!@#$%^&*()-_=+[]{}|;:,.<>?" for c in pw)

def no_triple_repeat(pw):
    """同じ文字が3回以上連続していないか"""
    for i in range(len(pw) - 2):
        if pw[i] == pw[i + 1] == pw[i + 2]:
            return False
    return True


def check_password(password):
    """パスワードを総合判定し、結果辞書を返す"""
    # 判定ルール: (項目名, 判定関数, 配点, 不合格時のアドバイス)
    rules = [
        ("大文字",   has_upper,          15, "大文字を含めてください"),
        ("小文字",   has_lower,          15, "小文字を含めてください"),
        ("数字",     has_digit,          15, "数字を含めてください"),
        ("特殊文字", has_special,        15, "特殊文字（!@#$%など）を含めてください"),
        ("連続文字", no_triple_repeat,   10, "同じ文字を3回以上連続させないでください"),
    ]

    score = 0
    results = []
    advices = []

    # 長さ判定（特別扱い）
    length = len(password)
    if length >= 12:
        score += 30
        results.append(("長さ", True, f"✓ OK ({length}文字)"))
    elif length >= 8:
        score += 20
        results.append(("長さ", True, f"⚠ 12文字以上推奨 ({length}文字)"))
        advices.append("12文字以上にするとさらに安全です")
    else:
        results.append(("長さ", False, f"✗ 短すぎます ({length}文字)"))
        advices.append("8文字以上にしてください")

    # 各ルールを順に判定
    for name, check_func, points, advice in rules:
        passed = check_func(password)
        if passed:
            score += points
            results.append((name, True, "✓ OK"))
        else:
            results.append((name, False, "✗ 不合格"))
            advices.append(advice)

    return score, results, advices


def display_result(password, score, results, advices):
    """結果を表示する"""
    print("\n--- 判定結果 ---")
    for name, passed, msg in results:
        print(f"{name:<16}: {msg}")

    labels = [(90, "非常に強い"), (70, "強い"), (40, "普通"), (0, "弱い")]
    label = next(label for threshold, label in labels if score >= threshold)

    print(f"\n総合スコア: {score}/100")
    print(f"強度: 【{label}】")

    if advices:
        print("\n--- アドバイス ---")
        for advice in advices:
            print(f"・{advice}")


# --- メイン処理 ---
print("===== パスワード強度チェッカー =====")
password = input("パスワードを入力してください: ")
score, results, advices = check_password(password)
display_result(password, score, results, advices)
```

**初心者向けとの違い:**
- 判定ルールをタプルのリストで定義 → ルール追加が1行で済む
- `any()` とジェネレータ式で文字判定を簡潔に
- `next()` でスコアからラベルを検索
- 判定ロジックと表示ロジックを完全分離
