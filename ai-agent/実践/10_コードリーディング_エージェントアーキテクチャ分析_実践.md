```python
class Tool:
    def __init__(self, name, description, function, parameters_schema):
        self.name = name
        self.description = description
        self.function = function
        self.parameters_schema = parameters_schema

    def execute(self, **kwargs):
        return self.function(**kwargs)


class Memory:
    def __init__(self, max_messages=50):
        self.messages = []
        self.max_messages = max_messages

    def add(self, role, content):
        self.messages.append({"role": role, "content": content})
        if len(self.messages) > self.max_messages:
            self.messages = self.messages[-self.max_messages:]

    def get_context(self):
        return self.messages


class Agent:
    def __init__(self, name, system_prompt, llm_client, tools=None,
                 max_iterations=10):
        self.name = name
        self.system_prompt = system_prompt
        self.llm = llm_client
        self.tools = {t.name: t for t in (tools or [])}
        self.memory = Memory()
        self.max_iterations = max_iterations

    def run(self, user_input):
        self.memory.add("user", user_input)

        for i in range(self.max_iterations):
            messages = [
                {"role": "system", "content": self.system_prompt},
                *self.memory.get_context()
            ]

            response = self.llm.chat(
                messages=messages,
                tools=self._get_tool_definitions()
            )

            if response.tool_calls:
                for tool_call in response.tool_calls:
                    tool_name = tool_call.name
                    tool_args = tool_call.arguments

                    if tool_name in self.tools:
                        result = self.tools[tool_name].execute(**tool_args)
                        self.memory.add("tool", f"[{tool_name}]: {result}")
                    else:
                        self.memory.add("tool",
                            f"[{tool_name}]: Error - Tool not found")
            else:
                self.memory.add("assistant", response.content)
                return response.content

        return "最大反復回数に達しました。処理を中断します。"

    def _get_tool_definitions(self):
        return [
            {
                "name": t.name,
                "description": t.description,
                "parameters": t.parameters_schema
            }
            for t in self.tools.values()
        ]


class MultiAgentOrchestrator:
    def __init__(self, agents, router_prompt):
        self.agents = {a.name: a for a in agents}
        self.router_prompt = router_prompt
        self.llm = agents[0].llm  # ルーター用にも同じLLMを使用
        self.conversation_log = []

    def route(self, user_input):
        prompt = f"""{self.router_prompt}

利用可能なエージェント:
{chr(10).join(f'- {name}: {agent.system_prompt[:100]}' for name, agent in self.agents.items())}

ユーザー入力: {user_input}

どのエージェントに割り振るべきですか？エージェント名のみ答えてください。"""

        response = self.llm.chat(messages=[{"role": "user", "content": prompt}])
        return response.content.strip()

    def run(self, user_input):
        agent_name = self.route(user_input)

        if agent_name not in self.agents:
            return f"エラー: エージェント '{agent_name}' が見つかりません"

        agent = self.agents[agent_name]
        result = agent.run(user_input)

        self.conversation_log.append({
            "input": user_input,
            "routed_to": agent_name,
            "output": result
        })

        return result
```

### 3. 識別した設計パターン

#### パターン1: エージェントループ（ReActパターンの変形）

```
while not done:
    response = llm.chat(context)
    if response.has_tool_calls:
        execute_tools(response.tool_calls)
    else:
        return response  # 完了
```

- **利点**: シンプルで理解しやすい。ツール呼び出しと最終回答の分岐が明確
- **欠点**: 明示的な「思考（Thought）」ステップがない。LLMの判断過程が記録されない

#### パターン2: ストラテジーパターン（Tool クラス）

Tool クラスは `execute` メソッドを共通インターフェースとして、内部で異なる `function` を呼び出します。

- **利点**: 新しいツールの追加が容易。統一されたインターフェースで扱える
- **欠点**: すべてのツールが同じインターフェース（`**kwargs`）で抽象化されるため、型安全性が低い

#### パターン3: ルーターパターン（MultiAgentOrchestrator）

LLMを使ってユーザーの入力を適切なエージェントにルーティングします��

- **利点**: ルーティングロジックをハードコードせず、LLMの判断に委ねられる
- **欠点**: LLMの判断が誤る可能性がある。ルーティングの再現性が低い

```python

    self.messages = self.messages[-self.max_messages:]
```

**改善案**: 重要なメッセージ（ユーザーの最初の質問、ツール結果）を優先保持する。

#### 問題点3: ルーティング結果の検証がない


LLMが余計な文字（「researcherです」「researcher を使いましょう」など）を返す可能性があります。

**改善案**: 構造化出力またはファジーマッチングを使用する。

#### 問題点4: エージェント間の情報共有がない

`MultiAgentOrchestrator` が複数のエージェントを持っていますが、エージェント間で情報を共有する仕組みがありません。

**改善案**: 共有メモリまたはメッセージパッシング機構を追加する。

#### 問題点5: ツール実行にセキュリティ制限がない

どんな引数でもツールに渡せてしまいます。パラメータのバリデーションが行われていません。

**改善案**: `parameters_schema` を使って引数をバリデーションしてから実行する。

```python
# 改善後の Memory クラス

class Memory:
    def __init__(self, max_tokens=100000, summarize_threshold=80000):
        self.messages = []
        self.max_tokens = max_tokens
        self.summarize_threshold = summarize_threshold
        self.summary = None  # 古いメッセージの要約

    def add(self, role, content, importance="normal"):
        self.messages.append({
            "role": role,
            "content": content,
            "importance": importance,  # "critical" / "normal" / "low"
            "timestamp": datetime.now().isoformat()
        })
        self._manage_context()

    def _manage_context(self):
        """コンテキストウィンドウの管理"""
        total_tokens = self._count_tokens()

        if total_tokens > self.summarize_threshold:
            # 重要度の低いメッセージから要約
            critical = [m for m in self.messages if m["importance"] == "critical"]
            recent = self.messages[-10:]  # 直近10件は保持
            to_summarize = [m for m in self.messages
                          if m not in critical and m not in recent]

            if to_summarize:
                self.summary = self._summarize(to_summarize)
                self.messages = critical + recent

    def get_context(self):
        """LLMに渡すコンテキストを構築する"""
        context = []
        if self.summary:
            context.append({
                "role": "system",
                "content": f"これまでの会話の要約: {self.summary}"
            })
        context.extend([
            {"role": m["role"], "content": m["content"]}
            for m in self.messages
        ])
        return context
```

```python
class MultiAgentOrchestrator:
    def __init__(self, agents, router_prompt, fallback_agent=None):
        self.agents = {a.name: a for a in agents}
        self.router_prompt = router_prompt
        self.llm = agents[0].llm
        self.fallback_agent = fallback_agent or agents[0]
        self.conversation_log = []
        self.shared_context = {}  # エージェント間の共有コ��テキスト

    def route(self, user_input):
        """構造化出力を使った安全なルーティング"""
        agent_names = list(self.agents.keys())

        prompt = f"""{self.router_prompt}

利用可能なエージェント:
{chr(10).join(f'- {name}: {agent.system_prompt[:100]}' for name, agent in self.agents.items())}

ユーザー入��: {user_input}

以下のJSON形式で回答してください:
{{"agent": "エージェント名", "reason": "選択理由", "confidence": 0.0-1.0}}
"""

        response = self.llm.chat(
            messages=[{"role": "user", "content": prompt}],
            response_format={"type": "json_object"}  # 構造化出力を強制
        )

        try:
            result = json.loads(response.content)
            agent_name = result.get("agent", "")
            confidence = result.get("confidence", 0)

            # 信頼度が低い場合はフォールバ���ク
            if confidence < 0.5 or agent_name not in self.agents:
                return self.fallback_agent.name

            return agent_name
        except json.JSONDecodeError:
            return self.fallback_agent.name

    def run(self, user_input):
        """共有コンテキスト付きの実行"""
        agent_name = self.route(user_input)
        agent = self.agents[agent_name]

        # 共有コンテキストをエージェントに渡す
        if self.shared_context:
            context_str = json.dumps(self.shared_context, ensure_ascii=False)
            enriched_input = f"{user_input}\n\n[共有コンテキスト]: {context_str}"
        else:
            enriched_input = user_input

        result = agent.run(enriched_input)

        # 結果を共有コンテキストに保存
        self.shared_context[agent_name] = {
            "last_output": result[:500],  # 直近の出力（500文字まで）
            "timestamp": datetime.now().isoformat()
        }

        self.conversation_log.append({
            "input": user_input,
            "routed_to": agent_name,
            "output": result,
            "timestamp": datetime.now().isoformat()
        })

        return result
```

### 2. 既存フレームワークとの比較

| 観点 | 分析対象コード | LangChain/LangGraph | CrewAI | Claude Agent SDK |
|------|-------------|--------------------|---------|--------------------|
| エージェントループ | 単純なwhileループ | LCELチェーン / グラフベースの状態遷移 | タスクベースの逐次/並列実行 | イベント駆動のエージェントループ |
| ツール管理 | dict で保持、バリデーションなし | Toolクラスで型ヒント付き、自動スキーマ生成 | タスクごとにツールを割り当て | ツール定義をJSON Schemaで管理 |
| メモリ | 単純なスライディングウィンドウ | ConversationBufferMemory等の複数戦略 | 共有メモリ + タスク結果の自動受け渡し | コンテキストウィンドウの自動管理 |
| マルチエージェント | LLMルーティング | LangGraphで明示的な状態遷移グラフ | 役割ベースの自動委譲 | オーケストレーション or ハンドオフ |
| エラー処理 | なし | リトライ機構付き | タスクレベルのエラー処理 | ツール実行の自動リトライ |

**主な違いと示唆**:

1. **LangGraph のグラフベースアプローチ**: 分析対象コードの単純なループに対し、LangGraphは状態遷移グラフとしてワークフローを定義します。これにより、分岐・合流・ループの条件を明示的にコントロールでき、複雑なワークフローに適しています。

2. **CrewAI の役割ベース設計**: 分析対象コードのルーティングはメッセージ単位ですが、CrewAI はタスク全体をエージェントに割り当てるため、より自律的な実行が可能です。

3. **Claude Agent SDK のイベント駆動**: 分析対象コードのポーリング型ループに対し、Claude Agent SDKはイベント駆動型で、ツール実行の結果をイベントとして処理します。これにより、非同期処理やストリーミングとの相性が良くなっています。

### 4. パフォーマンスの分析

| 問題 | 影響 | 改善案 |
|------|------|--------|
| 毎回全メッセージをLLMに送信 | トークン使用量が増大、コスト増 | 要約ベースのメモリ管理（上記参照） |
| ルーティングで毎回LLMを呼び出す | レイテンシが増加（ルーティング+実行で2回のLLM呼び出し） | キャッシュ or ルールベースの事前フィルタリング |
| ツール結果を文字列として保存 | 構造化データの情報が失われる | JSON形式でツール結果を保存し、必要に応じてパースする |
| 同期的な逐次処理 | 並列可能なツール呼び出しも逐次実行 | `asyncio.gather` で並列実行可能なツールは同時実行 |

**プロの知見**: コードリーディングで最も重要なスキルは「なぜこう書かれたのか」を考えることです。一見非効率に見える設計にも理由がある場合があります（シンプルさの優先、初期プロトタイプとしての割り切りなど）。批判するだけでなく、設計者の意図を推測した上で「この用途ならこう改善できる」と提案する姿勢が重要です。
