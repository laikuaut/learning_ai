# 実践課題06：星評価コンポーネント ★3

> **難易度**: ★★★☆☆（中級）
> **前提知識**: 第3章（ディレクティブ）、第5章（カスタムディレクティブ）
> **課題の種類**: ミニプロジェクト
> **学習目標**: カスタムディレクティブで再利用可能なUI部品を作成し、スコープ分離（Isolated Scope）の `@`・`=`・`&` バインディングを実践する

---

## 完成イメージ

星評価ディレクティブを作成し、複数の場所で再利用します。

```
┌──────────────────────────────────────┐
│  レストランレビュー                     │
│                                      │
│  ┌────────────────────────────────┐  │
│  │ イタリアンレストラン Bella        │  │
│  │ 料理:    ★★★★☆ (4/5)          │  │
│  │ サービス: ★★★☆☆ (3/5)          │  │
│  │ 雰囲気:  ★★★★★ (5/5)          │  │
│  │ [レビューを送信]                 │  │
│  └────────────────────────────────┘  │
│                                      │
│  ┌────────────────────────────────┐  │
│  │ 寿司処 鶴亀                     │  │
│  │ 料理:    ☆☆☆☆☆ (0/5)          │  │
│  │ サービス: ☆☆☆☆☆ (0/5)          │  │
│  │ 雰囲気:  ☆☆☆☆☆ (0/5)          │  │
│  │ [レビューを送信]                 │  │
│  └────────────────────────────────┘  │
│                                      │
│  送信済みレビュー: 1件                 │
└──────────────────────────────────────┘
```

---

## 課題の要件

1. `<star-rating>` カスタムディレクティブを作成する
2. スコープ分離で以下のバインディングを使う
   - `=`（双方向）: 現在の評価値
   - `@`（文字列）: ラベル（「料理」「サービス」など）
   - `&`（コールバック）: 評価変更時に親へ通知
3. 星をクリックして評価を変更できる
4. 星にホバーするとプレビュー表示が変わる
5. `max` 属性で星の最大数を変更可能にする（デフォルト5）
6. 読み取り専用モード（`readonly`）をサポートする
7. 複数のレストランカードでディレクティブを再利用する

---

## ステップガイド

<details>
<summary>ステップ1：ディレクティブ定義オブジェクト（DDO）の骨格を作る</summary>

```javascript
app.directive('starRating', function() {
  return {
    restrict: 'E',           // 要素として使用
    scope: {
      rating: '=',           // 双方向バインディング
      label: '@',            // 文字列バインディング
      onRate: '&'            // コールバック
    },
    template: '...', 
    link: function(scope, element, attrs) {
      scope.max = parseInt(attrs.max) || 5;
      // ...
    }
  };
});
```

</details>

<details>
<summary>ステップ2：テンプレートで星を表示する</summary>

```html
<span ng-repeat="star in stars track by $index"
      ng-click="!readonly && setRating($index + 1)"
      ng-mouseenter="!readonly && preview($index + 1)"
      ng-mouseleave="!readonly && clearPreview()"
      style="cursor: pointer; font-size: 24px;">
  {{ ($index < (hoverValue || rating)) ? '★' : '☆' }}
</span>
```

</details>

<details>
<summary>ステップ3：コールバック (&) で親に通知する</summary>

```javascript
scope.setRating = function(value) {
  scope.rating = value;
  // & バインディングで親のコールバックを呼ぶ
  scope.onRate({ value: value });
};
```

テンプレートでの使用:
```html
<star-rating rating="restaurant.food" label="料理"
             on-rate="onRatingChange(restaurant.name, '料理', value)">
</star-rating>
```

</details>

</head>
<body ng-controller="ReviewCtrl">

  <h1>レストランレビュー</h1>

  <div class="card" ng-repeat="restaurant in restaurants">
    <div class="card-title">{{ restaurant.name }}</div>

    <star-rating rating="restaurant.food" label="料理"
                 on-rate="onRatingChange(restaurant.name, '料理', value)">
    </star-rating>

    <star-rating rating="restaurant.service" label="サービス"
                 on-rate="onRatingChange(restaurant.name, 'サービス', value)">
    </star-rating>

    <star-rating rating="restaurant.atmosphere" label="雰囲気"
                 on-rate="onRatingChange(restaurant.name, '雰囲気', value)">
    </star-rating>

    <button class="submit-btn" ng-click="submitReview(restaurant)"
            ng-disabled="restaurant.submitted || !isComplete(restaurant)">
      {{ restaurant.submitted ? '送信済み' : 'レビューを送信' }}
    </button>

    <div class="review-summary" ng-if="restaurant.submitted">
      平均: {{ getAverage(restaurant) | number:1 }} / 5
    </div>
  </div>

  <!-- 読み取り専用の例 -->
  <div class="card" ng-if="submittedReviews.length > 0">
    <div class="card-title">送信済みレビュー</div>
    <div ng-repeat="review in submittedReviews" style="margin-bottom: 12px; padding-bottom: 12px; border-bottom: 1px solid #eee;">

      <star-rating rating="review.food" label="料理" readonly="true"></star-rating>
      <star-rating rating="review.service" label="サービス" readonly="true"></star-rating>
      <star-rating rating="review.atmosphere" label="雰囲気" readonly="true"></star-rating>
    </div>
  </div>

  <script>
    var app = angular.module('reviewApp', []);

    // カスタムディレクティブ: star-rating
    app.directive('starRating', function() {
      return {
        restrict: 'E',
        scope: {
          rating: '=',
          label: '@',
          onRate: '&',
          readonly: '@'
        },
        template:
          '<div class="rating-row">' +
            '<span class="rating-label">{{ label }}</span>' +
            '<span class="stars">' +
              '<span ng-repeat="star in stars track by $index" ' +
                'class="star" ' +
                'ng-class="{filled: $index < (hoverValue || rating), empty: $index >= (hoverValue || rating), hover: hoverValue && $index < hoverValue, readonly: isReadonly}" ' +
                'ng-click="!isReadonly && setRating($index + 1)" ' +
                'ng-mouseenter="!isReadonly && preview($index + 1)" ' +
                'ng-mouseleave="!isReadonly && clearPreview()">' +
                '{{ $index < (hoverValue || rating) ? "★" : "☆" }}' +
              '</span>' +
            '</span>' +
            '<span class="rating-value">({{ rating || 0 }}/{{ max }})</span>' +
          '</div>',
        link: function(scope, element, attrs) {
          scope.max = parseInt(attrs.max) || 5;
          scope.isReadonly = (attrs.readonly === 'true');
          scope.hoverValue = 0;

          // 星の配列を作成
          scope.stars = [];
          for (var i = 0; i < scope.max; i++) {
            scope.stars.push(i);
          }

          scope.setRating = function(value) {
            scope.rating = value;
            if (scope.onRate) {
              scope.onRate({ value: value });
            }
          };

          scope.preview = function(value) {
            scope.hoverValue = value;
          };

          scope.clearPreview = function() {
            scope.hoverValue = 0;
          };
        }
      };
    });

    app.controller('ReviewCtrl', function($scope) {
      $scope.restaurants = [
        { name: 'イタリアンレストラン Bella', food: 0, service: 0, atmosphere: 0, submitted: false },
        { name: '寿司処 鶴亀', food: 0, service: 0, atmosphere: 0, submitted: false },
        { name: 'カフェ モーニング', food: 0, service: 0, atmosphere: 0, submitted: false }
      ];
      $scope.submittedReviews = [];

      $scope.onRatingChange = function(name, category, value) {
        console.log(name + ' の ' + category + ' を ' + value + ' に変更');
      };

      $scope.isComplete = function(restaurant) {
        return restaurant.food > 0 && restaurant.service > 0 && restaurant.atmosphere > 0;
      };

      $scope.getAverage = function(restaurant) {
        return (restaurant.food + restaurant.service + restaurant.atmosphere) / 3;
      };

      $scope.submitReview = function(restaurant) {
        restaurant.submitted = true;
        $scope.submittedReviews.push({
          name: restaurant.name,
          food: restaurant.food,
          service: restaurant.service,
          atmosphere: restaurant.atmosphere
        });
      };
    });
  </script>

</body>
</html>

```html
<!DOCTYPE html>
<html ng-app="reviewApp">
<head>
  <meta charset="UTF-8">
  <title>レストランレビュー（改良版）</title>
  <script src="https://ajax.googleapis.com/ajax/libs/angularjs/1.8.3/angular.min.js"></script>
  <style>
    body { font-family: sans-serif; background: #f0f2f5; max-width: 580px; margin: 0 auto; padding: 40px 20px; }

    .btn-submit:disabled { background: #ccc; cursor: not-allowed; }

    .review-card .comment { color: #555; font-size: 13px; font-style: italic; margin-top: 6px; }

  </style>
</head>
<body ng-controller="ReviewCtrl as vm">

  <h1>レストランレビュー</h1>

  <div class="card" ng-repeat="r in vm.restaurants" ng-if="!r.submitted">
    <div class="card-title">{{ r.name }}</div>
    <div class="card-subtitle">{{ r.genre }} ・ {{ r.area }}</div>

    <!-- カスタムディレクティブを再利用 -->
    <star-rating ng-model="r.food" label="料理" max="5"
                 on-change="vm.log(r.name, '料理', value)"></star-rating>
    <star-rating ng-model="r.service" label="サービス" max="5"
                 on-change="vm.log(r.name, 'サービス', value)"></star-rating>

    <div class="card-title">送信済みレビュー（{{ vm.reviews.length }}件）</div>
    <div class="review-card" ng-repeat="rev in vm.reviews">
      <div class="name">{{ rev.name }}</div>
      <star-rating ng-model="rev.food" label="料理" readonly="true"></star-rating>
      <star-rating ng-model="rev.service" label="サービス" readonly="true"></star-rating>
      <star-rating ng-model="rev.atmosphere" label="雰囲気" readonly="true"></star-rating>
      <span class="avg">平均 {{ vm.avg(rev) | number:1 }}</span>
      <div class="comment" ng-if="rev.comment">「{{ rev.comment }}」</div>
    </div>
  </div>

  <div class="empty" ng-if="vm.allSubmitted() && vm.reviews.length === 0">
    レビューするレストランがありません
  </div>

  <script>
    var app = angular.module('reviewApp', []);

    /*
     * star-rating ディレクティブ（改良版）
     * - ngModel との連携（ng-model で値を受け渡し）
     * - require: 'ngModel' でモデルコントローラにアクセス
     * - readonly 対応
     */
    app.directive('starRating', function() {
      return {
        restrict: 'E',
        scope: {
          label: '@',
          max: '@',
          readonly: '@',
          onChange: '&'
        },
        require: 'ngModel',
        template:
          '<div class="rating-row">' +
            '<span class="rating-label">{{ label }}</span>' +
            '<span>' +
              '<span ng-repeat="s in range track by $index" ' +
                'class="star" ' +
                'ng-class="{on: $index < displayValue(), off: $index >= displayValue(), ro: isReadonly}" ' +
                'ng-click="!isReadonly && rate($index + 1)" ' +
                'ng-mouseenter="!isReadonly && hover($index + 1)" ' +
                'ng-mouseleave="!isReadonly && leave()">★</span>' +
            '</span>' +
            '<span class="val">({{ currentRating || 0 }}/{{ maxStars }})</span>' +
          '</div>',
        link: function(scope, element, attrs, ngModel) {
          scope.maxStars = parseInt(scope.max) || 5;
          scope.isReadonly = (scope.readonly === 'true');
          scope.hoverVal = 0;
          scope.currentRating = 0;

          // 星の配列を生成
          scope.range = new Array(scope.maxStars);

          // ngModel から値を受け取る
          ngModel.$render = function() {
            scope.currentRating = ngModel.$viewValue || 0;
          };

          scope.displayValue = function() {
            return scope.hoverVal || scope.currentRating;
          };

          scope.rate = function(val) {
            scope.currentRating = val;
            ngModel.$setViewValue(val);
            if (scope.onChange) {
              scope.onChange({ value: val });
            }
          };

          scope.hover = function(val) { scope.hoverVal = val; };
          scope.leave = function() { scope.hoverVal = 0; };
        }
      };
    });

    app.controller('ReviewCtrl', function() {
      var vm = this;

      vm.restaurants = [
        { name: 'イタリアンレストラン Bella', genre: 'イタリアン', area: '渋谷', food: 0, service: 0, atmosphere: 0, comment: '', submitted: false },
        { name: '寿司処 鶴亀', genre: '和食', area: '銀座', food: 0, service: 0, atmosphere: 0, comment: '', submitted: false },
        { name: 'カフェ モーニング', genre: 'カフェ', area: '代官山', food: 0, service: 0, atmosphere: 0, comment: '', submitted: false }
      ];
      vm.reviews = [];

      vm.log = function(name, cat, value) {
        console.log(name + ': ' + cat + ' → ' + value);
      };

      vm.isComplete = function(r) {
        return r.food > 0 && r.service > 0 && r.atmosphere > 0;
      };

      vm.avg = function(r) {
        return (r.food + r.service + r.atmosphere) / 3;
      };

      vm.submit = function(r) {
        r.submitted = true;
        vm.reviews.push({
          name: r.name,
          food: r.food,
          service: r.service,
          atmosphere: r.atmosphere,
          comment: r.comment
        });
      };

      vm.allSubmitted = function() {
        return vm.restaurants.every(function(r) { return r.submitted; });
      };
    });
  </script>

</body>
</html>
```

**初心者向けとの違い:**
- `require: 'ngModel'` でAngularJSのモデルコントローラと連携（`$render`, `$setViewValue`）
- 星は常に `★` を表示し、色の切り替えで状態を表現（`☆` と `★` の切り替えではなく）
- ホバー時に星がスケールアップするCSSアニメーション
- コメント入力機能を追加（`transclude` の代わりにテンプレート内で対応）
- レストラン情報にジャンル・エリアを追加してリアルなデータ構造に
- `Array.prototype.every` で全件送信済みかを判定
