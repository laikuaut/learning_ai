## 課題の要件

1. `socket` モジュールを使ってTCP接続を試みることでポートの開閉を判定する
2. 指定した範囲のポートをスキャンする
3. 開いているポートのサービス名を可能な限り表示する
4. スキャンの所要時間を計測して表示する
5. 結果のサマリを表示する

## ステップガイド

<details>
<summary>ステップ1：1つのポートの開閉を調べる</summary>

`socket.connect_ex()` を使うと、接続成功なら `0`、失敗ならエラーコードが返ります。

```python
import socket

def check_port(host, port, timeout=0.5):
    """指定ポートが開いているか調べる"""
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    sock.settimeout(timeout)
    result = sock.connect_ex((host, port))
    sock.close()
    return result == 0
```

</details>

<details>
<summary>ステップ2：サービス名を取得する</summary>

`socket.getservbyport()` でポート番号からサービス名を取得できます。

```python
def get_service_name(port):
    """ポート番号からサービス名を取得する"""
    try:
        return socket.getservbyport(port, "tcp")
    except OSError:
        return "不明"
```

</details>

<details>
<summary>ステップ3：範囲をループしてスキャンする</summary>

`for` ループで指定範囲を走査し、結果をリストに記録します。

```python
import time

start_time = time.time()
open_ports = []
for port in range(start_port, end_port + 1):
    if check_port(host, port):
        service = get_service_name(port)
        open_ports.append((port, service))
        print(f"  ポート {port:<5} ({service:<10}): 開いています")
elapsed = time.time() - start_time
```

<summary>解答例（初心者向け）</summary>

```python
# ポートスキャナ
# 学べる内容：TCPソケット接続、ポート番号、サービス名解決
# 実行方法：python port_scanner.py
# 注意：自分が管理するホストに対してのみ使用してください

import socket
import time


def check_port(host, port, timeout=0.5):
    """指定ポートが開いているか調べる"""
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    sock.settimeout(timeout)
    result = sock.connect_ex((host, port))
    sock.close()
    return result == 0


def get_service_name(port):
    """ポート番号からサービス名を取得する"""
    try:
        return socket.getservbyport(port, "tcp")
    except OSError:
        return "不明"


# --- メイン処理 ---
print("===== ポートスキャナ =====")
print()

host = input("対象ホスト（デフォルト: localhost）: ").strip()
if host == "":
    host = "localhost"

# IPアドレスを解決
try:
    ip_address = socket.gethostbyname(host)
except socket.gaierror:
    print(f"エラー: ホスト '{host}' を解決できません。")
    exit()

# スキャン対象のポート（よく使うポートのみ）
well_known_ports = [
    20, 21, 22, 23, 25, 53, 80, 110, 143, 443,
    993, 995, 3306, 3389, 5432, 5900, 8080, 8443,
]

print(f"対象ホスト: {host} ({ip_address})")
print(f"スキャン対象: よく使うポート {len(well_known_ports)}個")
print()
print("スキャン開始...")
print()

start_time = time.time()
open_ports = []
closed_count = 0

for port in well_known_ports:
    service = get_service_name(port)
    if check_port(host, port):
        open_ports.append((port, service))
        print(f"  ポート {port:<5} ({service:<12}): 開いています")
    else:
        closed_count += 1

elapsed = time.time() - start_time

print()
print(f"スキャン完了（所要時間: {elapsed:.1f}秒）")
print()
print("【結果サマリ】")
print(f"  スキャン対象    : {host} ({ip_address})")
print(f"  開いているポート: {len(open_ports)}個")
print(f"  閉じているポート: {closed_count}個")

if open_ports:
    print()
    print("【開いているポート一覧】")
    for port, service in open_ports:
        print(f"  {port:<6} {service}")
```

<details>
<summary>解答例（改良版 ─ 範囲スキャンと進捗表示）</summary>

任意の範囲をスキャンでき、進捗表示も行うバージョンです。

```python
# ポートスキャナ（改良版）
# 学べる内容：TCPソケット、ポート番号、進捗表示
# 実行方法：python port_scanner_v2.py
# 注意：自分が管理するホストに対してのみ使用してください

import socket
import time
import sys


def check_port(host, port, timeout=0.3):
    """指定ポートが開いているか調べる"""
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    sock.settimeout(timeout)
    result = sock.connect_ex((host, port))
    sock.close()
    return result == 0


def get_service_name(port):
    """ポート番号からサービス名を取得する"""
    try:
        return socket.getservbyport(port, "tcp")
    except OSError:
        return "---"


def scan_ports(host, start_port, end_port, timeout=0.3):
    """指定範囲のポートをスキャンする"""
    results = {"open": [], "closed": 0}
    total = end_port - start_port + 1

    for i, port in enumerate(range(start_port, end_port + 1), 1):
        # 進捗表示（同じ行を上書き）
        progress = int(i / total * 50)
        bar = "█" * progress + "░" * (50 - progress)
        sys.stdout.write(f"\r  [{bar}] {i}/{total} ポート {port}")
        sys.stdout.flush()

        if check_port(host, port, timeout):
            service = get_service_name(port)
            results["open"].append((port, service))
        else:
            results["closed"] += 1

    print()  # 進捗表示の改行
    return results


def display_results(host, ip_address, start_port, end_port, results, elapsed):
    """スキャン結果を表示する"""
    print()
    print(f"スキャン完了（所要時間: {elapsed:.1f}秒）")
    print()
    print("【結果サマリ】")
    print(f"  スキャン対象    : {host} ({ip_address})")
    print(f"  スキャン範囲    : {start_port} 〜 {end_port}")
    print(f"  開いているポート: {len(results['open'])}個")
    print(f"  閉じているポート: {results['closed']}個")

    if results["open"]:
        print()
        print("【開いているポート一覧】")
        print(f"  {'ポート':<8}{'サービス':<14}{'説明'}")
        print(f"  {'─' * 36}")
        for port, service in results["open"]:
            print(f"  {port:<8}{service:<14}")


def main():
    print("===== ポートスキャナ =====")
    print()
    print("※ 自分が管理するホストに対してのみ使用してください")
    print()

    host = input("対象ホスト（デフォルト: localhost）: ").strip() or "localhost"

    try:
        ip_address = socket.gethostbyname(host)
    except socket.gaierror:
        print(f"エラー: ホスト '{host}' を解決できません。")
        return

    print()
    print("スキャンモードを選んでください:")
    print("  1. よく使うポートのみ（高速）")
    print("  2. 範囲指定")
    mode = input("選択 (1/2): ").strip()

    if mode == "2":
        start_port = int(input("開始ポート: "))
        end_port = int(input("終了ポート: "))
    else:
        # よく使うポートをスキャン
        start_port = 1
        end_port = 1024

    print()
    print(f"対象: {host} ({ip_address})")
    print(f"範囲: ポート {start_port} 〜 {end_port}")
    print()
    print("スキャン開始...")

    start_time = time.time()
    results = scan_ports(host, start_port, end_port)
    elapsed = time.time() - start_time

    display_results(host, ip_address, start_port, end_port, results, elapsed)


if __name__ == "__main__":
    main()
```

- 関数分割で見通しの良い構造
- `sys.stdout.write()` で同じ行を上書き更新する技法
