## 課題の要件

1. ユーザーから送信元/宛先情報とデータを入力で受け取る
2. アプリケーション層からデータリンク層まで、各層のヘッダ付加をシミュレートする
3. 各層でのPDU（Protocol Data Unit）の構造を視覚的に表示する
4. カプセル化によるサイズの増加を棒グラフで表示する
5. 主要なヘッダフィールドの値を表示する

```python
# パケット構造シミュレータ
# 学べる内容：カプセル化、OSI各層のヘッダ構造、PDU
# 実行方法：python packet_simulator.py

import random


def display_box(title, fields, payload_line, total_size):
    """ボックス形式でPDU構造を表示する"""
    width = 44
    print(f"  ┌{'─' * width}┐")
    print(f"  │ {title:<{width - 1}}│")
    for label, value in fields:
        line = f"  {label:<14}: {value}"
        print(f"  │{line:<{width}}│")
    print(f"  ├{'─' * width}┤")
    print(f"  │ {payload_line:<{width - 1}}│")
    print(f"  └{'─' * width}┘")
    print(f"  合計: {total_size} bytes")
    print()


def display_bar(label, size, max_size):
    """サイズを棒グラフで表示する"""
    bar_length = int(size / max_size * 50)
    bar = "█" * bar_length
    print(f"  {label:<20}: {bar} {size} bytes")


# --- メイン処理 ---
print("===== パケット構造シミュレータ =====")
print()

# --- ユーザー入力 ---
src_ip = input("送信元IP（デフォルト: 192.168.1.10）: ").strip() or "192.168.1.10"
src_port = input("送信元ポート（デフォルト: 50001）: ").strip() or "50001"
src_port = int(src_port)

dst_ip = input("宛先IP（デフォルト: 93.184.216.34）: ").strip() or "93.184.216.34"

dst_port = int(dst_port)
dst_mac = input("宛先MAC（デフォルト: 11:22:33:44:55:66）: ").strip() or "11:22:33:44:55:66"

data = input("送信データ（デフォルト: Hello, Server!）: ").strip() or "Hello, Server!"

print()
print("=" * 50)
print("===== カプセル化の過程 =====")
print("=" * 50)
print()

# --- 第7層：アプリケーション層 ---
app_size = len(data.encode("utf-8"))
print("【第7層 アプリケーション層】")
print(f'  データ: "{data}"')
print(f"  サイズ: {app_size} bytes")
print()

        ("送信元ポート", str(src_port)),
        ("宛先ポート", str(dst_port)),

    segment_size,

packet_size = ip_header_size + segment_size

print(f"  ├{'─' * width}┤")
```

<summary>解答例（改良版 ─ クラスで構造化）</summary>

各層をクラスで表現し、カプセル化の過程をオブジェクト指向で実装したバージョンです。

```python
# パケット構造シミュレータ（改良版）
# 学べる内容：カプセル化、OSI各層、オブジェクト指向設計
# 実行方法：python packet_simulator_v2.py

import random


class Layer:
    """OSIモデルの各層を表す基底クラス"""

    def __init__(self, name, layer_num, header_size):
        self.name = name
        self.layer_num = layer_num
        self.header_size = header_size
        self.fields = {}
        self.payload_size = 0

    def total_size(self):
        return self.header_size + self.payload_size

    def display(self):
        width = 46
        print(f"【第{self.layer_num}層 {self.name}】")
        print(f"  ┌{'─' * width}┐")
        header_title = f"ヘッダ ({self.header_size} bytes)"
        print(f"  │ {header_title:<{width - 1}}│")
        for label, value in self.fields.items():
            line = f"  {label:<16}: {value}"
            print(f"  │{line:<{width}}│")
        print(f"  ├{'─' * width}┤")
        payload_line = f"ペイロード ({self.payload_size} bytes)"
        print(f"  │ {payload_line:<{width - 1}}│")
        print(f"  └{'─' * width}┘")
        print(f"  PDU合計: {self.total_size()} bytes\n")


class ApplicationLayer(Layer):
    def __init__(self, data):
        super().__init__("アプリケーション層", 7, 0)
        self.data = data
        self.payload_size = len(data.encode("utf-8"))
        self.fields = {"データ": f'"{data[:30]}"'}

    def display(self):
        print(f"【第{self.layer_num}層 {self.name}】")
        print(f'  データ: "{self.data}"')
        print(f"  サイズ: {self.payload_size} bytes\n")


class TransportLayer(Layer):
    def __init__(self, src_port, dst_port, payload_size):
        super().__init__("トランスポート層（TCP）", 4, 20)
        self.payload_size = payload_size
        self.fields = {
            "送信元ポート": str(src_port),
            "宛先ポート": str(dst_port),
            "シーケンス番号": str(random.randint(1000, 99999)),
            "ACK番号": "0",
            "ウィンドウサイズ": "65535",
            "フラグ": "PSH, ACK",
        }


class NetworkLayer(Layer):
    def __init__(self, src_ip, dst_ip, payload_size):
        super().__init__("ネットワーク層（IP）", 3, 20)
        self.payload_size = payload_size
        self.fields = {
            "バージョン": "4 (IPv4)",
            "ヘッダ長": "5 (20 bytes)",
            "送信元IP": src_ip,
            "宛先IP": dst_ip,
            "TTL": "64",
            "プロトコル": "6 (TCP)",
            "全長": str(20 + payload_size),
        }


class DataLinkLayer(Layer):
    def __init__(self, src_mac, dst_mac, payload_size):
        super().__init__("データリンク層（Ethernet）", 2, 14)
        self.payload_size = payload_size
        self.fcs_size = 4
        self.fields = {
            "宛先MAC": dst_mac,
            "送信元MAC": src_mac,
            "タイプ": "0x0800 (IPv4)",
        }

    def total_size(self):
        return self.header_size + self.payload_size + self.fcs_size

    def display(self):
        super().display()
        print(f"  ※ FCS {self.fcs_size} bytes を含む合計: {self.total_size()} bytes\n")


def simulate_encapsulation(src_ip, src_port, src_mac, dst_ip, dst_port, dst_mac, data):
    """カプセル化の全過程をシミュレートする"""
    # 各層を生成
    app = ApplicationLayer(data)
    transport = TransportLayer(src_port, dst_port, app.payload_size)
    network = NetworkLayer(src_ip, dst_ip, transport.total_size())
    datalink = DataLinkLayer(src_mac, dst_mac, network.total_size())

    layers = [app, transport, network, datalink]

    # 表示
    print("\n" + "=" * 50)
    print("===== カプセル化の過程 =====")
    print("=" * 50 + "\n")

    for layer in layers:
        layer.display()

    # サイズ推移
    sizes = [
        ("アプリケーション層", app.payload_size),
        ("トランスポート層", transport.total_size()),
        ("ネットワーク層", network.total_size()),
        ("データリンク層", datalink.total_size()),
    ]
    max_size = max(s for _, s in sizes)

    print("===== サイズ推移 =====")
    prev = 0
    for name, size in sizes:
        bar_len = int(size / max_size * 50)
        bar = "█" * bar_len
        added = f" (+{size - prev})" if prev > 0 else ""
        print(f"  {name:<20}: {bar} {size}B{added}")
        prev = size

    # オーバーヘッド
    overhead = datalink.total_size() - app.payload_size
    ratio = overhead / datalink.total_size() * 100
    print(f"\n  オーバーヘッド: {overhead}B / {datalink.total_size()}B ({ratio:.1f}%)")


def main():
    print("===== パケット構造シミュレータ =====\n")

    src_ip = input("送信元IP [192.168.1.10]: ").strip() or "192.168.1.10"
    src_port = int(input("送信元ポート [50001]: ").strip() or "50001")
    src_mac = input("送信元MAC [AA:BB:CC:DD:EE:01]: ").strip() or "AA:BB:CC:DD:EE:01"
    dst_ip = input("宛先IP [93.184.216.34]: ").strip() or "93.184.216.34"
    dst_port = int(input("宛先ポート [80]: ").strip() or "80")
    dst_mac = input("宛先MAC [11:22:33:44:55:66]: ").strip() or "11:22:33:44:55:66"
    data = input("送信データ [Hello, Server!]: ").strip() or "Hello, Server!"

    simulate_encapsulation(src_ip, src_port, src_mac, dst_ip, dst_port, dst_mac, data)


if __name__ == "__main__":
    main()
```

**初心者向けとの違い:**
- オブジェクト指向（クラスの継承）で各層をモデル化
- `Layer` 基底クラスに共通処理をまとめ、各層は固有の情報だけを定義
- 新しい層やプロトコルを追加する際の拡張性が高い
- メソッドのオーバーライド（`total_size()`, `display()`）の実用例
