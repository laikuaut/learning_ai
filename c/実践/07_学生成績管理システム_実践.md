## 課題の要件

1. 学生情報（名前・3科目の点数）を構造体で管理する
2. 最大20人まで登録できる
3. 各学生の平均点を自動計算する
4. 平均点に基づく評価（A/B/C/D/F）を表示する
5. 平均点の高い順にソートできる
6. 名前の一部で検索できる
7. 全体の統計情報（平均・最高・最低）を表示する

## 解答例

<details>
<summary>解答例（初心者向け）</summary>

```c
/*
 * 学生成績管理システム
 * 学べる内容：構造体、構造体配列、関数分割、ソート
 * コンパイル：gcc -o student_mgr student_mgr.c
 * 実行：./student_mgr
 */
#include <stdio.h>
#include <string.h>

#define MAX_STUDENTS 20
#define NAME_LEN 50

/* --- 構造体定義 --- */
typedef struct {
    char name[NAME_LEN];
    int math;
    int english;
    int japanese;
    double average;
} Student;

/* --- 評価を返す --- */
const char *get_grade(double avg) {
    if (avg >= 90.0) return "A";
    if (avg >= 80.0) return "B";
    if (avg >= 70.0) return "C";
    if (avg >= 60.0) return "D";
    return "F";
}

/* --- 一覧表示 --- */
void show_all(Student students[], int count) {
    int i;
    if (count == 0) {
        printf("学生が登録されていません。\n");
        return;
    }
    printf("\n--- 学生一覧 ---\n");
    printf("No.  名前          数学  英語  国語  平均   評価\n");
    printf("------------------------------------------------------\n");
    for (i = 0; i < count; i++) {
        printf("%2d.  %-14s %4d  %4d  %4d  %5.1f   %s\n",
               i + 1,
               students[i].name,
               students[i].math,
               students[i].english,
               students[i].japanese,
               students[i].average,
               get_grade(students[i].average));
    }
}

/* --- 平均点でソート（降順）--- */
void sort_by_average(Student students[], int count) {
    int i, j;
    Student temp;
    for (i = 0; i < count - 1; i++) {
        for (j = 0; j < count - 1 - i; j++) {
            if (students[j].average < students[j + 1].average) {
                temp = students[j];
                students[j] = students[j + 1];
                students[j + 1] = temp;
            }
        }
    }
}

/* --- 名前で検索 --- */
void search_by_name(Student students[], int count, const char *keyword) {
    int i;
    int found = 0;
    for (i = 0; i < count; i++) {
        if (strstr(students[i].name, keyword) != NULL) {
            printf("  %s: 数学%d 英語%d 国語%d 平均%.1f %s\n",
                   students[i].name,
                   students[i].math,
                   students[i].english,
                   students[i].japanese,
                   students[i].average,
                   get_grade(students[i].average));
            found++;
        }
    }
    if (found == 0) {
        printf("該当する学生が見つかりませんでした。\n");
    }
}

/* --- 統計情報 --- */
void show_stats(Student students[], int count) {
    int i;
    double total = 0;
    int max_idx = 0, min_idx = 0;

    if (count == 0) {
        printf("学生が登録されていません。\n");
        return;
    }

    for (i = 0; i < count; i++) {
        total += students[i].average;
        if (students[i].average > students[max_idx].average) {
            max_idx = i;
        }
        if (students[i].average < students[min_idx].average) {
            min_idx = i;
        }
    }

    printf("\n--- 統計情報 ---\n");
    printf("登録人数: %d\n", count);
    printf("全体平均: %.1f点\n", total / count);
    printf("最高平均: %s (%.1f点)\n", students[max_idx].name, students[max_idx].average);
    printf("最低平均: %s (%.1f点)\n", students[min_idx].name, students[min_idx].average);
}

/* --- メイン --- */
int main(void) {
    Student students[MAX_STUDENTS];
    int count = 0;
    int choice;
    char buf[256];

    printf("===== 学生成績管理 =====\n");

    while (1) {
        printf("\n1. 学生を追加\n");
        printf("2. 一覧表示\n");
        printf("3. 成績順にソート\n");
        printf("4. 名前で検索\n");
        printf("5. 統計情報\n");
        printf("0. 終了\n\n");
        printf("> ");
        scanf("%d", &choice);

        switch (choice) {
            case 0:
                printf("終了します。\n");
                return 0;

            case 1:
                if (count >= MAX_STUDENTS) {
                    printf("これ以上登録できません（最大%d人）。\n", MAX_STUDENTS);
                    break;
                }
                /* 改行を読み飛ばす */
                while (getchar() != '\n');

                printf("名前: ");
                fgets(students[count].name, NAME_LEN, stdin);
                /* 末尾の改行を除去 */
                students[count].name[strcspn(students[count].name, "\n")] = '\0';

                printf("数学: ");
                scanf("%d", &students[count].math);
                printf("英語: ");
                scanf("%d", &students[count].english);
                printf("国語: ");
                scanf("%d", &students[count].japanese);

                students[count].average =
                    (students[count].math + students[count].english + students[count].japanese) / 3.0;

                printf("%sさんを追加しました。（平均: %.1f点）\n",
                       students[count].name, students[count].average);
                count++;
                break;

            case 2:
                show_all(students, count);
                break;

            case 3:
                sort_by_average(students, count);
                printf("平均点の高い順にソートしました。\n");
                show_all(students, count);
                break;

            case 4:
                while (getchar() != '\n');
                printf("検索キーワード: ");
                fgets(buf, sizeof(buf), stdin);
                buf[strcspn(buf, "\n")] = '\0';
                search_by_name(students, count, buf);
                break;

            case 5:
                show_stats(students, count);
                break;

            default:
                printf("無効な選択です。\n");
                break;
        }
    }

    return 0;
}
```

<details>
<summary>解答例（改良版 ─ ファイル保存・読込付き）</summary>

第8章（ファイル操作）の知識を活用して、データの保存と読み込みを追加したバージョンです。

```c
/*
 * 学生成績管理システム（改良版）
 * 学べる内容：構造体、ファイルI/O、CSV形式の読み書き
 * コンパイル：gcc -o student_mgr2 student_mgr2.c
 * 実行：./student_mgr2
 */
#include <stdio.h>
#include <string.h>

#define MAX_STUDENTS 20
#define NAME_LEN 50
#define FILENAME "students.csv"

typedef struct {
    char name[NAME_LEN];
    int math;
    int english;
    int japanese;
    double average;
} Student;

const char *get_grade(double avg) {
    if (avg >= 90.0) return "A";
    if (avg >= 80.0) return "B";
    if (avg >= 70.0) return "C";
    if (avg >= 60.0) return "D";
    return "F";
}

void calc_average(Student *s) {
    s->average = (s->math + s->english + s->japanese) / 3.0;
}

void show_all(Student students[], int count) {
    int i;
    if (count == 0) {
        printf("学生が登録されていません。\n");
        return;
    }
    printf("\nNo.  名前          数学  英語  国語  平均   評価\n");
    printf("------------------------------------------------------\n");
    for (i = 0; i < count; i++) {
        printf("%2d.  %-14s %4d  %4d  %4d  %5.1f   %s\n",
               i + 1, students[i].name,
               students[i].math, students[i].english,
               students[i].japanese, students[i].average,
               get_grade(students[i].average));
    }
}

/* CSV形式で保存 */
int save_to_file(Student students[], int count, const char *filename) {
    FILE *fp = fopen(filename, "w");
    int i;
    if (fp == NULL) {
        printf("ファイルを開けませんでした: %s\n", filename);
        return 0;
    }
    fprintf(fp, "名前,数学,英語,国語\n");  /* ヘッダ行 */
    for (i = 0; i < count; i++) {
        fprintf(fp, "%s,%d,%d,%d\n",
                students[i].name,
                students[i].math,
                students[i].english,
                students[i].japanese);
    }
    fclose(fp);
    printf("%s に %d 件保存しました。\n", filename, count);
    return 1;
}

/* CSV形式で読み込み */
int load_from_file(Student students[], int *count, const char *filename) {
    FILE *fp = fopen(filename, "r");
    char line[256];
    if (fp == NULL) {
        printf("ファイルが見つかりません: %s\n", filename);
        return 0;
    }

    *count = 0;
    /* ヘッダ行をスキップ */
    fgets(line, sizeof(line), fp);

    while (fgets(line, sizeof(line), fp) != NULL && *count < MAX_STUDENTS) {
        line[strcspn(line, "\n")] = '\0';
        if (sscanf(line, "%[^,],%d,%d,%d",
                   students[*count].name,
                   &students[*count].math,
                   &students[*count].english,
                   &students[*count].japanese) == 4) {
            calc_average(&students[*count]);
            (*count)++;
        }
    }
    fclose(fp);
    printf("%s から %d 件読み込みました。\n", filename, *count);
    return 1;
}

int main(void) {
    Student students[MAX_STUDENTS];
    int count = 0;
    int choice;
    char buf[256];
    int i;

    printf("===== 学生成績管理（改良版）=====\n");

    while (1) {
        printf("\n1.追加 2.一覧 3.ソート 4.検索 5.統計 6.保存 7.読込 0.終了\n> ");
        scanf("%d", &choice);

        if (choice == 0) {
            printf("終了します。\n");
            break;
        } else if (choice == 1) {
            if (count >= MAX_STUDENTS) {
                printf("最大人数に達しています。\n");
                continue;
            }
            while (getchar() != '\n');
            printf("名前: ");
            fgets(students[count].name, NAME_LEN, stdin);
            students[count].name[strcspn(students[count].name, "\n")] = '\0';
            printf("数学: "); scanf("%d", &students[count].math);
            printf("英語: "); scanf("%d", &students[count].english);
            printf("国語: "); scanf("%d", &students[count].japanese);
            calc_average(&students[count]);
            printf("%sさんを追加しました。（平均: %.1f点）\n",
                   students[count].name, students[count].average);
            count++;
        } else if (choice == 2) {
            show_all(students, count);
        } else if (choice == 3) {
            /* バブルソート（降順） */
            for (i = 0; i < count - 1; i++) {
                int j;
                for (j = 0; j < count - 1 - i; j++) {
                    if (students[j].average < students[j + 1].average) {
                        Student temp = students[j];
                        students[j] = students[j + 1];
                        students[j + 1] = temp;
                    }
                }
            }
            printf("ソートしました。\n");
            show_all(students, count);
        } else if (choice == 6) {
            save_to_file(students, count, FILENAME);
        } else if (choice == 7) {
            load_from_file(students, &count, FILENAME);
        }
    }

    return 0;
}
```

**初心者向けとの違い:**
- CSV形式でファイル保存・読み込み → プログラム終了後もデータが残る
- `sscanf` と `%[^,]` でCSVのパースを実現 → 区切り文字を使った文字列解析のテクニック
- `calc_average` を独立関数にして読み込み時にも再利用 → DRY（Don't Repeat Yourself）の原則
