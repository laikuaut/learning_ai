---

## 課題の要件

1. 情報資産（asset）の登録機能（名前、種類、重要度）を実装する
2. 脅威（threat）の登録機能（名前、種類、発生可能性）を実装する
3. 脆弱性（vulnerability）の登録機能（名前、関連資産、深刻度）を実装する
4. リスク値を「影響度 x 発生可能性」で算出し、リスクレベルを判定する
5. リスクマトリクス（5x5）をテキストで表示する
6. リスク対応方針（低減・回避・移転・受容）を提案する
7. 評価結果をテキストファイルにレポート出力する

## ステップガイド

<details>
<summary>ステップ1：リスク値の算出方法を理解する</summary>

リスク値は一般に以下の式で求めます。

```
リスク値 = 影響度（Impact） × 発生可能性（Likelihood）
```

影響度と発生可能性はそれぞれ1〜5の5段階で評価します。

| スコア | 影響度 | 発生可能性 |
|:---:|---|---|
| 1 | 軽微 | 極めて低い |
| 2 | 小さい | 低い |
| 3 | 中程度 | 中程度 |
| 4 | 大きい | 高い |
| 5 | 壊滅的 | 極めて高い |

リスク値の判定基準:
- 1〜4: 低（受容可能）
- 5〜9: 中
- 10〜16: 高
- 17〜25: 極高（即座に対応が必要）

```python
# リスクアセスメントツール
# 学べる内容：リスクアセスメント、脅威分析、セキュリティマネジメント
# 実行方法：python risk_assessment.py

from datetime import datetime

# --- データ格納 ---
assets = []       # 情報資産
threats = []      # 脅威
vulnerabilities = []  # 脆弱性
risk_entries = []     # リスク評価結果


def get_risk_level(risk_value):
    """リスク値からリスクレベルを返します。"""
    if risk_value >= 17:
        return "極高"
    elif risk_value >= 10:
        return "高"
    elif risk_value >= 5:
        return "中"
    else:
        return "低"


def get_risk_response(risk_value):
    """リスク値に応じた対応方針を返します。"""
    if risk_value >= 17:
        return "リスク回避（該当システムの停止・廃止を検討）"
    elif risk_value >= 10:
        return "リスク低減（対策の実施が必要）"
    elif risk_value >= 5:
        return "リスク移転（保険の検討）またはリスク低減"
    else:
        return "リスク受容（現状維持で監視）"


def register_asset():
    """情報資産を登録します。"""
    name = input("資産名: ")
    asset_type = input("種類 (データ/ハードウェア/ソフトウェア/ネットワーク): ")
    while True:
        try:
            importance = int(input("重要度 (1-5): "))
            if 1 <= importance <= 5:
                break
            print("1〜5の範囲で入力してください。")
        except ValueError:
            print("数値を入力してください。")

    assets.append({"name": name, "type": asset_type, "importance": importance})
    print(f"資産 '{name}' を登録しました。")


def register_threat():
    """脅威を登録します。"""
    name = input("脅威名: ")
    threat_type = input("種類 (技術的/人的/環境的): ")
    while True:
        try:
            likelihood = int(input("発生可能性 (1-5): "))
            if 1 <= likelihood <= 5:
                break
            print("1〜5の範囲で入力してください。")
        except ValueError:
            print("数値を入力してください。")

    threats.append({"name": name, "type": threat_type, "likelihood": likelihood})
    print(f"脅威 '{name}' を登録しました。")


def register_vulnerability():
    """脆弱性を登録します。"""
    if not assets:
        print("先に情報資産を登録してください。")
        return

    name = input("脆弱性名: ")
    print("関連する資産:")
    for i, asset in enumerate(assets):
        print(f"  {i + 1}. {asset['name']}")
    while True:
        try:
            asset_idx = int(input("番号: ")) - 1
            if 0 <= asset_idx < len(assets):
                break
            print("有効な番号を入力してください。")
        except ValueError:
            print("数値を入力してください。")

    while True:
        try:
            severity = int(input("深刻度 (1-5): "))
            if 1 <= severity <= 5:
                break
            print("1〜5の範囲で入力してください。")
        except ValueError:
            print("数値を入力してください。")

    vulnerabilities.append({
        "name": name,
        "asset": assets[asset_idx]["name"],
        "severity": severity,
    })
    print(f"脆弱性 '{name}' を登録しました。")


def evaluate_risks():
    """リスク評価を実行します。"""
    if not assets or not threats or not vulnerabilities:
        print("資産・脅威・脆弱性をすべて登録してからリスク評価を行ってください。")
        return

    risk_entries.clear()

    print("\n--- リスク評価結果 ---")
    for vuln in vulnerabilities:
        # 関連資産の重要度を取得
        asset_importance = 3  # デフォルト
        for asset in assets:
            if asset["name"] == vuln["asset"]:
                asset_importance = asset["importance"]
                break

        for threat in threats:
            # 影響度 = 資産の重要度 × 脆弱性の深刻度 / 5（5段階に正規化）
            impact = min(5, max(1, round(asset_importance * vuln["severity"] / 5)))
            likelihood = threat["likelihood"]
            risk_value = impact * likelihood
            risk_level = get_risk_level(risk_value)
            response = get_risk_response(risk_value)

            entry = {
                "asset": vuln["asset"],
                "threat": threat["name"],
                "vulnerability": vuln["name"],
                "impact": impact,
                "likelihood": likelihood,
                "risk_value": risk_value,
                "risk_level": risk_level,
                "response": response,
            }
            risk_entries.append(entry)

            print(f"\n  資産: {entry['asset']}")
            print(f"  脅威: {entry['threat']}  脆弱性: {entry['vulnerability']}")
            print(f"  影響度: {impact}  発生可能性: {likelihood}  リスク値: {risk_value} [{risk_level}]")
            print(f"  推奨対応: {response}")


def show_risk_matrix():
    """リスクマトリクスを表示します。"""
    print("\n--- リスクマトリクス（影響度 × 発生可能性） ---")
    print()
    print("影響度 \\  発生可能性")
    print("        1     2     3     4     5")
    print("      +-----+-----+-----+-----+-----+")

    for impact in range(5, 0, -1):
        row = f"  {impact}   |"
        for likelihood in range(1, 6):
            value = impact * likelihood
            level = get_risk_level(value)
            # レベルに応じた表示
            if level == "極高":
                marker = f"!{value:>2}!"
            elif level == "高":
                marker = f"*{value:>2}*"
            elif level == "中":
                marker = f" {value:>2} "
            else:
                marker = f" {value:>2} "
            row += f" {marker}|"
        print(row)
        print("      +-----+-----+-----+-----+-----+")

    print()
    print("  凡例: !XX! = 極高, *XX* = 高, その他 = 中/低")


def suggest_responses():
    """リスク対応方針を表示します。"""
    if not risk_entries:
        print("先にリスク評価を実行してください。")
        return

    print("\n--- リスク対応方針 ---")

    # リスク値が高い順にソート
    sorted_risks = sorted(risk_entries, key=lambda x: x["risk_value"], reverse=True)

    print("\n【リスク対応の4つの選択肢】")
    print("  1. リスク低減（Reduction）: 対策を実施してリスクを下げる")
    print("  2. リスク回避（Avoidance）: リスクの原因そのものを排除する")
    print("  3. リスク移転（Transfer）:  保険やアウトソーシングでリスクを他者に移す")
    print("  4. リスク受容（Acceptance）: リスクが小さい場合、そのまま受け入れる")

    print("\n【優先対応リスト（リスク値の降順）】")
    for i, entry in enumerate(sorted_risks, 1):
        print(f"\n  {i}. [{entry['risk_level']}] リスク値: {entry['risk_value']}")
        print(f"     資産: {entry['asset']}")
        print(f"     脅威: {entry['threat']} / 脆弱性: {entry['vulnerability']}")
        print(f"     推奨: {entry['response']}")


def export_report():
    """レポートをファイルに出力します。"""
    if not risk_entries:
        print("先にリスク評価を実行してください。")
        return

    filename = "risk_assessment_report.txt"
    with open(filename, "w", encoding="utf-8") as f:
        f.write("=" * 60 + "\n")
        f.write("  リスクアセスメントレポート\n")
        f.write(f"  作成日: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}\n")
        f.write("=" * 60 + "\n\n")

        f.write("--- 登録資産 ---\n")
        for asset in assets:
            f.write(f"  {asset['name']} (種類: {asset['type']}, 重要度: {asset['importance']})\n")

        f.write("\n--- 脅威 ---\n")
        for threat in threats:
            f.write(f"  {threat['name']} (種類: {threat['type']}, 発生可能性: {threat['likelihood']})\n")

        f.write("\n--- 脆弱性 ---\n")
        for vuln in vulnerabilities:
            f.write(f"  {vuln['name']} (資産: {vuln['asset']}, 深刻度: {vuln['severity']})\n")

        f.write("\n--- リスク評価結果 ---\n")
        sorted_risks = sorted(risk_entries, key=lambda x: x["risk_value"], reverse=True)
        for entry in sorted_risks:
            f.write(f"\n  [{entry['risk_level']}] リスク値: {entry['risk_value']}\n")
            f.write(f"  資産: {entry['asset']}\n")
            f.write(f"  脅威: {entry['threat']} / 脆弱性: {entry['vulnerability']}\n")
            f.write(f"  影響度: {entry['impact']} × 発生可能性: {entry['likelihood']}\n")
            f.write(f"  推奨対応: {entry['response']}\n")

        # サマリ
        high_risks = sum(1 for e in risk_entries if e["risk_value"] >= 10)
        f.write(f"\n--- サマリ ---\n")
        f.write(f"  総リスク項目: {len(risk_entries)}\n")
        f.write(f"  高リスク以上: {high_risks}\n")

    print(f"レポートを {filename} に出力しました。")


def load_sample_data():
    """サンプルデータを読み込みます。"""
    assets.clear()
    threats.clear()
    vulnerabilities.clear()

    # サンプル資産

# --- メイン処理 ---
print("===== リスクアセスメントツール =====")

while True:
    print("\n1. 情報資産の登録    2. 脅威の登録")
```

```python
# リスクアセスメントツール（改良版）
# 学べる内容：オブジェクト指向設計、リスクマネジメント、JSON出力
# 実行方法：python risk_assessment_v2.py

import json
from datetime import datetime


class Asset:
    """情報資産を表すクラスです。"""
    TYPES = ["データ", "ハードウェア", "ソフトウェア", "ネットワーク", "人的", "物理的"]

    def __init__(self, name, asset_type, importance):
        self.name = name
        self.asset_type = asset_type
        self.importance = max(1, min(5, importance))

    def to_dict(self):
        return {"name": self.name, "type": self.asset_type, "importance": self.importance}


class Threat:
    """脅威を表すクラスです。"""
    TYPES = ["技術的", "人的", "環境的"]

    def __init__(self, name, threat_type, likelihood):
        self.name = name
        self.threat_type = threat_type
        self.likelihood = max(1, min(5, likelihood))

    def to_dict(self):
        return {"name": self.name, "type": self.threat_type, "likelihood": self.likelihood}


class Vulnerability:
    """脆弱性を表すクラスです。"""

    def __init__(self, name, asset_name, severity):
        self.name = name
        self.asset_name = asset_name
        self.severity = max(1, min(5, severity))

    def to_dict(self):
        return {"name": self.name, "asset": self.asset_name, "severity": self.severity}


class RiskEntry:
    """1件のリスク評価結果を表すクラスです。"""

    LEVELS = {
        range(1, 5): ("低", "リスク受容"),
        range(5, 10): ("中", "リスク移転またはリスク低減"),
        range(10, 17): ("高", "リスク低減（対策の実施が必要）"),
        range(17, 26): ("極高", "リスク回避（即座に対応が必要）"),
    }

    def __init__(self, asset, threat, vulnerability, impact, likelihood):
        self.asset = asset
        self.threat = threat
        self.vulnerability = vulnerability
        self.impact = impact
        self.likelihood = likelihood
        self.risk_value = impact * likelihood

    @property
    def risk_level(self):
        for rng, (level, _) in self.LEVELS.items():
            if self.risk_value in rng:
                return level
        return "不明"

    @property
    def response(self):
        for rng, (_, resp) in self.LEVELS.items():
            if self.risk_value in rng:
                return resp
        return "要確認"

    def to_dict(self):
        return {
            "asset": self.asset,
            "threat": self.threat,
            "vulnerability": self.vulnerability,
            "impact": self.impact,
            "likelihood": self.likelihood,
            "risk_value": self.risk_value,
            "risk_level": self.risk_level,
            "response": self.response,
        }


class RiskAssessment:
    """リスクアセスメントを管理するクラスです。"""

    def __init__(self):
        self.assets = []
        self.threats = []
        self.vulnerabilities = []
        self.risk_entries = []

    def add_asset(self, asset):
        self.assets.append(asset)

    def add_threat(self, threat):
        self.threats.append(threat)

    def add_vulnerability(self, vuln):
        self.vulnerabilities.append(vuln)

    def evaluate(self):
        """全リスクを評価します。"""
        self.risk_entries.clear()
        asset_map = {a.name: a for a in self.assets}

        for vuln in self.vulnerabilities:
            asset = asset_map.get(vuln.asset_name)
            if not asset:
                continue
            for threat in self.threats:
                impact = min(5, max(1, round(asset.importance * vuln.severity / 5)))
                entry = RiskEntry(
                    asset.name, threat.name, vuln.name,
                    impact, threat.likelihood,
                )
                self.risk_entries.append(entry)

        self.risk_entries.sort(key=lambda e: e.risk_value, reverse=True)
        return self.risk_entries

    def get_summary(self):
        """サマリ統計を返します。"""
        if not self.risk_entries:
            return {}
        total = len(self.risk_entries)
        levels = {"極高": 0, "高": 0, "中": 0, "低": 0}
        for e in self.risk_entries:
            levels[e.risk_level] = levels.get(e.risk_level, 0) + 1
        avg = sum(e.risk_value for e in self.risk_entries) / total
        return {"total": total, "levels": levels, "avg_risk": round(avg, 1)}

    def print_matrix(self):
        """リスクマトリクスを表示します。"""
        print("\n  影響度 \\ 発生可能性  1    2    3    4    5")
        print("        +----+----+----+----+----+")
        for impact in range(5, 0, -1):
            row = f"    {impact}   |"
            for lh in range(1, 6):
                val = impact * lh
                entry = RiskEntry("", "", "", impact, lh)
                lvl = entry.risk_level
                sym = {"極高": "!", "高": "*", "中": " ", "低": " "}.get(lvl, " ")
                row += f"{sym}{val:>2}{sym} |"
            print(row)
            print("        +----+----+----+----+----+")

    def export_json(self, filename):
        """JSON形式でレポートを出力します。"""
        report = {
            "generated_at": datetime.now().isoformat(),
            "assets": [a.to_dict() for a in self.assets],
            "threats": [t.to_dict() for t in self.threats],
            "vulnerabilities": [v.to_dict() for v in self.vulnerabilities],
            "risks": [e.to_dict() for e in self.risk_entries],
            "summary": self.get_summary(),
        }
        with open(filename, "w", encoding="utf-8") as f:
            json.dump(report, f, ensure_ascii=False, indent=2)
        print(f"JSONレポートを {filename} に出力しました。")

    def load_sample(self):
        """サンプルデータを読み込みます。"""
        self.assets = [

def input_int(prompt, min_val, max_val):
    """範囲指定付き整数入力ヘルパーです。"""
    while True:
        try:
            val = int(input(prompt))
            if min_val <= val <= max_val:
                return val
            print(f"{min_val}〜{max_val}の範囲で入力してください。")
        except ValueError:
            print("数値を入力してください。")


def main():
    ra = RiskAssessment()
    print("===== リスクアセスメントツール =====")

    while True:
        print("\n1.資産登録  2.脅威登録  3.脆弱性登録  4.リスク評価")
        print("5.マトリクス  6.対応方針  7.レポート出力  8.サンプル  9.終了")
        choice = input("\n選択 (1-9): ")

        if choice == "1":
            name = input("資産名: ")
            atype = input("種類 (データ/ハードウェア/ソフトウェア/ネットワーク): ")
            imp = input_int("重要度 (1-5): ", 1, 5)
            ra.add_asset(Asset(name, atype, imp))
            print(f"資産 '{name}' を登録しました。")

        elif choice == "2":
            name = input("脅威名: ")
            ttype = input("種類 (技術的/人的/環境的): ")
            lh = input_int("発生可能性 (1-5): ", 1, 5)
            ra.add_threat(Threat(name, ttype, lh))
            print(f"脅威 '{name}' を登録しました。")

        elif choice == "3":
            if not ra.assets:
                print("先に資産を登録してください。")
                continue
            name = input("脆弱性名: ")
            for i, a in enumerate(ra.assets, 1):
                print(f"  {i}. {a.name}")
            idx = input_int("関連資産の番号: ", 1, len(ra.assets)) - 1
            sev = input_int("深刻度 (1-5): ", 1, 5)
            ra.add_vulnerability(Vulnerability(name, ra.assets[idx].name, sev))
            print(f"脆弱性 '{name}' を登録しました。")

        elif choice == "4":
            entries = ra.evaluate()
            if not entries:
                print("データが不足しています。")
                continue
            print(f"\n--- リスク評価結果（{len(entries)}件） ---")
            for e in entries:
                print(f"\n  [{e.risk_level}] リスク値: {e.risk_value}")
                print(f"  資産: {e.asset} / 脅威: {e.threat} / 脆弱性: {e.vulnerability}")
                print(f"  推奨: {e.response}")

        elif choice == "5":
            ra.print_matrix()

        elif choice == "6":
            if not ra.risk_entries:
                print("先にリスク評価を実行してください。")
                continue
            s = ra.get_summary()
            print(f"\n--- サマリ ---")
            print(f"  総項目: {s['total']}  平均リスク値: {s['avg_risk']}")
            for level, count in s["levels"].items():
                print(f"  {level}: {count}件")

        elif choice == "7":
            ra.export_json("risk_report.json")

        elif choice == "8":
            ra.load_sample()
            print("サンプルデータを読み込みました。")

        elif choice == "9":
            print("終了します。")
            break


if __name__ == "__main__":
    main()
```

- `Asset`, `Threat`, `Vulnerability`, `RiskEntry` をクラスで定義 → データの意味と操作を明確に
- `RiskAssessment` クラスで評価ロジックを集約 → 再利用しやすい設計
- JSON形式でレポート出力 → 他のツールとの連携が容易
- `to_dict()` メソッドでシリアライズに対応 → APIやファイル出力に統一的に使える
- `input_int()` ヘルパー関数で入力検証を共通化 → コードの重複を排除
