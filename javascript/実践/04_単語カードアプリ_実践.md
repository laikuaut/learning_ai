# 実践課題04：単語カードアプリ ★2

> **難易度**: ★★☆☆☆（基礎）
> **前提知識**: 第1章〜第6章（変数・演算子・条件分岐・関数・配列・オブジェクト）
> **課題の種類**: ミニプロジェクト
> **学習目標**: オブジェクトの配列を活用したデータ管理と、関数による処理の分割を実践する

---

## 完成イメージ

```
===== 単語カード =====

┌─────────────────────────────┐
│                             │
│         apple               │  ← クリックすると裏面が表示
│                             │
└─────────────────────────────┘

[前へ] カード 1 / 5 [次へ]

正解: 3 / 5  不正解: 2 / 5
[シャッフル] [リセット]
```

カードをクリックすると裏面（日本語訳）が表示され、「知ってた」「知らなかった」を記録できます。

---

## 課題の要件

1. HTMLファイル1つで完結すること
2. 5枚以上の単語カード（英単語と日本語訳のペア）をオブジェクトの配列で管理する
3. カードをクリックすると表/裏が切り替わる
4. 「前へ」「次へ」ボタンでカードを移動できる
5. 「正解」「不正解」ボタンで学習結果を記録し、集計を表示する
6. 「シャッフル」ボタンでカードの順番をランダムに並び替える

---

<details>
<summary>ステップ2：カード表示と裏返し機能を作る</summary>

現在何枚目かを変数で管理し、クリックで表/裏を切り替えます。

```javascript
let currentIndex = 0;
let isFlipped = false;

function flipCard() {
  isFlipped = !isFlipped;
  // isFlippedに応じてfrontまたはbackを表示
  showCard();
}

function showCard() {
  const card = cards[currentIndex];
  const text = isFlipped ? card.back : card.front;
  document.getElementById("cardText").textContent = text;
}
```

<details>
<summary>ステップ3：ナビゲーションとシャッフルを実装する</summary>

```javascript
function nextCard() {
  if (currentIndex < cards.length - 1) {
    currentIndex = currentIndex + 1;
    isFlipped = false;
    showCard();
  }
}

// シャッフル（Fisher-Yatesアルゴリズム）
function shuffle() {
  for (let i = cards.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    // cards[i] と cards[j] を入れ替え
    const temp = cards[i];
    cards[i] = cards[j];
    cards[j] = temp;
  }
}
```

```html
<body>
  <h1>単語カード</h1>
  <p class="hint">カードをクリックして裏返す</p>

  <div id="card" onclick="flipCard()">
    <span id="cardText"></span>
  </div>

  <div class="nav">
    <button onclick="prevCard()">前へ</button>
    <span id="progress">1 / 5</span>
    <button onclick="nextCard()">次へ</button>
  </div>

  <div class="actions">
    <button class="correct" onclick="markResult(true)">知ってた</button>
    <button class="wrong" onclick="markResult(false)">知らなかった</button>
  </div>

  <div class="actions">
    <button onclick="shuffleCards()">シャッフル</button>
    <button onclick="resetAll()">リセット</button>
  </div>

  <p id="stats"></p>

  <script>
    // カードデータ（オブジェクトの配列）
    const cards = [
      { front: "apple",       back: "りんご" },
      { front: "computer",    back: "コンピュータ" },
      { front: "beautiful",   back: "美しい" },
      { front: "run",         back: "走る" },
      { front: "happy",       back: "幸せな" },
      { front: "important",   back: "重要な" },
      { front: "environment", back: "環境" },
    ];

    // 現在の状態を管理する変数
    let currentIndex = 0;
    let isFlipped = false;
    let correctCount = 0;
    let wrongCount = 0;

    // カードを表示する関数
    function showCard() {
      const card = cards[currentIndex];
      const cardEl = document.getElementById("card");

      if (isFlipped) {
        document.getElementById("cardText").textContent = card.back;
        cardEl.className = "flipped";
      } else {
        document.getElementById("cardText").textContent = card.front;
        cardEl.className = "";
      }

      // 進捗を更新
      document.getElementById("progress").textContent =
        (currentIndex + 1) + " / " + cards.length;
    }

    // カードを裏返す関数
    function flipCard() {
      isFlipped = !isFlipped;
      showCard();
    }

    // 次のカードに進む関数
    function nextCard() {
      if (currentIndex < cards.length - 1) {
        currentIndex = currentIndex + 1;
        isFlipped = false;
        showCard();
      }
    }

    // 前のカードに戻る関数
    function prevCard() {
      if (currentIndex > 0) {
        currentIndex = currentIndex - 1;
        isFlipped = false;
        showCard();
      }
    }

    // 結果を記録する関数
    function markResult(isCorrect) {
      if (isCorrect) {
        correctCount = correctCount + 1;
      } else {
        wrongCount = wrongCount + 1;
      }
      updateStats();

      // 次のカードに自動で進む
      if (currentIndex < cards.length - 1) {
        currentIndex = currentIndex + 1;
        isFlipped = false;
        showCard();
      }
    }

    // 統計を更新する関数
    function updateStats() {
      document.getElementById("stats").textContent =
        "正解: " + correctCount + " / " + cards.length +
        "   不正解: " + wrongCount + " / " + cards.length;
    }

    // カードをシャッフルする関数（Fisher-Yatesアルゴリズム）
    function shuffleCards() {
      for (let i = cards.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1));
        const temp = cards[i];
        cards[i] = cards[j];
        cards[j] = temp;
      }
      currentIndex = 0;
      isFlipped = false;
      showCard();
    }

    // 全てリセットする関数
    function resetAll() {
      currentIndex = 0;
      isFlipped = false;
      correctCount = 0;
      wrongCount = 0;
      updateStats();
      showCard();
    }

    // 初期表示
    showCard();
    updateStats();
  </script>
</body>
</html>
```

<summary>解答例（改良版 ─ 第7章・第12章の知識を先取り）</summary>

文字列操作やES6+の機能（分割代入、アロー関数、スプレッド構文）を活用した改良版です。

```html
<!DOCTYPE html>
<html lang="ja">
<head>
  <meta charset="UTF-8">
  <title>単語カード（改良版）</title>
  <style>
    body { font-family: sans-serif; text-align: center; margin-top: 30px; background: #f5f5f5; }
    #card {
      width: 340px; height: 200px; margin: 20px auto;
      background: white; border: 2px solid #333; border-radius: 12px;
      display: flex; flex-direction: column; align-items: center; justify-content: center;
      cursor: pointer; transition: all 0.3s;
      box-shadow: 2px 4px 10px rgba(0,0,0,0.1);
    }
    #card.flipped { background: #e3f2fd; border-color: #1565c0; }
    .card-label { font-size: 12px; color: #999; margin-bottom: 8px; }
    .card-text { font-size: 32px; font-weight: bold; }
    button { font-size: 14px; margin: 4px; padding: 8px 16px; cursor: pointer;
             border-radius: 4px; border: 1px solid #ccc; }
    .correct { background: #c8e6c9; }
    .wrong { background: #ffcdd2; }
    .progress-bar { width: 340px; height: 6px; background: #ddd; margin: 10px auto; border-radius: 3px; }
    .progress-fill { height: 100%; background: #4caf50; border-radius: 3px; transition: width 0.3s; }
    #stats { margin: 10px; color: #555; }
  </style>
</head>
<body>
  <h1>単語カード</h1>

  <div id="card" onclick="app.flip()">
    <div class="card-label" id="cardLabel">英語</div>
    <div class="card-text" id="cardText"></div>
  </div>

  <div class="progress-bar"><div class="progress-fill" id="progressBar"></div></div>
  <p id="progress"></p>

  <div>
    <button onclick="app.prev()">前へ</button>
    <button class="correct" onclick="app.mark(true)">知ってた</button>
    <button class="wrong" onclick="app.mark(false)">知らなかった</button>
    <button onclick="app.next()">次へ</button>
  </div>

  <div style="margin-top: 16px;">
    <button onclick="app.shuffle()">シャッフル</button>
    <button onclick="app.reset()">リセット</button>
  </div>

  <p id="stats"></p>

  <script>
    const app = {
      // カードデータ
      cards: [
        { front: "apple",       back: "りんご",       result: null },
        { front: "computer",    back: "コンピュータ", result: null },
        { front: "beautiful",   back: "美しい",       result: null },
        { front: "run",         back: "走る",         result: null },
        { front: "happy",       back: "幸せな",       result: null },
        { front: "important",   back: "重要な",       result: null },
        { front: "environment", back: "環境",         result: null },
      ],

      index: 0,
      flipped: false,

      // カード表示
      flip() {
        this.flipped = !this.flipped;
        this.render();
      },

      next() {
        if (this.index < this.cards.length - 1) {
          this.index++;
          this.flipped = false;
          this.render();
        }
      },

      prev() {
        if (this.index > 0) {
          this.index--;
          this.flipped = false;
          this.render();
        }
      },

      mark(correct) {
        this.cards[this.index].result = correct;
        // 自動で次へ
        if (this.index < this.cards.length - 1) {
          this.index++;
          this.flipped = false;
        }
        this.render();
      },

      shuffle() {
        // スプレッド構文ではなく直接シャッフル
        for (let i = this.cards.length - 1; i > 0; i--) {
          const j = Math.floor(Math.random() * (i + 1));
          const temp = this.cards[i];
          this.cards[i] = this.cards[j];
          this.cards[j] = temp;
        }
        this.index = 0;
        this.flipped = false;
        // 結果もリセット
        this.cards.forEach(c => { c.result = null; });
        this.render();
      },

      reset() {
        this.index = 0;
        this.flipped = false;
        this.cards.forEach(c => { c.result = null; });
        this.render();
      },

      render() {
        const card = this.cards[this.index];
        const cardEl = document.getElementById("card");

        // カードの表裏を表示
        document.getElementById("cardText").textContent =
          this.flipped ? card.back : card.front;
        document.getElementById("cardLabel").textContent =
          this.flipped ? "日本語" : "英語";
        cardEl.className = this.flipped ? "flipped" : "";

        // 進捗
        const num = this.index + 1;
        const total = this.cards.length;
        document.getElementById("progress").textContent = `${num} / ${total}`;

        // プログレスバー
        const answered = this.cards.filter(c => c.result !== null).length;
        const percent = Math.round((answered / total) * 100);
        document.getElementById("progressBar").style.width = percent + "%";

        // 統計
        const correct = this.cards.filter(c => c.result === true).length;
        const wrong = this.cards.filter(c => c.result === false).length;
        const remaining = total - correct - wrong;
        document.getElementById("stats").textContent =
          `正解: ${correct}  不正解: ${wrong}  残り: ${remaining}`;
      }
    };

    app.render();
  </script>
</body>
</html>
```

**初心者向けとの違い:**
- 各カードに `result` プロパティを持たせ、個別の正解/不正解を追跡
- `filter()` と `forEach()` で配列を操作（高階関数の活用）
- テンプレートリテラルで文字列を組み立て
- プログレスバーで学習進捗を視覚的に表示
- アロー関数（`=>`）でコールバックを簡潔に記述
