## 課題の要件

1. 設定ファイルから値を読み込むメインスクリプトを作成する
2. システム初期化（ホスト名、タイムゾーン、ディレクトリ構造）を行う
3. ユーザーとグループの作成・パーミッション設定を行う
4. セキュリティ関連の設定（ファイアウォールルール、パーミッション）を行う
5. サービスの設定ファイルを生成する
6. ヘルスチェック（サービス状態、ポート、ディスク）���行う
7. 全体の検証テストスクリプトを作成する

> **注意**: この課題はシミュレーション（simulation）です。実際のサーバーに変更を加えるのではなく、`/tmp/server_sim/` ディレクトリ内で構築作業を模擬します。`sudo` 権限がなくても実行できるように設計してください。

```bash
# ステップ2: ユーザー・グループセットアップ
source "$(dirname "$0")/common.sh"

log "INFO" "ステップ 2/6: ユーザーセットアップ"

GROUP_FILE="$SIM_ROOT/etc/group"

# ユーザー作成
PASSWD_FILE="$SIM_ROOT/etc/passwd"
echo "root:x:0:0:root:/root:/bin/bash" > "$PASSWD_FILE"

USER_CONF="$BASE_DIR/config/users.conf"
while IFS=: read -r username groups description; do
    # コメント行と空行をスキップ
    [[ "$username" =~ ^#.*$ ]] && continue
    [ -z "$username" ] && continue

    uid=$((1000 + RANDOM % 1000))
    home_dir="$SIM_ROOT/home/$username"
    mkdir -p "$home_dir/.ssh"
    chmod 700 "$home_dir/.ssh"

    echo "$username:x:$uid:1001:$description:$home_dir:/bin/bash" >> "$PASSWD_FILE"
    check_ok "ユーザー作成: $username ($groups) - $description"
done < "$USER_CONF"

log "INFO" "ステップ2 完了"
```

```bash
#!/bin/bash
# ステップ3: セキュリティ設定
source "$(dirname "$0")/common.sh"
load_config "server.conf"

log "INFO" "ステップ 3/6: セキュリティ設定"

# パーミッション設定
APP_ROOT="$SIM_ROOT/var/www/$APP_NAME"
LOG_ROOT="$SIM_ROOT/var/log/$APP_NAME"

chmod 755 "$APP_ROOT"
chmod 750 "$LOG_ROOT"
check_ok "パーミッション設定"

# SSH設定ファイル（シミュレーション）
mkdir -p "$SIM_ROOT/etc/ssh"
cat > "$SIM_ROOT/etc/ssh/sshd_config" << EOF
# SSH設定（セキュリティ強化版）
Port $SSH_PORT
PermitRootLogin no
PasswordAuthentication no
PubkeyAuthentication yes
MaxAuthTries 3
ClientAliveInterval 300
ClientAliveCountMax 2
AllowGroups $APP_GROUP $DEV_GROUP
EOF
check_ok "SSH設定"

# ファイアウォールルール
FIREWALL_CONF="$SIM_ROOT/etc/firewall.rules"
echo "# ファイアウォールルール（生成日時: $(date)）" > "$FIREWALL_CONF"
echo "# デフォルトポリシー: DROP" >> "$FIREWALL_CONF"

rule_count=0
while IFS=: read -r port proto description; do
    [[ "$port" =~ ^#.*$ ]] && continue
    [ -z "$port" ] && continue

    echo "ALLOW $proto $port  # $description" >> "$FIREWALL_CONF"
    ((rule_count++))
done < "$BASE_DIR/config/firewall.rules"
check_ok "ファイアウォール設定 ($rule_count ルール)"

log "INFO" "ステップ3 完了"
```

**scripts/04_service_setup.sh:**

```bash
#!/bin/bash
# ステップ4: サービスセットアップ
source "$(dirname "$0")/common.sh"
load_config "server.conf"

log "INFO" "ステップ 4/6: サービスセットアップ"

# Nginx設定の生成
NGINX_CONF="$SIM_ROOT/etc/nginx/sites-available/$APP_NAME"
cat > "$NGINX_CONF" << EOF
# Nginx設定 - $APP_NAME
# 生成日時: $(date '+%Y-%m-%d %H:%M:%S')

server {
    listen $WEB_PORT;
    server_name $HOSTNAME;

    location / {
        proxy_pass http://127.0.0.1:$APP_PORT;
        proxy_set_header Host \$host;
        proxy_set_header X-Real-IP \$remote_addr;
        proxy_set_header X-Forwarded-For \$proxy_add_x_forwarded_for;
    }

    location /static/ {
        alias /var/www/$APP_NAME/public/;
        expires 30d;
    }

    access_log /var/log/$APP_NAME/access.log;
    error_log /var/log/$APP_NAME/error.log;
}
EOF
check_ok "Nginx設定"

# シンボリックリンク（sites-enabled）
ln -sf "$NGINX_CONF" "$SIM_ROOT/etc/nginx/sites-enabled/$APP_NAME"
check_ok "Nginx有効化"

# アプリケーション systemd サービスファイル
mkdir -p "$SIM_ROOT/etc/systemd/system"
cat > "$SIM_ROOT/etc/systemd/system/$APP_NAME.service" << EOF
[Unit]
Description=$APP_NAME Web Application
After=network.target

[Service]
Type=simple
User=$APP_USER
Group=$APP_GROUP
WorkingDirectory=/var/www/$APP_NAME
ExecStart=/usr/bin/node /var/www/$APP_NAME/server.js
Restart=always
RestartSec=5
Environment=NODE_ENV=production
Environment=PORT=$APP_PORT

[Install]
WantedBy=multi-user.target
EOF
check_ok "systemdサービス設定"

log "INFO" "ステップ4 完了"
```

```bash
# 排他ロック
exec 9>"$LOCK_FILE"
flock -n 9 || die "別のセットアップが実行中です"
trap 'flock -u 9; rm -f "$LOCK_FILE"' EXIT

# 設定読み込み
source "$BASE_DIR/config/server.conf" 2>/dev/null || die "server.conf が見つかりません"

# --- ステップ関数（冪等性を保証） ---

step_1_system_init() {
    log "INFO" "ステップ 1/6: システム初期化"

    ensure_dir "$SIM_ROOT/etc"
    ensure_dir "$SIM_ROOT/var/www/$APP_NAME" 755
    ensure_dir "$SIM_ROOT/var/log/$APP_NAME" 750
    ensure_dir "$SIM_ROOT/var/backup"
    ensure_dir "$SIM_ROOT/etc/nginx/sites-available"
    ensure_dir "$SIM_ROOT/etc/nginx/sites-enabled"
    ensure_dir "$SIM_ROOT/etc/systemd/system"
    ensure_dir "$SIM_ROOT/etc/ssh"

    ensure_file "$SIM_ROOT/etc/hostname" "$HOSTNAME"
    ensure_file "$SIM_ROOT/etc/timezone" "$TIMEZONE"

    log "INFO" "  - ホスト名: $HOSTNAME"
    log "INFO" "  - タイムゾーン: $TIMEZONE"
    log "INFO" "  - ディレクトリ構造: OK"
}

step_2_user_setup() {
    log "INFO" "ステップ 2/6: ユーザーセットアップ"

    # グループファイル
    local group_content="root:x:0:\n${APP_GROUP}:x:1001:\n${DEV_GROUP}:x:1002:"
    ensure_file "$SIM_ROOT/etc/group" "$(echo -e "$group_content")"

    # ユーザーファイル
    local passwd_content="root:x:0:0:root:/root:/bin/bash"
    local uid=1000

    while IFS=: read -r username groups description; do
        [[ "$username" =~ ^#.*$ ]] && continue
        [ -z "$username" ] && continue

        ((uid++))
        passwd_content+="\n${username}:x:${uid}:1001:${description}:/home/${username}:/bin/bash"

        ensure_dir "$SIM_ROOT/home/$username/.ssh" 700
        log "INFO" "  - ユーザー: $username ($groups)"
    done < "$BASE_DIR/config/users.conf"

    ensure_file "$SIM_ROOT/etc/passwd" "$(echo -e "$passwd_content")" 644
    log "INFO" "  - ユーザーセットアップ: OK"
}

step_3_security_setup() {
    log "INFO" "ステップ 3/6: セキュリティ設定"

    # SSH設定
    local sshd_config="Port ${SSH_PORT}
PermitRootLogin no
PasswordAuthentication no
PubkeyAuthentication yes
MaxAuthTries 3
ClientAliveInterval 300
ClientAliveCountMax 2
AllowGroups ${APP_GROUP} ${DEV_GROUP}"

    ensure_file "$SIM_ROOT/etc/ssh/sshd_config" "$sshd_config" 600
    log "INFO" "  - SSH設定: OK"

    # ファイアウォール
    local fw_rules="# ���ァイアウォールルール - $(date '+%Y-%m-%d')\n# デフォルトポリシー: DROP\n"
    local rule_count=0

    while IFS=: read -r port proto desc; do
        [[ "$port" =~ ^#.*$ ]] && continue
        [ -z "$port" ] && continue
        fw_rules+="ALLOW ${proto} ${port}  # ${desc}\n"
        ((rule_count++))
    done < "$BASE_DIR/config/firewall.rules"

    ensure_file "$SIM_ROOT/etc/firewall.rules" "$(echo -e "$fw_rules")"
    log "INFO" "  - ファイアウォール: OK ($rule_count ルール)"
}

step_4_service_setup() {
    log "INFO" "ステップ 4/6: サービスセットアップ"

    # Nginx設定
    local nginx_conf="server {
    listen ${WEB_PORT};
    server_name ${HOSTNAME};

    location / {
        proxy_pass http://127.0.0.1:${APP_PORT};
        proxy_set_header Host \\\$host;
        proxy_set_header X-Real-IP \\\$remote_addr;
    }

    location /static/ {
        alias /var/www/${APP_NAME}/public/;
        expires 30d;
    }

    access_log /var/log/${APP_NAME}/access.log;
    error_log /var/log/${APP_NAME}/error.log;
}"

    ensure_file "$SIM_ROOT/etc/nginx/sites-available/$APP_NAME" "$nginx_conf"
    ln -sf "$SIM_ROOT/etc/nginx/sites-available/$APP_NAME" \
           "$SIM_ROOT/etc/nginx/sites-enabled/$APP_NAME"
    log "INFO" "  - Nginx設定: OK"

    # systemd サービス
    local service_unit="[Unit]
Description=${APP_NAME} Web Application
After=network.target

[Service]
Type=simple
User=${APP_USER}
Group=${APP_GROUP}
WorkingDirectory=/var/www/${APP_NAME}
ExecStart=/usr/bin/node /var/www/${APP_NAME}/server.js
Restart=always
RestartSec=5
Environment=NODE_ENV=production PORT=${APP_PORT}

[Install]
WantedBy=multi-user.target"

    ensure_file "$SIM_ROOT/etc/systemd/system/${APP_NAME}.service" "$service_unit"
    log "INFO" "  - systemdサービス: OK"
}

step_5_deploy() {
    log "INFO" "ステップ 5/6: デプロイ"

    local app_root="$SIM_ROOT/var/www/$APP_NAME"

    cat > "$app_root/server.js" << 'APPJS'
const http = require('http');
const port = process.env.PORT || 3000;
const server = http.createServer((req, res) => {
    res.statusCode = 200;
    res.setHeader('Content-Type', 'text/html');
    res.end('<html><body><h1>Hello from MyWebApp!</h1></body></html>');
});
server.listen(port, () => console.log(`Server running on port ${port}`));
APPJS

    cat > "$app_root/package.json" << PKGJSON
{"name":"$APP_NAME","version":"1.0.0","main":"server.js","scripts":{"start":"node server.js"}}
PKGJSON

    ensure_dir "$app_root/public"
    echo "<html><body><h1>Static Page</h1></body></html>" > "$app_root/public/index.html"
    log "INFO" "  - アプリケーション: OK"

    # バックアップスクリプト
    cat > "$SIM_ROOT/var/backup/backup.sh" << 'BACKUP'
#!/bin/bash
DATE=$(date '+%Y%m%d_%H%M%S')
BACKUP_DIR="$(dirname "$0")"
tar -czf "$BACKUP_DIR/${DATE}.tar.gz" -C "$(dirname "$0")/../www" . 2>/dev/null
find "$BACKUP_DIR" -name "*.tar.gz" -mtime +7 -delete 2>/dev/null
BACKUP
    chmod +x "$SIM_ROOT/var/backup/backup.sh"
    log "INFO" "  - バックアップ: OK"
}

step_6_health_check() {
    log "INFO" "ステップ 6/6: ヘルスチェック"
    local errors=0

    # 必須ファイルチェック
    local required_files=(
        "$SIM_ROOT/etc/hostname"
        "$SIM_ROOT/etc/nginx/sites-available/$APP_NAME"
        "$SIM_ROOT/etc/nginx/sites-enabled/$APP_NAME"
        "$SIM_ROOT/etc/systemd/system/${APP_NAME}.service"
        "$SIM_ROOT/etc/ssh/sshd_config"
        "$SIM_ROOT/etc/firewall.rules"
        "$SIM_ROOT/var/www/$APP_NAME/server.js"
        "$SIM_ROOT/var/www/$APP_NAME/package.json"
    )

    for f in "${required_files[@]}"; do
        if [ -f "$f" ]; then
            log "INFO" "  - $(basename "$f"): OK"
        else
            log "ERROR" "  - $(basename "$f"): MISSING"
            ((errors++))
        fi
    done

    # ��ィスク使用量
    local disk_usage
    disk_usage=$(du -sh "$SIM_ROOT" | cut -f1)
    log "INFO" "  - ディスク使用量: $disk_usage"

    local file_count
    file_count=$(find "$SIM_ROOT" -type f | wc -l)
    log "INFO" "  - 総ファイル数: $file_count"

    if [ "$errors" -eq 0 ]; then
        log "INFO" "ヘルスチェック: 全項目 OK"
    else
        log "ERROR" "ヘルスチェック: $errors 件のエラー"
    fi

    return "$errors"
}

# --- テスト関数 ---
run_tests() {
    log "INFO" "===== テスト実行 ====="
    local passed=0 failed=0

    assert_eq() {
        local desc="$1" expected="$2" actual="$3"
        if [ "$expected" = "$actual" ]; then
            echo "  PASS: $desc"
            ((passed++))
        else
            echo "  FAIL: $desc (期待: '$expected', 実際: '$actual')"
            ((failed++))
        fi
    }

    assert_file_exists() {
        local desc="$1" filepath="$2"
        if [ -f "$filepath" ]; then
            echo "  PASS: $desc"
            ((passed++))
        else
            echo "  FAIL: $desc (ファイルなし: $filepath)"
            ((failed++))
        fi
    }

    assert_contains() {
        local desc="$1" filepath="$2" pattern="$3"
        if grep -q "$pattern" "$filepath" 2>/dev/null; then
            echo "  PASS: $desc"
            ((passed++))
        else
            echo "  FAIL: $desc ('$pattern' が見つかりません)"
            ((failed++))
        fi
    }

    # テストケース
    assert_eq "ホスト名" "$HOSTNAME" "$(cat "$SIM_ROOT/etc/hostname")"
    assert_eq "タイムゾーン" "$TIMEZONE" "$(cat "$SIM_ROOT/etc/timezone")"
    assert_file_exists "Nginx設定" "$SIM_ROOT/etc/nginx/sites-available/$APP_NAME"
    assert_file_exists "systemdサービス" "$SIM_ROOT/etc/systemd/system/${APP_NAME}.service"
    assert_file_exists "SSH設定" "$SIM_ROOT/etc/ssh/sshd_config"
    assert_contains "SSH: rootログイン禁止" "$SIM_ROOT/etc/ssh/sshd_config" "PermitRootLogin no"
    assert_contains "SSH: パスワード認証禁止" "$SIM_ROOT/etc/ssh/sshd_config" "PasswordAuthentication no"
    assert_contains "Nginx: プロキシ設定" "$SIM_ROOT/etc/nginx/sites-available/$APP_NAME" "proxy_pass"
    assert_file_exists "アプリケーション" "$SIM_ROOT/var/www/$APP_NAME/server.js"
    assert_contains "ファイアウォール: SSH" "$SIM_ROOT/etc/firewall.rules" "ALLOW tcp 22"
    assert_contains "ファイアウォール: HTTP" "$SIM_ROOT/etc/firewall.rules" "ALLOW tcp 80"

    local perm
    perm=$(stat -c "%a" "$SIM_ROOT/etc/ssh/sshd_config" 2>/dev/null || stat -f "%OLp" "$SIM_ROOT/etc/ssh/sshd_config" 2>/dev/null)
    assert_eq "sshd_config パーミッション" "600" "$perm"

    echo ""
    echo "テスト結果: $passed 成功, $failed 失敗 (合計 $((passed + failed)))"

    return "$failed"
}

# --- メイン ---
main() {
    log "INFO" "===== サーバーセットアップ開始 ====="
    log "INFO" "シミュレーション先: $SIM_ROOT"

    ensure_dir "$SIM_ROOT"

    local steps=(
        step_1_system_init
        step_2_user_setup
        step_3_security_setup
        step_4_service_setup
        step_5_deploy
        step_6_health_check
    )

    local total=${#steps[@]}
    local errors=0

    for ((i = 0; i < total; i++)); do
        local step_num=$((i + 1))

        # 特定ステップのみ実行
        if [ -n "$TARGET_STEP" ] && [ "$step_num" != "$TARGET_STEP" ]; then
            continue
        fi

        echo ""
        if ! "${steps[$i]}"; then
            ((errors++))
        fi
    done

    echo ""
    log "INFO" "===== セットアップ完了 ====="
    log "INFO" "ステップ: $total, エラー: $errors"
    log "INFO" "ログ: $LOG_FILE"

    # テスト実行
    if [ "$RUN_TEST" = true ]; then
        echo ""
        run_tests
    fi

    # ディレクトリ構造を表示
    echo ""
    echo "--- ディレクトリ構造 ---"
    find "$SIM_ROOT" -type f | sort | while read -r f; do
        echo "  $f"
    done

    return "$errors"
}

main
```

**初心者向けとの違い:**
- **冪等性**（idempotency）: `ensure_dir`/`ensure_file` 関数で、何度実行しても同じ結果になる
- **引数による制御**: `--step` で特定ステップのみ実行、`--test` でテスト実行、`--cleanup` で環境削除
- **排他ロック**: `flock` で同時実行を防止
- **テスト機能内蔵**: `assert_eq`/`assert_file_exists`/`assert_contains` でセットアップ結果を自動検証
- **関数型設計**: 各ステップが独立した関数で、メインから配列で呼び出す
- **設定と処理の分離**: 設定ファイルを変更するだけで異なるサーバーに対応可能
