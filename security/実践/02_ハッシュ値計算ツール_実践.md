## ステップガイド

<details>
<summary>ステップ1：hashlibの基本的な使い方を確認する</summary>

`hashlib` はPythonの標準ライブラリです。

```python
import hashlib

# 文字列をバイト列に変換してからハッシュ化します
text = "Hello, World!"
hash_object = hashlib.sha256(text.encode("utf-8"))

# 16進数表記で取得
hex_digest = hash_object.hexdigest()
print(hex_digest)
```

ポイント：`hashlib` に渡すデータは**バイト列**（bytes型）でなければなりません。文字列は `.encode("utf-8")` でバイト列に変換します。

</details>

<details>
<summary>ステップ3：ファイルのハッシュ値を計算する</summary>

ファイルは大きい場合があるので、少しずつ読み込んで更新します。

```python
def hash_file(filepath, algorithm="sha256"):
    h = hashlib.new(algorithm)
    with open(filepath, "rb") as f:
        while True:
            chunk = f.read(4096)
            if not chunk:
                break
            h.update(chunk)
    return h.hexdigest()
```

<summary>解答例（初心者向け）</summary>

```python
# ハッシュ値計算ツール
# 学べる内容：ハッシュ関数、hashlib、完全性の検証
# 実行方法：python hash_tool.py

import hashlib

def calculate_hashes(text):
    """文字列のハッシュ値を3種類計算して表示します。"""
    data = text.encode("utf-8")

    md5_hash = hashlib.md5(data).hexdigest()
    sha1_hash = hashlib.sha1(data).hexdigest()
    sha256_hash = hashlib.sha256(data).hexdigest()

    print("\n--- ハッシュ値 ---")
    print(f"MD5:    {md5_hash}")
    print(f"SHA-1:  {sha1_hash}")
    print(f"SHA-256: {sha256_hash}")


def hash_file(filepath):
    """ファイルのハッシュ値を計算して表示します。"""
    try:
        with open(filepath, "rb") as f:
            data = f.read()

        print(f"\n--- {filepath} のハッシュ値 ---")
        print(f"MD5:    {hashlib.md5(data).hexdigest()}")
        print(f"SHA-1:  {hashlib.sha1(data).hexdigest()}")
        print(f"SHA-256: {hashlib.sha256(data).hexdigest()}")
    except FileNotFoundError:
        print(f"エラー: ファイル '{filepath}' が見つかりません。")


def tamper_detection_demo():
    """改ざん検知のデモを行います。"""
    print("\n--- 改ざん検知デモ ---")
    original = "振込金額は100万円です"
    tampered = "振込金額は999万円です"

    original_hash = hashlib.sha256(original.encode("utf-8")).hexdigest()
    tampered_hash = hashlib.sha256(tampered.encode("utf-8")).hexdigest()

    print(f"元の文字列: {original}")
    print(f"ハッシュ値:  {original_hash}")
    print()
    print(f"改ざん後:   {tampered}")
    print(f"ハッシュ値:  {tampered_hash}")
    print()

    if original_hash == tampered_hash:
        print("結果: ハッシュ値が一致しています。改ざんはありません。")
    else:
        print("結果: ハッシュ値が異なります。改ざんが検知されました！")


# --- メイン処理 ---
print("===== ハッシュ値計算ツール =====")

while True:
    print("\n1. 文字列のハッシュ値を計算する")
    print("2. ファイルのハッシュ値を計算する")
    print("3. 改ざん検知デモ")
    print("4. 終了")

    choice = input("\n選択 (1-4): ")

    if choice == "1":
        text = input("文字列を入力してください: ")
        calculate_hashes(text)

    elif choice == "2":
        filepath = input("ファイルパスを入力してください: ")
        hash_file(filepath)

    elif choice == "3":
        tamper_detection_demo()

    elif choice == "4":
        print("終了します。")
        break

    else:
        print("1〜4の番号を入力してください。")
```

<summary>解答例（改良版 ─ 大きなファイル対応とHMAC）</summary>

```python
# ハッシュ値計算ツール（改良版）
# 学べる内容：チャンク読み込み、HMAC、ハッシュの実務的な使い方
# 実行方法：python hash_tool_v2.py

import hashlib
import hmac
import os

ALGORITHMS = ["md5", "sha1", "sha256"]
CHUNK_SIZE = 8192  # 8KB ずつ読み込む


def calculate_hashes(text):
    """文字列のハッシュ値を複数アルゴリズムで計算します。"""
    data = text.encode("utf-8")
    print("\n--- ハッシュ値 ---")
    for algo in ALGORITHMS:
        h = hashlib.new(algo, data)
        print(f"{algo.upper():<8} {h.hexdigest()}")


def hash_file(filepath):
    """ファイルのハッシュ値をチャンク読み込みで計算します。
    大きなファイルでもメモリを圧迫しません。
    """
    if not os.path.isfile(filepath):
        print(f"エラー: ファイル '{filepath}' が見つかりません。")
        return

    # アルゴリズムごとにハッシュオブジェクトを用意
    hashers = {algo: hashlib.new(algo) for algo in ALGORITHMS}

    with open(filepath, "rb") as f:
        while True:
            chunk = f.read(CHUNK_SIZE)
            if not chunk:
                break
            for h in hashers.values():
                h.update(chunk)

    file_size = os.path.getsize(filepath)
    print(f"\n--- {filepath} ({file_size:,} bytes) のハッシュ値 ---")
    for algo, h in hashers.items():
        print(f"{algo.upper():<8} {h.hexdigest()}")


def tamper_detection_demo():
    """改ざん検知のデモを行います。"""
    print("\n--- 改ざん検知デモ ---")

    original = "振込金額は100万円です"
    tampered = "振込金額は999万円です"

    orig_hash = hashlib.sha256(original.encode("utf-8")).hexdigest()
    tamp_hash = hashlib.sha256(tampered.encode("utf-8")).hexdigest()

    print(f"元の文字列: {original}")
    print(f"SHA-256:    {orig_hash}")
    print(f"\n改ざん後:   {tampered}")
    print(f"SHA-256:    {tamp_hash}")

    match = "一致" if orig_hash == tamp_hash else "不一致"
    print(f"\n判定: {match}")
    if orig_hash != tamp_hash:
        print("→ ハッシュ値が異なるため、改ざんが検知されました。")

    # HMAC のデモ
    print("\n--- HMAC（鍵付きハッシュ）デモ ---")
    secret_key = b"my-secret-key-2024"
    msg = "重要なメッセージ".encode("utf-8")
    mac = hmac.new(secret_key, msg, hashlib.sha256).hexdigest()
    print(f"メッセージ: 重要なメッセージ")
    print(f"HMAC-SHA256: {mac}")
    print("※ HMACは秘密鍵を知らなければ正しい値を計算できません。")
    print("  単純なハッシュと異なり、送信者の認証にも使えます。")


def main():
    print("===== ハッシュ値計算ツール =====")

    while True:
        print("\n1. 文字列のハッシュ値を計算する")
        print("2. ファイルのハッシュ値を計算する")
        print("3. 改ざん検知デモ")
        print("4. 終了")

        choice = input("\n選択 (1-4): ")

        if choice == "1":
            text = input("文字列を入力してください: ")
            calculate_hashes(text)
        elif choice == "2":
            filepath = input("ファイルパスを入力してください: ")
            hash_file(filepath)
        elif choice == "3":
            tamper_detection_demo()
        elif choice == "4":
            print("終了します。")
            break
        else:
            print("1〜4の番号を入力してください。")


if __name__ == "__main__":
    main()
```

**初心者向けとの違い:**
- チャンク読み込みで大きなファイルにも対応 → メモリ効率が良い
- HMAC（Hash-based Message Authentication Code）のデモを追加 → 認証付きハッシュの概念を学べる
- `os.path.isfile()` で事前チェック → より丁寧なエラーハンドリング
- 定数 `ALGORITHMS` でアルゴリズム一覧を管理 → 追加・変更が容易
