## 課題の要件

1. OS情報（ホスト名、OS名、カーネルバージョン、アーキテクチャ、稼働時間）を表示する
2. CPU情報（モデル名、コア数、プロセス数）を表示する
3. メモリ情報（合計、使用中、空き、使用率バー）を表示する
4. ディスク情報（主要パーティションの使用状況とバー）を表示する
5. ネットワーク情報（ホスト名、IPアドレス）を表示する
6. 各セクションを関数として分離する
7. `-o` オプションでファイルに出力できる

## 解答例

<details>
<summary>解答例（初心者向け）</summary>

```bash
#!/usr/bin/env bash
# システム情報レポーター
# 学べる内容：関数、コマンド置換、リダイレクト、パイプ処理
# 実行方法：chmod +x sysinfo.sh && ./sysinfo.sh [-o ファイル名]

# --- オプション解析 ---
output_file=""
if [ "${1:-}" = "-o" ] && [ -n "${2:-}" ]; then
    output_file="$2"
fi

# --- 進捗バー生成 ---
progress_bar() {
    local percent=$1
    local width=20
    local filled=$((percent * width / 100))
    local empty=$((width - filled))
    local bar=""
    for ((i = 0; i < filled; i++)); do bar+="█"; done
    for ((i = 0; i < empty; i++)); do bar+="░"; done
    echo "[${bar}] ${percent}%"
}

# --- ヘッダー ---
show_header() {
    echo ""
    echo "======================================"
    echo "  システム情報レポート"
    echo "  生成日時: $(date '+%Y-%m-%d %H:%M:%S')"
    echo "======================================"
}

# --- OS情報 ---
show_os_info() {
    echo ""
    echo "■ OS情報"
    printf "  %-18s %s\n" "ホスト名:" "$(hostname)"
    printf "  %-18s %s\n" "OS:" "$(uname -s)"
    printf "  %-18s %s\n" "カーネル:" "$(uname -r)"
    printf "  %-18s %s\n" "アーキテクチャ:" "$(uname -m)"

    # 稼働時間
    if command -v uptime > /dev/null 2>&1; then
        local uptime_str
        uptime_str=$(uptime -p 2>/dev/null || uptime | sed 's/.*up //' | sed 's/,.*//')
        printf "  %-18s %s\n" "稼働時間:" "$uptime_str"
    fi
}

# --- CPU情報 ---
show_cpu_info() {
    echo ""
    echo "■ CPU情報"

    if [ -f /proc/cpuinfo ]; then
        local model
        model=$(grep -m1 'model name' /proc/cpuinfo | cut -d: -f2 | sed 's/^ //')
        local cores
        cores=$(grep -c 'processor' /proc/cpuinfo)
        printf "  %-18s %s\n" "モデル名:" "$model"
        printf "  %-18s %s\n" "コア数:" "$cores"
    elif command -v sysctl > /dev/null 2>&1; then
        # macOS
        local model
        model=$(sysctl -n machdep.cpu.brand_string 2>/dev/null || echo "不明")
        local cores
        cores=$(sysctl -n hw.ncpu 2>/dev/null || echo "不明")
        printf "  %-18s %s\n" "モデル名:" "$model"
        printf "  %-18s %s\n" "コア数:" "$cores"
    fi

    local proc_count
    proc_count=$(ps aux 2>/dev/null | wc -l | tr -d '[:space:]')
    printf "  %-18s %s\n" "プロセス数:" "$proc_count"
}

# --- メモリ情報 ---
show_memory_info() {
    echo ""
    echo "■ メモリ情報"

    if command -v free > /dev/null 2>&1; then
        # Linux
        local total used free_mem percent
        total=$(free -m | awk '/^Mem:/ {print $2}')
        used=$(free -m | awk '/^Mem:/ {print $3}')
        free_mem=$(free -m | awk '/^Mem:/ {print $4 + $6}')
        percent=$((used * 100 / total))

        printf "  %-18s %s MB\n" "合計:" "$total"
        printf "  %-18s %s MB (%d%%)\n" "使用中:" "$used" "$percent"
        printf "  %-18s %s MB\n" "空き:" "$free_mem"
        printf "  %s\n" "$(progress_bar "$percent")"
    elif command -v vm_stat > /dev/null 2>&1; then
        # macOS
        local pages_free pages_active page_size total_bytes
        page_size=$(sysctl -n hw.pagesize)
        total_bytes=$(sysctl -n hw.memsize)
        local total_mb=$((total_bytes / 1024 / 1024))
        pages_active=$(vm_stat | awk '/Pages active/ {gsub(/\./,""); print $3}')
        local used_mb=$((pages_active * page_size / 1024 / 1024))
        local free_mb=$((total_mb - used_mb))
        local percent=$((used_mb * 100 / total_mb))

        printf "  %-18s %s MB\n" "合計:" "$total_mb"
        printf "  %-18s %s MB (%d%%)\n" "使用中:" "$used_mb" "$percent"
        printf "  %-18s %s MB\n" "空き:" "$free_mb"
        printf "  %s\n" "$(progress_bar "$percent")"
    fi
}

# --- ディスク情報 ---
show_disk_info() {
    echo ""
    echo "■ ディスク情報"

    df -h 2>/dev/null | awk 'NR>1 && /^\// {
        used = $3
        total = $2
        percent = $5
        gsub(/%/, "", percent)
        mount = $6
        printf "  %-14s %s / %s (%s%%)\n", mount, used, total, percent
    }'
}

# --- ネットワーク情報 ---
show_network_info() {
    echo ""
    echo "■ ネットワーク情報"
    printf "  %-18s %s\n" "ホスト名:" "$(hostname)"

    local ip_addr
    if command -v ip > /dev/null 2>&1; then
        ip_addr=$(ip -4 addr show 2>/dev/null | grep -oP 'inet \K[\d.]+' | grep -v '127.0.0.1' | head -1)
    elif command -v ifconfig > /dev/null 2>&1; then
        ip_addr=$(ifconfig 2>/dev/null | grep 'inet ' | grep -v '127.0.0.1' | awk '{print $2}' | head -1)
    fi
    printf "  %-18s %s\n" "IPアドレス:" "${ip_addr:-不明}"
}

# --- フッター ---
show_footer() {
    echo ""
    echo "======================================"
    echo "  レポート完了"
    echo "======================================"
}

# --- メイン処理 ---
generate_report() {
    show_header
    show_os_info
    show_cpu_info
    show_memory_info
    show_disk_info
    show_network_info
    show_footer
}

if [ -n "$output_file" ]; then
    generate_report > "$output_file"
    echo "レポートを $output_file に保存しました。"
else
    generate_report
fi
```

<summary>解答例（改良版 ─ セクション選択とJSON出力対応）</summary>

```bash
#!/usr/bin/env bash
# システム情報レポーター（改良版）
# セクション選択、JSON出力、閾値アラート対応
# 実行方法：chmod +x sysinfo.sh && ./sysinfo.sh [-o FILE] [-s SECTIONS] [-j]
#   -o FILE     : ファイルに保存
#   -s SECTIONS : 表示セクション（os,cpu,mem,disk,net のカンマ区切り）
#   -j          : JSON形式で出力

set -euo pipefail

# --- デフォルト設定 ---
output_file=""
sections="os,cpu,mem,disk,net"
json_mode=false
WARN_THRESHOLD=80
CRIT_THRESHOLD=90

# --- オプション解析 ---
while getopts ":o:s:j" opt; do
    case "$opt" in
        o) output_file="$OPTARG" ;;
        s) sections="$OPTARG" ;;
        j) json_mode=true ;;
        *) echo "使い方: $0 [-o FILE] [-s os,cpu,mem,disk,net] [-j]" >&2; exit 1 ;;
    esac
done

# --- ユーティリティ ---
progress_bar() {
    local percent=$1
    local width=${2:-20}
    local filled=$((percent * width / 100))
    local empty=$((width - filled))
    local bar=""
    local i
    for ((i = 0; i < filled; i++)); do bar+="█"; done
    for ((i = 0; i < empty; i++)); do bar+="░"; done

    local color=""
    local reset=""
    if [ -t 1 ] && ! $json_mode; then
        reset='\033[0m'
        if [ "$percent" -ge "$CRIT_THRESHOLD" ]; then
            color='\033[31m'  # 赤
        elif [ "$percent" -ge "$WARN_THRESHOLD" ]; then
            color='\033[33m'  # 黄
        else
            color='\033[32m'  # 緑
        fi
    fi
    echo -e "${color}[${bar}] ${percent}%${reset}"
}

# セクションが有効かチェック
section_enabled() {
    [[ ",$sections," == *",$1,"* ]]
}

# --- 各セクション（テキスト出力） ---
show_os_info() {
    echo ""
    echo "■ OS情報"

    local os_name kernel arch
    os_name=$(uname -s)
    kernel=$(uname -r)
    arch=$(uname -m)

    # ディストリビューション情報
    local distro=""
    if [ -f /etc/os-release ]; then
        distro=$(. /etc/os-release && echo "${PRETTY_NAME:-$NAME}")
    fi

    printf "  %-18s %s\n" "ホスト名:" "$(hostname)"
    if [ -n "$distro" ]; then
        printf "  %-18s %s (%s)\n" "OS:" "$os_name" "$distro"
    else
        printf "  %-18s %s\n" "OS:" "$os_name"
    fi
    printf "  %-18s %s\n" "カーネル:" "$kernel"
    printf "  %-18s %s\n" "アーキテクチャ:" "$arch"
}

show_cpu_info() {
    echo ""
    echo "■ CPU情報"

    local model="不明" cores="不明"
    if [ -f /proc/cpuinfo ]; then
        model=$(grep -m1 'model name' /proc/cpuinfo | cut -d: -f2 | sed 's/^ //')
        cores=$(grep -c 'processor' /proc/cpuinfo)
    elif command -v sysctl > /dev/null 2>&1; then
        model=$(sysctl -n machdep.cpu.brand_string 2>/dev/null || echo "不明")
        cores=$(sysctl -n hw.ncpu 2>/dev/null || echo "不明")
    fi

    printf "  %-18s %s\n" "モデル名:" "$model"
    printf "  %-18s %s\n" "コア数:" "$cores"
    printf "  %-18s %s\n" "プロセス数:" "$(ps aux 2>/dev/null | wc -l | tr -d '[:space:]')"
}

show_memory_info() {
    echo ""
    echo "■ メモリ情報"

    local total=0 used=0 percent=0

    if command -v free > /dev/null 2>&1; then
        total=$(free -m | awk '/^Mem:/ {print $2}')
        used=$(free -m | awk '/^Mem:/ {print $3}')
        percent=$((used * 100 / total))
    elif command -v sysctl > /dev/null 2>&1; then
        local total_bytes page_size pages_active
        total_bytes=$(sysctl -n hw.memsize 2>/dev/null || echo 0)
        total=$((total_bytes / 1024 / 1024))
        page_size=$(sysctl -n hw.pagesize 2>/dev/null || echo 4096)
        pages_active=$(vm_stat 2>/dev/null | awk '/Pages active/ {gsub(/\./,""); print $3}')
        used=$((pages_active * page_size / 1024 / 1024))
        [ "$total" -gt 0 ] && percent=$((used * 100 / total))
    fi

    local free_mem=$((total - used))
    printf "  %-18s %s MB\n" "合計:" "$total"
    printf "  %-18s %s MB (%d%%)\n" "使用中:" "$used" "$percent"
    printf "  %-18s %s MB\n" "空き:" "$free_mem"
    printf "  %s\n" "$(progress_bar "$percent")"
}

show_disk_info() {
    echo ""
    echo "■ ディスク情報"

    while IFS= read -r line; do
        local mount used total percent_str
        mount=$(echo "$line" | awk '{print $6}')
        used=$(echo "$line" | awk '{print $3}')
        total=$(echo "$line" | awk '{print $2}')
        percent_str=$(echo "$line" | awk '{print $5}')
        local percent=${percent_str%\%}

        printf "  %-14s %s / %s  %s\n" "$mount" "$used" "$total" "$(progress_bar "$percent")"
    done < <(df -h 2>/dev/null | awk 'NR>1 && /^\// {print}')
}

show_network_info() {
    echo ""
    echo "■ ネットワーク情報"
    printf "  %-18s %s\n" "ホスト名:" "$(hostname)"

    local ip_addr="不明"
    if command -v ip > /dev/null 2>&1; then
        ip_addr=$(ip -4 addr show 2>/dev/null | grep -oP 'inet \K[\d.]+' | grep -v '127.0.0.1' | head -1)
    elif command -v ifconfig > /dev/null 2>&1; then
        ip_addr=$(ifconfig 2>/dev/null | grep 'inet ' | grep -v '127.0.0.1' | awk '{print $2}' | head -1)
    fi
    printf "  %-18s %s\n" "IPアドレス:" "${ip_addr:-不明}"
}

# --- レポート生成 ---
generate_report() {
    echo ""
    echo "======================================"
    echo "  システム情報レポート"
    echo "  生成日時: $(date '+%Y-%m-%d %H:%M:%S')"
    echo "======================================"

    section_enabled "os"   && show_os_info
    section_enabled "cpu"  && show_cpu_info
    section_enabled "mem"  && show_memory_info
    section_enabled "disk" && show_disk_info
    section_enabled "net"  && show_network_info

    echo ""
    echo "======================================"
    echo "  レポート完了"
    echo "======================================"
}

# --- 実行 ---
if [ -n "$output_file" ]; then
    generate_report > "$output_file"
    echo "レポートを $output_file に保存しました。"
else
    generate_report
fi
```

**初心者向けとの違い:**
- `getopts` でオプション解析 → `-o`, `-s`, `-j` の組み合わせに対応
- セクション選択 `-s os,mem` → 必要な情報だけ表示
- 閾値によるバーの色分け → 警告レベル（80%以上で黄、90%以上で赤）
- `/etc/os-release` からディストリビューション情報を取得
- `set -euo pipefail` で堅牢化
- プロセス置換 `< <(df ...)` でディスク情報をパース → サブシェル変数問題を回避
