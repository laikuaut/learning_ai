## 1. 条件分岐の基本

JSXの中では`if文`を直接書けません。代わりに**式（expression）** を使います。

### 方法1：三項演算子

```jsx
function StatusBadge({ isOnline }) {
  return (
    <span style={{
      color: isOnline ? 'green' : 'gray',
      fontWeight: 'bold',
    }}>
      {isOnline ? "オンライン" : "オフライン"}
    </span>
  );
}

// 使い方
<StatusBadge isOnline={true} />   // → オンライン（緑）
<StatusBadge isOnline={false} />  // → オフライン（灰色）
```

### 方法2：&&（論理AND）演算子

条件がtrueの時だけ表示したい場合に使います。

```jsx
function Notification({ count }) {
  return (
    <div>
      <h2>お知らせ</h2>
      {count > 0 && <p>未読メッセージが{count}件あります。</p>}
    </div>
  );
}

// count=3 → 「未読メッセージが3件あります。」が表示される
// count=0 → 何も表示されない
```

> **注意**：`{0 && <p>表示</p>}` は `0` が画面に表示されてしまいます。数値を条件にする場合は `{count > 0 && ...}` のように比較演算子を使いましょう。

### 方法3：早期リターン（if文 + return）

JSXの外でif文を使い、早めに別のUIを返します。

```jsx
function UserProfile({ user }) {
  // ユーザーがいない場合
  if (!user) {
    return <p>ユーザーが見つかりません。</p>;
  }

  // ユーザーがいる場合
  return (
    <div>
      <h2>{user.name}</h2>
      <p>{user.email}</p>
    </div>
  );
}
```

### 方法4：変数に代入してから使う

```jsx
function WeatherIcon({ weather }) {
  let icon;
  let message;

  if (weather === "sunny") {
    icon = "☀️";
    message = "いい天気です！";
  } else if (weather === "cloudy") {
    icon = "☁️";
    message = "曇りです。";
  } else if (weather === "rainy") {
    icon = "🌧️";
    message = "傘を忘れずに。";
  } else {
    icon = "❓";
    message = "天気情報がありません。";
  }

  return (
    <div>
      <span style={{ fontSize: '48px' }}>{icon}</span>
      <p>{message}</p>
    </div>
  );
}
```

### 各方法の使い分け

| 方法 | 使いどころ |
|------|-----------|
| 三項演算子 `? :` | AかBのどちらかを表示する場合 |
| `&&` | 条件を満たす時だけ表示する場合 |
| 早期リターン | コンポーネントの表示内容を大きく切り替える場合 |
| 変数代入 | 条件が3つ以上ある場合 |

## 2. 実践的な条件分岐

### ローディング・エラー・データの3状態

```jsx
import { useState } from 'react';

function DataDisplay() {
  const [status, setStatus] = useState("idle"); // idle, loading, success, error
  const [data, setData] = useState(null);

  const fetchData = () => {
    setStatus("loading");

    // 擬似的なデータ取得（2秒後に成功/失敗をランダムで）
    setTimeout(() => {
      if (Math.random() > 0.3) {
        setData({ message: "データを取得しました！", timestamp: new Date().toLocaleString() });
        setStatus("success");
      } else {
        setStatus("error");
      }
    }, 2000);
  };

  return (
    <div>
      <button onClick={fetchData} disabled={status === "loading"}>
        データを取得
      </button>

      {status === "loading" && <p>読み込み中...</p>}

      {status === "error" && (
        <p style={{ color: 'red' }}>エラーが発生しました。もう一度お試しください。</p>
      )}

      {status === "success" && data && (
        <div style={{ padding: '16px', backgroundColor: '#e8f5e9', borderRadius: '4px' }}>
          <p>{data.message}</p>
          <small>取得時刻：{data.timestamp}</small>
        </div>
      )}
    </div>
  );
}
```

## 3. リストのレンダリング（map）

配列データを表示するには `Array.map()` を使います。

### 基本的なリスト表示

```jsx
function FruitList() {
  const fruits = ["りんご", "みかん", "バナナ", "ぶどう", "メロン"];

  return (
    <ul>
      {fruits.map((fruit, index) => (
        <li key={index}>{fruit}</li>
      ))}
    </ul>
  );
}
```

### オブジェクト配列のリスト表示

```jsx
function UserList() {
  const users = [
    { id: 1, name: "田中太郎", age: 28, role: "エンジニア" },
    { id: 2, name: "鈴木花子", age: 32, role: "デザイナー" },
    { id: 3, name: "佐藤次郎", age: 25, role: "マネージャー" },
  ];

  return (
    <div>
      <h2>メンバー一覧</h2>
      {users.map(user => (
        <div key={user.id} style={{
          padding: '12px',
          margin: '8px 0',
          border: '1px solid #ddd',
          borderRadius: '4px',
        }}>
          <h3>{user.name}</h3>
          <p>年齢：{user.age}歳 / 役職：{user.role}</p>
        </div>
      ))}
    </div>
  );
}
```

## 4. keyの重要性

### keyとは？

`key`はリスト内の各要素を一意に識別するための属性です。Reactが効率的にDOMを更新するために必要です。

```jsx
// keyあり（推奨）：Reactが各要素を正しく追跡できる
{users.map(user => (
  <div key={user.id}>{user.name}</div>
))}
```

### keyのルール

| ルール | 説明 |
|--------|------|
| 一意であること | 兄弟要素の中で重複しないこと |
| 安定していること | レンダリングのたびに変わらないこと |
| IDを使う（推奨） | データのIDが最適 |
| indexは最終手段 | 並び替え・追加・削除があると問題になる |

### indexをkeyにすると起きる問題

```jsx
import { useState } from 'react';

function BadKeyExample() {
  const [items, setItems] = useState(["A", "B", "C"]);

  const addToFront = () => {
    setItems(["New", ...items]);
  };

  return (
    <div>
      <button onClick={addToFront}>先頭に追加</button>
      {/* indexをkeyにすると、先頭に追加した時に
          全ての要素が再マッチングされ、入力内容がずれる */}
      {items.map((item, index) => (
        <div key={index}>
          <span>{item}: </span>
          <input type="text" />
        </div>
      ))}
    </div>
  );
}
```

```jsx
// 正しい方法：一意のIDをkeyにする
function GoodKeyExample() {
  const [items, setItems] = useState([
    { id: 1, text: "A" },
    { id: 2, text: "B" },
    { id: 3, text: "C" },
  ]);
  const [nextId, setNextId] = useState(4);

  const addToFront = () => {
    setItems([{ id: nextId, text: "New" }, ...items]);
    setNextId(nextId + 1);
  };

  return (
    <div>
      <button onClick={addToFront}>先頭に追加</button>
      {items.map(item => (
        <div key={item.id}>
          <span>{item.text}: </span>
          <input type="text" />
        </div>
      ))}
    </div>
  );
}
```

## 5. リストのフィルタリングと並び替え

### フィルタリング

```jsx
import { useState } from 'react';

function ProductFilter() {
  const products = [
    { id: 1, name: "ノートPC", category: "電子機器", price: 89800 },
    { id: 2, name: "マウス", category: "電子機器", price: 3980 },
    { id: 3, name: "デスク", category: "家具", price: 29800 },
    { id: 4, name: "チェア", category: "家具", price: 49800 },
    { id: 5, name: "モニター", category: "電子機器", price: 39800 },
    { id: 6, name: "本棚", category: "家具", price: 19800 },
  ];

  const [selectedCategory, setSelectedCategory] = useState("all");

  // カテゴリでフィルタリング
  const filteredProducts = selectedCategory === "all"
    ? products
    : products.filter(p => p.category === selectedCategory);

  return (
    <div>
      <h2>商品一覧</h2>

      {/* フィルタボタン */}
      <div style={{ marginBottom: '16px' }}>
        <button onClick={() => setSelectedCategory("all")}>すべて</button>
        <button onClick={() => setSelectedCategory("電子機器")}>電子機器</button>
        <button onClick={() => setSelectedCategory("家具")}>家具</button>
      </div>

      <p>{filteredProducts.length}件の商品</p>

      {filteredProducts.map(product => (
        <div key={product.id} style={{
          padding: '8px',
          margin: '4px 0',
          backgroundColor: '#f9f9f9',
        }}>
          <strong>{product.name}</strong>
          <span>（{product.category}）</span>
          <span style={{ float: 'right' }}>
            ¥{product.price.toLocaleString()}
          </span>
        </div>
      ))}
    </div>
  );
}
```

### 並び替え

```jsx
import { useState } from 'react';

function SortableList() {
  const [sortOrder, setSortOrder] = useState("name-asc");

  const items = [
    { id: 1, name: "バナナ", price: 200 },
    { id: 2, name: "りんご", price: 150 },
    { id: 3, name: "みかん", price: 100 },
    { id: 4, name: "ぶどう", price: 400 },
    { id: 5, name: "メロン", price: 800 },
  ];

  const sortedItems = [...items].sort((a, b) => {
    switch (sortOrder) {
      case "name-asc":
        return a.name.localeCompare(b.name);
      case "name-desc":
        return b.name.localeCompare(a.name);
      case "price-asc":
        return a.price - b.price;
      case "price-desc":
        return b.price - a.price;
      default:
        return 0;
    }
  });

  return (
    <div>
      <select value={sortOrder} onChange={(e) => setSortOrder(e.target.value)}>
        <option value="name-asc">名前（昇順）</option>
        <option value="name-desc">名前（降順）</option>
        <option value="price-asc">価格（安い順）</option>
        <option value="price-desc">価格（高い順）</option>
      </select>

      <ul>
        {sortedItems.map(item => (
          <li key={item.id}>
            {item.name} - ¥{item.price}
          </li>
        ))}
      </ul>
    </div>
  );
}
```

## 6. 条件分岐とリストの組み合わせ

### 空リストの処理

```jsx
function TaskList({ tasks }) {
  if (tasks.length === 0) {
    return (
      <div style={{ textAlign: 'center', color: '#999', padding: '40px' }}>
        <p>タスクがありません。</p>
        <p>新しいタスクを追加してみましょう！</p>

  return (
    <ul>

        }}>
          {task.text}
          {task.completed && <span>（完了）</span>}
        </li>
      ))}
    </ul>
  );
}
```

## 7. 実践：検索可能な連絡先リスト

```jsx
import { useState } from 'react';

function ContactList() {
  const contacts = [
    { id: 1, name: "田中太郎", email: "tanaka@example.com", department: "開発部" },
    { id: 2, name: "鈴木花子", email: "suzuki@example.com", department: "デザイン部" },
    { id: 3, name: "佐藤次郎", email: "sato@example.com", department: "営業部" },
    { id: 4, name: "山田美咲", email: "yamada@example.com", department: "開発部" },
    { id: 5, name: "高橋健一", email: "takahashi@example.com", department: "営業部" },
    { id: 6, name: "伊藤さくら", email: "ito@example.com", department: "デザイン部" },
  ];

  const [searchText, setSearchText] = useState("");
  const [filterDept, setFilterDept] = useState("all");

  // 検索とフィルタの適用
  const filteredContacts = contacts.filter(contact => {
    const matchesSearch = contact.name.includes(searchText)
      || contact.email.includes(searchText);
    const matchesDept = filterDept === "all" || contact.department === filterDept;
    return matchesSearch && matchesDept;
  });

  // ユニークな部署一覧を取得
  const departments = [...new Set(contacts.map(c => c.department))];

  return (
    <div style={{ maxWidth: '600px', margin: '0 auto', padding: '20px' }}>
      <h1>連絡先</h1>

      {/* 検索バー */}
      <input
        type="text"
        value={searchText}
        onChange={(e) => setSearchText(e.target.value)}
        placeholder="名前またはメールで検索..."
        style={{ width: '100%', padding: '8px', marginBottom: '8px' }}
      />

      {/* 部署フィルタ */}
      <select
        value={filterDept}
        onChange={(e) => setFilterDept(e.target.value)}
        style={{ padding: '8px', marginBottom: '16px' }}
      >
        <option value="all">すべての部署</option>
        {departments.map(dept => (
          <option key={dept} value={dept}>{dept}</option>
        ))}
      </select>

      {/* 検索結果数 */}
      <p>{filteredContacts.length}件の連絡先</p>

      {/* リスト表示 */}
      {filteredContacts.length === 0 ? (
        <p style={{ color: '#999' }}>該当する連絡先がありません。</p>
      ) : (
        filteredContacts.map(contact => (
          <div key={contact.id} style={{
            padding: '12px',
            margin: '8px 0',
            border: '1px solid #e0e0e0',
            borderRadius: '8px',
          }}>
            <h3 style={{ margin: '0 0 4px 0' }}>{contact.name}</h3>
            <p style={{ margin: '2px 0', color: '#666' }}>{contact.email}</p>
            <span style={{
              backgroundColor: '#e3f2fd',
              padding: '2px 8px',
              borderRadius: '12px',
              fontSize: '12px',
            }}>
              {contact.department}
            </span>
          </div>
        ))
      )}
    </div>
  );
}

export default ContactList;
```

## まとめ

| 項目 | ポイント |
|------|---------|
| 三項演算子 | `条件 ? A : B` で2択の表示を切り替え |
| && 演算子 | `条件 && <JSX>` で条件付き表示 |
| 早期リターン | コンポーネントの表示を大きく切り替える場合に有効 |
| map | `配列.map(item => <JSX>)` でリストを表示 |
| key | リスト要素に一意の識別子を付与（IDが最適） |
| filter | `配列.filter()` で条件に合う要素だけ表示 |
| sort | `[...配列].sort()` で並び替え（元配列はコピーする） |
| 空リスト対応 | `length === 0` で空の場合のUIを表示 |
